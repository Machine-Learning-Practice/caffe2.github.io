68  // NOTE: This function preserves invariants of sparse_dim/dense_dim with respect to
69  // indices and values.
70  //
71  // NOTE: This function supports the following cases:
72  // 1. When we keep the number of dense dimensions unchanged, and NOT shrinking the size of
73  // any of the dense dimensions.
74  // 2. When we keep the number of sparse dimensions unchanged, and NOT shrinking the size of
75  // any of the sparse dimensions.
76  // 3. When the sparse tensor has zero nnz, in which case we are free to change the shapes of
77  // both its sparse and dense dimensions.
78  //
79  // This function DOESN'T support (and will throw an error) the following cases:
80  // 1. When we attempt to change the number of sparse dimensions on a non-empty sparse tensor
81  // (such an operation will invalidate the indices stored).
82  // 2. When we attempt to change the number of dense dimensions on a non-empty sparse tensor
83  // (such an operation will behave differently from an equivalent dense tensor's resize method,
84  // and for API consistency we don't support it).
85  // 3. When we attempt to shrink the size of any of the dense dimensions on a non-empty sparse tensor
86  // (such an operation will behave differently from an equivalent dense tensor's resize method,
87  // and for API consistency we don't support it).
88  // 4. When we attempt to shrink the size of any of the sparse dimensions on a non-empty sparse tensor
89  // (this could make some of the stored indices out-of-bound and thus unsafe).

91  AT_CHECK(allow_tensor_metadata_change(), "resize_ is not allowed on Tensor created from .data or .detach()");
92  AT_CHECK(sparse_dim + dense_dim == static_cast<int64_t>(size.size()), "number of dimensions must be sparse_dim (", sparse_dim, ") + dense_dim (", dense_dim, "), but got ", size.size());
93  if (nnz() > 0) {
94  auto alt_options_msg = "You could try the following options:\n\
95 1. If you need an empty sparse tensor of this size, call `x = torch.sparse_coo_tensor(size)`.\n\
96 2. If you need to resize this tensor, you have the following options:\n\
97  1. For both sparse and dense dimensions, keep the number of them constant and the size of them non-shrinking, and then try the same call again.\n\
98  2. Or, create a new sparse tensor with the correct indices and values from this sparse tensor.";
99 
100  AT_CHECK(sparse_dim == sparse_dim_,
101  "changing the number of sparse dimensions (from ", sparse_dim_, " to ", sparse_dim, ") on a non-empty sparse tensor is not supported.\n", alt_options_msg);
102 
103  AT_CHECK(dense_dim == dense_dim_,
104  "changing the number of dense dimensions (from ", dense_dim_, " to ", dense_dim, ") on a non-empty sparse tensor is not supported.\n", alt_options_msg);

106  bool shrinking_sparse_dims = false;
107  bool shrinking_dense_dim = false;
108  auto sparse_size_original = sizes().slice(0, sparse_dim);
109  auto sparse_size_new = size.slice(0, sparse_dim);
110  for (int64_t i = 0; i < sparse_dim; i++) {
111  if (sparse_size_new[i] < sparse_size_original[i]) {
112  shrinking_sparse_dims = true;
113  break;
114  }
115  }
116  auto dense_size_original = sizes().slice(sparse_dim);
117  auto dense_size_new = size.slice(sparse_dim);
118  for (int64_t i = 0; i < dense_dim; i++) {
119  if (dense_size_new[i] < dense_size_original[i]) {
120  shrinking_dense_dim = true;
121  break;
122  }
123  }
124 
125  AT_CHECK(!shrinking_sparse_dims,
126  "shrinking the size of sparse dimensions (from ", sparse_size_original, " to ", sparse_size_new, ") on a non-empty sparse tensor is not supported.\n", alt_options_msg);
127 
128  AT_CHECK(!shrinking_dense_dim,
129  "shrinking the size of dense dimensions (from ", dense_size_original, " to ", dense_size_new, ") on a non-empty sparse tensor is not supported.\n", alt_options_msg);
130  }

134  std::vector<int64_t> values_size = {nnz};
