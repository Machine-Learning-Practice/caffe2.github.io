211 // Despite its name, this function returns the number of *CUDA* GPUs.
212 static inline size_t getNumGPUs() {
213  // WARNING: DO NOT ADD LOGIC TO HANDLE OTHER DEVICE TYPES TO THIS
214  // FUNCTION. If you are interested in interrogating the number of
215  // devices for a specific device type, add that function to the
216  // relevant library (e.g., similar to at::cuda::device_count())
217  if (hasCUDA() && hasHIP()) {
218  throw std::runtime_error(
219  "Enabling both CUDA and HIP in ATen is not supported, as HIP masquerades "
220  "to be CUDA (e.g., when you say CUDA, on a HIP build of ATen, this actually "
221  "means HIP. Rebuild PyTorch with one or the other disabled.");
222  } else if (hasCUDA()) {
223  return detail::getCUDAHooks().getNumGPUs();
224  } else if (hasHIP()) {
225  return detail::getHIPHooks().getNumGPUs();
226  } else {
227  return 0;
228  }
229 }
