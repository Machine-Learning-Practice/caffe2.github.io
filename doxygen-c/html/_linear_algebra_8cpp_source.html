16 // Helper function for det methods.
17 // For pivoted LU factorization A = P * L * U. Since we always have det(L) = 1,
18 // det(P) = \pm 1, this method returns a 3-tuple:
19 // (det(P), diag(U), info),
20 // where info helps us identify singular matrices.
21 static inline std::tuple<double, Tensor, int> _lu_det_P_diag_U_info(const Tensor& self) {
22  Tensor p, lu, info;
23  std::tie(lu, p, info) = self.unsqueeze(0).btrifact_with_info();
24  p.squeeze_(0);
25  lu.squeeze_(0);
26  int int_info = info.squeeze_().item<int32_t>();
27  AT_CHECK(int_info >= 0, "LU factorization (getrf) failed with info = ", int_info);
28  auto n = self.size(0);
29  auto num_exchanges = (at::arange(1, n + 1, p.type()) != p).nonzero().size(0);
30  if (num_exchanges % 2 == 1) {
31  return std::make_tuple(-1., lu.diag(), int_info);
32  } else {
33  return std::make_tuple(1., lu.diag(), int_info);
34  }
35 }

37 Tensor det(const Tensor& self) {
38  AT_CHECK(at::isFloatingType(self.scalar_type()) &&
39  self.dim() == 2 && self.size(0) == self.size(1),
40  "det(", self.type(), "{", self.sizes(), "}): expected a 2D square tensor "
41  "of floating types");

43  Tensor diag_U;
44  int info;
45  std::tie(det_P, diag_U, info) = _lu_det_P_diag_U_info(self);
46  if (info > 0) {

48  } else {
49  return diag_U.prod().mul_(det_P);
50  }

53 Tensor logdet(const Tensor& self) {
54  AT_CHECK(at::isFloatingType(self.scalar_type()) &&
55  self.dim() == 2 && self.size(0) == self.size(1),
56  "logdet(", self.type(), "{", self.sizes(), "}): expected a 2D square tensor "

58  double det_P;
59  Tensor diag_U, det;
60  int info;
61  std::tie(det_P, diag_U, info) = _lu_det_P_diag_U_info(self);
62  if (info > 0) {
63  det = at::zeros({}, self.type());
64  } else {
65  det = diag_U.prod().mul_(det_P);
66  }
67  if (det.sign().item<double>() <= 0) {
68  return det.log_(); // in order to get proper -inf (det=0) or nan (det<0)
69  } else {
70  return diag_U.abs_().log_().sum();
71  }

212 template <typename scalar_t, bool is_bmm>
213 inline void baddbmm_cpu_kernel(const Tensor& result, const Tensor& self, const Tensor& mat2, Scalar beta_, Scalar alpha_) {
214  int64_t bs = result.size(0);
215  int64_t is = result.size(1);
216  int64_t js = result.size(2);
217  int64_t ks = self.size(2);
218 
219  scalar_t alpha = alpha_.to<scalar_t>();
220  scalar_t beta = beta_.to<scalar_t>();
221 
222  auto r0 = result.accessor<scalar_t, 3>();
223  auto s0 = self.accessor<scalar_t, 3>();
224  auto m0 = mat2.accessor<scalar_t, 3>();
225 
226  int64_t grain_size = std::min(internal::GRAIN_SIZE / (is * js * ks), (int64_t)1);
227  parallel_for(0, bs, grain_size, [&](int64_t b_begin, int64_t b_end) {
228  for (int64_t b = b_begin; b < b_end; b++) {
229  auto r1 = r0[b];
230  auto s1 = s0[b];
231  auto m1 = m0[b];
232  for (int64_t i = 0; i < is; i++) {
233  auto r2 = r1[i];
234  auto s2 = s1[i];
235  for (int64_t j = 0; j < js; j++) {
236  scalar_t &r = r2[j];
237  if (is_bmm) {
238  r = 0;
239  for (int64_t k = 0; k < ks; k++) {
240  r += s2[k] * m1[k][j];
241  }
242  } else {
243  r *= beta;
244  for (int64_t k = 0; k < ks; k++) {
245  r += alpha * s2[k] * m1[k][j];
246  }
247  }
248  }
249  }
250  }
251  });
252 }

254 // This tries to apply some optimizations to bmm/baddbmm:
255 // - When the operand size is small, computation are parallelized over the batch
256 // dimension using OMP and naive matrix multiplication is applied.
257 // - When the operand size is larger than the threshold, if compiled with MKL, MKL's batch gemm is used.
258 // - Otherwise, we use a series of matrix multiplications.
259 // The threshold of 400 for the first has not been thoroughly benchmarked yet and may have room for further
260 // optimization, it likely depends on the characteristics of the CPU, MKL will be different from non-MKL etc.,
261 // but this seems to be a first starting point.
262 
263 static inline Tensor& bmm_out_or_baddbmm_(Tensor& self_or_result, const Tensor& batch1, const Tensor& batch2, Scalar beta, Scalar alpha, bool is_bmm_out) {
264  // is_bmm_out: true for bmm_out, false for baddbmm_
265  // self_or_result is "self" for baddbmm_ and "result" for bmm_out
266  CheckedFrom c = (is_bmm_out ? "bmm" : "baddbmm");
267  TensorArg self_arg(self_or_result, is_bmm_out ? "self" : "result", 0);
268  TensorArg b1_arg(batch1, "batch1", 1);
269  TensorArg b2_arg(batch2, "batch2", 2);
270  checkBackend(c, {self_or_result, batch1, batch2}, Backend::CPU);
271  checkDim(c, b1_arg, 3);
272  checkDim(c, b2_arg, 3);
273 
274  int64_t bs = batch1.size(0);
275  checkSize(c, b2_arg, 0, bs);
276  int64_t contraction_size = batch1.size(2);
277  int64_t res_rows = batch1.size(1);
278  int64_t res_cols = batch2.size(2);
279  checkSize(c, b2_arg, 1, contraction_size);
280 
281  if (is_bmm_out) {
282  self_or_result.resize_({bs, res_rows, res_cols});
283  } else {
284  checkSize(c, self_arg, 0, bs);
285  checkSize(c, self_arg, 1, res_rows);
286  checkSize(c, self_arg, 2, res_cols);
287  }
288 
289  // handle pathological cases that blas may not like
290  if (self_or_result.numel() == 0) {
291  return self_or_result;
292  } else if (contraction_size == 0) {
293  return self_or_result.zero_();
294  }
295 
296  auto batch_items_contiguous_or_transposed = [&](const Tensor& t) {
297  return (t.stride(2) == 1 && t.stride(1) == t.size(2))
298  || (t.stride(1) == 1 && t.stride(2) == t.size(1));
299  };
300 
301  if (contraction_size * res_rows * res_cols < 400) {
302  if (is_bmm_out) {
303  AT_DISPATCH_ALL_TYPES(batch1.scalar_type(), "bmm", [&] {
304  baddbmm_cpu_kernel<scalar_t, true>(self_or_result, batch1, batch2, beta, alpha);
305  });
306  } else {
307  AT_DISPATCH_ALL_TYPES(batch1.scalar_type(), "baddbmm", [&] {
308  baddbmm_cpu_kernel<scalar_t, false>(self_or_result, batch1, batch2, beta, alpha);
309  });
310  }
311  } else if (at::hasMKL() && at::native::is_floating_point(self_or_result)
312  && batch_items_contiguous_or_transposed(batch1)
313  && batch_items_contiguous_or_transposed(batch2)
314  && self_or_result.is_contiguous()) {
315  at::native::_baddbmm_mkl_(self_or_result, batch1, batch2, beta, alpha);
316  } else { // split along batch dimension
317  if (is_bmm_out) {
318  for (int64_t b = 0; b < bs; b++) {
319  auto r = self_or_result.select(0, b);
320  at::native::mm_out(r, batch1.select(0, b), batch2.select(0, b));
321  }
322  } else {
323  for (int64_t b = 0; b < bs; b++) {
324  self_or_result.select(0, b).addmm_(batch1.select(0, b), batch2.select(0, b), beta, alpha);
325  }
326  }
327  }
328  return self_or_result;
329 }

372 /*
373 Matrix product of two Tensors.
374 The behavior depends on the dimensionality of the Tensors as follows:
375 - If both Tensors are 1-dimensional, the dot product (scalar) is returned.
376 - If both arguments are 2-dimensional, the matrix-matrix product is returned.
377 - If the first argument is 1-dimensional and the second argument is 2-dimensional,
378  a 1 is prepended to its dimension for the purpose of the matrix multiply.
379  After the matrix multiply, the prepended dimension is removed.
380 - If the first argument is 2-dimensional and the second argument is 1-dimensional,
381  the matrix-vector product is returned.
382 - If both arguments are at least 1-dimensional and at least one argument is
383  N-dimensional (where N > 2), then a batched matrix multiply is returned. If the first
384  argument is 1-dimensional, a 1 is prepended to its dimension for the purpose of the
385  batched matrix multiply and removed after. If the second argument is 1-dimensional, a
386  1 is appended to its dimension for the purpose of the batched matrix multiple and removed after.
387  The non-matrix (i.e. batch) dimensions are broadcasted (and thus
388  must be broadcastable). For example, if tensor1 is a (j x 1 x n x m) Tensor
389  and tensor2 is a (k x m x p) Tensor, the returned tensor will be an (j x k x n x p) Tensor.
390 */
391 Tensor matmul(
392  c10::optional<Tensor> out_opt,
393  const Tensor& tensor1,
394  const Tensor& tensor2) {
395  auto dim_tensor1 = tensor1.dim();
396  auto dim_tensor2 = tensor2.dim();
397  auto has_out = out_opt.has_value();
398  Tensor out = out_opt.value_or(Tensor());
399 
400  if (dim_tensor1 == 1 && dim_tensor2 == 1) {
401  return has_out ? at::native::dot_out(out, tensor1, tensor2) : tensor1.dot(tensor2);
402  } else if (dim_tensor1 == 2 && dim_tensor2 == 1) {
403  return has_out ? at::native::mv_out(out, tensor1, tensor2) : tensor1.mv(tensor2);
404  } else if (dim_tensor1 == 1 && dim_tensor2 == 2) {
405  return has_out ? at::native::mm_out(out, tensor1.unsqueeze(0), tensor2).squeeze_(0)
406  : tensor1.unsqueeze(0).mm(tensor2).squeeze_(0);
407  } else if (dim_tensor1 == 2 && dim_tensor2 == 2) {
408  return has_out ? at::native::mm_out(out, tensor1, tensor2) : tensor1.mm(tensor2);
409  } else if (dim_tensor1 >= 3 && (dim_tensor2 == 1 || dim_tensor2 == 2)) {
410  // optimization: use mm instead of bmm by folding tensor1's batch into
411  // its leading matrix dimension.
412 
413  Tensor t2 = dim_tensor2 == 1 ? tensor2.unsqueeze(-1) : tensor2;
414  auto size1 = tensor1.sizes();
415  auto size2 = t2.sizes();
416  std::vector<int64_t> output_size;
417  output_size.insert(output_size.end(), size1.begin(), size1.end() - 1);
418  if (dim_tensor2 > 1) {
419  output_size.push_back(size2[dim_tensor2 - 1]);
420  }
421 
422  // fold the batch into the first dimension
423  Tensor t1 = tensor1.contiguous().view({-1, size1[size1.size() - 1]});
424  Tensor output = has_out ? at::_unsafe_view(at::mm_out(out, t1, t2), output_size)
425  : at::_unsafe_view(t1.mm(t2), output_size);
426  return has_out ? out.set_(output) : output;
427  } else if ((dim_tensor1 >= 1 && dim_tensor2 >= 1) && (dim_tensor1 >= 3 || dim_tensor2 >= 3)) {
428  // We are multiplying b1 x n x m1 by x2 x m2 x p (where b1 can be a list);
429  // we track m1 vs m2 separately even though they must match for nicer error messages
430  int64_t n = dim_tensor1 > 1 ? tensor1.size(-2) : 1;
431  int64_t m1 = tensor1.size(-1);
432  IntArrayRef batch_tensor1(tensor1.sizes().data(), std::max<int64_t>(dim_tensor1 - 2, 0));
433  int64_t m2 = dim_tensor2 > 1 ? tensor2.size(-2) : 1;
434  int64_t p = tensor2.size(-1);
435  IntArrayRef batch_tensor2(tensor2.sizes().data(), std::max<int64_t>(dim_tensor2 - 2, 0));
436 
437  // expand the batch portion (i.e. cut off matrix dimensions and expand rest)
438  std::vector<int64_t> expand_batch_portion = infer_size(batch_tensor1, batch_tensor2);
439 
440  std::vector<int64_t> tensor1_expand_size(expand_batch_portion);
441  tensor1_expand_size.insert(tensor1_expand_size.end(), {n, m1});
442 
443  std::vector<int64_t> tensor2_expand_size(expand_batch_portion);
444  tensor2_expand_size.insert(tensor2_expand_size.end(), {m2, p});
445 
446  int expand_batch_product = std::accumulate(expand_batch_portion.begin(), expand_batch_portion.end(),
447  1, std::multiplies<int64_t>());
448 
449  std::vector<int64_t> tensor1_bmm_view({expand_batch_product});
450  tensor1_bmm_view.insert(tensor1_bmm_view.end(), {n, m1});
451 
452  std::vector<int64_t> tensor2_bmm_view({expand_batch_product});
453  tensor2_bmm_view.insert(tensor2_bmm_view.end(), {m2, p});
454 
455  // flatten expanded batches
456  Tensor tensor1_expanded = tensor1.expand(tensor1_expand_size).contiguous().view(tensor1_bmm_view);
457  Tensor tensor2_expanded = tensor2.expand(tensor2_expand_size).contiguous().view(tensor2_bmm_view);
458 
459  // reshape batches back into result
460  std::vector<int64_t> output_shape(expand_batch_portion);
461  if (dim_tensor1 > 1) {
462  output_shape.push_back(n);
463  }
464  if (dim_tensor2 > 1) {
465  output_shape.push_back(p);
466  }
467 
468  Tensor output = has_out ? at::_unsafe_view(at::bmm_out(out, tensor1_expanded, tensor2_expanded), output_shape)
469  : at::_unsafe_view(tensor1_expanded.bmm(tensor2_expanded), output_shape);
470 
471  return has_out ? out.set_(output) : output;
472  }
473 
474  AT_ERROR("both arguments to matmul need to be at least 1D, but they are ",
475  dim_tensor1, "D and ", dim_tensor2, "D");
476 }

487 Tensor matrix_power(const Tensor& a, int64_t n) {
488  AT_CHECK(a.dim() >= 2 && at::isFloatingType(a.scalar_type()),
489  "matrix_power(", a.type(), "{", a.sizes(), "}): expected a tensor "
490  "of floating types with dim at least 2");
491  if (n == 0) {
492  return a.clone().copy_(at::eye(a.size(-2), a.options()).expand_as(a));
493  } else if (n < 0) {
494  Tensor a_ = at::inverse(a);
495  n *= -1;
496  return at::native::matrix_power(a_, n);
497  } else if (n == 1) {
498  return a.clone();
499  } else if (n == 2) {
500  return at::native::matmul(a, a);
501  } else if (n == 3) {
502  return at::native::matmul(at::native::matmul(a, a), a);
503  }
504 
505  // This is a binary decomposition of n.
506  // Moving from the least significant bit to the most significant bit
507  // This is done to reduce the number of matrix multiplications
508  // by raising the input matrix in powers of 2
509  // The total number of matrix multiplications are
510  // number of bits + number of bits that equal 1 ~ O(log n)
511  // instead of O(n)
512  Tensor result, z;
513  int64_t r;
514  while (n > 0) {
515  z = (!z.defined()) ? a.clone() : at::native::matmul(z, z);
516  r = n % 2;
517  n = n / 2;
518  if (r == 1) {
519  result = (!result.defined()) ? z.clone() : at::native::matmul(result, z);
520  }
521  }
522  return result;
523 }

582 // Why the separate implementation for 3 matrices?
583 // The logic for three matrices is much faster when done directly
584 // Requires 1 comparison to 4 comparisons and lesser arithmetic operations
585 static inline Tensor _chain_matmul_three_matrices(TensorList matrices) {
586  int64_t a = matrices[0].size(0); // This is the first dimension
587  int64_t b = matrices[1].size(0); // This is the common dimension between the first two matrices
588  int64_t c = matrices[2].size(0); // This is the common dimension between the last two matrices
589  int64_t d = matrices[2].size(1); // This is the last dimension
590 
591  // The matrices are of size (a x b), (b x c), (c x d)
592  // cost_1 is the cost of parenthesizing (a x b) and (b x c) and then combining (c x d)
593  // cost_2 is the cost of parenthesizing (b x c) and (c x d) and then combining (a x b)
594  int64_t cost_1 = (a * c) * (b + d);
595  int64_t cost_2 = (b * d) * (a + c);
596 
597  if (cost_1 > cost_2) {
598  return at::mm(matrices[0], at::mm(matrices[1], matrices[2]));
599  } else {
600  return at::mm(at::mm(matrices[0], matrices[1]), matrices[2]);
601  }
602 }

604 Tensor chain_matmul(TensorList matrices) {
605  checkAllSameDim(matrices, 2);
606 
607  if (matrices.size() == 1) {
608  return matrices[0];
609  } else if (matrices.size() == 2) {
610  return at::mm(matrices[0], matrices[1]);
611  } else if (matrices.size() == 3) {
612  return _chain_matmul_three_matrices(matrices);
613  } else {
614 
615  // Following the algorithm in Chapter 15.2 : Introduction to Algorithms, Cormen et al.
616  // Minor modifications have been made to accommodate zero-indexing
617  auto n = matrices.size();
618 
619  // Dim vector - the length of which is n + 1. Note that for matrix multiplication, there
620  // needs to a common dimension between the multiplicands, hence for n matrices, there are
621  // n + 1 values. The values p_{i} and p_{i + 1} correspond to the dimensions of matrix i in
622  // the chain (zero-indexed)
623  std::vector<int64_t> p;
624  p.push_back(matrices[0].size(0));
625  for (int64_t i = 0; i < n; i++) {
626  p.push_back(matrices[i].size(1));
627  }
628 
629  // Cost matrix - an element m[i, j] of this matrix corresponds to the minimum cost of
630  // parenthesizing matrices A_{i} to A_{j}. By this definition m[i, i] = 0 for all i
631  // m[i, j] is filled using the substructure property of the algorithm, meaning:
632  // m[i, j] = min_{i <= k < j} m[i, k] + m[k, j] + p_{i-1}p_{k}p_{j}
633  std::vector<std::vector<int64_t>> m(n, std::vector<int64_t>(n, 0));
634 
635  // Auxiliary table for constructing the order
636  // s[i, j] stores the index k at which the optimal split is obtained
637  std::vector<std::vector<int64_t>> s(n, std::vector<int64_t>(n));
638 
639  // j and q are used repetitively in the algorithm below
640  int64_t j, q;
641 
642  for (int64_t l = 1; l < n; l++) {
643  for (int64_t i = 0; i < n - l; i++) {
644  j = i + l;
645  m[i][j] = std::numeric_limits<int64_t>::max();
646  for (int64_t k = i; k < j; k++) {
647  q = m[i][k] + m[k + 1][j] + p[i] * p[k + 1] * p[j + 1];
648  if (q < m[i][j]) {
649  m[i][j] = q;
650  s[i][j] = k;
651  }
652  }
653  }
654  }
655 
656  // We use the result from the algorithm to compute the matrix chain product via recursion
657  return _chain_matmul_general(matrices, s, 0, n - 1);
658  }
659 }
