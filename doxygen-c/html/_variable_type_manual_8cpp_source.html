100 // WARNING: YOU MUST DO THE NEXT TWO STATIC INITIALIZERS IN THIS ORDER.
101 //
102 // If you do it in the other order, this is what can happen if
103 // these static initializers are called before Context is
104 // initialized:
105 //
106 // - VariableHooks::registerVariableTypeFor will be activated
107 // to register a variable type
108 //
109 // - We run the constructor of VariableTypeRegistry, which
110 // calls at::globalContext()
111 //
112 // - Context is not initialized yet, so we call the constructor
113 // of Context
114 //
115 // - We register CPU types, calling VariableHooks::registerVariableTypeFor
116 //
117 // - We register the CPU type as a variable type
118 //
119 // - In VariableTypeRegistry, we try to register the Variable type AGAIN!!
120 // Disaster.
121 //
122 static VariableTypeRegistry registry;

238 Tensor & VariableType::s_copy_(Tensor & self, const Tensor & src, bool non_blocking) const {
239  jit::Value* output = nullptr;
240  if(torch::jit::tracer::isTracing()) {
241  const jit::tracer::TracingState& state = *jit::tracer::getTracingState();
242  auto& graph = state.graph;
243  if (state.force_outplace) {
244  // if you have no views of self, then an in place copy is equivalent to
245  // making sure we expand src to the same size as self
246  jit::Node* node = graph->create(jit::aten::expand_as, /*num_outputs=*/1);
247  jit::tracer::addInputs(node, "src", src);
248  jit::tracer::addInputs(node, "self", self);
249  graph->insertNode(node);
250  jit::tracer::ensureUniqueIfOutOfPlaced("copy_ (possibly due to an assignment)", self);
251  output = node->output();
252  } else {
253  output = graph->insert(
254  jit::aten::copy_,
255  {jit::tracer::getValueTrace(self), jit::tracer::getValueTrace(src)});
256  }
257  }
258  // TODO: once copy is exposed in Declarations.yaml we may be able to bind
259  // it automatically
260  auto& self_ = unpack(self, "self", 0);
261  auto& src_ = unpack(src, "src", 1);
262  check_inplace(self);
263  std::shared_ptr<CopyBackwards> grad_fn;
264  auto requires_grad = compute_requires_grad(self, src);
265  requires_grad &= isFloatingPoint(self.scalar_type());
266  if (requires_grad) {
267  grad_fn = std::make_shared<CopyBackwards>();
268  grad_fn->set_next_edges(collect_next_edges(self, src));
269  grad_fn->src_type = &src.type();
270  grad_fn->src_device = src.device();
271  }
272  {
273  at::AutoNonVariableTypeMode non_var_type_mode(true);
274  if (self.is_sparse() && src.is_sparse()) baseType->copy_sparse_to_sparse_(self_, src_, non_blocking);
275  else if (!self.is_sparse() && !src.is_sparse()) baseType->s_copy_(self_, src_, non_blocking);
276  else AT_ERROR("copy_() between dense and sparse Tensors is not implemented! Found self type = ", self.type(), " and src type = ", src.type());
277  }
278  increment_version(self);
279  rebase_history(as_variable_ref( self ), std::move(grad_fn));
280  if(torch::jit::tracer::isTracing()) {
281  jit::tracer::setOutput(output, self);
282  }
283  return self;
284 }
