19  LongTensor _to_csr(const int64_t* indices, int64_t dim, int64_t nnz) {
20  int64_t h, i, hp0, hp1;
21  LongTensor csr = native::zeros({dim + 1}, kLong);
22 
23  // TODO: eliminate this conditional when zero-size dims supported correctly
24  if (nnz > 0) {
25  auto csr_accessor = csr.accessor<int64_t, 1>();
26  // Convert the sparse matrix to CSR format
27 #pragma omp parallel for private(i, h, hp0, hp1) schedule(static) if (nnz > 10000)
28  for (i=0; i<nnz; i++) {
29  hp0 = indices[i];
30  hp1 = (i+1 == nnz) ? dim : indices[i+1];
31  if (hp0 != hp1) for (h = hp0; h < hp1; h++) {
32  csr_accessor[h+1] = i+1;
33  }
34  }
35  }
36  return csr;
37  }

188 SparseTensor& add_out_sparse_cpu(SparseTensor& r, const SparseTensor& t, const SparseTensor& src, Scalar value) {
189  AT_ASSERT(r.is_sparse());
190  AT_ASSERT(t.is_sparse());
191  AT_ASSERT(!t.is_cuda()); // the dispatch argument
192  AT_CHECK(!r.is_cuda(), "add: expected 'out' to be CPU tensor, but got CUDA tensor");
193  AT_CHECK(!src.is_cuda(), "add: expected 'other' to be a CPU tensor, but got a CUDA tensor");
194 
195  AT_CHECK(t.sizes().equals(src.sizes()), "add: expected sizes of 'self' and 'other' to match, but ", t.sizes(), " != ", src.sizes());
196 
197  if (src._nnz() == 0) {
198  return copy_sparse_to_sparse_(r, t);
199  }
200  if (t._nnz() == 0) {
201  return mul_out_sparse_scalar(r, src, value);
202  }
203 
204  AT_CHECK(is_same_density(t, src), "add: expected 'self' and 'other' to have same density, but 'self' has ", t.sparse_dim(), " sparse dimensions while 'other' has ", src.sparse_dim(), " sparse dimensions");
205 
206  // saving those because they can be overwritten when doing in-place operations
207  int64_t t_nnz = t._nnz(), s_nnz = src._nnz(), max_nnz = t_nnz + s_nnz;
208  bool t_coalesced = t.is_coalesced(), s_coalesced = src.is_coalesced();
209  int64_t sparse_dim = src.sparse_dim();
210  LongTensor t_indices = t._indices();
211  Tensor t_values = t._values();
212  LongTensor src_indices = src._indices();
213  Tensor s_values = src._values();
214  LongTensor r_indices = at::empty({sparse_dim, max_nnz}, t_indices.options());
215  Tensor r_values = new_values_with_size_of(s_values, max_nnz).zero_();
216  r.resize_as_(src);
217  get_sparse_impl(r)->set_indices_and_values_unsafe(r_indices, r_values);
218 
219  int64_t blockSize = r_values.stride(0);
220  int64_t cmp, d;
221  int64_t r_i = 0, t_i = 0, s_i = 0;
222 
223  // NB: relies on nnz tests above
224  auto t_indices_accessor = t_indices.accessor<int64_t, 2>();
225  auto r_indices_accessor = r_indices.accessor<int64_t, 2>();
226  auto src_indices_accessor = src_indices.accessor<int64_t, 2>();
227 
228  AT_DISPATCH_ALL_TYPES(
229  t_values.scalar_type(), "cadd_sparse", [&] {
230  scalar_t* t_values_ptr = t_values.data<scalar_t>();
231  scalar_t* s_values_ptr = s_values.data<scalar_t>();
232  scalar_t* r_values_ptr = r_values.data<scalar_t>();
233  scalar_t cast_value = value.to<scalar_t>();
234  while (t_i < t_nnz || s_i < s_nnz) {
235  if (t_i >= t_nnz) {
236  cmp = -1;
237  } else if (s_i >= s_nnz) {
238  cmp = 1;
239  } else {
240  cmp = 0;
241  for (d = 0; d < sparse_dim; d++) {
242  if (t_indices_accessor[d][t_i] < src_indices_accessor[d][s_i]) {
243  cmp = 1;
244  break;
245  }
246  if (t_indices_accessor[d][t_i] > src_indices_accessor[d][s_i]) {
247  cmp = -1;
248  break;
249  }
250  }
251  }
252  if (cmp >= 0) {
253  for (d = 0; d < sparse_dim; d++) {
254  r_indices_accessor[d][r_i] = t_indices_accessor[d][t_i];
255  }
256  if (t_values.numel() > 0) { // We add all elements from t_values to r_values only if t_values is not an empty tensor
257  THBlas_axpy<scalar_t>(blockSize, 1,
258  t_values_ptr + t_i * blockSize, 1,
259  r_values_ptr + r_i * blockSize, 1);
260  }
261  t_i++;
262  }
263  if (cmp <= 0) {
264  for (d = 0; d < sparse_dim; d++) {
265  r_indices_accessor[d][r_i] = src_indices_accessor[d][s_i];
266  }
267  if (s_values.numel() > 0) { // We add all elements from s_values to r_values only if s_values is not an empty tensor
268  THBlas_axpy<scalar_t>(blockSize, cast_value,
269  s_values_ptr + s_i * blockSize, 1,
270  r_values_ptr + r_i * blockSize, 1);
271  }
272  s_i++;
273  }
274  r_i++;
275  }
276  }
277  );
278 
279  get_sparse_impl(r)->set_nnz_and_narrow(r_i);
280  // TODO: I think it may be possible to track inside the loop and
281  // detect when we are uncoalesced (e.g., by observing that an
282  // index goes backwards) which may be more precise than using the
283  // coalesced flag here. But this is easy.
284  return r._coalesced_(t_coalesced && s_coalesced);
285 }

309  }

361 SparseTensor& mul_out_sparse_cpu(SparseTensor& r, const Tensor& t_, const Tensor& src_) {
362  if (src_.dim() == 0) {
363  return mul_out_sparse_zerodim(r, t_, src_);
364  } else if (t_.dim() == 0) {
365  return mul_out_sparse_zerodim(r, src_, t_);
366  }
367 
368  AT_CHECK(t_.sizes().equals(src_.sizes()), "mul operands have incompatible sizes");
369  AT_ASSERT(!t_.is_cuda()); // dispatch argument
370  AT_CHECK(!r.is_cuda(), "mul: expected 'out' to be CPU tensor, but got CUDA tensor");
371  AT_CHECK(!src_.is_cuda(), "mul: expected 'other' to be a CPU tensor, but got a CUDA tensor");
372 
373  AT_CHECK(t_.sizes().equals(src_.sizes()), "mul: expected 'self' and 'other' to have same sizes, but ", t_.sizes(), " != ", src_.sizes());
374 
375  if (src_._nnz() == 0 || t_._nnz() == 0) {
376  r.resize_as_(src_);
377  return r.zero_();
378  }
379 
380  SparseTensor t = t_.coalesce();
381  SparseTensor src = src_.coalesce();
382 
383  // saving those because they can be overwritten when doing in-place operations
384  int64_t t_nnz = t._nnz(), s_nnz = src._nnz();
385  int64_t max_nnz = std::min(t_nnz, s_nnz); // multiply by zero is zero, and can be dropped
386  int64_t sparse_dim = src.sparse_dim();
387  LongTensor t_indices = t._indices();
388  Tensor t_values = t._values();
389  LongTensor src_indices = src._indices();
390  Tensor s_values = src._values();
391  LongTensor r_indices = at::empty({sparse_dim, max_nnz}, t_indices.options());
392  Tensor r_values = new_values_with_size_of(t_values, max_nnz).zero_();
393  r.resize_as_(src);
394  get_sparse_impl(r)->set_indices_and_values_unsafe(r_indices, r_values);
395 
396  int64_t match, d;
397  int64_t r_i = 0, t_i = 0, s_i = 0;
398 
399  // NB: relies on nnz test above
400  auto t_indices_accessor = t_indices.accessor<int64_t, 2>();
401  auto r_indices_accessor = r_indices.accessor<int64_t, 2>();
402  auto src_indices_accessor = src_indices.accessor<int64_t, 2>();
403 
404  // Check if we can find matching indices, and if so, write an
405  // entry to the result indices vector. Returns true if matching
406  // indices were found.
407  auto index_preamble = [&]() {
408  match = 1;
409  for (d = 0; d < sparse_dim; d++) {
410  if (t_indices_accessor[d][t_i] < src_indices_accessor[d][s_i]) {
411  t_i++;
412  match = 0;
413  break;
414  }
415  if (t_indices_accessor[d][t_i] > src_indices_accessor[d][s_i]) {
416  s_i++;
417  match = 0;
418  break;
419  }
420  }
421  if (!match) return false;
422  for (d = 0; d < sparse_dim; d++) {
423  r_indices_accessor[d][r_i] = t_indices_accessor[d][t_i];
424  }
425  return true;
426  };
427 
428  if (t_values.dim() > 1) {
429  while (t_i < t_nnz && s_i < s_nnz) {
430  if (!index_preamble()) continue;
431  r_values.select(0, r_i).addcmul_(t_values.select(0, t_i), s_values.select(0, s_i));
432  r_i++;
433  t_i++;
434  s_i++;
435  }
436  } else {
437  AT_DISPATCH_ALL_TYPES(
438  r_values.scalar_type(), "mul_out_sparse", [&] {
439  auto r_accessor = r_values.accessor<scalar_t, 1>();
440  auto t_accessor = t_values.accessor<scalar_t, 1>();
441  auto s_accessor = s_values.accessor<scalar_t, 1>();
442 
443  while (t_i < t_nnz && s_i < s_nnz) {
444  if (!index_preamble()) continue;
445  r_accessor[r_i] = t_accessor[t_i] * s_accessor[s_i];
446  r_i++;
447  t_i++;
448  s_i++;
449  }
450  }
451  );
452  }
453 
454  get_sparse_impl(r)->set_nnz_and_narrow(r_i);
455  return r._coalesced_(true);
456 }

465 template <typename scalar_t>
466 void s_addmm_out_sparse_dense_worker(int64_t nnz, int64_t dim_i, int64_t dim_j, int64_t dim_k, Tensor& r, Scalar beta, const Tensor& t, Scalar alpha, const Tensor& indices, const Tensor& values, const Tensor& dense) {
467  int64_t i;
468 
469  // r_ = alpha * sparse * dense
470  scalar_t cast_alpha = alpha.to<scalar_t>();
471  scalar_t cast_beta = beta.to<scalar_t>();
472  if (cast_beta == 0) {
473  r.zero_();
474  } else if (cast_beta == 1) {
475  if (!is_same_tensor(r, t)) {
476  r.copy_(t);
477  }
478  } else {
479  at::mul_out(r, t, scalar_to_tensor(beta));
480  }
481 
482  auto indices_accessor = indices.accessor<int64_t, 2>();
483 
484  auto values_accessor = values.accessor<scalar_t, 1>();
485  scalar_t* dense_ptr = dense.data<scalar_t>();
486  scalar_t* r_ptr = r.data<scalar_t>();
487 
488  int64_t dense_stride0 = dense.stride(0);
489  int64_t dense_stride1 = dense.stride(1);
490  int64_t r_stride0 = r.stride(0);
491  int64_t r_stride1 = r.stride(1);
492  for (i = 0; i < nnz; i++) {
493  scalar_t val = values_accessor[i];
494  int64_t row = indices_accessor[0][i];
495  int64_t col = indices_accessor[1][i];
496  if (col >= 0 && col < dim_j && row >= 0 && row < dim_i) {
497  THBlas_axpy<scalar_t>(dim_k,
498  cast_alpha * val,
499  dense_ptr + col * dense_stride0, dense_stride1,
500  r_ptr + row * r_stride0, r_stride1);
501  } else {
502  if (col < 0 || col >= dim_j) {
503  AT_ERROR("addmm: index out of column bound: ", col, " not between 1 and ", dim_j);
504  } else {
505  AT_ERROR("addmm: index out of row bound: ", row, " not between 1 and ", dim_i);
506  }
507  }
508  }

689 
690 SparseTensor& _sspaddmm_out_cpu(
691  SparseTensor& r,
692  const SparseTensor& t,
693  const SparseTensor& sparse_,
694  const Tensor& dense,
695  Scalar beta,
696  Scalar alpha
697 ) {
698  AT_ASSERT(!t.is_cuda()); // dispatch argument
699  AT_CHECK(!r.is_cuda(), "sspaddmm: expected 'out' to be CPU tensor, but got CUDA tensor");
700  AT_CHECK(!sparse_.is_cuda(), "sspaddmm: expected 'mat1' to be a CPU tensor, but got a CUDA tensor");
701  AT_CHECK(!dense.is_cuda(), "sspaddmm: expected 'mat2' to be a CPU tensor, but got a CUDA tensor");
702 
703  AT_CHECK(sparse_.sparse_dim() == 2,
704  "sspaddmm: Argument #2: matrices expected, got ", sparse_.sparse_dim(), "D tensor");
705  AT_CHECK(sparse_.dense_dim() == 0,
706  "sspaddmm: Argument #2: scalar values expected, got ", sparse_.dense_dim(), "D values");
707  AT_CHECK(dense.dim() == 2,
708  "sspaddmm: Argument #2: matrices expected, got ", dense.dim(), "D tensor");
709 
710  SparseTensor sparse = sparse_.coalesce();
711 
712  // ixj * jxk = ixk
713  int64_t dim_i = sparse.size(0);
714  int64_t dim_j = sparse.size(1);
715  int64_t dim_k = dense.size(1);
716 
717  // NB: This has to occur before the checks, because r may alias t.
718  // See test_saddmm
719  get_sparse_impl(r)->raw_resize_(2, 0, {dim_i, dim_k});
720 
721  AT_CHECK(dense.size(0) == dim_j,
722  "sspaddmm: Argument #3: Expected dim 0 size ", dim_j, ", got ", dense.size(0));
723  AT_CHECK(t.size(0) == dim_i,
724  "sspaddmm: Argument #1: Expected dim 0 size ", dim_i, ", got ", t.size(0));
725  AT_CHECK(t.size(1) == dim_k,
726  "sspaddmm: Argument #1: Expected dim 1 size ", dim_k, ", got ", t.size(1));
727 
728  int64_t nnz = sparse._nnz();
729  LongTensor indices = sparse._indices();
730  Tensor values = sparse._values();
731 
732  LongTensor csr = _to_csr(indices.data<int64_t>(), dim_i, nnz);
733 
734  int64_t t_nnz = t._nnz();
735  int64_t r_nnz = nnz * dim_k + t_nnz;
736  LongTensor newi = at::empty({2, r_nnz}, kLong);
737  LongTensor newv = native::zeros({r_nnz}, values.options());
738 
739  if (t_nnz != 0) {
740  LongTensor narrowi = newi.narrow(1, 0, t_nnz);
741  Tensor narrowv = newv.narrow(0, 0, t_nnz);
742 
743  narrowi.copy_(t._indices());
744  narrowv.copy_(t._values());
745  newv.mul_(beta);
746  }
747 
748  // sparse = sparse * dense
749  int64_t p = t_nnz;
750 
751  auto csr_accessor = csr.accessor<int64_t, 1>();
752  auto indices_accessor = indices.accessor<int64_t, 2>();
753  auto newi_accessor = newi.accessor<int64_t, 2>();
754 
755  int64_t dense_stride0 = dense.stride(0);
756  int64_t dense_stride1 = dense.stride(1);
757  int64_t newv_stride0 = newv.stride(0);
758 
759  AT_DISPATCH_ALL_TYPES(
760  values.scalar_type(), "sspmm", [&] {
761  auto values_accessor = values.accessor<scalar_t, 1>();
762  scalar_t* dense_ptr = dense.data<scalar_t>();
763  scalar_t* newv_ptr = newv.data<scalar_t>();
764  scalar_t cast_alpha = alpha.to<scalar_t>();
765 
766  for (int64_t h = 0; h < dim_i; h++) {
767  int64_t i_start = csr_accessor[h];
768  int64_t i_end = csr_accessor[h+1];
769  for (int64_t i = i_start; i < i_end; i++) {
770  scalar_t val = values_accessor[i];
771  int64_t col = indices_accessor[1][i];
772  if (col >= 0 && col < dim_j) {
773  THBlas_axpy<scalar_t>(dim_k,
774  cast_alpha * val,
775  dense_ptr + col * dense_stride0, dense_stride1,
776  newv_ptr + p * newv_stride0, 1);
777  } else {
778  AT_ERROR("index out of bound. sspmm: ", col, " not between 1 and ", dim_j);
779  }
780  }
781  // Fill up the indices with the right values
782  if (i_start != i_end) {
783  for (int64_t i = 0; i < dim_k; i++) {
784  newi_accessor[0][p+i] = h;
785  newi_accessor[1][p+i] = i;
786  }
787  p += dim_k;
788  }
789  }
790  }
791  );
792 
793  // to avoid a clone
794  get_sparse_impl(r)->set_indices_and_values_unsafe(newi, newv);
795  get_sparse_impl(r)->set_nnz_and_narrow(p);
796 
797  return r;
798 }

844

847  auto dims_to_sum_v = dims_to_sum.vec();
848  maybe_wrap_dims(dims_to_sum_v, input_dim);

852  IntArrayRef sizes = input.sizes();
853  const int64_t sparse_dim = input.sparse_dim();

855 
856  auto dims_to_keep_v = std::vector<int64_t>();
857  auto dense_dims_to_sum_v = std::vector<int64_t>();
858  for (int64_t d = 0; d < input_dim; d++) {
859  if (dims_to_sum_b[d]) {

868  const bool sum_dense_dim = (dense_dims_to_sum_v.size() > 0);
869

871  Tensor new_values;
872  if (sum_dense_dim) {
873  new_values = values.sum(dense_dims_to_sum_v);

875  else {
876  new_values = values.clone();
877  }
878 
879  if (sum_all_sparse_dim) {
880  // return a dense tensor if sum over all sparse dims
881  new_values = new_values.sum(0);
882  return new_values;
883  }
884  else { // !sum_all_sparse_dim

889  }
890  else {
891  new_indices = at::empty({sparse_dim - sparse_dims_to_sum_size, input._nnz()}, indices.options());

893  int64_t d = dims_to_keep_v[i];
894  if (d < sparse_dim) new_indices[i].copy_(indices[d]);
895  else break;
896  }
897  }
898

900  int64_t new_sparse_dim = new_indices.size(0);
901  int64_t new_dense_dim = new_values.dim() - 1; // exclude nnz dim
902  std::vector<int64_t> new_sizes;
903  for (auto d : dims_to_keep_v) new_sizes.emplace_back(sizes[d]);
904  if (sum_all_sparse_dim) new_sizes.emplace(new_sizes.begin(), 1);
905 
906  // use coalesce() to do sum reduction
907  SparseTensor new_sparse = at::_sparse_coo_tensor_with_dims_and_tensors(new_sparse_dim, new_dense_dim, new_sizes, new_indices, new_values, input.options());

914 // --------------------------------------------------------------------
915 // NOTE [ sparse.sum() backward ]
916 //
917 // When sum over sparse_dim, backward scatters gradients from grad tensor to input tensor.
918 // Grad and input need to align indices over sparse_dim that are not summed (given
919 // input.spares_dim >= grad.sparse_dim). Implementation here compares each pair of
920 // indices between grad and input. When a matching indices pair (input_i, grad_i) is found,
921 // copy grad.values[grad_i] -> input_grad.values[input_i]. E.g.,
922 //
923 // input.sparse_dim = [5, 5]
924 // input.indices = [[0, 0, 1, 2, 2, 3, 4, 4],
925 // [1, 4, 4, 0, 1, 3, 2, 4]]
926 // input.values = [0, 1, 2, 3, 4, 5, 6, 7]
927 // ...
928 // sparse.sum(input, [0])
929 // backward(...)
930 // ...
931 // grad.indices = [[0, 1, 2, 3]]
932 // grad.values = [1, 2, 0, 4]
933 //
934 // # after indices matching
935 // input grad
936 // [[0, 1], -> [1]
937 // [0, 4], -> [ ]
938 // [1, 4], -> [ ]
939 // [2, 0], -> [0]
940 // [2, 1], -> [1]
941 // [3, 3], -> [3]
942 // [4, 2], -> [2]
943 // [4, 4]]) -> [ ]
944 //
945 // input_grad.indices = [[0, 0, 1, 2, 2, 3, 4, 4],
946 // [1, 4, 4, 0, 1, 3, 2, 4]]
947 // input_grad.values = [2, 0, 0, 1, 2, 4, 0, 0]
948 //
949 // Note that we allow input to be uncoalesced in the forward,
950 // we have to coalesce input at the backward, because grad-of-input
951 // take the same indices as input, if input is not coalesced, then
952 // coalescing grad-of-input may add up grad values for a duplicate indices,
953 // and hence generates a wrong grad-of-input.
954 //
955 // Other edge cases:
956 // - assign zero values to input gradients if cannot find matched indices at grad
957 // - grad.values might have zeros
958 // --------------------------------------------------------------------
959 Tensor _sparse_sum_backward_cpu(const Tensor& grad_, const SparseTensor& input_, IntArrayRef dims_to_sum) {
960  AT_CHECK(!grad_.is_cuda(), "_sparse_sum_backward_cpu: expected 'grad_' to be CPU tensor, but got CUDA tensor");
961  AT_CHECK(!input_.is_cuda(), "_sparse_sum_backward_cpu: expected 'input_' to be CPU tensor, but got CUDA tensor");
962 
963  auto input = input_.coalesce();
964  const int64_t input_dim = input.dim();
965  auto dims_to_sum_b = dim_list_to_bitset(dims_to_sum, input_dim);
966  auto dims_to_sum_v = dims_to_sum.vec();
967  maybe_wrap_dims(dims_to_sum_v, input_dim);
968 
969  LongTensor input_indices = input._indices();
970  Tensor input_values = input._values();
971  IntArrayRef input_sizes = input.sizes();
972  const int64_t input_sparse_dim = input.sparse_dim();
973  const int64_t input_dense_dim = input.dense_dim();
974  const int64_t input_nnz = input._nnz();
975 
976  int64_t sparse_dims_to_sum_size = 0;
977  auto sparse_dims_to_keep_v = std::vector<int64_t>();
978  auto dense_dims_to_sum_v = std::vector<int64_t>();
979  for (int64_t d = 0; d < input_dim; d++) {
980  if (dims_to_sum_b[d]) {
981  if (d < input_sparse_dim) sparse_dims_to_sum_size ++;
982  else dense_dims_to_sum_v.emplace_back(d + 1 - input_sparse_dim);
983  }
984  else {
985  if (d < input_sparse_dim) sparse_dims_to_keep_v.emplace_back(d);
986  }
987  }
988 
989  const bool sum_all_sparse_dim = (input_sparse_dim == sparse_dims_to_sum_size);
990  const bool sum_dense_dim = (dense_dims_to_sum_v.size() > 0);
991  const bool sum_sparse_dim = (sparse_dims_to_sum_size > 0);
992 
993  if (sum_all_sparse_dim) {
994  AT_CHECK(!grad_.is_sparse(), "_sparse_sum_backward_cpu: expected grad_ Tensor to be dense since all sparse dims are summed");
995  auto grad_input_values = grad_;
996  auto expand_size = input_values.sizes().vec();
997  if (sum_dense_dim) {
998  auto dense_expand_size = std::vector<int64_t>(expand_size);
999  dense_expand_size.erase(dense_expand_size.begin());
1000  AT_ASSERT(dense_expand_size.size() == (input_values.dim() - 1));
1001  for (auto d : dense_dims_to_sum_v) grad_input_values = grad_input_values.unsqueeze(d - 1); // -1 since grad has no nnz dim
1002  grad_input_values = grad_input_values.expand(dense_expand_size);
1003  }
1004  grad_input_values = grad_input_values.expand(expand_size).clone();
1005  return at::_sparse_coo_tensor_with_dims_and_tensors(input_sparse_dim, input_dense_dim, input_sizes, input_indices.clone(), grad_input_values, input.options().dtype(grad_.dtype())); // convert to grad dtype
1006  }
1007  else {
1008  AT_CHECK(grad_.is_sparse(), "_sparse_sum_backward_cpu: expected grad_ Tensor to be sparse, but got dense");
1009  auto grad = grad_.coalesce();
1010  LongTensor grad_indices = grad._indices();
1011  Tensor grad_values = grad._values();
1012  const int64_t grad_sparse_dim = grad.sparse_dim();
1013  const int64_t grad_nnz = grad._nnz();
1014 
1015  Tensor grad_values_expand = grad_values;
1016  if (sum_dense_dim) {
1017  auto expand_size = input_values.sizes().vec();
1018  if (sum_sparse_dim) expand_size[0] = grad_values.size(0);
1019  for (auto d : dense_dims_to_sum_v) grad_values_expand = grad_values_expand.unsqueeze(d);
1020  grad_values_expand = grad_values_expand.expand(expand_size).clone();
1021  }
1022 
1023  Tensor grad_input_values;
1024  if (sum_sparse_dim) {
1025  // see NOTE [ sparse.sum() backward ]
1026  grad_input_values = at::zeros_like(input_values, grad_values.options());
1027 
1028  // get flatten indices for grad and input
1029  auto grad_sparse_dim_to_keep_v = std::vector<int64_t>(grad_sparse_dim);
1030  std::iota(grad_sparse_dim_to_keep_v.begin(), grad_sparse_dim_to_keep_v.end(), 0);
1031 
1032  auto grad_indices_1D = flatten_indices_by_dims(grad_indices, grad.sizes(), grad_sparse_dim_to_keep_v); // flatten indices on all sparse_dim of grad, output indices is coalesced and sorted
1033  auto grad_indices_1D_accessor = grad_indices_1D.accessor<int64_t, 1>();
1034  auto input_indices_1D = flatten_indices_by_dims(input_indices, input_sizes, sparse_dims_to_keep_v);
1035  auto input_indices_1D_accessor = input_indices_1D.accessor<int64_t, 1>();
1036 
1037  // binary search to find matching indices
1038  int64_t i;
1039  #pragma omp parallel for private(i)
1040  for (i = 0; i < input_nnz; i++) {
1041  int64_t input_idx = input_indices_1D_accessor[i];
1042  int64_t l = 0, r = grad_nnz - 1;
1043  while (l <= r) {
1044  int64_t m = l + (r - l) / 2;
1045  if (grad_indices_1D_accessor[m] == input_idx) {
1046  grad_input_values[i].copy_(grad_values_expand[m]);
1047  break;
1048  }
1049  if (grad_indices_1D_accessor[m] < input_idx) {
1050  l = m + 1;
1051  }
1052  else {
1053  r = m - 1;
1054  }
1055  }
1056  }
1057  }
1058  else {
1059  grad_input_values = grad_values_expand;
1060  }
1061  return at::_sparse_coo_tensor_with_dims_and_tensors(input_sparse_dim, input_dense_dim, input_sizes, input_indices.clone(), grad_input_values, grad.options());
1062  }
