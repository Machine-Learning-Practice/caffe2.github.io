60  void Register(
61  const SrcType& key,
62  Creator creator,
63  const RegistryPriority priority = REGISTRY_DEFAULT) {
64  std::lock_guard<std::mutex> lock(register_mutex_);
65  // The if statement below is essentially the same as the following line:
66  // CHECK_EQ(registry_.count(key), 0) << "Key " << key
67  // << " registered twice.";
68  // However, CHECK_EQ depends on google logging, and since registration is
69  // carried out at static initialization time, we do not want to have an
70  // explicit dependency on glog's initialization function.
71  if (registry_.count(key) != 0) {
72  auto cur_priority = priority_[key];
73  if (priority > cur_priority) {
74  std::string warn_msg =
75  "Overwriting already registered item for key " + KeyStrRepr(key);
76  fprintf(stderr, "%s\n", warn_msg.c_str());
77  registry_[key] = creator;
78  priority_[key] = priority;
79  } else if (priority == cur_priority) {
80  std::string err_msg =
81  "Key already registered with the same priority: " + KeyStrRepr(key);
82  fprintf(stderr, "%s\n", err_msg.c_str());
83  if (terminate_) {
84  std::exit(1);
85  } else {
86  throw std::runtime_error(err_msg);
87  }
88  } else {
89  std::string warn_msg =
90  "Higher priority item already registered, skipping registration of " +
91  KeyStrRepr(key);
92  fprintf(stderr, "%s\n", warn_msg.c_str());
93  }
94  } else {
95  registry_[key] = creator;
96  priority_[key] = priority;
97  }
98  }

203 // Note on C10_IMPORT and C10_EXPORT below: we need to explicitly mark DECLARE
204 // as import and DEFINE as export, because these registry macros will be used
205 // in downstream shared libraries as well, and one cannot use *_API - the API
206 // macro will be defined on a per-shared-library basis. Semantically, when one
207 // declares a typed registry it is always going to be IMPORT, and when one
208 // defines a registry (which should happen ONLY ONCE and ONLY IN SOURCE FILE),
209 // the instantiation unit is always going to be exported.
210 //
211 // The only unique condition is when in the same file one does DECLARE and
212 // DEFINE - in Windows compilers, this generates a warning that dllimport and
213 // dllexport are mixed, but the warning is fine and linker will be properly
214 // exporting the symbol. Same thing happens in the gflags flag declaration and
215 // definition caes.
216 #define C10_DECLARE_TYPED_REGISTRY( \
217  RegistryName, SrcType, ObjectType, PtrType, ...) \
218  C10_IMPORT ::c10::Registry<SrcType, PtrType<ObjectType>, ##__VA_ARGS__>* \
219  RegistryName(); \
220  typedef ::c10::Registerer<SrcType, PtrType<ObjectType>, ##__VA_ARGS__> \
221  Registerer##RegistryName
222 
223 #define C10_DEFINE_TYPED_REGISTRY( \
224  RegistryName, SrcType, ObjectType, PtrType, ...) \
225  C10_EXPORT ::c10::Registry<SrcType, PtrType<ObjectType>, ##__VA_ARGS__>* \
226  RegistryName() { \
227  static ::c10::Registry<SrcType, PtrType<ObjectType>, ##__VA_ARGS__>* \
228  registry = new ::c10:: \
229  Registry<SrcType, PtrType<ObjectType>, ##__VA_ARGS__>(); \
230  return registry; \
231  }
