55  // types can be unified
56  void intersectRefinements(const Refinements& other) {
57  Refinements ret;
58  for (const auto& name_mapping : mappings_) {
59  const auto& name = name_mapping.first;

64  if (maybe_unified_type) {

73  // return the union of the values to type mappings in a and b whose
74  // types can be unified
75  void unionRefinements(const Refinements& other) {
76  Refinements ret;
77  for (const auto& name_mapping : mappings_) {
78  const auto& name = name_mapping.first;
79  const auto& mapping = name_mapping.second;
80  TypePtr t_1 = mapping.first;
81  if (auto other_mapping = other.getRefinement(name_mapping.first)) {
82  TypePtr t_2 = other_mapping->first;
83  c10::optional<TypePtr> maybe_unified_type = c10::nullopt;
84  if (t_1->isSubtypeOf(t_2)) {
85  maybe_unified_type = t_1;
86  } else if (t_2->isSubtypeOf(t_1)) {
87  maybe_unified_type = t_2;
88  }
89  if (maybe_unified_type) {
90  ret.setRefinement(
91  name, TypeAndRange(*maybe_unified_type, mapping.second));
92  }
93  } else {
94  ret.setRefinement(name, mapping);
95  }
96  }
97 
98  for (auto& name_mapping : other.mappings_) {
99  if (!getRefinement(name_mapping.first)) {
100  ret.setRefinement(name_mapping.first, name_mapping.second);
101  }
102  }
103 
104  mappings_ = std::move(ret.mappings_);
105  }

166 // Auxiliary data structure for desugaring variable binding into our always
167 // explicitly scoped language as we descend down nested control structures in
168 // the frontend (which themselves don't introduce scopes)
169 //
170 // The algorithm is roughly as follows:
171 // 1) While emitting a block within a control operator, add inputs and outputs
172 // from the block for each value referenced (both "reads" and "writes").
173 // This sets the value up as a candidate loop carried dependency.
174 // 2) When we reach the end of the block, examine all the values in the current
175 // scope's value map. If the name also resides in an outer scope with a
176 // different Value*, this is a true loop-carried dependency. If not, this
177 // value was not assigned to. Replace all references to the block input
178 // with the Value* pointed to in the tightest enclosing scope. Then delete
179 // that block input and output.
180 // 3) When we emit the actual control operator, take all of the loop-carried
181 // dependency values as inputs and return them as outputs from the control
182 // op
183 //
184 // Note that an alternative implementation could only add the loop-carried dep
185 // inputs and outputs when we see a value that is mutated. This, however
186 // requires replacing all references to that value *within the current
187 // block* with a new input. That is to say: we need to traverse the pre-
188 // decessor nodes and replace inputs that reference that value with the
189 // newly-created input. This could be made less expensive with a change to
190 // the IR API, but for now we choose to pessimisitically create inputs and
191 // delete unnecessary ones later with replaceAllusesWith().

260  SugaredValuePtr createCapturedInput(Value* orig, const std::string& name) {
261  // insert the captured input alphabetically in the capture list.
262  // this ensures consistency of the order of loop-carried dependencies
263  // even when the use in the loop is in a different order
264  size_t insert_pos = 0;
265  while (insert_pos < captured_inputs.size() &&
266  name > captured_inputs[insert_pos]) {
267  insert_pos++;
268  }
269  captured_inputs.insert(captured_inputs.begin() + insert_pos, name);
270 
271  // Create the input
272  const size_t loop_carried_block_inputs_offset = 1;
273  Value* new_input =
274  b->insertInput(loop_carried_block_inputs_offset + insert_pos)
275  ->setType(orig->type());
276 
277  // Associate this name with this value
278  auto sv = std::make_shared<SimpleValue>(new_input);
279  value_table[name] = sv;
280 
281  return sv;
282  }

319  void setSugaredVar(
320  const SourceRange& loc,
321  const std::string& name,
322  SugaredValuePtr value) {
323  Value* as_simple_value = asSimple(value);
324  if (as_simple_value && !as_simple_value->hasUniqueName() &&
325  meaningfulName(name) &&
326  // note: if the value wasn't defined in this block, we might be giving a
327  // name only used inside this block to a value outside of this. this is
328  // not normally helpful for debugging and causes import/export jitter.
329  as_simple_value->node()->owningBlock() == block()) {
330  as_simple_value->setUniqueName(name);
331  }
332  // prevent re-assignment involving any sugared values
333  // any reassignment like:
334  // a = ...
335  // while ...
336  // a = ..
337  // requires 'a' to be first-class in the graph since its value depends on
338  // control flow
339  if (auto parent = findInParentFrame(name)) {
340  if (!as_simple_value) {
341  throw ErrorReport(loc)
342  << "Cannot re-assign '" << name << "' to a value of type "
343  << value->kind() << " because " << name
344  << " is not a first-class value. Only reassignments to first-class values are allowed";
345  }
346  Value* simple_parent = asSimple(parent);
347  if (!simple_parent) {
348  throw ErrorReport(loc)
349  << "Cannot re-assign '" << name << "' because it has type "
350  << value->kind() << " and " << name
351  << " is not a first-class value. Only reassignments to first-class values are allowed";
352  }
353  if (!as_simple_value->type()->isSubtypeOf(
354  unshapedType(simple_parent->type()))) {
355  std::stringstream errMsg;
356  errMsg << "variable '" << name << "' previously has type "
357  << simple_parent->type()->str()
358  << " but is now being assigned to a value of type "
359  << as_simple_value->type()->str();
360  // Special-cased error msg if we're trying to assign to a tensor list.
361  if (simple_parent->type()->kind() == TypeKind::ListType &&
362  as_simple_value->type()->kind() == TypeKind::ListType) {
363  errMsg << "\n. (Note: empty lists are constructed as Tensor[]; "
364  << "if you want an empty list of a different type, "
365  << "use `torch.jit.annotate(List[T], [])`, "
366  << "where `T` is the type of elements in the list)";
367  }
368  throw ErrorReport(loc) << errMsg.str();
369  }
370  }
371  if (as_simple_value)
372  createCapturedInputIfNeeded(loc, name);
373  value_table[name] = std::move(value);
374  }

439  // Given that after emitting statements in a block, we've added block inputs
440  // for all value references and assignments, delete inputs for which there was
441  // no assignment, only references.
442  void deleteExtraInputs(const SourceRange& loc) {
443  // note: skip i == 0, it is the loop trip count for inputs
444  // and the loop condition for outputs.
445  // captured_inputs is indexed by i - 1 since it only contains loop
446  // carried dependencies
447  // inputs: loop_counter, lcd0, lcd1, ...
448  // outputs: loop_condition, lcd0, lcd1, ...
449  // captured_inputs: lcd0, lcd1, ...
450  AT_ASSERT(b->inputs().size() == b->outputs().size());
451  AT_ASSERT(b->inputs().size() == captured_inputs.size() + 1);
452  for (size_t i = b->inputs().size() - 1; i > 0; i--) {
453  // nothing changed along this loop
454  if (b->inputs()[i] == b->outputs()[i]) {
455  auto name = captured_inputs[i - 1];
456  Value* orig = findInParentFrame(name)->asValue(loc, method);
457  b->inputs()[i]->replaceAllUsesWith(orig);
458  b->eraseInput(i);
459  b->eraseOutput(i);
460  captured_inputs.erase(captured_inputs.begin() + i - 1);
461  }
462  }

595  std::vector<IValue> evaluateDefaults(
596  const SourceRange& r,
597  const std::vector<Expr>& default_types,
598  const std::vector<Expr>& default_exprs) {
599  std::vector<IValue> default_values;
600  if (default_exprs.empty())
601  return default_values;
602  // To evaluate the default expressions, we create a graph with no inputs,
603  // and whose returns are the default values we need.
604  // We then run constant prop on this graph and check the results are
605  // constant. This approach avoids having to have separate handling of
606  // default arguments from standard expressions by piecing together existing
607  // machinery for graph generation, constant propgation, and constant
608  // extraction.
609  auto tuple_type = Subscript::create(
610  r,
611  Var::create(r, Ident::create(r, "Tuple")),
612  List<Expr>::create(r, default_types));
613  auto blank_decl = Decl::create(
614  r, List<Param>::create(r, {}), Maybe<Expr>::create(r, tuple_type));
615 
616  auto tuple_expr =
617  TupleLiteral::create(r, List<Expr>::create(r, default_exprs));
618  auto ret = Return::create(r, tuple_expr);

631  std::vector<Argument> parseArgsFromDecl(
632  const Decl& decl,
633  const c10::optional<Self>& self) {
634  auto params_begin = decl.params().begin();
635  auto params_end = decl.params().end();
636  if (self) {
637  ++params_begin;
638  }

640 
641  std::vector<Expr> default_types;
642  std::vector<Expr> default_exprs;
643  // gather any non-empty default arguments
644  for (auto it = params_begin; it != params_end; ++it) {
645  auto param = *it;
646  auto def = param.defaultValue();
647  if (def.present()) {
648  default_types.emplace_back(param.type());
649  default_exprs.emplace_back(def.get());
650  }
651  }
652  auto default_values =
653  evaluateDefaults(decl.range(), default_types, default_exprs);
654 
655  auto defaults_it = default_values.begin();
656  for (auto it = params_begin; it != params_end; ++it) {

661 
662  // BroadcastList list can only appear at the argument level

664  typeParser_.parseBroadcastList(decl_arg.type())) {
665  type = maybe_broad_list->first;
666  N = maybe_broad_list->second;
667  } else {
668  type = typeParser_.parseTypeFromExpr(decl_arg.type());

670  }
671  c10::optional<IValue> default_value = c10::nullopt;

673  default_value = *defaults_it++;
674  }
675  auto arg = Argument(
676  decl_arg.ident().name(),
677  type,
678  N,
679  default_value,
680  decl_arg.kwarg_only());
681  retval.push_back(arg);
682  }
683  return retval;

781  std::pair<std::shared_ptr<Graph>, Value*> lambdaLift(Block* block) {
782  auto subgraph = std::make_shared<Graph>();
783  // note: type is set later on pack_context and context when we know it
784  Node* pack_context =
785  graph->insertNode(graph->create(prim::TupleConstruct, {}, 1));
786  Value* context = subgraph->addInput("context");
787  // cannot use createTupleUnpack because the type is not known yet
788  Node* unpack_context =
789  subgraph->insertNode(subgraph->create(prim::TupleUnpack, {context}, 0));
790 
791  std::unordered_map<Value*, Value*> captures;
792  auto env = [&](Value* v) -> Value* {
793  auto it = captures.find(v);
794  if (it != captures.end()) {
795  return it->second;
796  }
797  pack_context->addInput(v);
798  Value* r = unpack_context->addOutput()->copyMetadata(v);
799  captures[v] = r;
800  return r;
801  };
802  subgraph->block()->cloneFrom(block, env);
803  auto context_type = TupleType::create(
804  fmap(pack_context->inputs(), [](Value* v) { return v->type(); }));
805  pack_context->output()->setType(context_type);
806  context->setType(context_type);
807  return std::make_pair(std::move(subgraph), pack_context->output());
808  }
809  // XXX - right now closures are used _only_ for defining gradients internally
810  // There are several unfinished aspects that make them unusable generally
811  // 1. We do not have a type, ivalue, operator to represent prim::Function, so
812  // closure_node has type None
813  // and any graphs that contain it cannot be run
814  // 2. There is no export logic for it yet, so it cannot be
815  // exported/python_printed
816  // 3. There is nothing preventing the assignment of already existing variables
817  // inside the closures
818  // the changes to those variables will just get forgotten.
819  // 4. There is no parsing support in frontend.py, this is intentional since it
820  // prevents people from accidentally using this feature.
821  void emitClosure(const Def& def) {
822  Node* closure_node = graph->insertNode(graph->create(prim::Function, 1));
823  closure_node->output()->setType(
824  NoneType::get()); // it is not a real thing yet, so just say the type is
825  // none.

828  WithInsertPoint guard(block);
829  pushFrame(block, /*starts_def=*/true);
830  emitDef(
831  def,
832  c10::nullopt,
833  block); // ignore schema return, we just wont use it for now since we
834  // never create a Method for the closure
835  popFrame(/*ends_def=*/true);

843  auto tup =

849  void emitReturn(const Return& stmt) {
850  Value* result = emitExpr(stmt.expr());
851  TypePtr result_type = def_stack_.back().declared_return_type_;
852  // result type is annotated, every return must convert to that type
853  if (result_type) {
854  // this guard skips implicit conversion from None -> Tensor for the return
855  // type. otherwise forgetting a return a function returning a tensor will
856  // cause a None to be converted to a tensor.
857  if (!(result_type->isSubtypeOf(TensorType::get()) &&
858  result->type()->isSubtypeOf(NoneType::get()))) {
859  result = tryConvertToType(
860  stmt.range(),
861  *graph,
862  result_type,
863  result,
864  /*allow_conversions=*/true);
865  }
866 
867  if (!result->type()->isSubtypeOf(result_type)) {
868  throw ErrorReport(stmt.range())
869  << "Return value was annotated as having type "
870  << result_type->python_str() << " but is actually of type "
871  << result->type()->python_str();
872  }
873  } else {
874  result_type = def_stack_.back().merged_return_type_;
875  if (!result_type) {
876  result_type = result->type();
877  }
878  if (!unifyTypes(result_type, result->type())) {
879  throw ErrorReport(stmt.range())
880  << "Previous return statement returned a value of type "
881  << result_type->python_str()
882  << " but this return statement returns a value of type "
883  << result->type()->python_str();
884  }
885  }
886  AT_ASSERT(result_type);
887  def_stack_.back().merged_return_type_ = result_type;
888  environment_stack->setVar(stmt.range(), "$return", result);
889  }

891  void emitStatements(
892  List<Stmt>::const_iterator begin,
893  List<Stmt>::const_iterator end) {
894  for (; begin != end; ++begin) {
895  auto stmt = *begin;
896  switch (stmt.kind()) {
897  case TK_IF:
898  emitIf(If(stmt));
899  break;
900  case TK_WHILE:
901  emitWhile(While(stmt));
902  break;
903  case TK_FOR:
904  emitFor(For(stmt));
905  break;
906  case TK_ASSIGN:
907  emitAssignment(Assign(stmt));
908  break;
909  case TK_AUG_ASSIGN:
910  emitAugAssignment(AugAssign(stmt));
911  break;
912  case TK_GLOBAL:
913  for (auto ident : Global(stmt).names()) {
914  const auto& name = Ident(ident).name();
915  environment_stack->setVar(
916  ident.range(), name, graph->addInput(name));
917  }
918  break;
919  case TK_EXPR_STMT: {
920  auto expr = ExprStmt(stmt).expr();
921  emitSugaredExpr(expr, 0);
922  } break;
923  case TK_RAISE:
924  emitRaise(Raise(stmt).range());
925  break;
926  case TK_ASSERT:
927  emitAssert(Assert(stmt));
928  break;
929  case TK_RETURN: {
930  emitReturn(Return(stmt));
931  } break;
932  case TK_PASS:
933  // Emit nothing for pass
934  break;
935  case TK_DEF:
936  emitClosure(Def(stmt));
937  break;
938  default:
939  throw ErrorReport(stmt)
940  << "Unrecognized statement kind " << kindToString(stmt.kind());
941  }
942  }

1034  Value* emitShortCircuitIf(
1035  const SourceRange& loc,
1036  const TreeRef& first_expr,
1037  const TreeRef& second_expr,
1038  bool is_or) {
1039  const auto first_bool_info = findRefinements(first_expr);
1040  Value* first_value = emitCond(Expr(first_expr));
1041 
1042  // if the second expr in the short circuit is not evaluated,
1043  // than the first expression is False if the short circuit
1044  // is an `and` and True if the short circuit is an `or`.
1045  // `False and expr` -> False, `True or expr` -> True
1046  //
1047  // inserting it as a constant makes optimization easier
1048 
1049  Value* first_value_returned;
1050 
1051  const Refinements* first_expr_refinements;
1052  const Refinements* second_expr_refinements;
1053  // if it's an OR the first expr is emitted in the true branch
1054  // and the second expr in the false branch, if it's an AND the opposite
1055  if (is_or) {
1056  first_value_returned = graph->insertConstant(true, nullptr, loc);
1057  first_expr_refinements = &first_bool_info.true_refinements_;
1058  second_expr_refinements = &first_bool_info.false_refinements_;
1059  } else {
1060  first_value_returned = graph->insertConstant(false, nullptr, loc);
1061  first_expr_refinements = &first_bool_info.false_refinements_;
1062  second_expr_refinements = &first_bool_info.true_refinements_;
1063  }
1064 
1065  auto get_first_expr = [&] {
1066  insertRefinements(*first_expr_refinements);
1067  return first_value_returned;
1068  };
1069 
1070  auto get_second_expr = [&] {
1071  insertRefinements(*second_expr_refinements);
1072  return emitCond(Expr(second_expr));
1073  };
1074 
1075  // if this is an OR, eval second expression if first expr is False
1076  // If this is an AND, eval second expression if first expr is True
1077  if (is_or) {
1078  return emitIfExpr(loc, first_value, get_first_expr, get_second_expr);
1079  } else {
1080  return emitIfExpr(loc, first_value, get_second_expr, get_first_expr);
1081  }
1082  }

1136  void emitIfElseBlocks(Value* cond_value, const If& stmt) {
1137  Node* n = graph->insertNode(create(prim::If, stmt.range(), 0));
1138  n->addInput(cond_value);
1139  const auto bool_info = findRefinements(stmt.cond());
1140  auto* true_block = n->addBlock();
1141  auto* false_block = n->addBlock();
1142 
1143  // Emit both blocks once to get the union of all mutated values
1144  auto save_true = emitSingleIfBranch(
1145  true_block, stmt.trueBranch(), bool_info.true_refinements_);
1146  auto save_false = emitSingleIfBranch(
1147  false_block, stmt.falseBranch(), bool_info.false_refinements_);
1148 
1149  // In python, every variable assigned in an if statement escapes
1150  // the scope of the if statement (all variables are scoped to the function).
1151  // Script is a subset of python: we consider variables to be in scope
1152  // as long as there is a definition of the variable along all paths
1153  // through the if statemnent
1154  // ----
1155  // if ...:
1156  // a =
1157  // else:
1158  // ...
1159  // ... = a # error, a is not defined along all paths
1160  // ----
1161  // if ...:
1162  // a =
1163  // else:
1164  // a =
1165  // ... = a # OK, a is defined along all paths
1166  // ----
1167  // a = ...
1168  // if ...:
1169  // a =
1170  // ... = a # OK, a is defined along all paths
1171 
1172  // ordered set, because we want deterministic graph output
1173  std::set<std::string> mutated_variables;
1174 
1175  for (auto& v : save_true->definedVariables()) {
1176  if (save_false->findInAnyFrame(v)) {
1177  mutated_variables.insert(v);
1178  }
1179  }
1180  for (auto& v : save_false->definedVariables()) {
1181  if (save_true->findInAnyFrame(v)) {
1182  mutated_variables.insert(v);
1183  }
1184  }
1185 
1186  // Register outputs in each block
1187  for (const auto& x : mutated_variables) {
1188  auto tv = save_true->getVar(x, stmt.range());
1189  auto fv = save_false->getVar(x, stmt.range());
1190  auto unified = unifyTypes(tv->type(), fv->type());
1191 
1192  // attempt to unify the types. we allow variables to be set to different
1193  // types in each branch as long as that variable is not already in scope,
1194  // or if that variable does not get used later. here, we save the error
1195  // so that the error message will be more informative in the case that is
1196  // used later. When a is accessed in (a + 1), the error will get printed
1197  // if cond:
1198  // a = 1
1199  // else:
1200  // a = tensor
1201  // b = a + 1
1202  //
1203  if (!unified) {
1204  ErrorReport error(stmt);
1205  error << "Type mismatch: " << x << " is set to type "
1206  << tv->type()->str() << " in the true branch"
1207  << " and type " << fv->type()->str() << " in the false branch";
1208  if (save_true->findInParentFrame(x) ||
1209  save_false->findInParentFrame(x)) {
1210  throw error;
1211  } else {
1212  // error gets saved in the lowest environment because all
1213  // variables are scoped to the function. doesn't matter if this
1214  // accessed through save_true or save_false
1215  save_true->setVariableTypeError(x, error.what());
1216  continue;
1217  }
1218  }
1219  true_block->registerOutput(tv);
1220  false_block->registerOutput(fv);
1221  environment_stack->setVar(
1222  stmt.range(), x, n->addOutput()->setType(*unified));
1223  }

1226  void emitIf(const If& stmt) {
1227  // NOTE: emitIf checks on If stmt condition to see if the cond AST kind ==
1228  // is/is not, for such cases we do meta programming and disable emitting the
1229  // corresponding branches
1230  Expr cond = stmt.cond();
1231 
1232  if (cond.kind() != TK_IS && cond.kind() != TK_ISNOT) {
1233  // emit normal IF stmt for cases except TK_IS and TK_ISNOT
1234  Value* cond_value = emitCond(cond);
1235  emitIfElseBlocks(cond_value, stmt);
1236  return;
1237  }
1238  // meta programming on AST for is/is not cases and emit branches base on the
1239  // possible output of cond
1240  auto cond_op = BinOp(cond);
1241  SugaredValuePtr lhs_val = emitSugaredExpr(cond_op.lhs(), 1);
1242  SugaredValuePtr rhs_val = emitSugaredExpr(cond_op.rhs(), 1);
1243 
1244  List<Stmt> always_none_branch =
1245  cond.kind() == TK_IS ? stmt.trueBranch() : stmt.falseBranch();
1246  List<Stmt> never_none_branch =
1247  cond.kind() == TK_IS ? stmt.falseBranch() : stmt.trueBranch();
1248 
1249  auto lhs_none = lhs_val->isNone();
1250  auto rhs_none = rhs_val->isNone();
1251 
1252  // Dispatch logic (A: ALWAYS, N: NEVER, M: MAYBE):
1253  //
1254  // AA, -> emit always_none_branch
1255  // AN , NA-> emit never_none_branch
1256  // MA, MM, MN, NM, NN, AM -> emit both conditional branches
1257 
1258  if (lhs_none == ALWAYS && rhs_none == ALWAYS) {
1259  // None is/is not None: only emit the always_none_branch
1260  emitStatements(always_none_branch);
1261  } else if (
1262  (lhs_none == ALWAYS && rhs_none == NEVER) ||
1263  (lhs_none == NEVER && rhs_none == ALWAYS)) {
1264  // lhs_val/rhs_val with A/M: only emit never_none_branch
1265  emitStatements(never_none_branch);
1266  } else {
1267  // all other cases for lhs_val and rhs_val
1268  // emit the whole If stmt as usual, finish emitCond first
1269  auto lhs_range = cond_op.lhs().get()->range();
1270  auto rhs_range = cond_op.rhs().get()->range();
1271 
1272  auto kind = getNodeKind(cond.kind(), cond.get()->trees().size());
1273  Value* cond_value = emitBuiltinCall(
1274  cond.get()->range(),
1275  *method.graph(),
1276  kind,
1277  c10::nullopt,
1278  {lhs_val->asValue(lhs_range, method),
1279  rhs_val->asValue(rhs_range, method)},
1280  {},
1281  /*required=*/true);
1282  emitIfElseBlocks(cond_value, stmt);
1283  }
1284  }

1286  // *********************** Loop Operators ************************************
1287  // Emits a loop operators conforming to the semantics specified at
1288  // https://github.com/onnx/onnx/blob/master/docs/Operators.md#experimental-loop
1289  // TODO: implement scan_outputs
1290 
1291  // the format of the Loop instruction is:
1292  // loop_carried_outputs* = Loop(max_trip_count, start_condition,
1293  // loop_carried_inputs*)
1294  // block0(loop_counter, loop_carried_block*) {
1295  // <body>
1296  // -> (continue_condition, loop_carried_block_outputs*)
1297  // }
1298  // all loop_carried_... lists are the same length and represent the value of
1299  // loop-carried variables whose definitions are updated as the loop executes
1300  // in a way that ensure single static assignment.
1301 
1302  void emitLoopCommon(
1303  SourceRange range,
1304  c10::optional<Expr> max_trip_count,
1305  c10::optional<Expr> cond,
1306  const List<Stmt>& body,
1307  c10::optional<Ident> itr_ident,
1308  bool in_list = false) {
1309  Node* n = graph->insertNode(create(prim::Loop, range, 0));
1310  Value *max_trip_count_val, *cond_val;
1311  {
1312  WithInsertPoint guard(n);
1313  if (max_trip_count) {
1314  if (in_list) {
1315  auto listArg = emitExpr(max_trip_count.value());
1316 
1317  max_trip_count_val = emitBuiltinCall(
1318  max_trip_count->range(),
1319  *graph,
1320  aten::len,
1321  c10::nullopt,
1322  {listArg},
1323  {},
1324  /*required=*/true);
1325  } else {
1326  max_trip_count_val = ensureInt(
1327  max_trip_count->range(), emitExpr(max_trip_count.value()));
1328  }
1329  } else {
1330  max_trip_count_val = materializeConstant(
1331  std::numeric_limits<int64_t>::max(),
1332  *graph,
1333  range,
1334  integral_constants);
1335  }
1336  if (cond) {
1337  cond_val = emitCond(cond.value());
1338  } else {
1339  cond_val = graph->insertConstant(true, nullptr, range);
1340  }
1341  }
1342  n->addInput(max_trip_count_val);
1343  n->addInput(cond_val);
1344  auto* body_block = n->addBlock();
1345  Value* trip_count =
1346  body_block->addInput()->setType(IntType::get()); // Iteration num
1347 
1348  {
1349  pushFrame(body_block);
1350  WithInsertPoint guard(body_block);
1351  if (itr_ident) {
1352  if (in_list) {
1353  // set user's iterator variable to the current element
1354  auto listArg = emitExpr(max_trip_count.value());
1355  trip_count = emitBuiltinCall(
1356  max_trip_count->range(),
1357  *graph,
1358  aten::select,
1359  c10::nullopt,
1360  {listArg, trip_count},
1361  {},
1362  /*required=*/true);
1363  }
1364  environment_stack->setVar(
1365  itr_ident->range(), itr_ident->name(), trip_count);
1366  }
1367  emitStatements(body);
1368 
1369  // Also emit the conditional
1370  if (cond) {
1371  Value* body_cond_value = emitCond(cond.value());
1372  body_block->registerOutput(body_cond_value);
1373  } else {
1374  Value* cond_value_dummy = graph->insertConstant(true, nullptr, range);
1375  body_block->registerOutput(cond_value_dummy);
1376  }
1377 
1378  auto body_frame = popFrame();
1379  auto outer_frame = environment_stack;
1380 
1381  // Add block outputs to correspond to each captured input
1382  // some of these will be removed.
1383  for (const auto& x : body_frame->captured_inputs) {
1384  auto fv = body_frame->getValueInThisFrame(range, x);
1385  body_block->registerOutput(fv);
1386  }
1387 
1388  // Remove inputs for values that did not mutate within the
1389  // block
1390  body_frame->deleteExtraInputs(range);
1391 
1392  // register node inputs/outputs for the true loop carried deps,
1393  for (size_t i = 0; i < body_frame->captured_inputs.size(); ++i) {
1394  auto x = body_frame->captured_inputs[i];
1395  n->addInput(outer_frame->getVar(x, range));
1396  // body_block->inputs(): loop_counter, lcd0, lcd1, ...
1397  // captured_inputs: lcd0, lcd1, ...
1398  auto typ = body_block->inputs()[i + 1]->type();
1399  outer_frame->setVar(range, x, n->addOutput()->setType(typ));
1400  }
1401  }
1402  }

1516  // Validate that the `lhs` Expr's in an assignment statement are valid. That
1517  // is:
1518  //
1519  // 1) All lhs Expr's are either Var or Starred nodes
1520  // 2) There is at most one Starred node in the lhs Expr
1521  // 3) A Starred node can only appear when there is another non-Starred lhs
1522  // Expr. Concretely this means that `*abc = func()` is illegal. Unpacking
1523  // all outputs into a tuple is covered by `abc = func()`.

1529  num_normal_assign++;
1530  } else if (assignee.kind() == TK_STARRED) {
1531  num_starred++;

1534  << "subscript, or starred expression.";

1537

1539  throw ErrorReport(r)
1540  << "Only one starred expression is allowed on the lhs.";
1541  }

1544  throw ErrorReport(r) << "A Starred expression may only appear on the "
1545  << "lhs within the presence of another non-starred"
1546  << " expression.";
1547  }
1548

1814  if (starred_unpack)
1815  n_binders--;
1816  auto output = emitSugaredExpr(rhs, n_binders);
1817  auto outputs = output->asTuple(
1818  rhs.range(),
1819  method,
1820  starred_unpack ? c10::nullopt : c10::optional<size_t>{n_binders});
1821  if (outputs.size() < n_binders) {
1822  throw ErrorReport(tl)
1823  << "need " << (starred_unpack ? "at least " : "") << n_binders
1824  << " values to unpack but found only " << outputs.size();
1825  }
1826  if (outputs.size() > n_binders && !starred_unpack) {
1827  throw ErrorReport(tl) << "too many values to unpack: need " << n_binders
1828  << " but found " << outputs.size();

1830  int i = 0;
1831  for (auto assignee : tl.inputs()) {
1832  switch (assignee.kind()) {
1833  case TK_SUBSCRIPT:
1834  emitSubscriptAssign(
1835  rhs.range(),
1836  Subscript(assignee),
1837  NamedValue(
1838  rhs.range(), outputs.at(i)->asValue(rhs.range(), method)));
1839  i++;
1840  break;
1841  case TK_VAR:
1842  environment_stack->setSugaredVar(
1843  assignee.range(), Var(assignee).name().name(), outputs.at(i));
1844  i++;
1845  break;
1846  case TK_STARRED: {

1855  outputs_ref.slice(i, n_matched),

1860  environment_stack->setVar(var.range(), Var(var).name().name(), tup);
1861  i += n_matched;
1862  } break;
