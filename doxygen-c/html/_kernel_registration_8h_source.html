71 // ivalue_to_arg_type<T>: Take an IValue that is an argument to a kernel and
72 // cast it to the type that should be passed to the kernel function.
73 // Examples: If the IValue contains a plain type like an int, return that.
74 // If the IValue contains an IntList, return it as ArrayRef<int>.
75 template<class T>
76 struct ivalue_to_arg_type {
77  static T call(const IValue& v) {
78  return std::move(v).to<T>();
79  }
80 };

82 struct ivalue_to_arg_type<ArrayRef<T>> {
83  static ArrayRef<T> call(const IValue& v) {
84  return v.to<intrusive_ptr<ivalue::List<T>>>()->elements();
85  }
86 };

88 // call_with_ivalue_args: Take a function pointer and an ArrayRef<IValue>
89 // containing the arguments to call the function pointer with, and call it.
90 // The extra_args are appended as additional arguments at the end of the function call.
91 // Example:
92 // int myfunc(int a, ArrayRef<int> b, string c);
93 // int main() {
94 // std::vector<IValue> ivalue_args = {IValue(2), IntList::create(3, 4)};
95 // call_with_ivalue_args<decltype(myfunc), &myfunc>(ivalue_args, "extra_arg");
96 // }
97 template<class FuncType, FuncType* func, class... ExtraArgs, size_t... ivalue_arg_indices>
98 typename guts::function_traits<FuncType>::return_type call_with_ivalue_args_(ArrayRef<IValue> ivalue_args, guts::index_sequence<ivalue_arg_indices...>, ExtraArgs&&... extra_args) {
99  using IValueArgTypes = typename guts::function_traits<FuncType>::parameter_types;
100  return (*func)(ivalue_to_arg_type<guts::remove_cv_t<guts::remove_reference_t<guts::typelist::element_t<ivalue_arg_indices, IValueArgTypes>>>>::call(ivalue_args[ivalue_arg_indices])..., std::forward<ExtraArgs>(extra_args)...);
101 }

103 template<class FuncType, FuncType* func, class... ExtraArgs>
104 typename guts::function_traits<FuncType>::return_type call_with_ivalue_args(ArrayRef<IValue> ivalue_args, ExtraArgs&&... extra_args) {
105  constexpr size_t num_ivalue_args = guts::function_traits<FuncType>::number_of_parameters - sizeof...(ExtraArgs);
106  AT_ASSERTM(num_ivalue_args == ivalue_args.size(), "Wrong number of ivalue arguments");
107  return call_with_ivalue_args_<FuncType, func>(ivalue_args, guts::make_index_sequence<num_ivalue_args>(), std::forward<ExtraArgs>(extra_args)...);
108 }
