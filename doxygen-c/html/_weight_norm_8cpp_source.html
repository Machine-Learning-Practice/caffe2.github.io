19 Tensor norm_except_dim(const Tensor & v, int64_t pow, int64_t dim)
20 {
21  // I assume tensor.contiguous(), view(), norm(), etc. here will dispatch through VariableType.
22  if (dim == -1) {
23  return v.norm(pow);
24  } else if (dim == 0) {
25  std::vector<int64_t> output_size(v.dim(), 1);
26  output_size[0] = v.size(0);
27  return v.contiguous().view({v.size(0), -1}).norm(pow, 1).view(output_size);
28  } else if (dim == v.dim() - 1) {
29  std::vector<int64_t> output_size(v.dim(), 1);
30  output_size[v.dim() - 1] = v.size(v.dim() - 1);
31  return v.contiguous().view({-1, v.size(v.dim() - 1)}).norm(pow, 0).view(output_size);
32  } else {
33  // To consider: at::native::norm_except_dim is probably fine as well,
34  // and would avoid an additional dynamic dispatch.
35  return at::norm_except_dim(v.transpose(0, dim), pow, 0).transpose(0, dim); // optimize?
36  }

66 // Differentiable backward path, an alternative to weight_norm_cuda_backward, to be used
67 // when backward is itself creating a graph.
68 // The GradMode::is_enabled() check must be performed within Functions.cpp; that's why we
69 // define a separate function here, instead of inlining it in weight_norm_cuda_backward.
70 std::tuple<Tensor, Tensor> _weight_norm_differentiable_backward
71  (const Tensor & grad_w,
72  const Tensor & saved_v,
73  const Tensor & saved_g,
74  const Tensor & saved_norms,
75  int64_t dim)
76 {
77  // In Functions.cpp, the HardshrinkBackward object supplies "grad.contiguous()"
78  // as the first argument, so grad_w should be contiguous here.
79  // All these checks should succeed:
80  AT_CHECK(grad_w.is_contiguous(), "grad_w must be contiguous");
81  AT_CHECK(saved_v.is_contiguous(), "saved_v must be contiguous");
82  AT_CHECK(saved_g.is_contiguous(), "saved_g must be contiguous");
83  AT_CHECK(saved_norms.is_contiguous(), "saved_norms must be contiguous");
84 
85  int64_t last_dim = saved_v.dim() - 1;
86  int64_t last_size = saved_v.size(last_dim);
87 
88  // Like weight_norm_fused_backward, weight_norm_differentiable_backward should only ever be called
89  // through a WeightNormFusedBackward object, so we expect that dim == 0 || dim == saved_v.size(-1)
90  AT_CHECK(dim == 0 || dim == last_dim, "Expected dim to be the first or last dimension");
91 
92  // saved_g and saved_norms are already shaped to broadcast over the correct dimensions
93 
94  // ...but saved_norms might be Float when saved_g and saved_v are half.
95  // To consider: saved_norms.to(..., True /*non_blocking*/);
96  auto norms = saved_norms.to(saved_g.scalar_type());
97 
98  std::vector<int64_t> bcast_size(saved_v.dim(), 1);
99 
100  // Analytic backward path using differentiable primitive ops
101  if (dim == 0) {
102  bcast_size[0] = saved_v.size(0);
103  auto per_dim_sums = (grad_w*saved_v).view({saved_v.size(0), -1}).sum(1).view(bcast_size);
104  auto grad_v = (saved_g/norms)*(grad_w - saved_v*(per_dim_sums/(norms*norms)));
105  auto grad_g = per_dim_sums/norms;
106  return std::tuple<Tensor, Tensor>{grad_v, grad_g};
107  } else { // dim == last_dim
108  bcast_size[last_dim] = last_size;
109  auto per_dim_sums = (grad_w*saved_v).view({-1, last_size}).sum(0).view(bcast_size);
110  auto grad_v = (saved_g/norms)*(grad_w - saved_v*(per_dim_sums/(norms*norms)));
111  auto grad_g = per_dim_sums/norms;
112  return std::tuple<Tensor, Tensor>{grad_v, grad_g};
113  }
