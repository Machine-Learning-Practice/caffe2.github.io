49 PyObject* THPFInfo_pynew(PyTypeObject* type, PyObject* args, PyObject* kwargs) {
50  HANDLE_TH_ERRORS
51  static torch::PythonArgParser parser({
52  "finfo(ScalarType type)",
53  "finfo()",
54  });
55 
56  torch::ParsedArgs<1> parsed_args;
57  auto r = parser.parse(args, kwargs, parsed_args);
58  AT_CHECK(r.idx < 2, "Not a type");
59  at::ScalarType scalar_type;
60  if (r.idx == 1) {
61  scalar_type = torch::tensors::get_default_tensor_type().scalarType();
62  // The default tensor type can only be set to a floating point type/
63  AT_ASSERT(at::isFloatingType(scalar_type));
64  } else {
65  scalar_type = r.scalartype(0);
66  if (!at::isFloatingType(scalar_type)) {
67  return PyErr_Format(
68  PyExc_TypeError,
69  "torch.finfo() requires a floating point input type. Use torch.iinfo to handle '%s'",
70  type->tp_name);
71  }
72  }
73  return THPFInfo_New(scalar_type);
74  END_HANDLE_TH_ERRORS
75 }
