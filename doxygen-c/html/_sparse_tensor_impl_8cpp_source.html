21 // An empty dense tensor defaults to a 1-dimensional tensor of size [0]
22 // (recall, it is not a 0-dimensional tensor, because such a tensor would
23 // a scalar and have one element)
24 //
25 // Thus, an empty sparse tensor should be a 1-dimensional tensor of size [0].
26 // Furthermore, we have dim == sparse_dim + dense_dim; since this is a sparse
27 // tensor, let us say that an empty sparse tensor has sparse_dim == 1 and
28 // dense_dim == 0. (There is a degree of freedom here, but given that this
29 // is a sparse dimension, it seems reasonable to demand that sparse_dim > 0).
30 //
31 // This means that we allocate a [1,0] size indices tensor and a [0] size
32 // values tensor for such an empty tensor.
33 SparseTensorImpl::SparseTensorImpl(at::TensorTypeId type_id, const caffe2::TypeMeta& data_type)
34  : TensorImpl(type_id, data_type, nullptr, false)
35  , sparse_dim_(1)
36  , dense_dim_(0)
37  , indices_(at::empty({1, 0}, at::initialTensorOptions().device(sparseTensorIdToDeviceType(type_id)).dtype(ScalarType::Long)))
38  , values_(at::empty({0}, at::initialTensorOptions().device(sparseTensorIdToDeviceType(type_id)).dtype(data_type))) {}
