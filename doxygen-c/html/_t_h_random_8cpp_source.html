138 /* Macros for the Mersenne Twister random generator... */
139 /* Period parameters */
140 /* #define n 624 */
141 /* #define m 397 */
142 #define MATRIX_A 0x9908b0dfUL /* constant vector a */
143 #define UMASK 0x80000000UL /* most significant w-r bits */
144 #define LMASK 0x7fffffffUL /* least significant r bits */
145 #define MIXBITS(u,v) ( ((u) & UMASK) | ((v) & LMASK) )
146 #define TWIST(u,v) ((MIXBITS(u,v) >> 1) ^ ((v)&1UL ? MATRIX_A : 0UL))

149 void THRandom_manualSeed(THGenerator *_generator, uint64_t the_seed_)
150 {
151  int j;
152 
153  /* This ensures reseeding resets all of the state (i.e. state for Gaussian numbers) */
154  THGenerator *blank = THGenerator_newUnseeded();
155  THGenerator_copy(_generator, blank);
156  THGenerator_free(blank);
157 
158  _generator->gen_state.the_initial_seed = the_seed_;
159  _generator->gen_state.state[0] = _generator->gen_state.the_initial_seed & 0xffffffffUL;
160  for(j = 1; j < n; j++)
161  {
162  _generator->gen_state.state[j] = (1812433253UL * (_generator->gen_state.state[j-1] ^ (_generator->gen_state.state[j-1] >> 30)) + j);
163  /* See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. */
164  /* In the previous versions, mSBs of the seed affect */
165  /* only mSBs of the array state[]. */
166  /* 2002/01/09 modified by makoto matsumoto */
167  _generator->gen_state.state[j] &= 0xffffffffUL; /* for >32 bit machines */
168  }
169  _generator->gen_state.left = 1;
170  _generator->gen_state.seeded = 1;
171 }

178 void THRandom_nextState(THGenerator *_generator)
179 {
180  uint64_t *p = _generator->gen_state.state;
181  int j;
182 
183  _generator->gen_state.left = n;
184  _generator->gen_state.next = 0;
185 
186  for(j = n-m+1; --j; p++)
187  *p = p[m] ^ TWIST(p[0], p[1]);
188 
189  for(j = m; --j; p++)
190  *p = p[m-n] ^ TWIST(p[0], p[1]);
191 
192  *p = p[m-n] ^ TWIST(p[0], _generator->gen_state.state[0]);
193 }

267 double THRandom_normal(THGenerator *_generator, double mean, double stdv)
268 {
269  THArgCheck(stdv > 0, 2, "standard deviation must be strictly positive");
270 
271  /* This is known as the Box-Muller method */
272  if(!_generator->gen_state.normal_is_valid)
273  {
274  _generator->gen_state.normal_x = uniform_double(_generator);
275  _generator->gen_state.normal_y = uniform_double(_generator);
276  _generator->gen_state.normal_rho = sqrt(-2. * log(1.0-_generator->gen_state.normal_y));
277  _generator->gen_state.normal_is_valid = 1;
278  }
279  else
280  _generator->gen_state.normal_is_valid = 0;
281 
282  if(_generator->gen_state.normal_is_valid)
283  return _generator->gen_state.normal_rho*cos(2.*M_PI*_generator->gen_state.normal_x)*stdv+mean;
284  else
285  return _generator->gen_state.normal_rho*sin(2.*M_PI*_generator->gen_state.normal_x)*stdv+mean;
286 }
