7 // Note [im2col/col2im output padding]
8 // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
9 // Our implementations of im2col and col2im take both the input height/width as
10 // well as a seemingly redundant output height/width. In principle, you could
11 // compute the output height/width by using the convolution shape formulas. So,
12 // what's up with that?
13 //
14 // The trouble arises when one runs the backward of a transposed convolution
15 // with output_padding >= stride. (BTW, output_padding is known as adj inside
16 // THNN.) Let's consider a simple case where we have kernel=2, dilation=2,
17 // stride=1, output_padding=1 for a 4x4 input:
18 //
19 // Input: X
20 //
21 // Output: X.X.
22 // ....
23 // X.X.
24 // ....
25 //
26 // If we compute backwards of output with a standard convolution on the output
27 // with the same parameters, we would end up with a 2x2 grad_input (because you
28 // can slide the stencil over to the right once and down once). But that is all
29 // out-of-bounds if you're computing backwards for a 1x1 input.
30 //
31 // "Now Edward," you might say, "the real problem is that you set output_padding
32 // >= stride, surely an error should have been raised in this case." To
33 // understand why it is useful to handle this case, we have to understand how we
34 // compute the weight gradient of a convolution. Suppose we have a convolution
35 // with kernel=2, stride=2 on a 5x5 input. Let us see all the contributions of
36 // weight[0][0] (which we have labeled w) in the output:
37 //
38 // Input: a.b.. Weight: w.
39 // ..... ..
40 // c.d..
41 // .....
42 // .....
43 //
44 // Output: [ aw+... bw+... ]
45 // [ cw+... dw+... ]
46 //
47 // From this diagram, it easy to see that we can compute the weight gradient
48 // by performing a *dilated* convolution between the input and the
49 // output gradients with kernel=2, dilation=2, stride=1. But there's a rub: if
50 // we do a dilated convolution directly, we'll end up with a 3x3 weight
51 // gradient, when we clearly wanted a 2x2. So how do we avoid going out
52 // of bounds? We could add a notion of 'output_padding' for non-transposed
53 // convolution, but another simple and effective fix is to just accept
54 // the desired output size directly, and compute only within those bounds.
55 //
56 //
57 // ALSO do vol2col

59 static void THNN_(im2col)(const scalar_t* data_im, const int64_t channels,
60  const int64_t height, const int64_t width,
61  const int64_t output_height, const int64_t output_width,
62  const int64_t kernel_h, const int64_t kernel_w,
63  const int64_t pad_h, const int64_t pad_w,
64  const int64_t stride_h, const int64_t stride_w,
65  const int64_t dilation_h, const int64_t dilation_w,
66  scalar_t* data_col) {
67  const int64_t height_col = output_height;
68  const int64_t width_col = output_width;
69  const int64_t channels_col = channels * kernel_h * kernel_w;
70  for (int64_t c_col = 0; c_col < channels_col; ++c_col) {
71  int64_t w_offset = c_col % kernel_w;
72  int64_t h_offset = (c_col / kernel_w) % kernel_h;
73  int64_t c_im = c_col / kernel_h / kernel_w;
74  for (int64_t h_col = 0; h_col < height_col; ++h_col) {
75  int64_t h_im = h_col * stride_h - pad_h + h_offset * dilation_h;
76  for (int64_t w_col = 0; w_col < width_col; ++w_col) {
77  int64_t w_im = w_col * stride_w - pad_w + w_offset * dilation_w;
78  data_col[(c_col * height_col + h_col) * width_col + w_col] =
79  (h_im >= 0 && w_im >= 0 && h_im < height && w_im < width) ?
80  data_im[(c_im * height + h_im) * width + w_im] : 0;
81  }
82  }
83  }

86 static void THNN_(col2im)(const scalar_t* data_col, const int64_t channels,
87  const int64_t height, const int64_t width,
88  const int64_t output_height, const int64_t output_width,
89  const int64_t kernel_h, const int64_t kernel_w,
90  const int64_t pad_h, const int64_t pad_w,
91  const int64_t stride_h, const int64_t stride_w,
92  const int64_t dilation_h, const int64_t dilation_w,
93  scalar_t* data_im) {
94  memset(data_im, 0, sizeof(scalar_t) * height * width * channels);
95  const int64_t height_col = output_height;
96  const int64_t width_col = output_width;
97  const int64_t channels_col = channels * kernel_h * kernel_w;
98  for (int64_t c_col = 0; c_col < channels_col; ++c_col) {
99  int64_t w_offset = c_col % kernel_w;
100  int64_t h_offset = (c_col / kernel_w) % kernel_h;
101  int64_t c_im = c_col / kernel_h / kernel_w;
102  for (int64_t h_col = 0; h_col < height_col; ++h_col) {
103  int64_t h_im = h_col * stride_h - pad_h + h_offset * dilation_h;
104  for (int64_t w_col = 0; w_col < width_col; ++w_col) {
105  int64_t w_im = w_col * stride_w - pad_w + w_offset * dilation_w;
106  if (h_im >= 0 && h_im < height && w_im >= 0 && w_im < width)
107  data_im[(c_im * height + h_im) * width + w_im] +=
108  data_col[(c_col * height_col + h_col) * width_col + w_col];
109  }
110  }
111  }

114 static inline void THNN_(Col2Im_shapeCheck)(
115  THNNState *state,
116  THTensor *input,
117  THTensor *gradOutput,
118  int64_t outputHeight, int64_t outputWidth,
119  int64_t kH, int64_t kW, int64_t dilationH, int64_t dilationW,
120  int64_t padH, int64_t padW, int64_t dH, int64_t dW) {
121 
122  THArgCheck(kW > 0 && kH > 0, 6,
123  "kernel size should be greater than zero, but got kH: %d kW: %d", kH, kW);
124  THArgCheck(dW > 0 && dH > 0, 12,
125  "stride should be greater than zero, but got dH: %d dW: %d", dH, dW);
126  THArgCheck(dilationW > 0 && dilationH > 0, 8,
127  "dilation should be greater than zero, but got dilationH: %d dilationW: %d", dilationH, dilationW);
128 
129  int64_t ndim = THTensor_(nDimensionLegacyNoScalars)(input);
130  THNN_ARGCHECK(!input->is_empty() && (ndim == 2 || ndim == 3), 2, input,
131  "Expected non-empty 2D or 3D input tensor, but got input of shape %s");
132 
133  int64_t batch_dim = (ndim == 3) ? 0 : -1;
134  int64_t nInputPlane = input->size(batch_dim + 1);
135 
136  if (nInputPlane % (kW * kH) != 0) {
137  THError("Expected size of input's dimension 1 to be divisible by the "
138  "product of kernel_size, but got input.size(1)=%lld and "
139  "kernel_size=(%d, %d).", (long long) nInputPlane, kH, kW);
140  }
141 
142  int64_t inputLength = input->size(batch_dim + 2);
143  int64_t nBlocksH = div_rtn<int64_t>(outputHeight + 2 * padH - dilationH * (kH - 1) - 1, dH) + 1;
144  int64_t nBlocksW = div_rtn<int64_t>(outputWidth + 2 * padW - dilationW * (kW - 1) - 1, dW) + 1;
145 
146  if (inputLength != (nBlocksH * nBlocksW)) {
147  THError("Given output_size=(%d, %d), kernel_size=(%d, %d), "
148  "dilation=(%d, %d), padding=(%d, %d), stride=(%d, %d), expected "
149  "size of input's dimension 2 to match the calculated number of "
150  "sliding blocks %lld * %lld = %lld, but got input.size(2)=%lld.",
151  outputHeight, outputWidth, kH, kW, dilationH, dilationW, padH, padW, dH, dW,
152  (long long) nBlocksH, (long long) nBlocksW,
153  (long long) (nBlocksH * nBlocksW), (long long) inputLength);
154  }
155 
156  if (outputWidth < 1 || outputHeight < 1) {
157  THError("Expected output spatial size to be positive, but got: output_size=(%d, %d).",
158  outputHeight, outputWidth);
159  }
160 }

190

192  THTensor *output_n = THTensor_(new)();
193 
194  int64_t height_col = (outputHeight + 2 * padH - (dilationH * (kH - 1) + 1)) / dH + 1;
195  int64_t width_col = (outputWidth + 2 * padW - (dilationW * (kW - 1) + 1)) / dW + 1;
196 
197  for (int64_t elt = 0; elt < batchSize; elt++) {
198  THTensor_(select)(input_n, input, 0, elt);
199  THTensor_(select)(output_n, output, 0, elt);
200 
201  THNN_(col2im)(

203  nOutputPlane,

207  padH, padW,
208  dH, dW,
209  dilationH, dilationW, output_n->data<scalar_t>());
210  }
211 
212  c10::raw::intrusive_ptr::decref(input_n);
213  c10::raw::intrusive_ptr::decref(output_n);
214

217  }
