26  // clip_coordinates_set_grad works similarly to clip_coordinates except that
27  // it also returns the `d output / d input` via pointer argument `grad_in`.
28  // This is useful in the backward pass of grid_sampler.
29  template<typename scalar_t>
30  static inline scalar_t clip_coordinates_set_grad(scalar_t in, int64_t clip_limit,
31  scalar_t *grad_in) {
32  if (in < static_cast<scalar_t>(0)) {
33  *grad_in = static_cast<scalar_t>(0);
34  return static_cast<scalar_t>(0);
35  } else {
36  scalar_t max = static_cast<scalar_t>(clip_limit - 1);
37  if (in > max) {
38  *grad_in = static_cast<scalar_t>(0);
39  return max;
40  } else {
41  *grad_in = static_cast<scalar_t>(1);
42  return in;
43  }
44  }

47  template<typename scalar_t>
48  static inline scalar_t reflect_coordinates(scalar_t in, int64_t clip_limit) {
49  if (clip_limit == static_cast<int64_t>(1)) {
50  return static_cast<scalar_t>(0);
51  }
52  in = std::fabs(in);
53  scalar_t max = static_cast<scalar_t>(clip_limit - 1);
54  // `fmod` returns same sign as `in`, which is positive after the `fabs` above.
55  scalar_t extra = std::fmod(in, max);
56  int flips = static_cast<int>(std::floor(in / max));
57  if (flips % 2 == 0) {
58  return extra;
59  } else {
60  return max - extra;
61  }
62  }

64  // reflect_coordinates_set_grad works similarly to reflect_coordinates except
65  // that it also returns the `d output / d input` via pointer argument
66  // `grad_in`.
67  // This is useful in the backward pass of grid_sampler.
68  template<typename scalar_t>
69  static inline scalar_t reflect_coordinates_set_grad(scalar_t in, int64_t clip_limit,
70  scalar_t *grad_in) {
71  if (clip_limit == static_cast<int64_t>(1)) {
72  *grad_in = static_cast<scalar_t>(0);
73  return static_cast<scalar_t>(0);
74  }
75  int grad_in_mult_;
76  if (in < static_cast<scalar_t>(0)) {
77  grad_in_mult_ = -1;
78  in = -in;
79  } else {
80  grad_in_mult_ = 1;
81  }
82  scalar_t max = static_cast<scalar_t>(clip_limit - 1);
83  // `fmod` returns same sign as `in`, which is positive after the `if` above.
84  scalar_t extra = std::fmod(in, max);
85  int flips = static_cast<int>(std::floor(in / max));
86  if (flips % 2 == 0) {
87  *grad_in = static_cast<scalar_t>(grad_in_mult_);
88  return extra;
89  } else {
90  *grad_in = static_cast<scalar_t>(-grad_in_mult_);
91  return max - extra;
92  }
93  }

122  template<typename scalar_t>
123  Tensor grid_sampler_3d_cpu_impl(const Tensor& input, const Tensor& grid,
124  GridSamplerInterpolation interpolation_mode,
125  GridSamplerPadding padding_mode) {
126  int64_t N = input.size(0);
127  int64_t C = input.size(1);
128  int64_t inp_D = input.size(2);
129  int64_t inp_H = input.size(3);
130  int64_t inp_W = input.size(4);
131  int64_t out_D = grid.size(1);
132  int64_t out_H = grid.size(2);
133  int64_t out_W = grid.size(3);
134  auto output = at::empty({N, C, out_D, out_H, out_W}, input.options());
135  int64_t inp_sN = input.stride(0);
136  int64_t inp_sC = input.stride(1);
137  int64_t inp_sD = input.stride(2);
138  int64_t inp_sH = input.stride(3);
139  int64_t inp_sW = input.stride(4);
140  int64_t grid_sN = grid.stride(0);
141  int64_t grid_sD = grid.stride(1);
142  int64_t grid_sH = grid.stride(2);
143  int64_t grid_sW = grid.stride(3);
144  int64_t grid_sCoor = grid.stride(4);
145  int64_t out_sN = output.stride(0);
146  int64_t out_sC = output.stride(1);
147  int64_t out_sD = output.stride(2);
148  int64_t out_sH = output.stride(3);
149  int64_t out_sW = output.stride(4);
150  scalar_t *inp_ptr = input.data<scalar_t>();
151  scalar_t *out_ptr = output.data<scalar_t>();
152  scalar_t *grid_ptr = grid.data<scalar_t>();
153  // loop over each output pixel
154  #ifdef _OPENMP
155  #pragma omp parallel for
156  #endif
157  for (int64_t n = 0; n < N; ++n) {
158  scalar_t *grid_ptr_N = grid_ptr + n * grid_sN;
159  scalar_t *inp_ptr_N = inp_ptr + n * inp_sN;
160  for (int64_t d = 0; d < out_D; ++d) {
161  for (int64_t h = 0; h < out_H; ++h) {
162  for (int64_t w = 0; w < out_W; ++w) {
163  // get the corresponding input x, y, z co-ordinates from grid
164  scalar_t *grid_ptr_NDHW = grid_ptr_N + d * grid_sD + h * grid_sH + w * grid_sW;
165  scalar_t ix = *grid_ptr_NDHW;
166  scalar_t iy = grid_ptr_NDHW[grid_sCoor];
167  scalar_t iz = grid_ptr_NDHW[2 * grid_sCoor];
168 
169  // normalize ix, iy, iz from [-1, 1] to [0, inp_W-1] & [0, inp_H-1] & [0, inp_D-1]
170  ix = ((ix + 1) / 2) * (inp_W - 1);
171  iy = ((iy + 1) / 2) * (inp_H - 1);
172  iz = ((iz + 1) / 2) * (inp_D - 1);
173 
174  if (padding_mode == GridSamplerPadding::Border) {
175  // clip coordinates to image borders
176  ix = clip_coordinates(ix, inp_W);
177  iy = clip_coordinates(iy, inp_H);
178  iz = clip_coordinates(iz, inp_D);
179  } else if (padding_mode == GridSamplerPadding::Reflection) {
180  // reflect coordinates by image borders
181  ix = reflect_coordinates(ix, inp_W);
182  iy = reflect_coordinates(iy, inp_H);
183  iz = reflect_coordinates(iz, inp_D);
184  }
185 
186  if (interpolation_mode == GridSamplerInterpolation::Bilinear) {
187  // get corner pixel values from (x, y, z)
188  // for 4d, we used north-east-south-west
189  // for 5d, we add top-bottom
190  int64_t ix_tnw = static_cast<int64_t>(std::floor(ix));
191  int64_t iy_tnw = static_cast<int64_t>(std::floor(iy));
192  int64_t iz_tnw = static_cast<int64_t>(std::floor(iz));
193 
194  int64_t ix_tne = ix_tnw + 1;
195  int64_t iy_tne = iy_tnw;
196  int64_t iz_tne = iz_tnw;
197 
198  int64_t ix_tsw = ix_tnw;
199  int64_t iy_tsw = iy_tnw + 1;
200  int64_t iz_tsw = iz_tnw;
201 
202  int64_t ix_tse = ix_tnw + 1;
203  int64_t iy_tse = iy_tnw + 1;
204  int64_t iz_tse = iz_tnw;
205 
206  int64_t ix_bnw = ix_tnw;
207  int64_t iy_bnw = iy_tnw;
208  int64_t iz_bnw = iz_tnw + 1;
209 
210  int64_t ix_bne = ix_tnw + 1;
211  int64_t iy_bne = iy_tnw;
212  int64_t iz_bne = iz_tnw + 1;
213 
214  int64_t ix_bsw = ix_tnw;
215  int64_t iy_bsw = iy_tnw + 1;
216  int64_t iz_bsw = iz_tnw + 1;
217 
218  int64_t ix_bse = ix_tnw + 1;
219  int64_t iy_bse = iy_tnw + 1;
220  int64_t iz_bse = iz_tnw + 1;
221 
222  // get surfaces to each neighbor:
223  scalar_t tnw = (ix_bse - ix) * (iy_bse - iy) * (iz_bse - iz);
224  scalar_t tne = (ix - ix_bsw) * (iy_bsw - iy) * (iz_bsw - iz);
225  scalar_t tsw = (ix_bne - ix) * (iy - iy_bne) * (iz_bne - iz);
226  scalar_t tse = (ix - ix_bnw) * (iy - iy_bnw) * (iz_bnw - iz);
227  scalar_t bnw = (ix_tse - ix) * (iy_tse - iy) * (iz - iz_tse);
228  scalar_t bne = (ix - ix_tsw) * (iy_tsw - iy) * (iz - iz_tsw);
229  scalar_t bsw = (ix_tne - ix) * (iy - iy_tne) * (iz - iz_tne);
230  scalar_t bse = (ix - ix_tnw) * (iy - iy_tnw) * (iz - iz_tnw);
231 
232  // calculate bilinear weighted pixel value and set output pixel
233  scalar_t *out_ptr_NCDHW = out_ptr + n * out_sN + d * out_sD + h * out_sH + w * out_sW;
234  scalar_t *inp_ptr_NC = inp_ptr_N;
235  for (int c = 0; c < C; ++c, out_ptr_NCDHW += out_sC, inp_ptr_NC += inp_sC) {
236  // (c, iz_tnw, iy_tnw, ix_tnw) * tnw + (c, iz_tne, iy_tne, ix_tne) * tne
237  // + (c, iz_tsw, iy_tsw, ix_tsw) * tsw + (c, iz_tse, iy_tse, ix_tse) * tse
238  // + (c, iz_bnw, iy_bnw, ix_bnw) * bnw + (c, iz_bne, iy_bne, ix_bne) * bne
239  // + (c, iz_bsw, iy_bsw, ix_bsw) * bsw + (c, iz_bse, iy_bse, ix_bse) * bse
240  *out_ptr_NCDHW = static_cast<scalar_t>(0);
241  if (within_bounds_3d(iz_tnw, iy_tnw, ix_tnw, inp_D, inp_H, inp_W)) {
242  *out_ptr_NCDHW += inp_ptr_NC[iz_tnw * inp_sD + iy_tnw * inp_sH + ix_tnw * inp_sW] * tnw;
243  }
244  if (within_bounds_3d(iz_tne, iy_tne, ix_tne, inp_D, inp_H, inp_W)) {
245  *out_ptr_NCDHW += inp_ptr_NC[iz_tne * inp_sD + iy_tne * inp_sH + ix_tne * inp_sW] * tne;
246  }
247  if (within_bounds_3d(iz_tsw, iy_tsw, ix_tsw, inp_D, inp_H, inp_W)) {
248  *out_ptr_NCDHW += inp_ptr_NC[iz_tsw * inp_sD + iy_tsw * inp_sH + ix_tsw * inp_sW] * tsw;
249  }
250  if (within_bounds_3d(iz_tse, iy_tse, ix_tse, inp_D, inp_H, inp_W)) {
251  *out_ptr_NCDHW += inp_ptr_NC[iz_tse * inp_sD + iy_tse * inp_sH + ix_tse * inp_sW] * tse;
252  }
253  if (within_bounds_3d(iz_bnw, iy_bnw, ix_bnw, inp_D, inp_H, inp_W)) {
254  *out_ptr_NCDHW += inp_ptr_NC[iz_bnw * inp_sD + iy_bnw * inp_sH + ix_bnw * inp_sW] * bnw;
255  }
256  if (within_bounds_3d(iz_bne, iy_bne, ix_bne, inp_D, inp_H, inp_W)) {
257  *out_ptr_NCDHW += inp_ptr_NC[iz_bne * inp_sD + iy_bne * inp_sH + ix_bne * inp_sW] * bne;
258  }
259  if (within_bounds_3d(iz_bsw, iy_bsw, ix_bsw, inp_D, inp_H, inp_W)) {
260  *out_ptr_NCDHW += inp_ptr_NC[iz_bsw * inp_sD + iy_bsw * inp_sH + ix_bsw * inp_sW] * bsw;
261  }
262  if (within_bounds_3d(iz_bse, iy_bse, ix_bse, inp_D, inp_H, inp_W)) {
263  *out_ptr_NCDHW += inp_ptr_NC[iz_bse * inp_sD + iy_bse * inp_sH + ix_bse * inp_sW] * bse;
264  }
265  }
266  } else if (interpolation_mode == GridSamplerInterpolation::Nearest) {
267  int64_t ix_nearest = static_cast<int64_t>(std::round(ix));
268  int64_t iy_nearest = static_cast<int64_t>(std::round(iy));
269  int64_t iz_nearest = static_cast<int64_t>(std::round(iz));
270 
271  // assign nearest neighor pixel value to output pixel
272  scalar_t *out_ptr_NCDHW = out_ptr + n * out_sN + d * out_sD + h * out_sH + w * out_sW;
273  scalar_t *inp_ptr_NC = inp_ptr_N;
274  for (int c = 0; c < C; ++c, out_ptr_NCDHW += out_sC, inp_ptr_NC += inp_sC) {
275  if (within_bounds_3d(iz_nearest, iy_nearest, ix_nearest, inp_D, inp_H, inp_W)) {
276  *out_ptr_NCDHW = inp_ptr_NC[iz_nearest * inp_sD + iy_nearest * inp_sH + ix_nearest * inp_sW];
277  } else {
278  *out_ptr_NCDHW = static_cast<scalar_t>(0);
279  }
280  }
281  }
282  }
283  }
284  }
285  }
286  return output;
287  }

420  scalar_t bne = (ix - ix_tsw) * (iy_tsw - iy) * (iz - iz_tsw);
421  scalar_t bsw = (ix_tne - ix) * (iy - iy_tne) * (iz - iz_tne);
422  scalar_t bse = (ix - ix_tnw) * (iy - iy_tnw) * (iz - iz_tnw);
423 
424  scalar_t gix = static_cast<scalar_t>(0), giy = static_cast<scalar_t>(0), giz = static_cast<scalar_t>(0);
425  scalar_t *gOut_ptr_NCDHW = gOut_ptr + n * gOut_sN + d * gOut_sD + h * gOut_sH + w * gOut_sW;
426  scalar_t *gInp_ptr_NC = gInp_ptr + n * gInp_sN;
427  scalar_t *inp_ptr_NC = inp_ptr_N;
428  // calculate bilinear weighted pixel value and set output pixel
429  for (int c = 0; c < C; ++c, gOut_ptr_NCDHW += gOut_sC, gInp_ptr_NC += gInp_sC, inp_ptr_NC += inp_sC) {
430  scalar_t gOut = *gOut_ptr_NCDHW;
431 
432  // calculate and set grad_input
433  safe_add_3d(gInp_ptr_NC, iz_tnw, iy_tnw, ix_tnw, gInp_sD, gInp_sH, gInp_sW, inp_D, inp_H, inp_W, tnw * gOut);
434  safe_add_3d(gInp_ptr_NC, iz_tne, iy_tne, ix_tne, gInp_sD, gInp_sH, gInp_sW, inp_D, inp_H, inp_W, tne * gOut);
435  safe_add_3d(gInp_ptr_NC, iz_tsw, iy_tsw, ix_tsw, gInp_sD, gInp_sH, gInp_sW, inp_D, inp_H, inp_W, tsw * gOut);
436  safe_add_3d(gInp_ptr_NC, iz_tse, iy_tse, ix_tse, gInp_sD, gInp_sH, gInp_sW, inp_D, inp_H, inp_W, tse * gOut);
437  safe_add_3d(gInp_ptr_NC, iz_bnw, iy_bnw, ix_bnw, gInp_sD, gInp_sH, gInp_sW, inp_D, inp_H, inp_W, bnw * gOut);
438  safe_add_3d(gInp_ptr_NC, iz_bne, iy_bne, ix_bne, gInp_sD, gInp_sH, gInp_sW, inp_D, inp_H, inp_W, bne * gOut);
439  safe_add_3d(gInp_ptr_NC, iz_bsw, iy_bsw, ix_bsw, gInp_sD, gInp_sH, gInp_sW, inp_D, inp_H, inp_W, bsw * gOut);
440  safe_add_3d(gInp_ptr_NC, iz_bse, iy_bse, ix_bse, gInp_sD, gInp_sH, gInp_sW, inp_D, inp_H, inp_W, bse * gOut);
441 
442  // calculate grad_grid
443  if (within_bounds_3d(iz_tnw, iy_tnw, ix_tnw, inp_D, inp_H, inp_W)) {
444  scalar_t tnw_val = inp_ptr_NC[iz_tnw * inp_sD + iy_tnw * inp_sH + ix_tnw * inp_sW];
445  gix -= tnw_val * (iy_bse - iy) * (iz_bse - iz) * gOut;
446  giy -= tnw_val * (ix_bse - ix) * (iz_bse - iz) * gOut;
447  giz -= tnw_val * (ix_bse - ix) * (iy_bse - iy) * gOut;
448  }
449  if (within_bounds_3d(iz_tne, iy_tne, ix_tne, inp_D, inp_H, inp_W)) {
450  scalar_t tne_val = inp_ptr_NC[iz_tne * inp_sD + iy_tne * inp_sH + ix_tne * inp_sW];
451  gix += tne_val * (iy_bsw - iy) * (iz_bsw - iz) * gOut;
452  giy -= tne_val * (ix - ix_bsw) * (iz_bsw - iz) * gOut;
453  giz -= tne_val * (ix - ix_bsw) * (iy_bsw - iy) * gOut;
454  }
455  if (within_bounds_3d(iz_tsw, iy_tsw, ix_tsw, inp_D, inp_H, inp_W)) {
456  scalar_t tsw_val = inp_ptr_NC[iz_tsw * inp_sD + iy_tsw * inp_sH + ix_tsw * inp_sW];
457  gix -= tsw_val * (iy - iy_bne) * (iz_bne - iz) * gOut;
458  giy += tsw_val * (ix_bne - ix) * (iz_bne - iz) * gOut;
459  giz -= tsw_val * (ix_bne - ix) * (iy - iy_bne) * gOut;
460  }
461  if (within_bounds_3d(iz_tse, iy_tse, ix_tse, inp_D, inp_H, inp_W)) {
462  scalar_t tse_val = inp_ptr_NC[iz_tse * inp_sD + iy_tse * inp_sH + ix_tse * inp_sW];
463  gix += tse_val * (iy - iy_bnw) * (iz_bnw - iz) * gOut;
464  giy += tse_val * (ix - ix_bnw) * (iz_bnw - iz) * gOut;
465  giz -= tse_val * (ix - ix_bnw) * (iy - iy_bnw) * gOut;
466  }
467  if (within_bounds_3d(iz_bnw, iy_bnw, ix_bnw, inp_D, inp_H, inp_W)) {
468  scalar_t bnw_val = inp_ptr_NC[iz_bnw * inp_sD + iy_bnw * inp_sH + ix_bnw * inp_sW];
469  gix -= bnw_val * (iy_tse - iy) * (iz - iz_tse) * gOut;
470  giy -= bnw_val * (ix_tse - ix) * (iz - iz_tse) * gOut;
471  giz += bnw_val * (ix_tse - ix) * (iy_tse - iy) * gOut;
472  }
473  if (within_bounds_3d(iz_bne, iy_bne, ix_bne, inp_D, inp_H, inp_W)) {
474  scalar_t bne_val = inp_ptr_NC[iz_bne * inp_sD + iy_bne * inp_sH + ix_bne * inp_sW];
475  gix += bne_val * (iy_tsw - iy) * (iz - iz_tsw) * gOut;
476  giy -= bne_val * (ix - ix_tsw) * (iz - iz_tsw) * gOut;
477  giz += bne_val * (ix - ix_tsw) * (iy_tsw - iy) * gOut;
478  }
479  if (within_bounds_3d(iz_bsw, iy_bsw, ix_bsw, inp_D, inp_H, inp_W)) {
480  scalar_t bsw_val = inp_ptr_NC[iz_bsw * inp_sD + iy_bsw * inp_sH + ix_bsw * inp_sW];
481  gix -= bsw_val * (iy - iy_tne) * (iz - iz_tne) * gOut;
482  giy += bsw_val * (ix_tne - ix) * (iz - iz_tne) * gOut;
483  giz += bsw_val * (ix_tne - ix) * (iy - iy_tne) * gOut;
484  }
485  if (within_bounds_3d(iz_bse, iy_bse, ix_bse, inp_D, inp_H, inp_W)) {
486  scalar_t bse_val = inp_ptr_NC[iz_bse * inp_sD + iy_bse * inp_sH + ix_bse * inp_sW];
487  gix += bse_val * (iy - iy_tnw) * (iz - iz_tnw) * gOut;
488  giy += bse_val * (ix - ix_tnw) * (iz - iz_tnw) * gOut;
489  giz += bse_val * (ix - ix_tnw) * (iy - iy_tnw) * gOut;
490  }
491  }
492 
493  // un-normalize grad_grid values back to [-1, 1] constraints
494  gix = gix * (inp_W - 1) / 2;
495  giy = giy * (inp_H - 1) / 2;
496  giz = giz * (inp_D - 1) / 2;
497 
498  // assuming grad_grid is contiguous
499  gGrid_ptr_NDHW[0] = gix_mult * gix;
500  gGrid_ptr_NDHW[1] = giy_mult * giy;
501  gGrid_ptr_NDHW[2] = giz_mult * giz;
502  } else if (interpolation_mode == GridSamplerInterpolation::Nearest) {
503  int64_t ix_nearest = static_cast<int64_t>(std::round(ix));
504  int64_t iy_nearest = static_cast<int64_t>(std::round(iy));
505  int64_t iz_nearest = static_cast<int64_t>(std::round(iz));
506 
507  // assign nearest neighor pixel value to output pixel
508  scalar_t *gOut_ptr_NCDHW = gOut_ptr + n * gOut_sN + d * gOut_sD + h * gOut_sH + w * gOut_sW;
509  scalar_t *gInp_ptr_NC = gInp_ptr + n * gInp_sN;
510  for (int c = 0; c < C; ++c, gOut_ptr_NCDHW += gOut_sC, gInp_ptr_NC += gInp_sC) {
511  // calculate and set grad_input
512  safe_add_3d(gInp_ptr_NC, iz_nearest, iy_nearest, ix_nearest,
513  gInp_sD, gInp_sH, gInp_sW, inp_D, inp_H, inp_W, *gOut_ptr_NCDHW);
514  }
515  }
516  }
517  }
518  }
519  }
520  return std::make_tuple(grad_input, grad_grid);
521  }

565 Tensor grid_sampler(const Tensor& input, const Tensor& grid,
566  int64_t interpolation_mode, int64_t padding_mode) {
567  AT_CHECK(
568  input.defined() && grid.defined(),
569  "grid_sampler(): expected input and grid to not be undefined, but input "
570  "is ", input, " and grid is ", grid);
571  auto input_opt = input.options();
572  auto grid_opt = grid.options();
573  AT_CHECK(
574  input_opt.device() == grid_opt.device(),
575  "grid_sampler(): expected input and grid to be on same device, but input "
576  "is on ", input_opt.device(), " and grid is on ", grid_opt.device());
577  AT_CHECK(
578  input_opt.dtype() == grid_opt.dtype(),
579  "grid_sampler(): expected input and grid to have same dtype, but input "
580  "has ", input_opt.dtype(), " and grid has ", grid_opt.dtype());
581  AT_CHECK(
582  input_opt.layout() == kStrided && grid_opt.layout() == kStrided,
583  "grid_sampler(): expected input and grid to have torch.strided layout, but "
584  "input has ", input_opt.layout(), " and grid has ", grid_opt.layout());
585  AT_CHECK(
586  (input.dim() == 4 || input.dim() == 5) && input.dim() == grid.dim(),
587  "grid_sampler(): expected 4D or 5D input and grid with same number of "
588  "dimensions, but got input with sizes ", input.sizes(),
589  " and grid with sizes ", grid.sizes());
590  AT_CHECK(
591  input.size(0) == grid.size(0),
592  "grid_sampler(): expected grid and input to have same batch size, but got "
593  "input with sizes ", input.sizes(), " and grid with sizes ", grid.sizes());
594  AT_CHECK(
595  grid.size(-1) == input.dim() - 2,
596  "grid_sampler(): expected grid to have size ", input.dim() - 2, " in last "
597  "dimension, but got grid with sizes ", grid.sizes());
598  for (int64_t i = 2; i < input.dim(); i++) {
599  AT_CHECK(input.size(i) > 0,
600  "grid_sampler(): expected input to have non-empty spatial dimensions, "
601  "but input has sizes ", input.sizes(), " with dimension ", i, " being "
602  "empty");
603  }
604  // cudnn does not support inputs larger than 1024
605  if (at::native::cudnn_is_acceptable(input) &&
606  at::native::cudnn_is_acceptable(grid) &&
607  static_cast<GridSamplerInterpolation>(interpolation_mode) == GridSamplerInterpolation::Bilinear &&
608  static_cast<GridSamplerPadding>(padding_mode) == GridSamplerPadding::Zeros &&
609  input.dim() == 4 &&
610  input.size(1) <= 1024) {
611  return cudnn_grid_sampler(input, grid);
612  }
613  if (input.dim() == 4) {
614  return at::grid_sampler_2d(input, grid, interpolation_mode, padding_mode);
615  } else {
616  return at::grid_sampler_3d(input, grid, interpolation_mode, padding_mode);
617  }
618 }
