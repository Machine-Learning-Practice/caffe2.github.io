86  void connect(typename WrappedGraph::NodeRef n) {
87  n->mutableData()->Index = Index;
88  n->mutableData()->LowLink = Index;
89  Index++;
90 
91  Stack.emplace_back(n);
92  n->mutableData()->OnStack = true;
93 
94  for (const auto& outEdge : n->getOutEdges()) {
95  typename WrappedGraph::NodeRef newNode = outEdge->head();
96  // Check if we've considered this node before.
97  if (newNode->data().Index == -1) {
98  connect(newNode);
99  n->mutableData()->LowLink =
100  std::min(n->data().LowLink, newNode->data().LowLink);
101  // Check if newNode is in the SCC.
102  } else if (newNode->data().OnStack) {
103  n->mutableData()->LowLink =
104  std::min(n->data().LowLink, newNode->data().Index);
105  }
106  }
107 
108  // If our node is a root node, pop it from the stack (we've found an SCC)
109  if (n->data().LowLink == n->data().Index) {
110  WrappedSubgraph wrappedSCC;
111 
112  typename WrappedGraph::NodeRef w;
113  do {
114  w = Stack.back();
115  w->mutableData()->OnStack = false;
116  Stack.pop_back();
117  wrappedSCC.addNode(w);
118  } while (w != n);
119 
120  // Add all the edges into the SCC.
121  // TODO include edges in the SCC in a smarter way.
122  const auto& sccNodes = wrappedSCC.getNodes();
123  for (const auto& sccNode : sccNodes) {
124  for (const auto& outEdge : sccNode->getOutEdges()) {
125  if (std::find(sccNodes.begin(), sccNodes.end(), outEdge->head()) !=
126  sccNodes.end()) {
127  wrappedSCC.addEdge(outEdge);
128  }
129  }
130  }
131  WrappedSCCs.emplace_back(wrappedSCC);
132  }
133  }
