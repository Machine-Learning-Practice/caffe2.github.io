65 static PyObject * THPModule_initNames(PyObject *self, PyObject *arg)
66 {
67  static std::vector<std::string> names;
68 
69  THPObjectPtr types(PySequence_Fast(arg, "expected a sequence"));
70  if (!types) return nullptr;
71 
72  int num_classes = PySequence_Fast_GET_SIZE(types.get());
73  names.reserve(names.size() + num_classes);
74  for (int i = 0; i < num_classes; i++) {
75  PyObject* obj = PySequence_Fast_GET_ITEM(types.get(), i);
76  THPUtils_assert(PyType_Check(obj), "expected a PyTypeObject");
77  PyTypeObject* type = (PyTypeObject*)obj;
78 
79  THPObjectPtr module_name(PyObject_GetAttrString(obj, "__module__"));
80  if (!module_name) return nullptr;
81  THPUtils_assert(THPUtils_checkString(module_name.get()),
82  "expected __module__ to be a string");
83  std::string name = THPUtils_unpackString(module_name.get());
84  names.push_back(name + "." + type->tp_name);
85  type->tp_name = names.back().c_str();
86  }
87  Py_RETURN_NONE;
88 }

178 PyObject *THPModule_safeCall(PyObject *_unused, PyObject *args, PyObject *kwargs)
179 {
180  PyObject *result = nullptr;
181  PyObject *args_slice = nullptr;
182  PyThreadState *thread_state = PyThreadState_Get();
183  Py_ssize_t num_args = args ? PyTuple_Size(args) : 0;
184  THPUtils_assert(num_args > 0, "expected at least one argument");
185  try {
186  args_slice = PyTuple_GetSlice(args, 1, num_args);
187  result = PyObject_Call(PyTuple_GET_ITEM(args, 0), args_slice, kwargs);
188  } catch (std::exception &e) {
189  PyEval_RestoreThread(thread_state);
190  Py_DECREF(args_slice);
191  PyErr_SetString(THPException_FatalError, e.what());
192  Py_LeaveRecursiveCall();
193  }
194  Py_DECREF(args_slice);
195  return result;
196 }

198 PyObject *THPModule_addDocStr(PyObject *_unused, PyObject *args)
199 {
200  // adds a __doc__ string to a function, similar to numpy's arr_add_docstring
201  static std::vector<std::string> all_docs;
202  PyObject *obj;
203  PyObject *doc_obj;
204  if (!PyArg_ParseTuple(args, "OO", &obj, &doc_obj)) {
205  return nullptr;
206  }
207 
208  const char* doc_str = "<invalid string>";
209  if (THPUtils_checkString(doc_obj)) {
210  all_docs.push_back(THPUtils_unpackString(doc_obj));
211  doc_str = all_docs.back().c_str();
212  }
213 
214  if (Py_TYPE(obj) == &PyCFunction_Type) {
215  PyCFunctionObject* f = (PyCFunctionObject *)obj;
216  if (f->m_ml->ml_doc) {
217  return PyErr_Format(PyExc_RuntimeError,
218  "function '%s' already has a docstring", f->m_ml->ml_name);
219  }
220  f->m_ml->ml_doc = doc_str;
221  } else if (strcmp(Py_TYPE(obj)->tp_name, "method_descriptor") == 0) {
222  PyMethodDescrObject* m = (PyMethodDescrObject *)obj;
223  if (m->d_method->ml_doc) {
224  return PyErr_Format(PyExc_RuntimeError,
225  "method '%s' already has a docstring", m->d_method->ml_name);
226  }
227  m->d_method->ml_doc = doc_str;
228  } else if (strcmp(Py_TYPE(obj)->tp_name, "getset_descriptor") == 0) {
229  //NOLINTNEXTLINE(cppcoreguidelines-pro-type-cstyle-cast)
230  PyGetSetDescrObject* m = (PyGetSetDescrObject *)obj;
231  if (m->d_getset->doc) {
232  //NOLINTNEXTLINE(cppcoreguidelines-pro-type-vararg)
233  return PyErr_Format(PyExc_RuntimeError,
234  "attribute '%s' already has a docstring", m->d_getset->name);
235  }
236  // This field is not const for python < 3.7 yet the content is
237  // never modified.
238  //NOLINTNEXTLINE(cppcoreguidelines-pro-type-const-cast)
239  m->d_getset->doc = const_cast<char *>(doc_str);
240  } else {
241  return PyErr_Format(PyExc_TypeError,
242  "don't know how to add docstring to type '%s'", Py_TYPE(obj)->tp_name);
243  }
244 
245  Py_INCREF(obj);
246  return obj;
247 }

326 PyObject *THPModule_fromDLPack(PyObject *_unused, PyObject *data)
327 {
328  using namespace torch::autograd;
329  HANDLE_TH_ERRORS
330  DLManagedTensor * dlMTensor = (DLManagedTensor *)PyCapsule_GetPointer(data, "dltensor");
331  THPUtils_assert(dlMTensor, "from_dlpack received an invalid capsule. "
332  "Note that DLTensor capsules can be consumed only once, "
333  "so you might have already constructed a tensor from it once.")
334  // atensor steals the ownership of the underlying storage. It also passes a
335  // destructor function that will be called when the underlying storage goes
336  // out of scope. When the destructor is called, the dlMTensor is destructed too.
337  auto atensor = make_variable(at::fromDLPack(dlMTensor), false);
338 
339  // It is possible that the call to at::fromDLPack is the very first
340  // call to create a Tensor in PyTorch. If so, then _lazy_init has
341  // not been called, and the attempt to call createPyObject will fail
342  // because cuda ATen types have not been registered in Python yet.
343  // so if we have a cuda tensor, then we need to make sure
344  // we have called _lazy_init here
345  if(atensor.is_cuda()) {
346  py::module::import("torch.cuda").attr("init")();
347  }
348  // Make sure this capsule will never be used again.
349  PyCapsule_SetName(data, "used_dltensor");
350  return THPVariable_Wrap(std::move(atensor));
351  END_HANDLE_TH_ERRORS
352 }
