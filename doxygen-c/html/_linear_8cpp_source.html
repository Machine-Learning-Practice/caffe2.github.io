26 // sumproduct_pair computes `(left*right).sum(sumdims)` by means of permutation and
27 // batch matrix multiplication
28 // its main purpose is to provide a pairwise reduction for einsum
29 static Tensor sumproduct_pair(const Tensor& left_, const Tensor& right_, IntArrayRef sum_dims_, bool keepdim) {
30  // assumes that tensors have been pre-unsqueezed (so that all dimensions match - after broadcasting)
31  // but makes no other assumptions on the order of dimensions
32  AT_CHECK(left_.dim()==right_.dim(), "number of dimensions must match");
33  if (sum_dims_.size() == 0)
34  return at::mul(left_, right_);
35  int64_t dim = left_.dim();
36  auto sum_dims = at::dim_list_to_bitset(sum_dims_, dim);
37  // dimensions that will be part of the output (i.e. not summed over) in three vectors
38  // dims in lro appear in left, right and output, similarly lo: left and output, ro: right and output
39  // also the sizes are kept track of for reshaping
40  std::vector<int64_t> lro, lo, ro;
41  int64_t lro_size = 1, lo_size = 1, ro_size = 1, sum_size = 1;
42  Tensor left = left_;
43  Tensor right = right_;
44  for (int64_t i = 0; i < dim; i++) {
45  auto sl = left.size(i)>1;
46  auto sr = right.size(i)>1;
47  if (sum_dims[i]) { // first dimensions that will be summed over after multiplication
48  if (sl && sr) { // dimensions nontrivially in both left and right must be of the same size
49  AT_CHECK(left.size(i)==right.size(i), "non-broadcast dimensions must match");
50  sum_size *= left.size(i);
51  } else if (sl) { // if it is only in one of left and right, we can sum right away
52  left = left.sum(i, true);
53  } else if (sr) {
54  right = right.sum(i, true);
55  }
56  } else if (sl && sr) { // now deal with dimensions dimensions that will be in the output
57  // dimensions nontrivially in both left and right must be of the same size
58  AT_CHECK(left.size(i)==right.size(i), "non-broadcast dimensions must match");
59  lro.push_back(i);
60  lro_size *= left.size(i);
61  } else if (sl) { // keep track of dimensions appearing only once
62  lo.push_back(i);
63  lo_size *= left.size(i);
64  } else {
65  ro.push_back(i);
66  ro_size *= right.size(i);
67  }
68  }
69  // we now work with the following permutations / shapes.
70  // the pipeline is permute inputs -> reshape inputs -> batch matrix mul -> reshape(view) output -> permute output
71  // output: "lro, lo, 1-for-summed-dims, ro" with orgiginal shape dimensions
72  // left: "lro, lo, summed" permuted with lpermutation and the three flattened
73  // right: "lro, summed, ro" permuted with rpermutation and the three flattened
74  // then the permuted output is a view of bmm(left, right)
75  // finally, opermutation reverts the permutation to the original order of dimensions
76  std::vector<int64_t> out_size;
77  for (auto& d : lro) out_size.push_back(left.size(d));
78  for (auto& d : lo) out_size.push_back(left.size(d));
79  for (auto& d : sum_dims_) { out_size.push_back(1); (void)(d); }; // avoid warining about not using d
80  for (auto& d : ro) out_size.push_back(right.size(d));
81 
82  std::vector<int64_t> lpermutation(lro);
83  lpermutation.insert(lpermutation.end(), lo.begin(), lo.end());
84  lpermutation.insert(lpermutation.end(), sum_dims_.begin(), sum_dims_.end());
85  lpermutation.insert(lpermutation.end(), ro.begin(), ro.end());
86 
87  std::vector<int64_t> rpermutation(lro);
88  rpermutation.insert(rpermutation.end(), sum_dims_.begin(), sum_dims_.end());
89  rpermutation.insert(rpermutation.end(), ro.begin(), ro.end());
90  rpermutation.insert(rpermutation.end(), lo.begin(), lo.end());
91 
92  std::vector<int64_t> opermutation(lro.size()+lo.size()+sum_dims_.size()+ro.size(), -1);
93  {
94  int64_t i = 0;
95 
96  for (auto it = lro.begin(); it != lro.end(); i++, it++) {
97  opermutation[*it] = i;
98  }
99  for (auto it = lo.begin(); it != lo.end(); i++, it++) {
100  opermutation[*it] = i;
101  }
102  for (auto it = sum_dims_.begin(); it != sum_dims_.end(); i++, it++) {
103  opermutation[*it] = i;
104  }
105  for (auto it = ro.begin(); it != ro.end(); i++, it++) {
106  opermutation[*it] = i;
107  }
108  }
109 
110  // now we can execute the operations above
111  left = left.permute(lpermutation).reshape({lro_size, lo_size, sum_size});
112  right = right.permute(rpermutation).reshape({lro_size, sum_size, ro_size});
113  Tensor result = at::bmm(left, right);
114  result = result.view(out_size).permute(opermutation);
115 
116  // finally squeeze summed dimensions if desired
117  if (! keepdim) {
118  for (int i = dim-1; i>=0; i--)
119  if (sum_dims[i])
120  result.squeeze_(i);
121  }
122  return result;
123 }

125 Tensor einsum(std::string eqn, TensorList tensors) {
126  constexpr size_t number_of_letters = 26;
127  std::string in_eqn;
128  size_t pos;
129  // The equation is given in terms of single lowercase letters ('a'..'z') and potentially an ellipsis.
130  // Internally, we represent it using indices from 0 to num_total_dimensions, with each letter
131  // mapped to an index and the ellipsis ('...') being mapped to a number of consequtive indices.
132  // The mapping of letters to internal indices is given in letter_mapping. A value of -1 means that
133  // the letter has not been assigned an index yet (because it has not been seen).
134  // The ellipsis is defined by first_ell_idx (the first index) and num_ell_idxes (the number of indices).
135  // A value of -1 for num_ell_idxes specifies that we have not seen an ellipsis yet.
136  // Note: The internal indices are NOT the dimensions used internally. There is a mapping to them below.
137 
138  std::array<std::int64_t, number_of_letters> letter_mapping; // map letter to internal (numerical) label
139  letter_mapping.fill(-1);
140  int64_t num_ell_idxes = -1;
141  int64_t first_ell_idx = 0;
142 
143  // The internal representation of the left hand side fo the equation (with ellipsis expanded) is stored in input_op_idxes.
144  // For each operand, we have a vector mapping each dimension to an internal index.
145  // We also keep track of the number of occurrences for each letter (to infer a right hand side if not given) and
146  // of the last occurence of each index.
147  std::vector<std::vector<int64_t>> input_op_idxes; // the parsed operand indices
148  std::array<std::int64_t, number_of_letters> num_letter_occurrences; // number of occurrence in the equation of this letter
149  num_letter_occurrences.fill(0);
150  std::vector<std::int64_t> last_idx_occurrence; // the last operator (left to right) using this index
151 
152  if ((pos = eqn.find("->")) != std::string::npos) { // check whether we have a right hand side. in_eq is the left hand side
153  in_eqn = eqn.substr(0, pos);
154  } else {
155  in_eqn = eqn;
156  }
157  // remove spaces for einsum compatibility (#9929)
158  in_eqn.erase(std::remove_if(in_eqn.begin(), in_eqn.end(), isspace), in_eqn.end());
159 
160  // next we parse in_eq (the left hand side) by iterating. It is a string of comma separated terms per index
161  int64_t operand = 0;
162  std::stringstream eqn_stream(in_eqn);
163  std::string term;
164  int64_t num_total_idxes = 0;
165  while (! eqn_stream.eof()) {
166  std::getline(eqn_stream, term, ','); // term = string with indices of current term
167  AT_CHECK((int64_t) tensors.size()>operand, "more operands in equation than tensors"); // we cannot have a longer equation than operands. We need to check here before we use the dimension
168 
169  int64_t ell_char_count = 0; // handling of ellipsis '...' is a bit tedious, we count the '.'
170  // if there is an ellipsis, the number of dimensions it represents must be total dim - letter dimensions
171  int64_t candidate_num_ell_idxes = tensors[operand].dim() - term.size() + 3;
172  int64_t dims_in_term = 0; // dimensions we have seen
173  std::vector<int64_t> current_op_idxes; // mapping of operand dimensions to indices for current term
174  for (auto &c : term) { // c = character with a single letter or '.'
175  if (c == '.') {
176  ell_char_count++;
177  AT_CHECK(ell_char_count <= 3, "can only have '.' in one ellispis '...' in term ", operand, " of the equation");
178  if (ell_char_count == 3) { // this completes the ellipsis
179  if (num_ell_idxes == -1) { // if we have not seen an ellipsis before, keep track of indices and size
180  first_ell_idx = num_total_idxes;
181  num_ell_idxes = candidate_num_ell_idxes;
182  num_total_idxes += num_ell_idxes;
183  }
184  else { // we have seen an ellipsis before, so we check compatibility
185  AT_CHECK(candidate_num_ell_idxes == num_ell_idxes,
186  "ellipsis must represent ", num_ell_idxes, " dimensions in all terms");
187  }
188  for (int64_t i = 0; i < num_ell_idxes; ++i) { // map ellipsis dimensions in operand to indices
189  current_op_idxes.push_back(first_ell_idx + i);
190  last_idx_occurrence.push_back(operand);
191  }
192  dims_in_term += num_ell_idxes; // keep track of dimensions
193  }
194  } else { // a letter (hopefully)
195  AT_CHECK((ell_char_count == 0) || (ell_char_count == 3), "'.' must only occur in ellipsis, operand ", operand);
196  AT_CHECK(('a' <= c) && (c <= 'z'), "only lowercase letters a-z allowed as indices");
197  int64_t letter_num = c-'a'; // letter_num = position in letter_mapping
198  if (letter_mapping[letter_num] == -1) { // new letter, add internal index and mapping
199  letter_mapping[letter_num] = num_total_idxes;
200  num_total_idxes++;
201  last_idx_occurrence.push_back(operand);
202  } else { // letter we have already seen
203  last_idx_occurrence[letter_mapping[letter_num]] = operand;
204  }
205  num_letter_occurrences[letter_num]++;
206  current_op_idxes.push_back(letter_mapping[letter_num]);
207  dims_in_term++;
208  }
209  }
210  AT_CHECK(dims_in_term == tensors[operand].dim(), "dimension mismatch for operand ", operand, ": equation ", dims_in_term, " tensor ", tensors[operand].dim());
211  input_op_idxes.push_back(std::move(current_op_idxes));
212  operand++;
213  }
214  // in the check below, we need ==, but > is captured above, so the error message can be specific that it is <.
215  AT_CHECK((int64_t) tensors.size()==operand, "more tensors than operands in equation");
216 
217  // the following parses or infers output (right hand side)
218  // it also assigns the idxes_to_preprocessed_dims (index -> dimension in preprocessed / output tensors)
219  // for the output indices. -1 means that the index has not been assigned a dimension yet
220  std::vector<int64_t> idxes_to_preprocessed_dims(num_total_idxes, -1); // the position of the index in the tensor dimensions
221  int64_t num_output_dims = 0;
222  if (pos != std::string::npos) { // parse the user provided right hand side
223  int64_t ell_char_count = 0;
224  for (auto &c : eqn.substr(pos+2)) {
225  if (c == '.') { // '.' as part of ellipsis
226  ell_char_count++;
227  AT_CHECK(ell_char_count <= 3, "can only have '.' in one ellispis '...' in right hand side of the equation");
228  if (ell_char_count == 3) { // ellipsis complete
229  AT_CHECK(num_ell_idxes >= 0, "ellipsis '...' may only appear in right hand side if it does in left hand side");
230  for (int64_t i = 0; i < num_ell_idxes; ++i) {
231  idxes_to_preprocessed_dims[first_ell_idx + i] = num_output_dims;
232  num_output_dims++;
233  }
234  }
235  } else if (! isspace(c)) { // letter (hopefully)
236  AT_CHECK((ell_char_count == 0) || (ell_char_count == 3), "'.' must only occur in ellipsis in the right hand side");
237  AT_CHECK(('a' <= c) && (c <= 'z'), "only lowercase letters a-z allowed as indices");
238  int64_t letter_num = c-'a';
239  AT_CHECK(idxes_to_preprocessed_dims[letter_mapping[letter_num]] == -1, "index ", c, "occurs twice in output");
240  idxes_to_preprocessed_dims[letter_mapping[letter_num]] = num_output_dims;
241  num_output_dims++;
242  }
243  }
244  } else { // create an inferred right hand side
245  // the ellipsis (if in the lhs) comes first
246  if (num_ell_idxes >= 0) {
247  for (int64_t i = 0; i < num_ell_idxes; ++i) {
248  idxes_to_preprocessed_dims[first_ell_idx + i] = num_output_dims;
249  num_output_dims++;
250  }
251  }
252  // then the indices that occur exactly once in alphabetic order
253  for (size_t idx = 0; idx < number_of_letters; idx++) {
254  if (num_letter_occurrences[idx] == 1) {
255  idxes_to_preprocessed_dims[letter_mapping[idx]] = num_output_dims;
256  num_output_dims++;
257  }
258  }
259  }
260  // now we assign the idxes_to_preprocessed_dims (index -> dimension in preprocessed / output tensors)
261  // for the non-output indices - those that are eventually summed over
262  int64_t position = num_output_dims;
263  for (int64_t i = 0; i < num_total_idxes; i++) {
264  if (idxes_to_preprocessed_dims[i]==-1) {
265  idxes_to_preprocessed_dims[i] = position;
266  position++;
267  }
268  }
269 
270  // we now "homogenize the dimensions", i.e.
271  // - take diagonals for duplicated indices
272  // - permute the dimensions to match the order given by idxes_to_preprocessed_dims
273  // - unsqueeze to create all dimensions for each index in each tensor where they are missing
274  // we also check that sizes match
275  // after this, all operands will have compatible shapes (i.e. all dimensions are aligned are broadcastable)
276  std::vector<Tensor> preprocessed_operands;
277  std::vector<std::int64_t> size_of_dims(num_total_idxes, -1); // keep track of sizes for each index, -1 means we have not seen a size yet
278  for (int64_t op = 0; op < (int64_t) tensors.size(); op++) {
279  auto preprocessed_op = tensors[op];
280  std::vector<int64_t> idx_to_dim(num_total_idxes, -1); // the dimension which the index refers to in the original tensor, -1 means it does not appear
281  std::vector<int64_t>& current_op_input_idxes = input_op_idxes[op];
282  int64_t dim = 0; // there are two dimension indices: dim is after taking diagonals, i is in input
283  for (size_t i = 0; i < current_op_input_idxes.size(); i++) {
284  auto idx = current_op_input_idxes[i];
285  auto dim_out = idxes_to_preprocessed_dims[idx];
286  if (idx_to_dim[dim_out] == -1) { // first appearance
287  idx_to_dim[dim_out] = dim;
288  if (size_of_dims[idx] == -1) { // keep track of sizes
289  size_of_dims[idx] = preprocessed_op.size(dim);
290  }
291  else {
292  AT_CHECK(size_of_dims[idx] == preprocessed_op.size(dim), "size of dimension does not match previous size, operand ", op, ", dim ", i);
293  }
294  dim++;
295  } else { // duplicate dimension in tensor --> take diagonal of idx_to_dim[dim_out] and dim and put the diagonal dimension to idx_to_dim[dim_out]
296  AT_CHECK(size_of_dims[idx] == preprocessed_op.size(dim), "size of dimension does not match previous size, operand ", op, ", dim ", i);
297  preprocessed_op = preprocessed_op.diagonal(0, idx_to_dim[dim_out], dim);
298  // diagonal moves the diagonal dimension to the back
299  // now we permute the last dim back to idx_to_dim[dim_out]
300  std::vector<int64_t> perm(preprocessed_op.dim(), 0);
301  for (int64_t d = 0; d < preprocessed_op.dim(); d++) {
302  if (d == idx_to_dim[dim_out]) {
303  perm[d] = preprocessed_op.dim() - 1;
304  } else {
305  perm[d] = d - (d > idx_to_dim[dim_out]);
306  }
307  }
308  preprocessed_op = preprocessed_op.permute(perm);
309  }
310  }
311  // now we permute the dimensions in the right order
312  std::vector<int64_t> permutation; // permutation for this tensor
313  for (auto &d : idx_to_dim) {
314  if (d > -1) {
315  permutation.push_back(d);
316  }
317  }
318  preprocessed_op = preprocessed_op.permute(permutation);
319  // finally, we insert dimensions for idxes not in the operand
320  for (size_t dim = 0; dim < idx_to_dim.size(); dim++) {
321  if (idx_to_dim[dim] == -1) {
322  preprocessed_op = preprocessed_op.unsqueeze(dim);
323  }
324  }
325  preprocessed_operands.push_back(preprocessed_op);
326  }
327 
328  // now we reduce the indices from left to right
329  // numpy allows to optimize the path using various
330  // algorithms (see eigen_path in numpy docs)
331  // we start with the leftmost operator and reduce indices that
332  // appear only there
333  Tensor result = preprocessed_operands[0];
334  for (int64_t idx = 0; idx < num_total_idxes; idx++) {
335  if ((last_idx_occurrence[idx] == 0)
336  && (idxes_to_preprocessed_dims[idx]>=num_output_dims)) {
337  result = result.sum(idxes_to_preprocessed_dims[idx], true);
338  }
339  }
340 
341  // now we process each tensor using sumproduct_pair
342  for (int64_t i = 1; i < (int64_t) preprocessed_operands.size(); i++) {
343  std::vector<int64_t> sum_dims;
344  for (int64_t idx = 0; idx < num_total_idxes; idx++) {
345  if ((last_idx_occurrence[idx] == i)
346  && (idxes_to_preprocessed_dims[idx]>=num_output_dims)) {
347  sum_dims.push_back(idxes_to_preprocessed_dims[idx]);
348  }
349  }
350  result = at::native::sumproduct_pair(result, preprocessed_operands[i], sum_dims, true);
351  }
352  // finally, we squeeze out all non-result dimensions
353  for (int64_t dim = num_total_idxes-1; dim >= num_output_dims; dim--)
354  result.squeeze_(dim);
355  return result;
356 }

358 // _trilinear computes a trilinear einstein sum with an unrolled dimension
359 // the result is `(i1.unsqueeze(expand1)*i2.unsqueeze(expand2)*i2.unsqueeze(expand3)).sum(sumdim)`
360 // the computation is unrolled in the unroll_dim dimension
361 // its main purpose is to unify the computations in bilinear and bilinear_backward
362 Tensor _trilinear(const Tensor& i1_, const Tensor& i2_, const Tensor& i3_,
363  IntArrayRef expand1_, IntArrayRef expand2_, IntArrayRef expand3_,
364  IntArrayRef sumdim_, int64_t unroll_dim) {
365  int64_t total_dim = i1_.dim()+expand1_.size();
366  AT_CHECK((unroll_dim >= 0) && (unroll_dim < total_dim), "unroll_dim must be in [0,", total_dim-1, "]");
367  auto expand1 = at::dim_list_to_bitset(expand1_, total_dim);
368  auto expand2 = at::dim_list_to_bitset(expand2_, total_dim);
369  auto expand3 = at::dim_list_to_bitset(expand3_, total_dim);
370  auto sumdim = at::dim_list_to_bitset(sumdim_, total_dim);
371  Tensor i1 = i1_;
372  Tensor i2 = i2_;
373  Tensor i3 = i3_;
374  std::vector<int64_t> output_size;
375  std::vector<int64_t> sum_dims_12, sum_dims_23;
376  int64_t unroll_size = -1;
377  // asserts...
378  for (int64_t i = 0; i < total_dim; i++) {
379  int64_t s = 0;
380  if (expand1[i]) {
381  i1 = i1.unsqueeze(i);
382  } else {
383  s = i1.size(i);
384  }
385  if (expand2[i]) {
386  i2 = i2.unsqueeze(i);
387  } else {
388  s = i2.size(i);
389  }
390  if (expand3[i]) {
391  i3 = i3.unsqueeze(i);
392  if (sumdim[i] && (i != unroll_dim))
393  sum_dims_12.push_back(i);
394  } else {
395  s = i3.size(i);
396  if (sumdim[i] && (i != unroll_dim))
397  sum_dims_23.push_back(i);
398  }
399  output_size.push_back(sumdim[i] ? 1 : s);
400  if (i == unroll_dim)
401  unroll_size = s;
402  }
403  int64_t slicemul1 = (expand1[unroll_dim] ? 0 : 1);
404  int64_t slicemul2 = (expand2[unroll_dim] ? 0 : 1);
405  int64_t slicemul3 = (expand3[unroll_dim] ? 0 : 1);
406 
407  auto output = at::zeros(output_size, i1.options());
408  if (! sumdim[unroll_dim]) {
409  for (int64_t k = 0; k < unroll_size; k++) {
410  Tensor buf = at::native::sumproduct_pair(i1.narrow(unroll_dim, k * slicemul1, 1),
411  i2.narrow(unroll_dim, k * slicemul2, 1),
412  sum_dims_12, true);
413  buf = at::native::sumproduct_pair(buf, i3.narrow(unroll_dim, k * slicemul3, 1), sum_dims_23, true);
414  output.narrow(unroll_dim, k, 1).add_(buf);
415  }
416  }
417  else {
418  for (int64_t k = 0; k < unroll_size; k++) {
419  Tensor buf = at::native::sumproduct_pair(i1.narrow(unroll_dim, k*slicemul1, 1),
420  i2.narrow(unroll_dim, k*slicemul2, 1), sum_dims_12, true);
421  buf = at::native::sumproduct_pair(buf, i3.narrow(unroll_dim, k*slicemul3, 1), sum_dims_23, true);
422  output.add_(buf);
423  }
424  }
425  for (int64_t i = output.dim()-1; i >= 0; i--)
426  if (sumdim[i])
427  output.squeeze_(i);
428  return output;
429 }

432  AT_CHECK(input1.dim() == input2.dim(), "bilinear(): input dimensions do not match: got ", input1.dim(), " and ", input2.dim());
433  for (int64_t i = 0; i < input1.dim() - 1; i++) {

435  "bilinear(): input batch dimensions do not match at dim ", i, ": got ", input1.size(i), " and ", input2.size(i));

439  input1.size(input1.dim() - 1), " but expected ", weight.size(1));

441  "bilinear(): input2 size does not match weight size: got ",

454  if (bias.defined()) {

460 // implements tensordot, a matrix-multiplication-like contraction, but the dimensions given
461 // in the two dimension lists
462 Tensor tensordot(const Tensor& input1, const Tensor& input2, IntArrayRef dims1, IntArrayRef dims2) {
463  AT_CHECK(dims1.size() == dims2.size(), "both dimension lists should have same length");
464  int64_t csize = 1; // total size of the contracted dimensions
465  Tensor t1 = input1;
466  Tensor t2 = input2;
467  for (size_t i = 0; i < dims1.size(); i++) {
468  int s1 = input1.size(dims1[i]);
469  int s2 = input2.size(dims2[i]);
470  if (s2 == 1) { // broadcasted dimensions can be summed right away
471  t1 = t1.sum(dims1[i], true);
472  } else if (s1 == 1) {
473  t2 = t2.sum(dims2[i], true);
474  } else {
475  AT_CHECK(s1 == s2, "contracted dimensions need to match, but first has size ", s1, " in dim ", dims1[i],
476  " and second has size ", s2, " in dim ", dims2[i]);
477  csize *= s1;
478  }
479  }
480 
481  auto cdims1 = at::dim_list_to_bitset(dims1, input1.dim());
482  auto cdims2 = at::dim_list_to_bitset(dims2, input2.dim());
483  std::vector<int64_t> p1, p2, rsizes; // p1, p2: input permutations, rsizes: sizes of the result
484  p1.reserve(input1.dim());
485  p2.reserve(input2.dim());
486  rsizes.reserve(input1.dim() + input2.dim() - (int64_t) dims1.size());
487  int64_t size1 = 1; // number of non-contracted elements in input1
488  int64_t size2 = 1; // number of non-contracted elements in input2
489 
490  // fill the permutations and compute sizes
491  for (int64_t i = 0; i < input1.dim(); i++) {
492  if (! cdims1[i]) {
493  p1.emplace_back(i);
494  size1 *= t1.size(i);
495  rsizes.emplace_back(t1.size(i));
496  }
497  }
498  for (size_t i = 0; i < dims1.size(); i++) {
499  p1.emplace_back(dims1[i]);
500  }
501  for (size_t i = 0; i < dims2.size(); i++) {
502  p2.emplace_back(dims2[i]);
503  }
504  for (int64_t i = 0; i < input2.dim(); i++) {
505  if (! cdims2[i]) {
506  p2.emplace_back(i);
507  size2 *= t2.size(i);
508  rsizes.emplace_back(t2.size(i));
509  }
510  }
511  // permut and reshape for matrix multiplication
512  t1 = t1.permute(p1).reshape({size1, csize});
513  t2 = t2.permute(p2).reshape({csize, size2});
514  // multiply and reshape to target size
515  return at::mm(t1, t2).reshape(rsizes);
516 }
