56 class InlineDeviceGuard {
57 public:
58  // Note [Omitted default constructor from RAII]
59  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
60  // In principle, we could add a default constructor to
61  // DeviceGuard which reads the current device and promises to
62  // restore to that device on exit. However, most cases where you
63  // would have written this, you probably meant to actually just
64  // use OptionalDeviceGuard (since you don't actually need the
65  // restore to happen if you don't ever actually set the device).
66  // We remove the constructor here to encourage you to think about
67  // what you actually want to happen.
68  explicit InlineDeviceGuard() = delete;

72  : impl_(device.type())
73  , original_device_(device.index() == -1 ? impl_.getDevice() : impl_.exchangeDevice(device))
74  , current_device_(device.index() == -1 ? original_device_ : device)
75  {}

145  auto index = device.index();

230  // TODO: Consider readding Tensor and TensorList constructors here, when
231  // Tensor moves to c10. (These are only valid on OptionalDeviceGuard,
232  // because a Tensor may be undefined, in which case we need an uninitialized
233  // tensor guard.)
234 
235  // Note [Move construction for RAII guards is tricky]
236  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
237  // In principle, move construction is useful for terminating
238  // the lifetime of a `OptionalDeviceGuard` early; for example:
239  //
240  // // current device is d0
241  // OptionalDeviceGuard g1(d1);
242  // // current device is d1
243  // {
244  // OptionalDeviceGuard g2(std::move(g1));
245  // }
246  // // current device is d0!!
247  //
248  // However, it's difficult to implement the move constructor
249  // in a way that works in all situations. For example, consider
250  // the following example:
251  //
252  // OptionalDeviceGuard g1(d1);
253  // {
254  // OptionalDeviceGuard g2(d2);
255  // {
256  // OptionalDeviceGuard g3(std::move(g1)); // !!!
257  // }
258  // }
259  //
260  // What should the current device be while g3 in scope... and what
261  // should it be after it goes out of scope? What about g2?
262  // There don't seem to be satisfactory answers for these questions.
263  //
264  // It's in principle possible to raise an error when this occurs
265  // by doing some extra thread-local bookkeeping. But why bother?
266  // Just don't provide the constructor.
267  InlineOptionalDeviceGuard(InlineOptionalDeviceGuard<T>&& other) = delete;
268 
269  // Note [Move assignment for RAII guards is tricky]
270  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
271  // Move assignment is deleted, because you need to know which guard was
272  // defined "first", as that guard's original_device_ wins--with the current
273  // representation, we have no way of telling which is the case. (Move
274  // construction does not have this problem, as one guard is always
275  // uninitialized.)
276  //
277  // We can make this clear by way of a pair of examples:
278  //
279  // Example 1:
280  //
281  // // initial device is n0
282  // {
283  // CUDAGuard g1(n1);
284  // {
285  // CUDAGuard g2(n2);
286  // // current device should be n2
287  // g1 = std::move(g2);
288  // // current device should still be n2
289  // }
290  // // current device should still be n2
291  // }
292  // // current device should be n0
293  //
294  // Example 2 (flip the order of the two guards):
295  //
296  // // initial device is n0
297  // {
298  // CUDAGuard g2(n2);
299  // {
300  // CUDAGuard g1(n1);
301  // // current device should be n1
302  // g1 = std::move(g2);
303  // // current device should be n2
304  // }
305  // // current device should be n0 (since g2 has been vacated)
306  // }
307  //
308  // In both examples, we need g1 to restore to n0 after move assignment.
309  // However, in example 1, this is determined by the restore value of g1
310  // (prior to the move). In example 2, however, it is determined by the the
311  // restore value of g2(!!). We don't know which one should win, without having
312  // a way of telling which guard was allocated first.
313  //
314  // We could solve this with an extra thread-local variable. But no one is
315  // actually using move-assignment. So just get rid of it.
316  InlineOptionalDeviceGuard& operator=(InlineOptionalDeviceGuard&& other) = delete;
