487 // Note [Caffe2 ConvPoolOpBase]
488 // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~
489 // To understand what is going on here, we have to talk a little bit about
490 // Caffe2's internals.
491 //
492 // First, it's important to know that all of Caffe2's pooling and convolution
493 // operators inherit from "ConvPoolOpBase", which is an abstract class that
494 // defines all of the attributes (kernels, dilations, strides, etc) which one
495 // sees on these operators. Unfortunately, Caffe2's documentation generator
496 // doesn't know how to handle cases like this, so for example, if you look at
497 // the docs for MaxPool at
498 // <https://caffe2.ai/docs/operators-catalogue.html#maxpool> you won't see any
499 // of the attributes. You have to go source diving to find the information; in
500 // particular, you want to look at:
501 // https://github.com/caffe2/caffe2/blob/master/caffe2/operators/conv_pool_op_base.h
502 // This class handles *global* pooling as well.
503 //
504 // Second, it's important to know what Caffe2 expects for padding, which can
505 // be somewhat difficult to understand from the code because Caffe2 handles
506 // both singular/pluralized spellings of padding, and there is also legacy
507 // padding business. The short version of the story is that, for NON-legacy
508 // padding (which is what we want to output), padding is expected to be
509 // *twice* the size of kernels. So if you have a 2D convolution, Caffe2
510 // will accept two values in 'kernels', but FOUR values in 'pads';
511 // furthermore, this is *mandatory.*
512 //
513 // Finally, ConvPoolOpBase is not the only class of it's kind; there is
514 // be tricked by the fact that Conv and ConvTranspose have similar
515 // parameters; they exercise different codepaths and need to be handled
516 // differently.
517 Caffe2Ops Caffe2Backend::CreateConvPoolOpBase(
518  OnnxNode* onnx_node,
519  const ConversionContext& ctx) {
520  const auto& node = onnx_node->node;
521  auto& attributes = onnx_node->attributes;
522  if (node.op_type().find("Global") == 0) {
523  auto* attr = attributes.AddRewrittenAttribute("global_pooling");
524  attr->set_i(1);
525  }
526 
527  if (attributes.HasAttribute("kernel_shape") &&
528  attributes.HasAttribute("pads")) {
529  auto kernel_shape =
530  attributes
531  .get<::google::protobuf::RepeatedField<::google::protobuf::int64>>(
532  "kernel_shape");
533  auto pads =
534  attributes
535  .get<::google::protobuf::RepeatedField<::google::protobuf::int64>>(
536  "pads");
537  if (kernel_shape.size() == pads.size()) {
538  // Caffe2 requires pads to be twice the size of kernels.
539  auto* attr = attributes.AddRewrittenAttribute("pads");
540  attr->mutable_ints()->CopyFrom(pads);
541  attr->mutable_ints()->MergeFrom(pads);
542  }
543  }
544 
545  return CommonOnnxNodeToCaffe2Ops(onnx_node, ctx);
546 }

634  OnnxNode* onnx_node,
635  const ConversionContext& ctx) {
636  const auto& node = onnx_node->node;
637  if (node.input_size() < 3 || node.output_size() < 1) {
638  CAFFE_THROW("Caffe2 Gemm should have 3 inputs and 1 output");
639  }
640 
641  Caffe2Ops ret;
642  auto input_a = node.input(0);
643  auto input_b = node.input(1);
644  auto input_c = node.input(2);
645  auto output = node.output(0);
646 
647  auto alpha = onnx_node->attributes.get<float>("alpha", 1.0);
648  auto beta = onnx_node->attributes.get<float>("beta", 1.0);
649  if (!AlmostEqual(alpha, 1)) {
650  auto scaled_a = dummy_->NewDummyName();
651  caffe2::Argument scale;
652  scale.set_name("scale");
653  scale.set_f(alpha);
654 
655  auto* c2_op = ret.ops.Add();
656  BuildOperator(c2_op, "Scale", {input_a}, {scaled_a}, {scale});
657  input_a = scaled_a;
658  }
659  if (!AlmostEqual(beta, 1)) {
660  auto scaled_c = dummy_->NewDummyName();
661  caffe2::Argument scale;
662  scale.set_name("scale");
663  scale.set_f(beta);
664 
665  auto* c2_op = ret.ops.Add();
666  BuildOperator(c2_op, "Scale", {input_c}, {scaled_c}, {scale});
667  input_c = scaled_c;
668  }
669 
670  auto trans_a = onnx_node->attributes.get<int64_t>("transA", 0L);
671  auto trans_b = onnx_node->attributes.get<int64_t>("transB", 0L);
672  // Support broadcast by default when opset_version > 6.
673  auto broadcast =
674  onnx_node->attributes.get<int64_t>("broadcast",
675  (ctx.opset_version() > 6) ? 1L : 0L);
676 
677  // If the c's shape information is available and c is a 1d tensor(except
678  // c is a scalar), use FC aggressively.
679  auto check_fc = [&]() -> bool {
680  const auto input_c_vi_iter = ctx.value_infos().find(node.input(2));
681 
682  if (input_c_vi_iter == ctx.value_infos().end()) {
683  return false;
684  }
685 
686  const auto input_c_shape =
687  input_c_vi_iter->second.type().tensor_type().shape();
688 
689  if (input_c_shape.dim_size() != 1) {
690  return false;
691  }
692 
693  // c is a scalar.
694  if (input_c_shape.dim(0).dim_value() == 1) {
695  const auto input_b_vi_iter = ctx.value_infos().find(node.input(1));
696 
697  // If the b's shape is not available, skip FC.
698  if (input_b_vi_iter == ctx.value_infos().end()) {
699  return false;
700  }
701  const auto input_b_shape =
702  input_b_vi_iter->second.type().tensor_type().shape();
703  int input_b_last_dim_index = (trans_b) ? 0 : 1;
704  // If b's last dim is not 1, skip FC.
705  if (input_b_shape.dim_size() <= input_b_last_dim_index ||
706  input_b_shape.dim(input_b_last_dim_index).dim_value() != 1) {
707  return false;
708  }
709  }
710 
711  return true;
712  };
713 
714  if (!trans_a && broadcast && check_fc()) {
715  auto* c2_op = ret.ops.Add();
716  if (trans_b) {
717  BuildOperator(c2_op, "FC", {input_a, input_b, input_c}, {output});
718  } else {
719  BuildOperator(c2_op, "FCTransposed", {input_a, input_b, input_c}, {output});
720  }
721  } else {
722  auto ab = dummy_->NewDummyName();
723  caffe2::Argument arg_trans_a;
724  arg_trans_a.set_name("trans_a");
725  arg_trans_a.set_i(trans_a);
726  caffe2::Argument arg_trans_b;
727  arg_trans_b.set_name("trans_b");
728  arg_trans_b.set_i(trans_b);
729 
730  auto* c2_op = ret.ops.Add();
731  BuildOperator(
732  c2_op, "MatMul", {input_a, input_b}, {ab}, {arg_trans_a, arg_trans_b});
733  c2_op = ret.ops.Add();
734  if (ctx.opset_version() >= 7) {
735  BuildOperator(c2_op, "Add", {ab, input_c}, {output});
736  } else {
737  caffe2::Argument arg_broadcast;
738  arg_broadcast.set_name("broadcast");
739  arg_broadcast.set_i(broadcast);
740  BuildOperator(c2_op, "Add", {ab, input_c}, {output}, {arg_broadcast});
741  }
742  }
743 
744  return ret;
745 }

975 // for a given index tensor (i.e. `starts` or `ends`):
976 // 1) Hilariously subtract 1 from the value if it is negative. This due to
977 // the behavior of Caffe2's slice operator not matching that of ONNX's slice

980 std::string Caffe2Backend::PreprocessSliceIndexTensor(OnnxNode* onnx_node,
981  Caffe2Ops& ret,
982  std::string indices_tensor,
983  std::string axes_tensor,
984  std::string rank_tensor,
985  std::string zero_tensor,
986  std::string one_tensor,
987  int default_value) {

992  value.set_name("value");
993  value.set_i(default_value);

1009  broadcast.set_name("broadcast");
1010  broadcast.set_i(1);

1017  caffe2::Argument broadcast;
1018  broadcast.set_name("broadcast");
1019  broadcast.set_i(1);
1020  auto c2_op = ret.ops.Add();
1021  BuildOperator(c2_op, "Sub", {indices_tensor, one_tensor}, {sub_one_tensor}, {broadcast});

1292 // This translator performs the basic translation of ONNX nodes into
1293 // Caffe2 operators. Besides doing a straightforward marshalling from
1294 // one format to another, it also does these extra things:
1295 //
1296 // - Renames operators based on 'renamed_operators'
1297 // - Renames attributes based on 'renamed_attrs' and
1298 // 'get_per_op_renamed_attrs'
1299 //
1300 // If you're writing a custom translator, consider calling this first,
1301 // and then fixing things up further.
1302 Caffe2Ops Caffe2Backend::CommonOnnxNodeToCaffe2Ops(
1303  OnnxNode* onnx_node,
1304  const ConversionContext& ctx) {
1305  Caffe2Ops ret;
1306  auto* c2_op = ret.ops.Add();
1307 
1308  const auto& node = onnx_node->node;
1309  c2_op->mutable_input()->MergeFrom(node.input());
1310  c2_op->mutable_output()->MergeFrom(node.output());
1311  c2_op->set_name(node.name());
1312 
1313  const auto onnx_op_type = node.op_type();
1314  auto broken_version = caffe2::get_default(
1315  get_broken_operators(), onnx_op_type, std::numeric_limits<int>::max());
1316  if (broken_version <= ctx.opset_version()) {
1317  CAFFE_THROW(
1318  "Don't know how to translate op ",
1319  onnx_op_type,
1320  " in ONNX operator set v",
1321  ctx.opset_version(),
1322  " (I only support prior to v",
1323  broken_version);
1324  }
1325  c2_op->set_type(
1326  caffe2::get_default(get_renamed_operators(), onnx_op_type, onnx_op_type));
1327  if (!IsOperator(c2_op->type())) {
1328  CAFFE_THROW(
1329  "Don't know how to translate op ", onnx_op_type);
1330  }
1331 
1332  auto mapper = [&, this](const std::string& k) {
1333  const auto it = get_per_op_renamed_attrs().find(onnx_op_type);
1334  if (it != get_per_op_renamed_attrs().end()) {
1335  const auto it_op = it->second.find(k);
1336  if (it_op != it->second.end()) {
1337  return it_op->second;
1338  }
1339  }
1340  const auto it_global = get_renamed_attrs().find(k);
1341  if (it_global != get_renamed_attrs().end()) {
1342  return it_global->second;
1343  }
1344  return k;
1345  };
1346  c2_op->mutable_arg()->MergeFrom(
1347  onnx_node->attributes.OnnxAttrToCaffe2Arg(mapper));
1348 
1349  return ret;
1350 }

1415 void Caffe2Backend::OnnxToCaffe2(
1416  caffe2::NetDef* init_net,
1417  caffe2::NetDef* pred_net,
1418  const ModelProto& onnx_model,
1419  const std::string& device,
1420  int opset_version,
1421  bool include_initializers,
1422  const std::vector<Caffe2Ops>& extras) {
1423  auto device_option = GetDeviceOption(Device(device));
1424 
1425 #ifndef C10_MOBILE
1426  ModelProto init_model = OptimizeOnnx(onnx_model, true);
1427  ModelProto pred_model = OptimizeOnnx(onnx_model, false);
1428 #else
1429  ModelProto init_model = ModelProto();
1430  ModelProto pred_model = onnx_model;
1431  pred_model.mutable_graph()->mutable_initializer()->Clear();
1432 #endif
1433 
1434  init_net->set_name(onnx_model.graph().name() + "_init");
1435  pred_net->set_name(onnx_model.graph().name() + "_predict");
1436 
1437  // Convert initializer if necessary
1438  if (include_initializers) {
1439  for (const auto& tp : onnx_model.graph().initializer()) {
1440  auto* c2_op = init_net->add_op();
1441  BuildTensorFillingOp(c2_op, tp);
1442  }
1443  }
1444 
1445  auto name_set = AllNamesInGraph(init_model.graph());
1446  auto name_set_pred = AllNamesInGraph(pred_model.graph());
1447  name_set.insert(name_set_pred.begin(), name_set_pred.end());
1448  dummy_->Reset(name_set);
1449 
1450  ValueInfoMap graph_value_infos{};
1451  for (const auto& vi : pred_model.graph().input()) {
1452  graph_value_infos[vi.name()].CopyFrom(vi);
1453  }
1454  for (const auto& vi : pred_model.graph().output()) {
1455  graph_value_infos[vi.name()].CopyFrom(vi);
1456  }
1457  for (const auto& vi : pred_model.graph().value_info()) {
1458  graph_value_infos[vi.name()].CopyFrom(vi);
1459  }
1460 
1461  size_t idx_extra = 0;
1462  auto converter = [&](const ModelProto& model, caffe2::NetDef* net) mutable {
1463  net->mutable_device_option()->CopyFrom(device_option);
1464  for (const auto& node : model.graph().node()) {
1465  auto* init_net_tmp = include_initializers ? init_net : net;
1466  // For RNN operators, we rely on Python code to convert them for us, and
1467  // we simply deserilize the string. This is hack and eventually we want to
1468  // get rid of this to have one flow. Note that we need to update the dummy
1469  // name generator to avoid having duplicated names between Python and C++
1470  // generated dummies
1471  if (get_rnn_operators().count(node.op_type())) {
1472  if (idx_extra < extras.size()) {
1473  const auto& c2ops = extras[idx_extra++];
1474  for (const auto& op : c2ops.init_ops) {
1475  UpdateNames(dummy_, op);
1476  }
1477  init_net_tmp->mutable_op()->MergeFrom(c2ops.init_ops);
1478  for (const auto& op : c2ops.ops) {
1479  UpdateNames(dummy_, op);
1480  }
1481  net->mutable_op()->MergeFrom(c2ops.ops);
1482  for (const auto& input : c2ops.interface_blobs) {
1483  dummy_->AddName(input);
1484  }
1485  net->mutable_external_input()->MergeFrom(c2ops.interface_blobs);
1486  } else {
1487  CAFFE_THROW(
1488  "Don't know how to convert ",
1489  node.op_type(),
1490  " without enough extra preconverted string");
1491  }
1492  } else {
1493  ValueInfoMap value_infos{};
1494  for (const auto& name : node.input()) {
1495  auto iter = graph_value_infos.find(name);
1496  if (iter != graph_value_infos.end()) {
1497  value_infos[name].CopyFrom(iter->second);
1498  }
1499  }
1500  auto onnx_node = OnnxNode(node);
1501  auto c2ops = OnnxNodeToCaffe2Ops(
1502  init_model, pred_model, {value_infos, opset_version}, &onnx_node);
1503  init_net_tmp->mutable_op()->MergeFrom(c2ops.init_ops);
1504  net->mutable_op()->MergeFrom(c2ops.ops);
1505  net->mutable_external_input()->MergeFrom(c2ops.interface_blobs);
1506  }
1507  }
1508 
1509  for (const auto& value : model.graph().output()) {
1510  net->add_external_output(value.name());
1511  }
1512  for (const auto& value : model.graph().input()) {
1513  net->add_external_input(value.name());
1514  }
1515  };
1516 
1517  converter(init_model, init_net);
1518  converter(pred_model, pred_net);
1519 }

1639 void Caffe2Backend::BuildTensorFillingOp(
1640  caffe2::OperatorDef* c2_op,
1641  const TensorProto& onnx_tensor,
1642  const std::string& output_name,
1643  const std::string& shape_name) {
1644  auto fill_name = output_name.empty() ? onnx_tensor.name() : output_name;
1645  CAFFE_ENFORCE(!fill_name.empty());
1646 
1647  if (onnx_tensor.has_segment()) {
1648  CAFFE_THROW("Currently not supporting loading segments.");
1649  }
1650 
1651  auto* c2_values = c2_op->add_arg();
1652  // if shape_name is empty, we generate GivenTensorFill
1653  // otherwise, we generate ConstantFill, which accept shape as input
1654  if (shape_name.empty()) {
1655  // GivenTensor*Fill uses values
1656  c2_values->set_name("values");
1657  if (onnx_tensor.data_type() == TensorProto::FLOAT) {
1658  c2_op->set_type("GivenTensorFill");
1659  auto* floats = c2_values->mutable_floats();
1660  if (!TryConvertingTensorRawValues<float>(onnx_tensor, floats)) {
1661  floats->CopyFrom(onnx_tensor.float_data());
1662  }
1663  } else if (onnx_tensor.data_type() == TensorProto::DOUBLE) {
1664  c2_op->set_type("GivenTensorDoubleFill");
1665  ::google::protobuf::RepeatedField<double> tmp;
1666  const ::google::protobuf::RepeatedField<double>* src = &tmp;
1667  if (!TryConvertingTensorRawValues<double>(onnx_tensor, &tmp)) {
1668  src = &onnx_tensor.double_data();
1669  }
1670  for (const auto i : *src) {
1671  c2_values->add_floats(i);
1672  }
1673  } else if (onnx_tensor.data_type() == TensorProto::INT64) {
1674  ConvertIntegralValueToCaffe2<::google::protobuf::int64>(c2_op, c2_values, onnx_tensor);
1675  } else if (onnx_tensor.data_type() == TensorProto::UINT32) {
1676  ConvertIntegralValueToCaffe2<::google::protobuf::uint64>(c2_op, c2_values, onnx_tensor);
1677  } else if (onnx_tensor.data_type() == TensorProto::BOOL) {
1678  ConvertIntegralValueToCaffe2<::google::protobuf::int8>(c2_op, c2_values, onnx_tensor);
1679  } else if (onnx_tensor.data_type() == TensorProto::UINT8) {
1680  ConvertIntegralValueToCaffe2<::google::protobuf::uint8>(c2_op, c2_values, onnx_tensor);
1681  } else if (onnx_tensor.data_type() == TensorProto::INT8) {
1682  ConvertIntegralValueToCaffe2<::google::protobuf::int8>(c2_op, c2_values, onnx_tensor);
1683  } else if (onnx_tensor.data_type() == TensorProto::UINT16) {
1684  ConvertIntegralValueToCaffe2<::google::protobuf::uint16>(c2_op, c2_values, onnx_tensor);
1685  } else if (onnx_tensor.data_type() == TensorProto::INT16) {
1686  ConvertIntegralValueToCaffe2<::google::protobuf::int16>(c2_op, c2_values, onnx_tensor);
1687  } else if (onnx_tensor.data_type() == TensorProto::INT32) {
1688  ConvertIntegralValueToCaffe2<::google::protobuf::int32>(c2_op, c2_values, onnx_tensor);
1689  } else if (onnx_tensor.data_type() == TensorProto::STRING) {
1690  c2_op->set_type("GivenTensorStringFill");
1691  auto* strings = c2_values->mutable_strings();
1692  strings->CopyFrom(onnx_tensor.string_data());
1693  } else {
1694  CAFFE_THROW("unrecognized tensor type: ", onnx_tensor.data_type());
1695  }
1696  auto* c2_shape = c2_op->add_arg();
1697  c2_shape->set_name("shape");
1698  for (const auto d : onnx_tensor.dims()) {
1699  c2_shape->add_ints(d);
1700  }
1701  } else {
1702  int value_size = 1;
1703  for (const auto d : onnx_tensor.dims()) {
1704  value_size *= d;
1705  }
1706  CAFFE_ENFORCE(value_size == 1);
1707  auto c2_input_as_shape = c2_op->add_arg();
1708  c2_input_as_shape->set_name("input_as_shape");
1709  c2_input_as_shape->set_i(1);
1710  c2_values->set_name("value");
1711  auto* c2_dtype = c2_op->add_arg();
1712  c2_dtype->set_name("dtype");
1713  if (onnx_tensor.data_type() == TensorProto::FLOAT) {
1714  c2_dtype->set_i(caffe2::TensorProto::FLOAT);
1715  if (onnx_tensor.float_data_size() > 0) {
1716  c2_values->set_f(onnx_tensor.float_data(0));
1717  } else {
1718  CAFFE_ENFORCE(onnx_tensor.raw_data().size() == sizeof(float));
1719  float f;
1720  memcpy(&f, onnx_tensor.raw_data().c_str(), sizeof(float));
1721  c2_values->set_f(f);
1722  }
1723  } else if (onnx_tensor.data_type() == TensorProto::DOUBLE) {
1724  c2_dtype->set_i(caffe2::TensorProto::DOUBLE);
1725  if (onnx_tensor.double_data_size() > 0) {
1726  c2_values->set_f(static_cast<float>(onnx_tensor.double_data(0)));
1727  } else {
1728  CAFFE_ENFORCE(onnx_tensor.raw_data().size() == sizeof(double));
1729  double d;
1730  memcpy(&d, onnx_tensor.raw_data().c_str(), sizeof(double));
1731  c2_values->set_f(static_cast<float>(d));
