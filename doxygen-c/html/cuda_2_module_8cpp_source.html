229 // We need to ensure that as long as a thread will NEVER loose the GIL as long as
230 // it holds the CUDA mutex. Otherwise another thread might be scheduled and try to
231 // e.g. allocate a new tensor which will cause a deadlock. It's enough to have a
232 // single global, because it can be only set once (cudaMutex is not recursive)
233 // by the thread that owns the mutex (obviously there can be only one such thread).
234 static PyGILState_STATE cudaMutexGILState;
235 
236 PyObject * THCPModule_cudaLockMutex(PyObject *module)
237 {
238  auto mutex = c10::cuda::CUDACachingAllocator::getFreeMutex();
239  // This has to be a busy loop because we **absolutely need to** hold the GIL
240  // or it's a recipe for a deadlock otherwise (if we let other Python threads
241  // run while we have the cudaMutex, but not the GIL, they might try to e.g.
242  // free a CUDA tensor and acquire the cudaMutex without giving up the GIL,
243  // because it happens deep within THC).
244  while (true) {
245  if (mutex->try_lock())
246  break;
247  {
248  AutoNoGIL no_gil;
249  std::this_thread::sleep_for(std::chrono::microseconds(10));
250  }
251  }
252 
253  cudaMutexGILState = PyGILState_Ensure();
254  Py_RETURN_NONE;
255 }
256 
257 PyObject * THCPModule_cudaUnlockMutex(PyObject *module)
258 {
259  auto mutex = c10::cuda::CUDACachingAllocator::getFreeMutex();
260  PyGILState_Release(cudaMutexGILState);
261  mutex->unlock();
262  Py_RETURN_NONE;
263 }

338  // Add class and method to torch.cuda
339  auto m = py::handle(module).cast<py::module>();
340  py::class_<cudaDeviceProp>(m, "_CudaDeviceProperties")

347  .def_readonly("total_memory", &cudaDeviceProp::totalGlobalMem)

361 static PyObject * THCPModule_initExtension(PyObject *self)
362 {
363  HANDLE_TH_ERRORS
364  state = at::globalContext().lazyInitCUDA();
365 
366  auto m = THPObjectPtr(PyImport_ImportModule("torch.cuda"));
367  if (!m) throw python_error();
368 
369  // Register Storage Python objects with DynamicTypes.cpp
370  THCPDoubleStorage_postInit(m);
371  THCPFloatStorage_postInit(m);
372  THCPHalfStorage_postInit(m);
373  THCPLongStorage_postInit(m);
374  THCPIntStorage_postInit(m);
375  THCPShortStorage_postInit(m);
376  THCPCharStorage_postInit(m);
377  THCPByteStorage_postInit(m);
378  THCPBoolStorage_postInit(m);
379 
380  bool has_magma = at::hasMAGMA();
381  if (has_magma) {
382  THCMagma_init(state);
383  }
384 
385  bool has_half = true;
386 
387  auto set_module_attr = [&](const char* name, PyObject* v) {
388  // PyObject_SetAttrString doesn't steal reference. So no need to incref.
389  if (PyObject_SetAttrString(m, name, v) < 0) {
390  throw python_error();
391  }
392  };
393 
394  set_module_attr("has_magma", has_magma ? Py_True : Py_False);
395  set_module_attr("has_half", has_half ? Py_True : Py_False);
396 
397  auto _state_cdata = THPObjectPtr(PyLong_FromVoidPtr(state));
398  if (!_state_cdata) throw python_error();
399  set_module_attr("_state_cdata", _state_cdata.get());
400 
401  bindCudaDeviceProperties(m);
402 
403  Py_RETURN_NONE;
404  END_HANDLE_TH_ERRORS
405 }
