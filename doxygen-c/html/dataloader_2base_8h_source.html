182  void worker_thread(Dataset& dataset) {
183  while (true) {
184  auto job = shuttle_.pop_job();
185  if (job.quit) {
186  break;
187  }
188  try {
189  auto batch = dataset.get_batch(std::move(*job.batch_request));
190  shuttle_.push_result({std::move(batch), job.sequence_number});
191  } catch (...) {
