216 template<typename scalar_t>
217 struct ComputeLocation<scalar_t, GridSamplerPadding::Border>
218  : ComputeLocationBase<scalar_t> {
219  using Vec = Vec256<scalar_t>;
220  using ComputeLocationBase<scalar_t>::unnormalize;
221  using ComputeLocationBase<scalar_t>::half_max_val;
222 
223  const scalar_t max_val;
224 
225  ComputeLocation(int64_t size)
226  : ComputeLocationBase<scalar_t>(size)
227  , max_val(static_cast<scalar_t>(size - 1)) {}
228 
229  inline Vec apply(const Vec &in) const {
230  return minimum(Vec(max_val), maximum(unnormalize(in), Vec(0)));
231  }
232  inline std::pair<Vec, Vec> apply_get_grad(const Vec &in) const {
233  using int_t = int_same_size_t<scalar_t>;
234  Vec max_val_vec(max_val), zeros(0);
235  auto indices = unnormalize(in);
236  auto bounded_lo = maximum(indices, zeros);
237  // Integral type equality comparison is very very fast because it just looks
238  // at the bits. Casting is free too. So we use the following pattern instead
239  // of comparison + blendv.
240  auto in_bound_lo = cast<scalar_t>(cast<int_t>(bounded_lo) == cast<int_t>(indices));
241  auto res = minimum(bounded_lo, max_val_vec);
242  auto in_bound_hi = cast<scalar_t>(cast<int_t>(res) == cast<int_t>(indices));
243  return std::make_pair(res, (in_bound_lo & in_bound_hi) & Vec(half_max_val));
244  }
245 };

247 template<typename scalar_t>
248 struct ComputeLocation<scalar_t, GridSamplerPadding::Reflection>
249  : ComputeLocationBase<scalar_t> {
250  using Vec = Vec256<scalar_t>;
251  using ComputeLocationBase<scalar_t>::unnormalize;
252  using ComputeLocationBase<scalar_t>::half_max_val;
253 
254  bool unit_size; // whether size == 1, just return 0 in this case
255  const scalar_t double_max_val;
256  const scalar_t neg_half_max_val;
257 
258  ComputeLocation(int64_t size)
259  : ComputeLocationBase<scalar_t>(size)
260  , unit_size(size == 1)
261  , double_max_val(static_cast<scalar_t>((size - 1) * 2))
262  , neg_half_max_val(-0.5 * static_cast<scalar_t>(size - 1)) {}
263 
264  inline Vec apply(const Vec &in) const {
265  if (unit_size) {
266  return Vec(0);
267  }
268  Vec double_max_val_vec(double_max_val);
269  auto abs_in = unnormalize(in).abs();
270  auto fdouble_flips = abs_in / double_max_val_vec;
271  auto double_flips = fdouble_flips.trunc();
272  auto extra = abs_in - double_flips * double_max_val_vec;
273  // Now we need to test if extra > max_val to find out if another flip is
274  // needed. The following comparison does that and returns the correct
275  // flipped value.
276  return minimum(extra, double_max_val_vec - extra);
277  }
278 
279  inline std::pair<Vec, Vec> apply_get_grad(const Vec &in) const {
280  if (unit_size) {
281  return std::make_pair(Vec(0), Vec(0));
282  }
283  Vec double_max_val_vec(double_max_val);
284  auto unnorm_in = unnormalize(in);
285  auto neg_in = unnorm_in < Vec(0);
286  auto abs_in = unnorm_in.abs();
287  auto fdouble_flips = abs_in / double_max_val_vec;
288  auto double_flips = fdouble_flips.trunc();
289 
290  auto extra = abs_in - double_flips * double_max_val_vec;
291  auto reflected_extra = double_max_val_vec - extra;
292  auto one_more_flip = extra > reflected_extra;
293 
294  return std::make_pair(
295  Vec::blendv(extra, reflected_extra, one_more_flip),
296  Vec::blendv(Vec(half_max_val), Vec(neg_half_max_val), one_more_flip ^ neg_in)
297  );
298  }
299 };

339  const ComputeLocation<scalar_t, padding> compute_W;
340  const bool must_in_bound = padding != GridSamplerPadding::Zeros;
341 
342  ApplyGridSample(const TensorAccessor<scalar_t, 4>& input)
343  : inp_H(input.size(2))
344  , inp_W(input.size(3))
345  , inp_sH(input.stride(2))
346  , inp_sW(input.stride(3))
347  , C(input.size(1))
348  , inp_sC(input.stride(1))
349  , compute_H(input.size(2))
350  , compute_W(input.size(3)) {}
351 
352  inline std::tuple<
353  Vec, Vec, Vec, Vec, // distances to 4 sides
354  Vec, Vec, Vec, Vec, // interpolation weights wrt 4 corners
355  Vec, Vec, Vec, Vec, // in_bound masks
356  iVec, iVec // y_n and x_w
357  >
358  compute_interp_params(const Vec& x, const Vec& y) const {
359  // get NE, NW, SE, SW pixel values from (x, y)
360  // assuming we get exact integer representation and just use scalar_t
361  // if we don't, the weights will be garbage anyways.
362  auto x_w = x.floor();
363  auto y_n = y.floor();
364 
365  // get distances to each side
366  auto w = x - x_w;
367  auto e = Vec(1) - w;
368  auto n = y - y_n;
369  auto s = Vec(1) - n;
370 
371  // get interpolation weights for each neighbor
372  // e.g., for the nw corder, the weight is `dist_to_south * dist_to_east`.
373  auto nw = s * e;
374  auto ne = s * w;
375  auto sw = n * e;
376  auto se = n * w;
377 
378  auto i_x_w = convert_to_int_of_same_size(x_w);
379  auto i_y_n = convert_to_int_of_same_size(y_n);
380  auto i_x_e = i_x_w + iVec(1);
381  auto i_y_s = i_y_n + iVec(1);
382 
383  // Use int comparison because it is much faster than float comp with AVX2
384  // (latency 1 cyc vs. 4 cyc on skylake)
385  // Avoid using the le and ge because those are not implemented in AVX2 and
386  // are actually simulated using multiple instructions.
387  auto w_mask = must_in_bound ? iVec(-1) // true = all ones
388  : (i_x_w > iVec(-1)) & (i_x_w < iVec(inp_W));
389  auto n_mask = must_in_bound ? iVec(-1) // true = all ones
390  : (i_y_n > iVec(-1)) & (i_y_n < iVec(inp_H));
391  auto e_mask = must_in_bound ? (i_x_e < iVec(inp_W))
392  : (i_x_e > iVec(-1)) & (i_x_e < iVec(inp_W));
393  auto s_mask = must_in_bound ? (i_y_s < iVec(inp_H))
394  : (i_y_s > iVec(-1)) & (i_y_s < iVec(inp_H));
395  auto nw_mask = cast<scalar_t>(must_in_bound ? iVec(-1) : (w_mask & n_mask));
396  auto ne_mask = cast<scalar_t>(e_mask & n_mask);
397  auto sw_mask = cast<scalar_t>(w_mask & s_mask);
398  auto se_mask = cast<scalar_t>(e_mask & s_mask);
399 
400  return std::make_tuple(
401  n, s, w, e,
402  nw, ne, sw, se,
403  nw_mask, ne_mask, sw_mask, se_mask,
404  i_y_n, i_x_w);
405  }

432  auto i_se_offset = i_sw_offset + iVec(inp_sW);

435  # pragma unroll
436  #endif
437  for (int64_t c = 0; c < C; ++c) {
438  auto inp_slice_C_ptr = inp_slice[c].data();
439 
440  // mask_gather zeros out the mask, so we need to make copies
441  Vec nw_mask_copy = nw_mask;
442  Vec ne_mask_copy = ne_mask;
443  Vec sw_mask_copy = sw_mask;
444  Vec se_mask_copy = se_mask;
445  auto nw_val = mask_gather<sizeof(scalar_t)>(Vec(0), inp_slice_C_ptr, i_nw_offset, nw_mask_copy);
446  auto ne_val = mask_gather<sizeof(scalar_t)>(Vec(0), inp_slice_C_ptr, i_ne_offset, ne_mask_copy);
447  auto sw_val = mask_gather<sizeof(scalar_t)>(Vec(0), inp_slice_C_ptr, i_sw_offset, sw_mask_copy);
448  auto se_val = mask_gather<sizeof(scalar_t)>(Vec(0), inp_slice_C_ptr, i_se_offset, se_mask_copy);
449 
450  auto interpolated = (nw_val * nw) + (ne_val * ne) + (sw_val * sw) + (se_val * se);
451  interpolated.store(out_slice[c].data() + offset, len);

455  inline void backward(TensorAccessor<scalar_t, 3>& gInp_slice,
456  TensorAccessor<scalar_t, 3>& gGrid_slice,
457  const TensorAccessor<scalar_t, 3>& gOut_slice,
458  const TensorAccessor<scalar_t, 3>& inp_slice,
459  int64_t offset, const Vec& grid_x, const Vec& grid_y,
460  int64_t len) const {
461  Vec x, y, gx_mult, gy_mult;
462  std::tie(x, gx_mult) = compute_W.apply_get_grad(grid_x);
463  std::tie(y, gy_mult) = compute_H.apply_get_grad(grid_y);
464 
465  Vec n, s, w, e, nw, ne, sw, se, nw_mask, ne_mask, sw_mask, se_mask;
466  iVec i_y_n, i_x_w;
467 
468  std::tie(
469  n, s, w, e, nw, ne, sw, se, nw_mask, ne_mask, sw_mask, se_mask,
470  i_y_n, i_x_w) = compute_interp_params(x, y);
471 
472  auto i_nw_offset = i_y_n * iVec(inp_sH) + i_x_w * iVec(inp_sW);
473  auto i_ne_offset = i_nw_offset + iVec(inp_sW);
474  auto i_sw_offset = i_nw_offset + iVec(inp_sH);
475  auto i_se_offset = i_sw_offset + iVec(inp_sW);
476 
477  auto i_gInp_nw_offset = i_y_n * iVec(inp_W) + i_x_w;
478  auto i_gInp_ne_offset = i_gInp_nw_offset + iVec(1);
479  auto i_gInp_sw_offset = i_gInp_nw_offset + iVec(inp_W);
480  auto i_gInp_se_offset = i_gInp_sw_offset + iVec(1);
481 
482  // When reading input values, we used mask_gather. Unfortunately, there is
483  // no mask_scatter_add (the backward of mask_gather) in Intel intrinsics.
484  // So we store the necessary vectors to temporary arrays and use the helper
485  // mask_scatter_add defined above.
486 
487  integer_t i_gInp_nw_offset_arr[iVec::size()];
488  integer_t i_gInp_ne_offset_arr[iVec::size()];
489  integer_t i_gInp_sw_offset_arr[iVec::size()];
490  integer_t i_gInp_se_offset_arr[iVec::size()];
491  i_gInp_nw_offset.store(i_gInp_nw_offset_arr);
492  i_gInp_ne_offset.store(i_gInp_ne_offset_arr);
493  i_gInp_sw_offset.store(i_gInp_sw_offset_arr);
494  i_gInp_se_offset.store(i_gInp_se_offset_arr);
495 
496  integer_t i_nw_mask_arr[iVec::size()];
497  integer_t i_ne_mask_arr[iVec::size()];
498  integer_t i_sw_mask_arr[iVec::size()];
499  integer_t i_se_mask_arr[iVec::size()];
500  nw_mask.store(i_nw_mask_arr);
501  ne_mask.store(i_ne_mask_arr);
502  sw_mask.store(i_sw_mask_arr);
503  se_mask.store(i_se_mask_arr);
504 
505  scalar_t gInp_corner_arr[Vec::size()];
506 
507  auto gx = Vec(0), gy = Vec(0);
508  #ifndef _MSC_VER
509  # pragma unroll
510  #endif
511  for (int64_t c = 0; c < C; ++c) {
512  auto inp_slice_C_ptr = inp_slice[c].data();
513  auto gInp_slice_C_ptr = gInp_slice[c].data();
514  auto gOut = Vec::loadu(gOut_slice[c].data() + offset, len);
515 
516  (nw * gOut).store(gInp_corner_arr);
517  mask_scatter_add(gInp_corner_arr, gInp_slice_C_ptr, i_gInp_nw_offset_arr, i_nw_mask_arr, len);
518  (ne * gOut).store(gInp_corner_arr);
519  mask_scatter_add(gInp_corner_arr, gInp_slice_C_ptr, i_gInp_ne_offset_arr, i_ne_mask_arr, len);
520  (sw * gOut).store(gInp_corner_arr);
521  mask_scatter_add(gInp_corner_arr, gInp_slice_C_ptr, i_gInp_sw_offset_arr, i_sw_mask_arr, len);
522  (se * gOut).store(gInp_corner_arr);
523  mask_scatter_add(gInp_corner_arr, gInp_slice_C_ptr, i_gInp_se_offset_arr, i_se_mask_arr, len);
524 
525  // mask_gather zeros out the mask, so we need to make copies
526  Vec nw_mask_copy = nw_mask;
527  Vec ne_mask_copy = ne_mask;
528  Vec sw_mask_copy = sw_mask;
529  Vec se_mask_copy = se_mask;
530  auto nw_val = mask_gather<sizeof(scalar_t)>(Vec(0), inp_slice_C_ptr, i_nw_offset, nw_mask_copy);
531  auto ne_val = mask_gather<sizeof(scalar_t)>(Vec(0), inp_slice_C_ptr, i_ne_offset, ne_mask_copy);
532  auto sw_val = mask_gather<sizeof(scalar_t)>(Vec(0), inp_slice_C_ptr, i_sw_offset, sw_mask_copy);
533  auto se_val = mask_gather<sizeof(scalar_t)>(Vec(0), inp_slice_C_ptr, i_se_offset, se_mask_copy);
534 
535  gx = gx + ((ne_val - nw_val) * s + (se_val - sw_val) * n) * gOut;
536  gy = gy + ((sw_val - nw_val) * e + (se_val - ne_val) * w) * gOut;
537  }
538 
539  gx = gx * gx_mult;
540  gy = gy * gy_mult;
541 
542  constexpr int64_t step = Vec::size();
543  auto interleaved_gGrid = interleave2(gx, gy);
544  auto gGrid_ptr = gGrid_slice.data() + offset * 2;
545  std::get<0>(interleaved_gGrid).store(gGrid_ptr,
546  std::min(len * 2, step));
547  std::get<1>(interleaved_gGrid).store(gGrid_ptr + step,
548  std::max(static_cast<int64_t>(0), len * 2 - step));
549  }

639  # pragma unroll
640  #endif
641  for (int64_t c = 0; c < C; ++c) {
642  mask_scatter_add(gOut_slice[c].data() + offset, gInp_slice[c].data(),
643  gInp_offset_arr, mask_arr, len);
644  }
645 
646  // grid has zero 0 gradient in Nearest mode
647  auto gGrid_ptr = gGrid_slice.data() + offset * 2;
648  std::memset(gGrid_ptr, 0, sizeof(scalar_t) * len * 2);
649  }

652 // ~~~~~~~~~~~~~~~~~~ grid_sample_2d_grid_slice_iterator ~~~~~~~~~~~~~~~~~~~~~~
653 // Function to apply a vectorized function on a grid slice tensor (without batch
654 // dimension).
655 // See NOTE [ Grid Sample CPU Kernels ] for details.
656 
657 template<typename scalar_t, typename ApplyFn>
658 static inline void grid_sample_2d_grid_slice_iterator(
659  const TensorAccessor<scalar_t, 3>& grid_slice, const ApplyFn &apply_fn) {
660  int64_t out_H = grid_slice.size(0);
661  int64_t out_W = grid_slice.size(1);
662  int64_t grid_sH = grid_slice.stride(0);
663  int64_t grid_sW = grid_slice.stride(1);
664  int64_t grid_sCoor = grid_slice.stride(2);
665  auto grid_ptr = grid_slice.data();
666 
667  using Vec = Vec256<scalar_t>;
668  using iVec = Vec256<int_same_size_t<scalar_t>>;
669  constexpr int64_t step = Vec::size();
670 
671  // Loop over each output pixel in grid.
672  // We consider the following three cases (after slicing out the batch
673  // dimension).
674  // See detailed discussions under each if-case.
675 
676  if (at::geometry_is_contiguous({out_H, out_W, 2}, {grid_sH, grid_sW, grid_sCoor})) {
677  // Case 1:
678  // Grid is contiguous.
679  // Strategy: Sequentially load two vectors at the same time, and get,
680  // e.g., {x0, y0, x1, y1}, {x2, y2, x3, y3}. Then we use
681  // at::vec256::deinterleave2 to get x and y vectors.
682  auto total_size = out_H * out_W;
683  for (int64_t spatial_offset = 0; spatial_offset < total_size; spatial_offset += step) {
684  auto grid_offset = spatial_offset * 2;
685  auto len = std::min(step, total_size - spatial_offset);
686  auto vec1 = Vec::loadu(grid_ptr + grid_offset,
687  std::min(step, len * 2));
688  auto vec2 = Vec::loadu(grid_ptr + grid_offset + step,
689  std::max(static_cast<int64_t>(0), len * 2 - step));
690  auto vec_xy_pair = deinterleave2(vec1, vec2);
691 
692  auto x = std::get<0>(vec_xy_pair);
693  auto y = std::get<1>(vec_xy_pair);
694 
695  // make sure that x and y are valid grid sample locations
696  if (len < step) {
697  x = Vec::set(Vec(0), x, len);
698  y = Vec::set(Vec(0), y, len);
699  }
700  apply_fn(x, y, spatial_offset, len);
701  }
702  } else if (grid_sW == 1 || out_W == 1) {
703  // Case 2:
704  // The W dimension is contiguous.
705  // This can be common, e.g., grid is from a conv net output of shape
706  // [N, 2, H, W].
707  // Strategy: Divide into two contiguous slices each of shape [H, W], and
708  // each containing x and y vectors. So we sequentially load a
709  // vector from each of them to get x and y vector
710 
711  // Function to apply along a contiguous W dimension (or flattened H x W).
712  auto line_fn = [&](const scalar_t *grid_ptr_x, const scalar_t *grid_ptr_y,
713  int64_t out_base_offset, int64_t total_size) {
714  for (int64_t i = 0; i < total_size; i += step) {
715  auto len = std::min(step, total_size - i);
716  auto x = Vec::loadu(grid_ptr_x + i, len);
717  auto y = Vec::loadu(grid_ptr_y + i, len);
718  // make sure that x and y are valid grid sample locations
719  if (len < step) {
720  x = Vec::set(Vec(0), x, len);
721  y = Vec::set(Vec(0), y, len);
722  }
723  apply_fn(x, y, out_base_offset + i, len);
724  }
725  };
726 
727  if (at::geometry_is_contiguous({out_H, out_W}, {grid_sH, grid_sW})) {
728  // If [H, W] is contiguous, apply line_fn once.
729  line_fn(grid_ptr, grid_ptr + grid_sCoor, 0, out_H * out_W);
730  } else {
731  // If only [W] is contiguous, apply line_fn once for each h slice.
732  auto grid_ptr_NH = grid_ptr;
733  for (int64_t h = 0; h < out_H; h++) {
734  line_fn(grid_ptr_NH, grid_ptr_NH + grid_sCoor, h * out_W, out_W);
735  grid_ptr_NH += grid_sH;
736  }
737  }
738  } else {
739  // Case 3:
740  // General case.
741  // Strategy: Do a for-loop over H, for each W slice, use
742  // at::vec256::gather to load the x and y vectors.
743  auto spatial_offset = 0;
744  auto i_offsets_delta = iVec(grid_sW * step);
745 
746  #ifndef _MSC_VER
747  # pragma unroll
748  #endif
749  for (int64_t h = 0; h < out_H; h++) {
750  auto grid_ptr_x = grid_ptr + h * grid_sH;
751  auto grid_ptr_y = grid_ptr_x + grid_sCoor;
752  auto i_offsets = iVec::arange(0, grid_sW);
753  #ifndef _MSC_VER
754  # pragma unroll
755  #endif
756  for (int64_t w = 0; w < out_W; w += step) {
757  auto len = std::min(step, out_W - w);
758  if (len < step) {
759  // prevents illegal memory access, sets the exceeding offsets to zero
760  i_offsets = iVec::set(iVec(0), i_offsets, len);
761  }
762  apply_fn(vec256::gather<sizeof(scalar_t)>(grid_ptr_x, i_offsets),
763  vec256::gather<sizeof(scalar_t)>(grid_ptr_y, i_offsets),
764  spatial_offset, len);
765 
766  i_offsets = i_offsets + i_offsets_delta;
767  spatial_offset += len;
768  }
769  }
770  }
771 }

782  auto H = grid.size(1);
783  auto W = grid.size(2);
784  auto output = at::empty({N, input.size(1), H, W}, input.options());
785  auto spatial_size = H * W;

787  : at::divup(at::internal::GRAIN_SIZE, spatial_size * 4 /* 2d * 2 tensors*/);
788 
789 #define HANDLE_CASE(interp, padding) \
790  case padding: { \
791  ApplyGridSample<scalar_t, 2, interp, padding> grid_sample(inp_acc); \
792  parallel_for(0, N, grain_size, [&](int64_t begin, int64_t end) { \
793  for (int64_t n = begin; n < end; n++) { \
794  auto out_slice = out_acc[n]; \
795  auto inp_slice = inp_acc[n]; \
796  grid_sample_2d_grid_slice_iterator( \
797  grid_acc[n], \
798  [&](const Vec256<scalar_t>& grid_x, const Vec256<scalar_t>& grid_y, \
799  int64_t spatial_offset, int64_t len) { \
800  grid_sample.forward(out_slice, inp_slice, spatial_offset, \
801  grid_x, grid_y, len); \
802  }); \
803  } \
804  }); \
805  return; \
806  }
807 
808 #define HANDLE_INTERP(interp) \
809  case interp: { \
810  switch (static_cast<GridSamplerPadding>(padding_mode)) { \
811  HANDLE_CASE(interp, GridSamplerPadding::Zeros); \
812  HANDLE_CASE(interp, GridSamplerPadding::Border); \
813  HANDLE_CASE(interp, GridSamplerPadding::Reflection); \
814  } \
815  return; \
