63 DataChannelNccl::DataChannelNccl(InitMethod::Config config, int timeout)
64  : _rank(config.rank),
65  _numProcesses(config.world_size),
66  _timeout(timeout),
67  _masterListeningSocket(-1),
68  _slaveSocket(-1) {
69  // Establish the socket connections from rank 0 to all others
70  if (_rank == 0) {
71  _masterListeningSocket = config.master.listen_socket;
72  _masterSendingSockets = std::vector<int>(_numProcesses - 1, -1);
73 
74  try {
75  for (rank_type i = 0; i < _numProcesses - 1; ++i) {
76  std::tie(_masterSendingSockets[i], std::ignore) =
77  accept(_masterListeningSocket, _timeout);
78  }
79  } catch (...) {
80  // Destroy the created sockets
81  _destroySockets();
82  throw std::runtime_error("Rank 0 cannot establish thelistening socket");
83  }
84 
85  } else {
86  _masterAddr = config.worker.master_addr;
87  _masterPort = config.worker.master_port;
88 
89  try {
90  _slaveSocket = connect(_masterAddr, _masterPort, true, _timeout);
91  } catch (...) {
92  // Destroy the created sockets
93  _destroySockets();
94  std::string errStr = "Rank: " + std::to_string(_rank) +
95  " cannot "
96  "connect to the master: " +
97  _masterAddr + ":" + std::to_string(_masterPort);
98  throw std::runtime_error(errStr);
99  }
100  }

170 // Helper function that destroys the CUDA event and NCCL communicator
171 void DataChannelNccl::_destroyNcclResources(THDGroup groupId) {
172  if (_groupNcclResources.find(groupId) != _groupNcclResources.end()) {
173  for (int i = 0; i < _groupDevices[groupId].size(); i++) {
174  // Devices used for this group ID
175  auto devices = getDevicesList(_groupDevices[groupId][i]);
176  // Guard GPU device
177  at::cuda::OptionalCUDAGuard gpuGuard;
178  // Destroy the CUDA events
179  size_t idx = 0;
180  for (auto& event : *(_groupNcclResources[groupId][i].ncclCudaEvents())) {
181  gpuGuard.set_index(devices[idx++]);
182  THCudaCheck(cudaEventSynchronize(event));
183  THCudaCheck(cudaEventDestroy(event));
184  }
185  // Destroy the communicators
186  for (auto& comm : *(_groupNcclResources[groupId][i].ncclComms())) {
187  NCCL_CHECK(ncclCommDestroy(comm));
188  }
189  }
190  }

229 NcclResourcePair DataChannelNccl::_getNcclResourcePair(
230  std::vector<at::Tensor>& input,
231  THDGroup groupId) {
232  if (input.empty()) {
233  throw std::runtime_error(
234  "Not able to create/get the Nccl Comm since "
235  "input tensor is empty");
236  }
237  // Get the deviceList String
238  std::string deviceList;
239  for (auto tensor : input) {
240  if (deviceList.empty()) {
241  deviceList = std::to_string(tensor.get_device());
242  } else {
243  deviceList += "," + std::to_string(tensor.get_device());
244  }
245  }
246 
247  int index = -1;
248 
249  if (_groupDevices.find(groupId) != _groupDevices.end()) {
250  auto pos = std::find(
251  _groupDevices[groupId].begin(),
252  _groupDevices[groupId].end(),
253  deviceList);
254  if (pos != _groupDevices[groupId].end())
255  index = pos - _groupDevices[groupId].begin();
256  }
257 
258  if (index >= 0) {
259  return std::make_pair(
260  _groupNcclResources[groupId][index].ncclComms(),
261  _groupNcclResources[groupId][index].ncclCudaEvents());
262  }
263 
264  // Add in the device list of the group
265  _groupDevices[groupId].push_back(deviceList);
266 
267  // NCCL communicator
268  auto comms =
269  std::unique_ptr<std::vector<ncclComm_t>>(new std::vector<ncclComm_t>());
270 
271  comms->resize(input.size());
272 
273  // Corresponding CUDA events
274  auto events =
275  std::unique_ptr<std::vector<cudaEvent_t>>(new std::vector<cudaEvent_t>());
276 
277  events->resize(input.size());
278 
279  // Create the unique NCCL ID and broadcast it
280  ncclUniqueId ncclId;
281  NCCL_CHECK(ncclGetUniqueId(&ncclId));
282 
283  // Broadcast so that each process can have a unique NCCL ID
284  broadcastUniqueNcclId(&ncclId);
285 
286  // Guard GPU device
287  at::cuda::OptionalCUDAGuard gpuGuard;
288 
289  // Now creating the CUDA events
290  for (size_t i = 0; i < input.size(); ++i) {
291  gpuGuard.set_index(input[i].get_device());
292  THCudaCheck(cudaEventCreate(&((*events)[i])));
293  }
294  // Create the communicator on each device of the input
295  NCCL_CHECK(ncclGroupStart());
296  for (size_t i = 0; i < input.size(); ++i) {
297  int nRanks = int(_numProcesses) * input.size();
298  gpuGuard.set_index(input[i].get_device());
299  NCCL_CHECK(ncclCommInitRank(
300  &((*comms)[i]), nRanks, ncclId, _rank * input.size() + i));
301  }
302  NCCL_CHECK(ncclGroupEnd());
303 
304  // Move into the hash table
305  if (_groupNcclResources.find(groupId) == _groupNcclResources.end())
306  _groupNcclResources.emplace(
307  std::make_pair(groupId, std::vector<NcclResources>()));
308 
309  _groupNcclResources[groupId].push_back(
310  NcclResources(std::move(comms), std::move(events)));
311 
312  return std::make_pair(
313  _groupNcclResources[groupId].back().ncclComms(),
314  _groupNcclResources[groupId].back().ncclCudaEvents());
315 }

317 // Helper function that checks the input and output tensors for validity
318 bool DataChannelNccl::_tensorCheckHelper(
319  const std::vector<at::Tensor>& input,
320  const std::vector<at::Tensor>& output,
321  size_t outputOverInput) {
322  if (input.size() != output.size()) {
323  throw std::runtime_error(
324  "Input tensor sequence should have the same "
325  "number of tensors as the output tensor sequence");
326  }
327 
328  if (input.size() == 0) {
329  // Return false saying this is a no-op
330  return false;
331  }
332 
333  if (input.size() > _numGPUs) {
334  throw std::runtime_error(
335  "The number of input tensors is larger than "
336  "the number of available GPUs");
337  }
338 
339  // To make sure each tensor is on separate devices
340  std::unordered_set<int> usedDevices;
341  usedDevices.reserve(input.size());
342 
343  uint64_t inputNumElement = input[0].numel();
344  auto elementType = input[0].scalar_type();
345 
346  for (size_t i = 0; i < input.size(); ++i) {
347  // Check to make sure it's a GPU dense tensor
348  if (!(input[i].is_cuda() && !input[i].is_sparse() &&
349  output[i].is_cuda() && !output[i].is_sparse())) {
350  throw std::runtime_error(
351  "Only CUDA dense tensor is supported for NCCL "
352  "collective operations");
353  }
354  // Check the tensor type is identical
355  if (input[i].scalar_type() != elementType ||
356  output[i].scalar_type() != elementType) {
357  throw std::runtime_error(
358  "Expecting all GPU tensors to have identical "
359  "type");
360  }
361  // Check the input tensor size is identical
362  if (input[i].numel() != inputNumElement) {
363  throw std::runtime_error(
364  "Expecting all input tensors to have identical "
365  "number of elements");
366  }
367  // Check the output tensor size equals to input tensor size
368  if (output[i].numel() != inputNumElement * outputOverInput) {
369  throw std::runtime_error(
370  "The number of elements of output tensor does "
371  "not match the number of elements of the input "
372  "tensor");
373  }
374  // Contiguous verification
375  if (!input[i].is_contiguous() || !output[i].is_contiguous()) {
376  throw std::runtime_error("Expecting all GPU tensors to be contiguous");
377  }
378 
379  bool inserted;
380  std::tie(std::ignore, inserted) = usedDevices.insert(input[i].get_device());
381  // Device verification, if the insertion didn't take place
382  if (!inserted) {
383  throw std::runtime_error("Expecting inputs on different GPU devices");
384  }
385 
386  // Now check the output device
387  if (input[i].get_device() != output[i].get_device()) {
388  throw std::runtime_error(
389  "Expecting input and output tensors to be on "
390  "the same device");
391  }
392  }
393  return true;
394 }

649
