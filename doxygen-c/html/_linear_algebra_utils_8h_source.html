8 /*
9  * Clones a Tensor so that the following conditions hold:
10  * If we think of a Tensor of having size (B, M, N), where B is any number
11  * of batch dimensions, then:
12  * - Each (M, N) matrix is in column major form
13  * - Let Tensor P have size (B, M, N) and Q have size (B, M', N').
14  * Then when laid out in memory, the M by N matrix starting at
15  * P.data_ptr()[b * M * N] is of the same corresponding batch as the M' by N'
16  * matrix starting at Q.data_ptr()[b * M' * N'].
17  */
18 static inline Tensor cloneBatchedColumnMajor(const Tensor& src) {
19  // If src is already in batched column major format, then
20  // this will be efficient (no reordering of the data will occur)
21  // because the first transpose will make the tensor contiguous,
22  // and cloning a contiguous tensor is fast.
23  auto result = src.transpose(-2, -1).clone();
24  result.transpose_(-2, -1);
25  return result;
26 }

45 /* Checks a necessary property for the triu and tril implementations, hence the name.
46  * Here batch contiguity is checked for tensors with greater than 4 dimensions.
47  * Contiguous tensors and tensors with less than 3 dimensions pass this check
48  */
49 static inline bool checkTrilTriuBatchContiguous(const Tensor& tensor) {
50  // Complete contiguity is the most desired property, which is why
51  // we return true if the tensor is contiguous
52  if (tensor.is_contiguous()) return true;
53 
54  int64_t dims = tensor.dim();
55 
56  // Tensors with dimension less than 4 are handled by default
57  if (dims <= 3) return true;
58 
59  int64_t expected_stride = tensor.size(-1) * tensor.size(-2);
60  for (int64_t i = dims - 3; i >= 0; i--) {
61  if (expected_stride != tensor.stride(i)) return false;
62  expected_stride *= tensor.size(i);
63  }
64  return true;
65 }
