30 BatchTensor::BatchTensor(
31  const std::vector<at::Tensor>& datalist,
32  at::Tensor dims) {
33  auto bs = datalist.size();
34  std::vector<int64_t> sizes(dims.size(0) + 1, 0),
35  mask_sizes(dims.size(0) + 1, 0);
36  sizes[0] = bs;
37  mask_sizes[0] = bs;
38  for (int64_t i = 1; i < dims.size(0) + 1; i++) {
39  for (const auto& x : datalist) {
40  sizes[i] = std::max(sizes[i], x.size(i));
41  }
42  mask_sizes[i] = *dims[i - 1].data<uint8_t>() ? sizes[i] : 1;
43  }
44  data = at::empty(sizes, datalist[0].options());
45  data.fill_(0);
46  mask = at::empty(mask_sizes, datalist[0].options().dtype(at::kByte));
47  mask.fill_(0);
48  for (std::size_t i = 0; i < datalist.size(); i++) {
49  auto data_item = data.narrow(0, i, 1);
50  auto mask_item = mask.narrow(0, i, 1);
51  for (int64_t j = 0; j < dims.size(0); j++) {
52  if (*dims[j].data<uint8_t>()) {
53  data_item = data_item.narrow(j + 1, 0, datalist[i].size(j + 1));
54  mask_item = mask_item.narrow(j + 1, 0, datalist[i].size(j + 1));
55  }
56  }
57  data_item += datalist[i];
58  mask_item.fill_(1);
59  }
60  this->dims = std::move(dims);
61 }

63 std::vector<at::Tensor> BatchTensor::examples() {
64  std::vector<at::Tensor> result;
65  // calculate number of valid entries in dth dimension of data
66  auto mask_sum = [](at::Tensor data, int d) -> int64_t {
67  data = data.sum(d, /*keepdim=*/true);
68  while (data.dim() >= 1)
69  data = data[0];
70  return *data.data<int64_t>();
71  };
72  for (int64_t i = 0; i < data.size(0); i++) {
73  auto data_tmp = data.narrow(0, i, 1);
74  for (int64_t d = 0; d < dims.size(0); d++) {
75  if (*dims[d].data<uint8_t>()) {
76  data_tmp = data_tmp.narrow(d + 1, 0, mask_sum(mask[i], d));
77  }
78  }
79  result.push_back(data_tmp);
80  }
81  return result;
82 }
