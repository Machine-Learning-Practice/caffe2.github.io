170  struct DispatchStrategy final {
171  // this is caching the index so we don't have to parse the schema inputs
172  // again and again for each dispatcher lookup.
173  // reverse_index means this is the distance from the first tensor argument
174  // to argument_list.end(), i.e. from the top of the stack.
175  // Since it is distance to end(), this means it's 1-indexed,
176  // i.e. '1' is the last argument.
177  size_t reverse_index_of_first_tensor_arg_;
178  bool first_tensor_arg_is_tensor_list_;
179 
180  TensorTypeId get_dispatch_key(const Stack* stack) const {
181  auto first_tensor_arg = torch::jit::peek(
182  *stack,
183  0,
184  reverse_index_of_first_tensor_arg_
185  );
186  if (first_tensor_arg_is_tensor_list_) {
187  const auto& tensor_list = first_tensor_arg.toTensorListRef();
188  if (tensor_list.size() == 0) {
189  throw std::runtime_error("Tried to dispatch based on an empty tensor list. When the first tensor argument of an operator is a tensor list, then it must not be empty.");
190  }
191  return tensor_list[0].type_id();
192  } else {
193  // TODO Avoid bumping the refcounter
194  return first_tensor_arg.toTensor().type_id();
195  }
196  }
197  };

199  static DispatchStrategy get_dispatch_strategy_(const FunctionSchema& schema) {
200  for (size_t i = 0; i < schema.arguments().size(); ++i) {
201  const auto& type = schema.arguments()[i].type();
202  if (type->isSubtypeOf(TensorType::get())) {
203  return {schema.arguments().size() - i, false};
204  }
205  if (type->isSubtypeOf(ListType::ofTensors())) {
206  return {schema.arguments().size() - i, true};
207  }
208  }
209 
210  AT_ERROR("Tried to create dispatch table for operator schema ", schema.name(), " that doesn't have tensor arguments.");
211  }
