29 std::string getRandomString() {
30  static constexpr char charset[] =
31  "0123456789"
32  "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
33  "abcdefghijklmnopqrstuvwxyz";
34  int fd;
35  uint8_t rand_bytes[num_rand_bytes];
36  ssize_t bytes_read;
37  SYSCHECK(fd = open("/dev/urandom", O_RDONLY));
38  SYSCHECK(bytes_read = read(fd, &rand_bytes, sizeof(rand_bytes)));
39  if (bytes_read != sizeof(rand_bytes))
40  throw std::runtime_error("failed to read from /dev/urandom");
41  SYSCHECK(::close(fd));
42 
43  std::string str;
44  str.reserve(num_rand_bytes);
45  for (uint8_t* byte = rand_bytes; byte != rand_bytes + num_rand_bytes;
46  ++byte) {
47  str.push_back(charset[(*byte) % (sizeof(charset) - 1)]);
48  }
49  return str;
50 }

83 bool isMulticastAddress(struct sockaddr* address) {
84  if (address->sa_family == AF_INET) {
85  struct sockaddr_in* address_ipv4 =
86  reinterpret_cast<struct sockaddr_in*>(address);
87  uint32_t host_addr = ntohl(address_ipv4->sin_addr.s_addr);
88  return (host_addr & 0xF0000000) == 0xE0000000;
89  } else if (address->sa_family == AF_INET6) {
90  struct sockaddr_in6* address_ipv6 =
91  reinterpret_cast<struct sockaddr_in6*>(address);
92  auto& addr_bytes = address_ipv6->sin6_addr.s6_addr;
93  // NOTE: address is in network byte order
94  return addr_bytes[0] == 0xff;
95  } else {
96  throw std::invalid_argument("unsupported address family");
97  }
98 }

100 int bindMulticastSocket(
101  struct sockaddr* address,
102  struct sockaddr_storage* sock_addr,
103  int timeout_sec = 1,
104  int ttl = 1) {
105  struct timeval timeout = {.tv_sec = timeout_sec, .tv_usec = 0};
106 
107  int socket, optval;
108  SYSCHECK(socket = ::socket(address->sa_family, SOCK_DGRAM, 0));
109  optval = 1;
110  SYSCHECK(
111  ::setsockopt(socket, SOL_SOCKET, SO_REUSEADDR, &optval, sizeof(int)));
112 
113  if (address->sa_family == AF_INET) {
114  struct sockaddr_in* sock_addr_ipv4 =
115  reinterpret_cast<struct sockaddr_in*>(sock_addr);
116  struct sockaddr_in* address_ipv4 =
117  reinterpret_cast<struct sockaddr_in*>(address);
118  std::memset(sock_addr_ipv4, 0, sizeof(*sock_addr_ipv4));
119  sock_addr_ipv4->sin_family = address->sa_family;
120  sock_addr_ipv4->sin_addr.s_addr = INADDR_ANY;
121  sock_addr_ipv4->sin_port = address_ipv4->sin_port;
122 
123  SYSCHECK(::bind(
124  socket,
125  reinterpret_cast<struct sockaddr*>(sock_addr_ipv4),
126  sizeof(*sock_addr_ipv4)));
127  SYSCHECK(::setsockopt(
128  socket, SOL_SOCKET, SO_RCVTIMEO, (char*)&timeout, sizeof(timeout)));
129 
130  struct ip_mreq mreq;
131  mreq.imr_multiaddr = address_ipv4->sin_addr;
132  mreq.imr_interface.s_addr = htonl(INADDR_ANY);
133  SYSCHECK(
134  ::setsockopt(socket, IPPROTO_IP, IP_MULTICAST_TTL, &ttl, sizeof(ttl)));
135  SYSCHECK(::setsockopt(
136  socket, IPPROTO_IP, IP_ADD_MEMBERSHIP, &mreq, sizeof(mreq)));
137 
138  sock_addr_ipv4->sin_addr = address_ipv4->sin_addr;
139  } else if (address->sa_family == AF_INET6) {
140  struct sockaddr_in6* sock_addr_ipv6 =
141  reinterpret_cast<struct sockaddr_in6*>(sock_addr);
142  struct sockaddr_in6* address_ipv6 =
143  reinterpret_cast<struct sockaddr_in6*>(address);
144  std::memset(sock_addr_ipv6, 0, sizeof(*sock_addr_ipv6));
145  sock_addr_ipv6->sin6_family = address->sa_family;
146  sock_addr_ipv6->sin6_addr = in6addr_any;
147  sock_addr_ipv6->sin6_port = address_ipv6->sin6_port;
148 
149  SYSCHECK(::bind(
150  socket,
151  reinterpret_cast<struct sockaddr*>(sock_addr_ipv6),
152  sizeof(*sock_addr_ipv6)));
153  SYSCHECK(::setsockopt(
154  socket, SOL_SOCKET, SO_RCVTIMEO, (char*)&timeout, sizeof(timeout)));
155 
156  struct ipv6_mreq mreq;
157  mreq.ipv6mr_multiaddr = address_ipv6->sin6_addr;
158  mreq.ipv6mr_interface = 0;
159  SYSCHECK(::setsockopt(
160  socket, IPPROTO_IPV6, IPV6_MULTICAST_HOPS, &ttl, sizeof(ttl)));
161  SYSCHECK(::setsockopt(
162  socket, IPPROTO_IPV6, IPV6_JOIN_GROUP, &mreq, sizeof(mreq)));
163 
164  sock_addr_ipv6->sin6_addr = address_ipv6->sin6_addr;
165  }
166 
167  return socket;
168 }

171 std::vector<MulticastMessage> getMessages(
172  struct sockaddr* addr,
173  rank_type world_size,
174  std::string group_name,
175  std::string packed_msg) {
176  struct sockaddr_storage sock_addr;
177  int socket = bindMulticastSocket(addr, &sock_addr);
178  // NOTE: Multicast membership is dropped on close
179  ResourceGuard socket_guard([socket]() { ::close(socket); });
180 
181  std::set<std::string> msgs = {packed_msg};
182 
183  char recv_message[max_msg_length];
184  if (packed_msg.length() + 1 > max_msg_length) {
185  throw std::logic_error("message too long for multicast init");
186  }
187 
188  auto broadcast = [socket, &sock_addr, &packed_msg]() {
189  SYSCHECK(::sendto(
190  socket,
191  packed_msg.c_str(),
192  packed_msg.size() + 1,
193  0,
194  reinterpret_cast<struct sockaddr*>(&sock_addr),
195  sock_addr.ss_family == AF_INET ? sizeof(struct sockaddr_in)
196  : sizeof(struct sockaddr_in6)));
197  };
198 
199  broadcast();
200 
201  // Wait for messages from all processes
202  while (msgs.size() < world_size) {
203  try {
204  SYSCHECK(::recv(socket, recv_message, sizeof(recv_message), 0));
205  std::string recv_message_str(recv_message);
206 
207  if (recv_message_str == packed_msg)
208  continue; // ignore multicast loopback
209 
210  // We should ignore messages coming from different group
211  auto recv_msg = MulticastMessage(recv_message_str);
212  if (recv_msg.group_name != group_name) {
213  continue;
214  }
215 
216  msgs.insert(
217  recv_message_str); // set will automatically deduplicate messages
218  } catch (const std::system_error& e) {
219  // Check if this was really a timeout from `recvfrom` or a different
220  // error.
221  if (errno != EAGAIN && errno != EWOULDBLOCK)
222  throw;
223  }
224 
225  broadcast();
226  }
227 
228  // Just to decrease the probability of packet loss deadlocking the system
229  std::this_thread::sleep_for(std::chrono::milliseconds(50));
230  broadcast();
231 
232  std::vector<MulticastMessage> unpacked_msgs;
233  for (auto& msg : msgs) {
234  unpacked_msgs.emplace_back(msg);
235  }
236 
237  return unpacked_msgs;
238 }

270 InitMethod::Config initTCPMulticast(
271  std::string group_name,
272  rank_type world_size,
273  int assigned_rank,
274  struct sockaddr* addr) {
275  InitMethod::Config config;
276 
277  int listen_socket;
278  port_type listen_port;
279  std::tie(listen_socket, listen_port) = listen();
280  ResourceGuard listen_socket_guard(
281  [listen_socket]() { ::close(listen_socket); });
282 
283  MulticastMessage msg{group_name, listen_port, assigned_rank};
284  std::string packed_msg = msg.pack();
285 
286  std::vector<MulticastMessage> msgs =
287  getMessages(addr, world_size, group_name, packed_msg);
288 
289  std::vector<MulticastMessage*> sorted_msgs(msgs.size());
290 
291  // Pre-fill sorted_msgs with processes that had their ranks assigned manually
292  for (auto& msg : msgs) {
293  if (msg.rank >= 0) {
294  if (sorted_msgs[msg.rank] != nullptr)
295  throw std::logic_error("more than one node have assigned same rank");
296  sorted_msgs[msg.rank] = &msg;
297  }
298  }
299 
300  // NOTE: msgs are already sorted lexicographically, so we can greedily
301  // insert them into free slots
302  size_t free_pos = 0;
303  for (auto& msg : msgs) {
304  if (msg.rank >= 0)
305  continue; // These were sorted in the previous loop
306  while (sorted_msgs[free_pos] != nullptr)
307  free_pos++;
308  sorted_msgs[free_pos] = &msg;
309  }
310 
311  auto& master_msg = *sorted_msgs[0];
312  for (size_t rank = 0; rank < sorted_msgs.size(); ++rank) {
313  if (packed_msg == sorted_msgs[rank]->pack()) {
314  config.rank = rank;
315  config.world_size = world_size;
316  if (config.rank == 0) {
317  listen_socket_guard.release();
318  config.master = {
319  .listen_socket = listen_socket,
320  .listen_port = master_msg.port,
321  };
322 
323  config.public_address = discoverWorkers(listen_socket, world_size);
324  } else {
325  std::string master_address;
326  std::tie(master_address, config.public_address) =
327  discoverMaster(master_msg.addresses, master_msg.port);
328  config.worker = {
329  .master_addr = master_address,
330  .master_port = master_msg.port,
331  };
332  }
333  break;
334  }
335  }
336 
337  return config;
338 }

342 InitMethod::Config initTCP(
343  std::string argument,
344  int world_size_r,
345  std::string group_name,
346  int rank) {
347  group_name.append("#"); // To make sure it's not empty
348  argument.erase(0, 6); // chop "tcp://"
349  rank_type world_size;
350  try {
351  world_size = convertToRank(world_size_r);
352  } catch (std::exception& e) {
353  if (world_size_r == -1) {
354  throw std::invalid_argument(
355  "world_size is not set - it is required for "
356  "`tcp://` init methods with this backend");
357  }
358  throw std::invalid_argument("invalid world_size");
359  }
360 
361  // Parse arguments
362  std::string address, str_port;
363  std::tie(address, str_port) = splitAddress(argument);
364 
365  // Resolve addr and select init method
366  struct addrinfo hints = {0};
367  hints.ai_family = AF_UNSPEC;
368  struct addrinfo* res;
369  if (::getaddrinfo(address.c_str(), str_port.c_str(), &hints, &res)) {
370  throw std::invalid_argument("invalid init address");
371  }
372  ResourceGuard res_guard([res]() { ::freeaddrinfo(res); });
373 
374  for (struct addrinfo* head = res; head != NULL; head = head->ai_next) {
375  if (head->ai_family != AF_INET && head->ai_family != AF_INET6)
376  continue;
377  try {
378  if (isMulticastAddress(head->ai_addr)) {
379  return initTCPMulticast(group_name, world_size, rank, head->ai_addr);
380  } else {
381  return initTCPMaster(address, str_port, world_size, rank);
382  }
383  } catch (std::exception& e) {
384  if (!head->ai_next)
385  throw;
386  }
387  }
388  throw std::runtime_error("failed to initialize THD using given address");
389 }
