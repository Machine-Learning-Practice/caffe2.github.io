92  double p = p_.value_or(2.0).toDouble();
93  Tensor self_scaled;
94  Tensor scale_v;
95  if (p == 0.0) {
96  return zeros_like(self);
97  } else if (p == 1.0) {
98  return self.sign() * grad;
99  } else if (p == 2.0) {
100  self_scaled = self;
101  scale_v = grad / norm;
102  } else if (std::isinf(p)) {
103  self_scaled = self.sign() * (self.abs() == norm).toType(self.type());
104  scale_v = grad.clone();
105  } else if (p < 2.0) {
106  self_scaled = self.sign() * self.abs().pow(p - 1);
107  scale_v = grad / norm.pow(p - 1);
108  } else {
109  self_scaled = self * self.abs().pow(p - 2);
110  scale_v = grad / norm.pow(p - 1);
111  }
112  // handle case at 0 where we return a subgradient containing 0
113  scale_v.masked_fill_(norm == 0, 0);
114  return self_scaled * scale_v;

280 Tensor cumprod_backward(const Tensor &grad, const Tensor &input, int64_t dim) {
281  /*
282  There are two algorithms to do this. The first one
283  is very efficient, but works only when there are no
284  nonzero elements in the input.
285 
286  The second one is much more complex, but it doesn't
287  assume anything on the input. The main downside is
288  that it takes time O(n^2), where n = input.size(self.dim)
289  (i.e. the length of the cumulative product). This is in
290  contrast to the forward pass and the efficient algorithm,
291  which are both O(n).
292 
293  The second algorithm is a simple application of the chain
294  rule. If x is an n-dimensional vector, and y = cumprod(x),
295  and F is the final cost, then
296 
297  dF / dx_k = sum_j (dF / dy_j) * (dy_j / dx_k) (1)
298 
299  The term dF / dy_j is just grad_output[j] (assuming again
300  everything is one-dimensional).
301 
302  The term (dy_j / dx_k) is easilly seen to be
303 
304  if j >= k
305  dy_j / dx_k = prod_{1 <= i <= j, i != k} x_i
306  else:
307  dy_j / dx_k = 0
308 
309  Note that the indicator (j>=k) can be taken out
310  by replacing the sum in (1) with a sum from
311  j = k to n.
312 
313  Thus,
314  df / dx_k = sum_{k <= j <= n} grad_output[j] * (dy_j / dx_k)
315 
316  with
317  dy_j / dx_k = prod_{1 <= i <= j, i != k} x_i (2)
318 
319  Note that this last term is just the cumulative product
320  with k omitted. Thus, if x_k (the input) is nonzero, we can
321  just express this as
322 
323  dy_j / dx_k = (prod_{1 <= i <= j} x_i) / x_k
324  = y_j / x_k
325 
326  So therefore,
327 
328  df / dx_k = sum_{k <= j <= n} grad_output[j] * y_j / x_k
329 
330  so
331 
332  grad_output = sum_scan_exclusiv(grad_output * output) / input
333 
334  If the input is nonzero, we need to calculate the dy_j / dx_k
335  by using the formula (2), called in the code omitted_products.
336 
337  The way the code calculates it is simply by noting that
338 
339  prod_{1 <= i <= j, i != k} x_i
340  = (prod_{1 <= i <= k} x_i) * (prod_{k + 1 <= i <= j} x_i)
341 
342  the first term is calculated as prods_until_k, which since
343  doesn't depend in j is easy to vectorize.
344 
345  The second term (indexed by j) is the cumulative product of
346  x_{k+1}, x_{k+2}, ..., x_n, and it's named in the code
347  prods_from_k_pkus_1, and it's calculated as a cumprod.
348 
349  In order to vectorize this properly, we need to add to
350  omitted_products the dimensions where k > j, and therefore
351  dy_j / dx_k = 0, which is done right after the assert.
352  */
353 
354  if (input.dim() == 0) {
355  return grad;
356  }
357  dim = at::maybe_wrap_dim(dim, input.sizes().size());
358  int64_t dim_size = input.size(dim);
359  if (dim_size == 1) {
360  return grad;
361  }
362 
363  // Simple case with nonzero elements in the input
364  if ((input != 0).all().item<uint8_t>()) {
365  Tensor result = at::cumprod(input, dim);
366  return sum_scan_exclusive(result * grad, dim) / input;
367  }
368 
369  auto ones_size = input.sizes().vec();
370  ones_size[dim] = 1;
371  Tensor ones = at::ones({1}, grad.options()).expand(ones_size);
372  Tensor grad_input = at::zeros(input.sizes(), grad.options());
373  Tensor prods_from_k_plus_1;
374  Tensor omitted_products;
375  for (int k = 0; k < dim_size; ++k) {
376  if (k == 0) {
377  prods_from_k_plus_1 = at::cumprod(input.slice(dim, k + 1), dim);
378  omitted_products = at::cat({ones, prods_from_k_plus_1}, dim);
379  } else if (k == dim_size - 1) {
380  Tensor prods_until_k = at::prod(input.slice(dim, 0, k), dim, true);
381  omitted_products = prods_until_k;
382  } else {
383  Tensor prods_until_k = at::prod(input.slice(dim, 0, k), dim, true);
384  prods_from_k_plus_1 = at::cumprod(input.slice(dim, k+1), dim);
385  omitted_products = prods_until_k.expand_as(prods_from_k_plus_1) * prods_from_k_plus_1;
386  omitted_products = at::cat({prods_until_k, omitted_products}, dim);
387  }
388 
389  // At this point omitted_products is the same size
390  // as input, except on the dimension dim where it's
391  // dim_size - k
392  AT_ASSERT(omitted_products.size(dim) == dim_size - k);
393 
394  grad_input.select(dim, k).copy_(
395  at::sum(grad.slice(dim, k) * omitted_products,dim));
396  }
397 
398  return grad_input;
399 }

543  return t.transpose(dim, 0).contiguous().view({t.size(dim), -1});
544  };

546  return t.contiguous().view(transposed_sizes).transpose(dim, 0);

550  // we need the flatten and unflatten business. TODO: simplify this when we
551  // add support for norm over multiple dimensions.
552  auto self_flat = flatten(self);
553  auto grad_flat = flatten(grad);
554  auto norm_flat = self_flat.norm(p, 1, true);
555  auto grad_output = (self_flat * grad_flat).sum(1, true);
556  auto nb = norm_backward(grad_output, self_flat, p, norm_flat, 1, true);
557  auto invnorm = (norm_flat + 1e-7).reciprocal();
558  auto grad_norm = unflatten(maxnorm * invnorm * (grad_flat - invnorm * nb));
559  auto norm = unflatten(norm_flat.expand_as(self_flat));

684 Tensor cholesky_backward(Tensor grad, bool upper, Tensor L) {
685  // cf. Iain Murray (2016); arXiv 1602.07527
686  if (upper) {
687  grad = grad.transpose(-1, -2);
688  } else {
689  L = L.transpose(-1, -2);
690  }
691 
692  auto phi = [](const Tensor & A) -> Tensor {
693  auto B = A.tril();
694  B = B - 0.5 * at::diag_embed(B.diagonal(0, -1, -2), 0, -2, -1);
695  return B;
696  };
697 
698  // make sure not to double-count variation, since
699  // only half of output matrix is unique
700  auto Lbar = grad.tril();
701 
702  auto P = phi(at::matmul(L, Lbar));
703  Tensor S;
704  std::tie(S, std::ignore) = at::solve(P + P.transpose(-1, -2), L);
705  std::tie(S, std::ignore) = at::solve(S.transpose(-1, -2), L);
706  S = phi(S);
707  if (upper) {
708  S = S.transpose(-1, -2);
709  }
710  return S;
711 }

753 Tensor glu_double_backward(const Tensor & grad, const Tensor & grad_output, const Tensor & input, int64_t dim) {
754  auto& gO = grad_output;
755  auto input_size = input.size(dim) / 2;
756  auto first_half = input.narrow(dim, 0, input_size);
757  auto second_half = input.narrow(dim, input_size, input_size);
758  auto sig_second_half = second_half.sigmoid();
759  auto one_sub_sig_second_half = 1 - sig_second_half;
760  auto sig_one_sub_sig = sig_second_half * one_sub_sig_second_half;
761 
762  auto ggI_first_half = grad.narrow(dim, 0, input_size);
763  auto ggI_second_half = grad.narrow(dim, input_size, input_size);
764  auto ggI_second_half_times_first_half = ggI_second_half * first_half;
765 
766  auto gI_first_half = ggI_second_half * gO * sig_one_sub_sig;
767  auto second_order_sh = sig_one_sub_sig * one_sub_sig_second_half - sig_second_half * sig_one_sub_sig;
768  auto gI_second_half = ggI_second_half_times_first_half * gO * second_order_sh + ggI_first_half * gO * sig_one_sub_sig;
769  return at::cat({gI_first_half, gI_second_half}, dim);
770 }

937 // NOTE [ as_strided Backward and layout-aware/agnostic autograd ]
938 //
939 // `storage_offset` is ignored for simplicity in this note. If you just want the
940 // full algorithm without explanation, scroll down to bottom of this note.
941 //
942 // Implementing the backward of as_strided is tricky because you have to deal
943 // with mappings that map one memory location to multiple indices, i.e., the
944 // output tensor has multiple indices pointing to **overlapping** memory
945 // addresses. This can happen in all in all sorts of weird cases. For example,
946 //
947 // x = torch.randn(15)
948 // x.as_strided([3, 3], [1, 0]) # "expand" case
949 // x.as_strided([3, 3], [2, 1]) # "size too large" case
950 // x.as_strided([3, 2], [3, 6]) # res[2, 0] points to 2*3 + 0*6 = 6
951 // # res[0, 1] points to 0*3 + 1*6 = 6
952 //
953 // Here is the general strategy we apply in implementing as_strided backward:
954 // 0. ??? (optimization step. we will talk about this later)
955 // 1. Create some underlying flattened tensor as if it is the base tensor
956 // representing the contiguous memory storage for both input and output.
957 // 2. Use the output geometry to scatter (or index_add) the gradients into
958 // this storage tensor.
959 // 3. ??? (fix for input tensor with overlapping memory. we will talk about
960 // this later)
961 // 4. Return the as_strided view of the storage tensor using input geometry.
962 //
963 // In step (2), if the output tensor does't have overlapping memory, we can
964 // safely scatter (`storage.as_strided(output_geometry).copy_(grad)`);
965 // otherwise, we must use `index_add` as gradients at different indices may need
966 // to be summed to a single location.
967 //
968 // For example, in this case:
969 //
970 // x = torch.randn(3)
971 // y = x.as_strided([3, 3], [1, 0]) # "expand" case
972 // # size [ 3, 3]
973 // # stride [ 1, 0]
974 // y.backward() # step (1): contiguous storagte tensor `s` of size 3, which
975 // is large enough to be used as underlying storage
976 // for `x` and `y`.
977 // s = [ 0, 0, 0]
978 // # step (2): since `y` has overlapping memory, index_add grad
979 // into `s` basing on `y`'s geometry, i.e.,
980 // s[i * y.stride(0) + j * y.stride(1)] += gy[i, j].
981 // s = [ 3, 3, 3]
982 // # step (4): as_strided view `s` using `x`'s geometry
983 // s = [ 3, 3, 3]
984 // grad_input = s.as_strided(x.size(), x.stride())
985 // = s.as_strided([3], [1])
986 // = [ 3, 3, 3]
987 //
988 // This is exactly what we would get if using `expand`. However, here the input
989 // tensor doesn't have overlapping memory. If it does, we must add an extra step
990 // before (4). Considering this case:
991 //
992 // t = torch.randn(3)
993 // x = t.expand(3, 3) # input with overlapping memory
994 // # size [3, 3]
995 // # stride [0, 1]
996 // y = x.as_strided([1], [1]) # contiguous output
997 // # size [1]
998 // # stride [1]
999 // y.backward() # step (1): contiguous storage tensor `s` of size 3, which
1000 // is large enough to be used as underlying storage
1001 // for `x` and `y`.
1002 // s = [ 0, 0, 0]
1003 // # step (2): scatter grad into `s` basing on `y`'s geometry
1004 // s = [ 1, 0, 0]
1005 // # step (4): as_strided view `s` using `x`'s geometry
1006 // s = [ 1, 0, 0]
1007 // grad_input = s.as_strided([3, 3], [0, 1])
1008 // = s.as_strided([3, 3], [0, 1])
1009 // = [[ 1, 0, 0],
1010 // [ 1, 0, 0],
1011 // [ 1, 0, 0]]
1012 // Is this result correct?
1013 //
1014 // `x.as_strided([1], [1])` call is obviously equivalent with
1015 // `x[(0,) * x.dim()].view(1)` for any `x`. But autograd through the second
1016 // gives gradient `[ [ 1, 0, 0], [ 0, 0, 0], [ 0, 0, 0]]`. For this specific
1017 // case, indexing `x` at any index in first column is also equivalent, and
1018 // yields a gradient of shape `[3 x 3]` containing eight 0's and one 1. There is
1019 // an `x.size(1)`-times difference between these gradients computed from other
1020 // PyTorch ops and the gradient we got from as_strided.
1021 //
1022 // You might conclude that the gradients from as_strided is wrong. However,
1023 // let's first see why they are actually reasonable. Consider the pointwise
1024 // perturbations by `delta` anywhere in the first column of `x`. It will lead to
1025 // a `delta` change in the same memory location, and then `y` will change by
1026 // `delta`. So one can say the gradient should be exactly 1 at the first column,
1027 // as given by our above procedure.
1028 //
1029 // In the above computation of numerical gradients, they only match the
1030 // analytical results because strides and memory locations are considered in the
1031 // forward pass, i.e., this op (including both forward and backward) is
1032 // layout-aware.
1033 //
1034 // However, in PyTorch, most (probably all) other ops (forward and backward) are
1035 // layout-agnostic. E.g.,
1036 //
1037 // t = torch.randn(1)
1038 // x = t.expand(2)
1039 // y = x.sum()
1040 // y.backward()
1041 //
1042 // Layout-agnostic autograd (as it is currently in PyTorch) will give you
1043 //
1044 // gy = 1
1045 // gx = [ 1, 1] # SumBackward: torch.ones_like(x)
1046 // gt = [ 2] # ExpandBackward: gx.sum()
1047 //
1048 // Note that `gx = [ 1, 1]`. However, if you perturb any value in `x` by `delta`
1049 // (the other will also change by `delta`), `y` will change by `2 * delta`. So
1050 // the gradients, if strides are taken into consideration, should be 2.
1051 //
1052 // Layout-aware autograd should give you
1053 //
1054 // gy = 1
1055 // gx = [ 2, 2] # Because the backward considers the fact that the input `x`
1056 // # is already expanded.
1057 // gt = [ 2] # Layout-aware backward of expand is just a slicing because
1058 // # the previous backward should have already taken care of
1059 // # strides and made sure that gradients are the same along the
1060 // # expanded dimension.
1061 //
1062 // As shown above, these two types are not compatible. Therefore, we must either
1063 // make as_strided layout-agnostic, or make all other ops layout-aware.
1064 //
1065 // It is difficult to support layout-aware autograd (at least in the current
1066 // codebase structure), because it would mean
1067 // 1. storing tensor geometries of every input tensor for backward
1068 // 2. depending on input geometry, the gradient computed from backward change
1069 // 3. ideally enforcing gradient of T to always have same strides as T
1070 // (although these two methods only differ when it comes to overlapping memory)
1071 //
1072 // Therefore, we must formulate `as_strided` in a layout-agnostic way, i.e.,
1073 // giving the same output regardless of the input layout. We consider
1074 // `input.stride()` as a separate independent fixed argument `input_stride`.
1075 // Then, `as_strided(input, size, stride)` can be thought of as:
1076 // 1. "Scatter" each value of `input` into a "storage" using storage location
1077 // computed from the value's index in `input`, `input.size()` and
1078 // `input_stride`, but if N values end up in the same location, the value
1079 // is average of those N values (they will be the same value anyways).
1080 //
1081 // Formal description:
1082 // Denote the set of all input indices that pointing to the same storage
1083 // location `storage[n]` as `S(n)`, i.e.,
1084 //
1085 // S(n) = { index : <index, input_stride> == n, index is valid given input.size() },
1086 //
1087 // where `<x, y>` is the dot product between `x` and `y`.
1088 //
1089 // Then, the process is:
1090 //
1091 // storage[n] = Avg { S(n) }
1092 //
1093 // Note that all values in `S(n)` are the same (they point to the same
1094 // memory location anyways, so this step doesn't change anything, but
1095 // effectively avoids having the denpendency on the layout of `input`.
1096 // I.e., the result holds fixed regardless of the layout of `input`, as
1097 // long as `input_stride` is fixed.
1098 //
1099 // NOTE: for forward pass, we can equivalently simply selet any one of
1100 // `S(n)` as `storage[n]`. However, cosnidering this as an average
1101 // operation makes backward easier (so all values in set
1102 // `{ grad_input[i] : i in S(n) }` are the same, and it can use the
1103 // same geometry as input).
1104 // 2. As usual, return the as_strided view of `storage` using required output
1105 // `size` and `stride`.
1106 //
1107 // To backward through this layout-agnostic version, we simply add the following
1108 // step:
1109 // .... (scatter gradients into the storage tensor using output geometry)
1110 // 3. For all storage location n, `storage[n] /= |S(n)|`.
1111 // .... (return as_strided view of the storage tensor using input geometry)
1112 //
1113 // Finally, we note that these general operations are expensive, so we apply the
1114 // following optimizations:
1115 // Add step (0): For all output dimension `d` with output stride 0, sum the
1116 // gradients along dimension `d` (don't keepdim), and remove
1117 // dimension `d` from output size and stride.
1118 // (An optimization for "expand" cases so we may avoid step (3))
1119 // Only apply step (3) when input tensor has overlapping memory.
1120 //
1121 // FULL ALGORITHM:
1122 // 0. For all output dimension `d` with output stride 0, sum the gradients
1123 // along dimension `d` (don't keepdim), and remove dimension `d` from
1124 // output size and stride.
1125 // 1. Create some underlying flattened tensor as if it is the base tensor
1126 // representing the contiguous memory storage for both input and output.
1127 // 2. Use the output geometry to scatter (or index_add) the gradients into
1128 // this storage tensor `storage`.
1129 // 3. If input tensor has overlapping memory,
1130 // For all storage location `i`, `storage[i] /= N(i)`, where `N(i)` is the
1131 // number of indices in input geometry pointing to the same storage
1132 // location `i` (i.e., `|S(i)|` in equations above).
1133 // 4. Return the as_strided view of the storage tensor using input geometry.
1134 //
1135 // See NOTE [ Detecting Memory Overlap Within A Strided Tensor ] on how to
1136 // roughly detech overlapping memory.
1137

1139 // NOTE [ Detecting Memory Overlap Within A Strided Tensor ]
1140 //
1141 // Checking memory overlap within a strided tensor is the special case of
1142 // detecting memory overlap of two strided tensors, where the two tensors start
1143 // at the same memory address. The later is HARD (see #8212).
1144 //
1145 // But even this special case isn't simple. This note describes a check for a
1146 // even more constrained simple case where we can be certain that there is no
1147 // overlap.
1148 //
1149 // The checking algorithm can be described as:
1150 // 0. Return [ pass check ] if any dimension has size 0
1151 // 1. Ignore all dimensions that have size 1
1152 // 2. If no remaining dimensions, return [ pass check ]
1153 // 3. Sort the remaining dimensions according to the strides decreasingly
1154 // 4. Check that for each dimension k,
1155 //
1156 // stride[k] > \sum_{ i > k } (size[i] - 1) * stride[i]
1157 //
1158 // That is equivalent to, after reording the dimensions so strides are
1159 // in decreasing order, checking that stride of each dimension is larger
1160 // than the maximum memory offset in a slice at that dimension.
1161 //
1162 // Obviously this check passes for contiguous tensors ( the dimensions will be
1163 // already sorted with LHS = stride[0] = \prod size[i] being exactly 1 larger
1164 // than RHS ). Similarly, the check passes for tensors contiguous in all but
1165 // the last dimension, and LHS = stride[0] = stride[-1] * \prod size[i] being
1166 // exactly stride[-1] larger than RHS. (*)
1167 //
1168 // We will show that these view operations, including all our view operations
1169 // *except for* general as_strided and unfold, also preserve this invariant:
1170 //
1171 // alias: Obviously preserves
1172 //
1173 // expand: All changed dimensions are removed in step (1)
1174 //
1175 // view: Consider the input dimensions as grouped into consecutive
1176 // dimension "blocks", where dimensions are contiguous in each one.
1177 // one. view only works when the output dimensions can also be
1178 // grouped into the same consecutive blocks of same ordering.
1179 //
1180 // NB: this means that the number of elements and stride of the
1181 // last dimension in each block is the same in input and
1182 // output. (**)
1183 //
1184 // Notation:
1185 // Consider a single such block B,
1186 // ... B_prev[-1]], [ B[0], ..., B[i], ..., B[k] = B[-1] ], [ B_next[0], ...
1187 // start--^^^^ ^^^^^^^^^^^^--end
1188 // Each B[i] denotes a dimension index such that B[i] = B[0] + i.
1189 //
1190 // We first show that in a tensor (i.e., input) satisfies the
1191 // invariant, after sorting, the dimensions within each block
1192 // still remain consecutive. (***)
1193 //
1194 // After removing dimensions of size 1, the dimensions within a
1195 // block is already sorted by strides in descending order. So
1196 // sorting all dimensions will not change the relative ordering
1197 // among them.
1198 //
1199 // Assume that some block B is not consecutive after sorting,
1200 // i.e., there exists a dimension d between B[0] and B[-1] in
1201 // sorted order.
1202 //
1203 // By (*), we know that
1204 // stride[B[0]]
1205 // = \sum_{i > 0} (size[B[i]] - 1) * stride[B[i]] + stride[B[-1]]
1206 // < \sum_{i > 0} (size[B[i]] - 1) * stride[B[i]] + stride[d]
1207 // <= \sum_{i > 0} (size[B[i]] - 1) * stride[B[i]] + (size[d] - 1) * stride[d]
1208 // <= \sum{j > B[0]} (size[j] - 1) * stride[j],
1209 //
1210 // where the first < comes from sorting and
1211 // the second <= comes from the fact that dimension d
1212 // exists after step (1) and
1213 // thus must have size greater
1214 // than 1
1215 // the third <= comes from the fact that each term in
1216 // the sum is non-negative
1217 //
1218 // Then we have a countradiction as the invariant must not be
1219 // satisfied at B[0]. So the original proposition is true.
1220 //
1221 // Now that we established the above claim (***), we consider the
1222 // view operation as first sorting the dimensions (i.e., blocks),
1223 // apply the original view (since it only cares dimensions being
1224 // consecutive and contiguous withtin each block), and then undo
1225 // the sort.
1226 //
1227 // Consider a single block B in the output,
1228 // ... ], [ B[0], ..., B[i], ..., B[k] = B[-1] ], [ ...
1229 // start--^^^^ ^^^^^^^^^^^^--end
1230 //
1231 // By (*), we know that for all i
1232 // stride[i] = stride[B[-1]] +
1233 // \sum_{j=i+1}^{k} (size[B[j]] - 1) * stride[B[j]]
1234 //
1235 // Then the invariant is obviously satisfied at every dimension
1236 // in this block if it is satisfied at dimnesion B[-1]. It only
1237 // remains to show that it is satisfied at the last dimension in
1238 // each block.
1239 //
1240 // Since the same blocks are present in both input and output
1241 // with the same ordering, we will abuse the notation in the
1242 // following statements.
1243 //
1244 // By (*), we know that the following holds for both input and
1245 // output, for any block B:
1246 // \sum_{i > B[-1]} (size[i] - 1) * stride[i]
1247 // = \sum_{block B' after B} \prod_{j in B'} size[B[j]] * stride[B'[-1]]
1248 // = \sum_{block B' after B} numel(B') * stride[B'[-1]].
1249 // ^^^^^^^^^^^^^^^^^^^^^^^|^^^^^^^^^^^^^^^^^^^^^^^^^^
1250 // By (**), we know that, this quantity in the above equation
1251 // remains the same in input and output. So both
1252 // \sum_{i > B[-1]} (size[i] - 1) * stride[i]
1253 // and
1254 // stride[B[-1]]
1255 // are the same in input and output.
1256 //
1257 // These two quantities are exactly the LHS and RHS of the
1258 // invariant inequality. Since by assumption the invariant is
1259 // satisfied in input at B[-1], it is also satisfied in output at
1260 // B[-1]. This concludes the proof.
1261 //
1262 // squeeze: Special case of view
1263 //
1264 // unsqueeze: Special case of view
1265 //
1266 // slice: Consider slicing dimension i with step = k >= 1.
1267 //
1268 // Let stride' and size' be the output strides and sizes. We have
1269 //
1270 // stride'[i] = k * stride[i]
1271 // size'[i] <= floor(size[i] / k)
1272 //
1273 // If size'[i] = 1, invariant is obviously satisfied as we are
1274 // just removing a dimension (afte step (1)).
1275 //
1276 // Assume size'[i] > 1.
1277 //
1278 // By assumption, the invariant is satisfied at every dimension
1279 // in input.
1280 //
1281 // For any dimension j, if stride[j] > stride[i], we have
1282 // stride'[j] = stride[j]
1283 // > (size[i] - 1) * stride[i]
1284 // = (size[i] / k * k - 1) * k * stride[i] / k
1285 // = (size[i] / k - 1 / k) * stride'[i]
1286 // >= (size'[i] - 1 / k) * stride'[i]
1287 // >= stride'[i].
1288 //
1289 // If stride[j] < stride[i], we have
1290 // stride'[j] = stride[j] < stride[i] <= stride'[i].
1291 //
1292 // So the sorting order remains unchanged after slice.
1293 //
1294 // Since
1295 // (size'[i] - 1) * stride'[i]
1296 // = (floor(size[i] / k) - 1) * k * stride[i]
1297 // <= (size[i] / k - 1) * k * stride[i]
1298 // = (size[i] - k) * stride[i]
1299 // <= (size[i] - 1) * * stride[i],
1300 // the term from this dimension i in the invariant inequality at
1301 // other dimensions can only decrease after slice. So the
1302 // invariant is preserved.
1303 //
1304 // narrow: Special case of slice
1305 //
1306 // select: narrow + squeeze
1307 //
1308 // permute: Sorting makes permutation of dimensions irrelevant
1309 //
1310 // transpose: Sorting makes swapping dimensions irrelevant
1311 //
1312 // diagonal: Effectively merging two dimensions i and j into a new
1313 // dimension k s.t.
1314 // stride'[k] = stride[i] + stride[j]
1315 // size'[k] <= min(size[i], size[j]),
1316 // where stride and size are on the input, and stride' and size'
1317 // are on the output.
1318 //
1319 // Assuming that size[i] > 1 and size[j] > 1. If any has size 1,
1320 // then this is unsqueeze on that dimension.
1321 //
1322 // WLOG, say stride[i] >= stride[j].
1323 //
1324 // Each dimension d in input with stride[d] > stride[j] has
1325 // stride'[d] = stride[d]
1326 // > (size[i] - 1) * stride[i] + (size[j] - 1) * stride[j]
1327 // >= stride[i] + stride[j]
1328 // = stride[k].
1329 // So, considering the sorted dimensions, this is effectively
1330 // removing i, and replacing j with k.
1331 //
1332 // For dimensions d with stride[i] < stride[d] < stride[j], the
1333 // term from dimension i is removed in the invariant inequality.
1334 // For dimensions d with stride[d] > stride[j], we have
1335 // (size'[k] - 1) * stride'[k]
1336 // <= (min(size[i], size[j]) - 1) * (stride[i] + stride[j])
1337 // <= (size[i] - 1) * stride[i] + (size[j] - 1) * stride[j],
1338 // so the term from i and j in the invariant can only decrease.
1339 //
1340 // So this is generally relaxing the constraint, and thus it
1341 // preserves it.

1343 // This implements steps (2)~(4) of the algorithm in
1344 // NOTE [ Detecting Memory Overlap Within A Strided Tensor ]
1345 // Helper for as_strided_backward
1346 static inline bool _maybe_overlapping_memory(IntArrayRef sizes, IntArrayRef strides) {
1347  if (sizes.size() > 0) {
1348  std::vector<std::size_t> argsort(sizes.size());
1349  std::iota(argsort.begin(), argsort.end(), 0);
1350  std::sort(argsort.begin(), argsort.end(),
1351  [&](std::size_t i, std::size_t j){ return strides[i] < strides[j]; });
1352 
1353  int64_t max_index_in_slice = 0;
1354  for (auto i : argsort) {
1355  auto stride_ = strides[i];
1356  if (stride_ <= max_index_in_slice) {
1357  return true;
1358  }
1359  max_index_in_slice += stride_ * (sizes[i] - 1);
1360  }
1361  }
1362  return false;
1363 }

1365 // Returns the minimum storage size needed to contain a tensor of sizes, strides, and storage_offset

1367 static inline int64_t _min_storage_size(IntArrayRef sizes, IntArrayRef strides, int64_t storage_offset) {
1368  int64_t storage_size = storage_offset + 1;

1371  auto size_i = sizes[i];

1373  return storage_offset;
1374  }
1375  storage_size += (size_i - 1) * strides[i];
1376  }
1377  return storage_size;

1381 Tensor as_strided_backward(Tensor grad, TensorGeometry input_geometry, IntArrayRef sizes, IntArrayRef strides, optional<int64_t> storage_offset_) {
1382  // For output geometry,
1383  // check for size 0 dimensions,
1384  // skip size 1 dimensions,
1385  // reduce grad on expanded dims (stride=0, size>1)
1386  // Step (0) for the algorithm in NOTE [ as_strided Backward and layout-aware/agnostic autograd ]
1387  // Step (0)~(1) for the algorithm in NOTE [ Detecting Memory Overlap Within A Strided Tensor ]
1388  // on output geometry
1389  auto storage_offset = storage_offset_.value_or(input_geometry.storage_offset());
1390  auto odim = grad.dim();
1391  std::vector<int64_t> out_sizes_, out_strides_;
1392  out_sizes_.reserve(odim);
1393  out_strides_.reserve(odim);
1394  for (int64_t i = odim - 1; i >= 0; i--) {
1395  auto size_i = sizes[i];
1396  auto stride_i = strides[i];
1397  if (size_i == 0) {
1398  return at::zeros(input_geometry.sizes(), grad.options());
1399  } else if (size_i == 1) {
1400  grad = grad.squeeze(i);
1401  } else if (stride_i == 0) {
1402  grad = grad.sum(i, false);
1403  } else {
1404  out_sizes_.insert(out_sizes_.begin(), size_i);
1405  out_strides_.insert(out_strides_.begin(), stride_i);
1406  }
1407  }
1408  // Step (2)~(4) for the algorithm in NOTE [ Detecting Memory Overlap Within A Strided Tensor ]
1409  // on output geometry
1410  auto out_maybe_overlap = _maybe_overlapping_memory(out_sizes_, out_strides_);
1411 
1412  // For input geometry,
1413  // check for size 0 dimensions,
1414  // skip size 1 dimensions,
1415  // Step (0)~(1) for the algorithm in NOTE [ Detecting Memory Overlap Within A Strided Tensor ]
1416  // on input geometry
1417  auto idim = input_geometry.dim();
1418  IntArrayRef inp_sizes = input_geometry.sizes(), inp_strides = input_geometry.strides();
1419  std::vector<int64_t> inp_sizes_, inp_strides_;
1420  inp_sizes_.reserve(idim);
1421  inp_strides_.reserve(idim);
1422  for (int64_t i = idim - 1; i >= 0; i--) {
1423  auto size_i = inp_sizes[i];
1424  auto stride_i = inp_strides[i];
1425  if (size_i == 0) {
1426  return at::zeros(input_geometry.sizes(), grad.options());
1427  } else if (size_i != 1) {
1428  inp_sizes_.insert(inp_sizes_.begin(), size_i);
1429  inp_strides_.insert(inp_strides_.begin(), stride_i);
1430  }
1431  }
1432  // Step (1)~(4) for the algorithm in NOTE [ Detecting Memory Overlap Within A Strided Tensor ]
1433  // on input geometry
1434  auto inp_maybe_overlap = _maybe_overlapping_memory(inp_sizes_, inp_strides_);
1435 
1436 
1437  // Rest of this function implements
1438  // Step (1)~(4) for the algorithm in NOTE [ as_strided Backward and layout-aware/agnostic autograd ]
1439  // TODO: Raise if not all output values are visible in input geometry.
1440  // Technically speaking, if you treat those values as constants, not
1441  // raising is fine, and mathematically correct. However, these values
1442  // really are contained in some base tensor, and by treating them as
1443  // constants we are ignoring this tight dependency. Therefore, it is
1444  // more sensible to raise here.
1445 
1446  // Step (1): create underlying tensor as "storage"
1447  auto shared_offset = std::min(input_geometry.storage_offset(), storage_offset);
1448  auto inp_effective_offset = input_geometry.storage_offset() - shared_offset;
1449  auto out_effective_offset = storage_offset - shared_offset;
1450  auto base_size = std::max(
1451  _min_storage_size(inp_sizes_, inp_strides_, inp_effective_offset),
1452  _min_storage_size(out_sizes_, out_strides_, out_effective_offset)
1453  );
1454  auto storage = at::zeros({base_size}, grad.options());
1455 
1456  // prepare indices tensor if we will do index_add_ later
1457  c10::optional<at::Tensor> flatten_full_indices;
1458  if (inp_maybe_overlap || out_maybe_overlap) {
1459  flatten_full_indices = at::arange(0, base_size, grad.options().dtype(at::kLong));
1460  }
1461 
1462  // Step (2): use output geometry to scatter gradients into storage
1463  if (out_maybe_overlap) {
1464  auto out_indices = flatten_full_indices->as_strided(out_sizes_, out_strides_, out_effective_offset);
1465  storage.index_add_(0, out_indices.reshape(-1), grad.reshape(-1));
1466  } else {
1467  // assume that new tensors have 0 storage offset
1468  storage.as_strided(out_sizes_, out_strides_, out_effective_offset).copy_(grad);
1469  }
1470 
1471  // Step (3): if input tensor has overlapping memory, divide scattered gradient
1472  // at storage[i] by the number of times i shows up in input geometry
1473  if (inp_maybe_overlap) {
1474  auto count = at::zeros_like(storage);
1475  auto inp_indices = flatten_full_indices->as_strided(inp_sizes_, inp_strides_, inp_effective_offset).reshape(-1);
1476  count.index_add_(0, inp_indices, at::ones({1}, grad.options()).expand_as(inp_indices));
1477  storage.div_(count); // this will give nan outside visible range
1478  }
1479  // Step (4): return as_strided view of the storage tensor with input geometry
1480  return storage.as_strided(inp_sizes, inp_strides, inp_effective_offset);
1481 }

1574 // https://j-towns.github.io/papers/svd-derivative.pdf
1575 //
1576 // This makes no assumption on the signs of sigma.
1577 Tensor svd_backward(const std::vector<torch::autograd::Variable> &grads, const Tensor& self,
1578  bool some, bool compute_uv, const Tensor& raw_u, const Tensor& sigma, const Tensor& raw_v) {
1579  AT_CHECK(compute_uv,
1580  "svd_backward: Setting compute_uv to false in torch.svd doesn't compute singular matrices, ",
1581  "and hence we cannot compute backward. Please use torch.svd(compute_uv=True)");
1582 
1583  auto m = self.size(0);
1584  auto n = self.size(1);
1585  auto k = sigma.size(0);
1586  auto gsigma = grads[1];
1587 
1588  auto u = raw_u;
1589  auto v = raw_v;
1590  auto gu = grads[0];
1591  auto gv = grads[2];
1592 
1593  if (!some) {
1594  // We ignore the free subspace here because possible base vectors cancel
1595  // each other, e.g., both -v and +v are valid base for a dimension.
1596  // Don't assume behavior of any particular implementation of svd.
1597  u = raw_u.narrow(1, 0, k);
1598  v = raw_v.narrow(1, 0, k);
1599  if (gu.defined()) {
1600  gu = gu.narrow(1, 0, k);
1601  }
1602  if (gv.defined()) {
1603  gv = gv.narrow(1, 0, k);
1604  }
1605  }
1606  auto vt = v.t();
1607 
1608  Tensor sigma_term;
1609  if (gsigma.defined()) {
1610  sigma_term = u.mm(gsigma.diag()).mm(vt);
1611  } else {
1612  sigma_term = at::zeros({1}, self.options()).expand_as(self);
1613  }
1614  // in case that there are no gu and gv, we can avoid the series of kernel
1615  // calls below
1616  if (!gv.defined() && !gu.defined()) {
1617  return sigma_term;
1618  }
1619 
1620  auto ut = u.t();
1621  auto im = at::eye(m, self.options());
1622  auto in = at::eye(n, self.options());
1623  auto sigma_mat = sigma.diag();
1624  auto sigma_mat_inv = sigma.pow(-1).diag();
1625  auto sigma_expanded_sq = sigma.pow(2).expand_as(sigma_mat);
1626  auto F = sigma_expanded_sq - sigma_expanded_sq.t();
1627  // The following two lines invert values of F, and fills the diagonal with 0s.
1628  // Notice that F currently has 0s on diagonal. So we fill diagonal with +inf
1629  // first to prevent nan from appearing in backward of this function.
1630  F.diagonal().fill_(INFINITY);
1631  F = F.pow(-1);
1632 
1633  Tensor u_term, v_term;
1634 
1635  if (gu.defined()) {
1636  u_term = u.mm(F.mul(ut.mm(gu) - gu.t().mm(u))).mm(sigma_mat);
1637  if (m > k) {
1638  u_term = u_term + (im - u.mm(ut)).mm(gu).mm(sigma_mat_inv);
1639  }
1640  u_term = u_term.mm(vt);
1641  } else {
1642  u_term = at::zeros({1}, self.options()).expand_as(self);
1643  }
1644 
1645  if (gv.defined()) {
1646  auto gvt = gv.t();
1647  v_term = sigma_mat.mm(F.mul(vt.mm(gv) - gvt.mm(v))).mm(vt);
1648  if (n > k) {
1649  v_term = v_term + sigma_mat_inv.mm(gvt.mm(in - v.mm(vt)));
1650  }
1651  v_term = u.mm(v_term);
1652  } else {
1653  v_term = at::zeros({1}, self.options()).expand_as(self);
1654  }
1655 
1656  return u_term + sigma_term + v_term;
1657 }

1771 // Generally speaking, fft's backward is ifft.
1772 Tensor fft_backward(const Tensor& self, const Tensor& grad, int64_t signal_ndim,
1773  bool complex_input, bool complex_output,
1774  bool inverse, IntArrayRef checked_signal_sizes,
1775  bool normalized, bool onesided,
1776  IntArrayRef output_sizes) {
1777  Tensor gI;
1778  if (!complex_input && complex_output) {
1779  // Forward is R2C
1780  // Do inverse C2C and project onto real plane because grad can be
1781  // asymmetrical so C2R can't be used.
1782  if (onesided) {
1783  // Forward is R2C (onesided)
1784  // Think of onesided R2C rfft as
1785  // 1. view as complex numbers (fill complex dim with zeros)
1786  // 2. C2C fft
1787  // 3. discard half of results
1788  // So backward is
1789  // 1. fill the other half with zeros (with `zero_grad_shape` below)
1790  // (C2C ifft only take twosided inputs so we need to fill here)
1791  // 2. inverse C2C ifft
1792  // 3. discard the complex dim
1793  int64_t zero_length = checked_signal_sizes[signal_ndim - 1] - grad.size(signal_ndim);
1794  auto complex_full_grad = grad;
1795  if (zero_length > 0) {
1796  std::vector<int64_t> zero_grad_shape(signal_ndim + 2);
1797  zero_grad_shape[0] = self.size(0);
1798  for (int64_t i = 1; i < signal_ndim; i++) {
1799  zero_grad_shape[i] = checked_signal_sizes[i - 1];
1800  }
1801  zero_grad_shape[signal_ndim] = zero_length;
1802  zero_grad_shape[signal_ndim + 1] = 2;
1803  complex_full_grad = at::cat({ grad, at::zeros(zero_grad_shape, grad.options()) }, signal_ndim);
1804  }
1805  gI = _fft_with_size(complex_full_grad, signal_ndim,
1806  /* complex_input */ true, /* complex_output */ true,
1807  !inverse, checked_signal_sizes, normalized,
1808  /* onesided */ false, complex_full_grad.sizes()).select(-1, 0);
1809  } else {
1810  gI = _fft_with_size(grad, signal_ndim, /* complex_input */ true,
1811  /* complex_output */ true, !inverse,
1812  checked_signal_sizes, normalized,
1813  /* onesided */ false, grad.sizes()).select(-1, 0);
1814  }
1815  } else if (complex_input && !complex_output && onesided) {
1816  // Forward is C2R (onesided)
1817  // Think of onesided C2R irfft as
1818  // 1. fill the other half by conjugate symmetry
1819  // 2. inverse C2C ifft
1820  // 3. discard the complex dimension
1821  // So backward is
1822  // 1. R2C rfft (essentially add dummy complex dimension, and dft)
1823  // 2. accumulate gradient by conjugate symmetry
1824  // since rfft results follow conjugate symmetry, we only need to
1825  // double some entries from onesided rfft results, i.e., the ones with
1826  // their reflected indices also landing out of the onesided range. So
1827  // consider the index of last dim:
1828  // i. idx = 0.
1829  // Reflected to (N - 0) % N = 0. Not doubled.
1830  // ii 0 < idx < floor(N/2) (last).
1831  // N > N - idx > ceil(N/2)
1832  // Reflected to ()
1833  // iii. idx = floor(N/2) = N/2 (last) when N even.
1834  // Reflected to (N - N/2) % N = N/2. Not doubled.
1835  // iv. idx = floor(N/2) = (N-1)/2 (last) when N odd.
1836  // Reflected to (N - (N-1)/2) % N = (N+1)/2. Doubled.
1837  // Therefore, needs to double
1838  // idx = 1, 2, ..., N/2 - 1 when N even
1839  // idx = 1, 2, ..., (N-1)/2 when N odd
1840  // that is
1841  // idx = 1, 2, ..., N - (floor(N/2) + 1)
1842  // = 1, 2, ..., N - onesided_length
1843  gI = _fft_with_size(grad, signal_ndim, /* complex_input */ false,
1844  /* complex_output */ true, /* inverse */ false,
1845  checked_signal_sizes, normalized, /* onesided */ true,
1846  self.sizes());
1847  int64_t double_length = checked_signal_sizes[signal_ndim - 1] - self.size(signal_ndim);
1848  if (double_length > 0) { // also covers case when signal size is zero
1849  gI.narrow(signal_ndim, 1, double_length).mul_(2);
1850  }
1851  } else {
1852  gI = _fft_with_size(grad, signal_ndim, complex_output, complex_input,
1853  !inverse, checked_signal_sizes, normalized, onesided,
1854  self.sizes());
1855  }
1856  if (normalized) {
1857  // If normalized, backward is exactly calling fft with inversed argument as
1858  // the forward because both are unitary.
1859  return gI;
1860  } else {
1861  // If not normalized, in backward, we need to upscale or downscale gI basing
1862  // on whether the forward is an inverse fft.
1863  auto signal_numel = std::accumulate(checked_signal_sizes.begin(),
1864  checked_signal_sizes.end(), 1, std::multiplies<int64_t>());
1865  if (!inverse) {
1866  return gI.mul_(static_cast<double>(signal_numel));
1867  } else {
1868  return gI.div_(static_cast<double>(signal_numel));
1869  }
1870  }
1871 }

1908 std::tuple<Tensor, Tensor, Tensor> batchnorm_double_backward(
1909  const Tensor & input,
1910  const Tensor & gamma,
1911  const Tensor & ggI,
1912  const Tensor & ggG,
1913  const Tensor & ggB,
1914  const Tensor & gO,
1915  const Tensor & running_mean,
1916  const Tensor & running_var,
1917  bool training,
1918  double eps,
1919  const Tensor & save_mean,
1920  const Tensor & save_invstd,
1921  std::array<bool,3> output_mask) {
1922 
1923  bool affine = gamma.defined();
1924  // TODO: Do we have a ScalarOrTensor type? Would such a thing exist?
1925  Tensor gamma_expanded;
1926  Tensor ggG_expanded, ggB_expanded;
1927  if (affine) {
1928  gamma_expanded = expand_as_dim1(gamma, input);
1929  if (ggG.defined()) {
1930  ggG_expanded = expand_as_dim1(ggG, input);
1931  }
1932  if (ggB.defined()) {
1933  ggB_expanded = expand_as_dim1(ggB, input);
1934  }
1935  } else {
1936  gamma_expanded = at::ones({}, input.options());
1937  }
1938 
1939  // define some terms we will reuse
1940  auto M = input.size(0);
1941  for (auto s : input.sizes().slice(2)) {
1942  M *= s;
1943  }
1944  // for half inputs, save_mean, save_invstd are float (ideally, we would cast
1945  // everything else, but not now)
1946  auto mu = unsqueeze_dim1(training ? save_mean.to(input.scalar_type()) : running_mean, input);
1947  auto input_sub_mu = input - mu;
1948  auto sigma2_eps_neg_1_2 = unsqueeze_dim1(
1949  training ? save_invstd.to(input.scalar_type())
1950  : running_var.add(Scalar(eps)).pow(-0.5),
1951  input);
1952  auto sigma2_eps_neg_1 = sigma2_eps_neg_1_2.pow(2);
1953  auto sigma2_eps_neg_3_2 = sigma2_eps_neg_1_2.pow(3);
1954 
1955  // calculate gI
1956  auto input_mu_sigma2_neg_3_2 = input_sub_mu * sigma2_eps_neg_3_2;
1957  auto gOinmu_sum = sum_exclude_dim1(gO * input_sub_mu);
1958  auto gO_sum = sum_exclude_dim1(gO);
1959 
1960  Tensor gI;
1961  if (ggI.defined() && training) {
1962  auto ggI_sum = sum_exclude_dim1(ggI);
1963  auto ggIinmu_sum = sum_exclude_dim1(ggI * input_sub_mu);
1964  auto all_sub = ((ggI_sum * gO_sum).div_(M)).sub_(sum_exclude_dim1(gO * ggI)).add_(
1965  (sigma2_eps_neg_1 * gOinmu_sum * ggIinmu_sum).mul_(3. / M));
1966  auto gI_0t = (input_mu_sigma2_neg_3_2 * all_sub).div_(M);
1967  auto gI_1t = (ggIinmu_sum * sigma2_eps_neg_3_2).div_(M) * (gO_sum.div(M) - gO);
1968  auto gI_2t = (gOinmu_sum * sigma2_eps_neg_3_2).div_(M) * (ggI_sum.div(M) - ggI);
1969  gI = gamma_expanded * (gI_0t.add_(gI_1t).add_(gI_2t));
1970  }
1971 
1972  // add contribution of gamma term to gI
1973  Tensor gI_G_term;
1974  if (affine && ggG.defined()) {
1975  if (training) {
1976  auto t0 = gO * sigma2_eps_neg_1_2;
1977  auto t1 = (sigma2_eps_neg_1_2 * gO_sum).div_(-M);
1978  auto t2 = (input_mu_sigma2_neg_3_2 * sum_exclude_dim1(gO * input_sub_mu)).div_(-M);
1979  gI_G_term = ggG_expanded * (t0.add_(t1).add_(t2));
1980  gI = gI.defined() ? gI.add_(gI_G_term) : gI_G_term;
1981  } else {
1982  gI_G_term = ggG_expanded * sigma2_eps_neg_1_2 * gO;
1983  gI = gI.defined() ? gI.add_(gI_G_term) : gI_G_term;
1984  }
1985  }
1986

1988  auto first_back_grad_input = [&](const Tensor& gO, const Tensor& gamma) -> Tensor {
1989  auto h0 = (gamma * sigma2_eps_neg_1_2).div_(M);
1990  auto h1 = (M * gO).sub_(sum_exclude_dim1(gO)).sub_(
1991  input_sub_mu.mul(sigma2_eps_neg_1) * sum_exclude_dim1(gO * input_sub_mu));
1992  return h0 * h1;
1993  };
1994 
1995  // calculate gG
1996  Tensor gG;
1997  if (affine && ggI.defined()) {
1998  if (training) {
1999  // gG is just the first backwards with the gamma term removed (then shaped properly)
2000  gG = ggI * first_back_grad_input(gO, at::ones({}, sigma2_eps_neg_1_2.options()));
2001  gG = sum_exclude_dim1(gG, false);
2002  } else {
2003  gG = sum_exclude_dim1(ggI * gO * sigma2_eps_neg_1_2, false);
2004  }
2005  }
2006 
2007  // calculate ggO
2008  Tensor ggO;
2009  // contribution of input term
2010  if (ggI.defined()) {
2011  if (training) {
2012  ggO = first_back_grad_input(ggI, gamma_expanded);
2013  } else {
2014  ggO = ggI * sigma2_eps_neg_1_2 * gamma_expanded;
2015  }
2016  }
2017  if (ggG.defined()) {
2018  auto ggO_G_term = ggG_expanded * input_sub_mu * sigma2_eps_neg_1_2;
2019  ggO = ggO.defined() ? ggO.add_(ggO_G_term) : ggO_G_term;
2020  }
2021  if (ggB.defined()) {
2022  auto ggO_B_term = ggB_expanded;
2023  ggO = ggO.defined() ? ggO.add_(ggO_B_term) : ggO_B_term;
2024  }

2027  if (output_mask[1] && !gG.defined()) {
2028  AT_ASSERTM(affine, "gamma should always be defined when it requires grad");
2029  gG = at::zeros_like(gamma);
2030  }
2031  if (output_mask[2] && !gI.defined()) gI = at::zeros_like(input);
2032
