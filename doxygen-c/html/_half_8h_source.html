81  /*
82  * Convert a 16-bit floating-point number in IEEE half-precision format, in bit representation, to
83  * a 32-bit floating-point number in IEEE single-precision format, in bit representation.
84  *
85  * @note The implementation doesn't use any floating-point operations.
86  */
87  inline uint32_t fp16_ieee_to_fp32_bits(uint16_t h) {
88  /*
89  * Extend the half-precision floating-point number to 32 bits and shift to the upper part of the 32-bit word:
90  * +---+-----+------------+-------------------+
91  * | S |EEEEE|MM MMMM MMMM|0000 0000 0000 0000|
92  * +---+-----+------------+-------------------+
93  * Bits 31 26-30 16-25 0-15
94  *
95  * S - sign bit, E - bits of the biased exponent, M - bits of the mantissa, 0 - zero bits.
96  */
97  const uint32_t w = (uint32_t) h << 16;
98  /*
99  * Extract the sign of the input number into the high bit of the 32-bit word:
100  *
101  * +---+----------------------------------+
102  * | S |0000000 00000000 00000000 00000000|
103  * +---+----------------------------------+
104  * Bits 31 0-31
105  */
106  const uint32_t sign = w & UINT32_C(0x80000000);
107  /*
108  * Extract mantissa and biased exponent of the input number into the bits 0-30 of the 32-bit word:
109  *
110  * +---+-----+------------+-------------------+
111  * | 0 |EEEEE|MM MMMM MMMM|0000 0000 0000 0000|
112  * +---+-----+------------+-------------------+
113  * Bits 30 27-31 17-26 0-16
114  */
115  const uint32_t nonsign = w & UINT32_C(0x7FFFFFFF);
116  /*
117  * Renorm shift is the number of bits to shift mantissa left to make the half-precision number normalized.
118  * If the initial number is normalized, some of its high 6 bits (sign == 0 and 5-bit exponent) equals one.
119  * In this case renorm_shift == 0. If the number is denormalize, renorm_shift > 0. Note that if we shift
120  * denormalized nonsign by renorm_shift, the unit bit of mantissa will shift into exponent, turning the
121  * biased exponent into 1, and making mantissa normalized (i.e. without leading 1).
122  */
123 #ifdef _MSC_VER
124  unsigned long nonsign_bsr;
125  _BitScanReverse(&nonsign_bsr, (unsigned long)nonsign);
126  uint32_t renorm_shift = (uint32_t)nonsign_bsr ^ 31;
127 #else
128  uint32_t renorm_shift = __builtin_clz(nonsign);
129 #endif
130  renorm_shift = renorm_shift > 5 ? renorm_shift - 5 : 0;
131  /*
132  * Iff half-precision number has exponent of 15, the addition overflows
133  * it into bit 31, and the subsequent shift turns the high 9 bits
134  * into 1. Thus inf_nan_mask == 0x7F800000 if the half-precision number
135  * had exponent of 15 (i.e. was NaN or infinity) 0x00000000 otherwise
136  */
137  const int32_t inf_nan_mask =
138  ((int32_t)(nonsign + 0x04000000) >> 8) & INT32_C(0x7F800000);
139  /*
140  * Iff nonsign is 0, it overflows into 0xFFFFFFFF, turning bit 31
141  * into 1. Otherwise, bit 31 remains 0. The signed shift right by 31
142  * broadcasts bit 31 into all bits of the zero_mask. Thus zero_mask ==
143  * 0xFFFFFFFF if the half-precision number was zero (+0.0h or -0.0h)
144  * 0x00000000 otherwise
145  */
146  const int32_t zero_mask = (int32_t)(nonsign - 1) >> 31;
147  /*
148  * 1. Shift nonsign left by renorm_shift to normalize it (if the input
149  * was denormal)
150  * 2. Shift nonsign right by 3 so the exponent (5 bits originally)
151  * becomes an 8-bit field and 10-bit mantissa shifts into the 10 high
152  * bits of the 23-bit mantissa of IEEE single-precision number.
153  * 3. Add 0x70 to the exponent (starting at bit 23) to compensate the
154  * different in exponent bias (0x7F for single-precision number less 0xF
155  * for half-precision number).
156  * 4. Subtract renorm_shift from the exponent (starting at bit 23) to
157  * account for renormalization. As renorm_shift is less than 0x70, this
158  * can be combined with step 3.
159  * 5. Binary OR with inf_nan_mask to turn the exponent into 0xFF if the
160  * input was NaN or infinity.
161  * 6. Binary ANDNOT with zero_mask to turn the mantissa and exponent
162  * into zero if the input was zero.
163  * 7. Combine with the sign of the input number.
164  */
165  return sign |
166  ((((nonsign << renorm_shift >> 3) + ((0x70 - renorm_shift) << 23)) |
167  inf_nan_mask) &
168  ~zero_mask);
169  }

171  /*
172  * Convert a 16-bit floating-point number in IEEE half-precision format, in bit representation, to
173  * a 32-bit floating-point number in IEEE single-precision format.
174  *
175  * @note The implementation relies on IEEE-like (no assumption about rounding mode and no operations on denormals)
176  * floating-point operations and bitcasts between integer and floating-point variables.
177  */
178  inline float fp16_ieee_to_fp32_value(uint16_t h) {
179  /*
180  * Extend the half-precision floating-point number to 32 bits and shift to the upper part of the 32-bit word:
181  * +---+-----+------------+-------------------+
182  * | S |EEEEE|MM MMMM MMMM|0000 0000 0000 0000|
183  * +---+-----+------------+-------------------+
184  * Bits 31 26-30 16-25 0-15
185  *
186  * S - sign bit, E - bits of the biased exponent, M - bits of the mantissa, 0 - zero bits.
187  */
188  const uint32_t w = (uint32_t) h << 16;
189  /*
190  * Extract the sign of the input number into the high bit of the 32-bit word:
191  *
192  * +---+----------------------------------+
193  * | S |0000000 00000000 00000000 00000000|
194  * +---+----------------------------------+
195  * Bits 31 0-31
196  */
197  const uint32_t sign = w & UINT32_C(0x80000000);
198  /*
199  * Extract mantissa and biased exponent of the input number into the high bits of the 32-bit word:
200  *
201  * +-----+------------+---------------------+
202  * |EEEEE|MM MMMM MMMM|0 0000 0000 0000 0000|
203  * +-----+------------+---------------------+
204  * Bits 27-31 17-26 0-16
205  */
206  const uint32_t two_w = w + w;
207 
208  /*
209  * Shift mantissa and exponent into bits 23-28 and bits 13-22 so they become mantissa and exponent
210  * of a single-precision floating-point number:
211  *
212  * S|Exponent | Mantissa
213  * +-+---+-----+------------+----------------+
214  * |0|000|EEEEE|MM MMMM MMMM|0 0000 0000 0000|
215  * +-+---+-----+------------+----------------+
216  * Bits | 23-31 | 0-22
217  *
218  * Next, there are some adjustments to the exponent:
219  * - The exponent needs to be corrected by the difference in exponent bias between single-precision and half-precision
220  * formats (0x7F - 0xF = 0x70)
221  * - Inf and NaN values in the inputs should become Inf and NaN values after conversion to the single-precision number.
222  * Therefore, if the biased exponent of the half-precision input was 0x1F (max possible value), the biased exponent
223  * of the single-precision output must be 0xFF (max possible value). We do this correction in two steps:
224  * - First, we adjust the exponent by (0xFF - 0x1F) = 0xE0 (see exp_offset below) rather than by 0x70 suggested
225  * by the difference in the exponent bias (see above).
226  * - Then we multiply the single-precision result of exponent adjustment by 2**(-112) to reverse the effect of
227  * exponent adjustment by 0xE0 less the necessary exponent adjustment by 0x70 due to difference in exponent bias.
228  * The floating-point multiplication hardware would ensure than Inf and NaN would retain their value on at least
229  * partially IEEE754-compliant implementations.
230  *
231  * Note that the above operations do not handle denormal inputs (where biased exponent == 0). However, they also do not
232  * operate on denormal inputs, and do not produce denormal results.
233  */
234  const uint32_t exp_offset = UINT32_C(0xE0) << 23;
235  // const float exp_scale = 0x1.0p-112f;
236  uint32_t scale_bits = (uint32_t) 15 << 23;
237  float exp_scale_val;
238  std::memcpy(&exp_scale_val, &scale_bits, sizeof(exp_scale_val));
239  const float exp_scale = exp_scale_val;
240  const float normalized_value = fp32_from_bits((two_w >> 4) + exp_offset) * exp_scale;
241 
242  /*
243  * Convert denormalized half-precision inputs into single-precision results (always normalized).
244  * Zero inputs are also handled here.
245  *
246  * In a denormalized number the biased exponent is zero, and mantissa has on-zero bits.
247  * First, we shift mantissa into bits 0-9 of the 32-bit word.
248  *
249  * zeros | mantissa
250  * +---------------------------+------------+
251  * |0000 0000 0000 0000 0000 00|MM MMMM MMMM|
252  * +---------------------------+------------+
253  * Bits 10-31 0-9
254  *
255  * Now, remember that denormalized half-precision numbers are represented as:
256  * FP16 = mantissa * 2**(-24).
257  * The trick is to construct a normalized single-precision number with the same mantissa and thehalf-precision input
258  * and with an exponent which would scale the corresponding mantissa bits to 2**(-24).
259  * A normalized single-precision floating-point number is represented as:
260  * FP32 = (1 + mantissa * 2**(-23)) * 2**(exponent - 127)
261  * Therefore, when the biased exponent is 126, a unit change in the mantissa of the input denormalized half-precision
262  * number causes a change of the constructud single-precision number by 2**(-24), i.e. the same ammount.
263  *
264  * The last step is to adjust the bias of the constructed single-precision number. When the input half-precision number
265  * is zero, the constructed single-precision number has the value of
266  * FP32 = 1 * 2**(126 - 127) = 2**(-1) = 0.5
267  * Therefore, we need to subtract 0.5 from the constructed single-precision number to get the numerical equivalent of
268  * the input half-precision number.
269  */
270  const uint32_t magic_mask = UINT32_C(126) << 23;
271  const float magic_bias = 0.5f;
272  const float denormalized_value = fp32_from_bits((two_w >> 17) | magic_mask) - magic_bias;
273 
274  /*
275  * - Choose either results of conversion of input as a normalized number, or as a denormalized number, depending on the
276  * input exponent. The variable two_w contains input exponent in bits 27-31, therefore if its smaller than 2**27, the
277  * input is either a denormal number, or zero.
278  * - Combine the result of conversion of exponent and mantissa with the sign of the input number.
279  */
280  const uint32_t denormalized_cutoff = UINT32_C(1) << 27;
281  const uint32_t result = sign |
282  (two_w < denormalized_cutoff ? fp32_to_bits(denormalized_value) : fp32_to_bits(normalized_value));
283  return fp32_from_bits(result);
284  }

286  /*
287  * Convert a 32-bit floating-point number in IEEE single-precision format to a 16-bit floating-point number in
288  * IEEE half-precision format, in bit representation.
289  *
290  * @note The implementation relies on IEEE-like (no assumption about rounding mode and no operations on denormals)
291  * floating-point operations and bitcasts between integer and floating-point variables.
292  */
293  inline uint16_t fp16_ieee_from_fp32_value(float f) {
294  // const float scale_to_inf = 0x1.0p+112f;
295  // const float scale_to_zero = 0x1.0p-110f;
296  uint32_t scale_to_inf_bits = (uint32_t) 239 << 23;
297  uint32_t scale_to_zero_bits = (uint32_t) 17 << 23;
298  float scale_to_inf_val, scale_to_zero_val;
299  std::memcpy(&scale_to_inf_val, &scale_to_inf_bits, sizeof(scale_to_inf_val));
300  std::memcpy(&scale_to_zero_val, &scale_to_zero_bits, sizeof(scale_to_zero_val));
301  const float scale_to_inf = scale_to_inf_val;
302  const float scale_to_zero = scale_to_zero_val;
303 
304  float base = (fabsf(f) * scale_to_inf) * scale_to_zero;
305 
306  const uint32_t w = fp32_to_bits(f);
307  const uint32_t shl1_w = w + w;
308  const uint32_t sign = w & UINT32_C(0x80000000);
309  uint32_t bias = shl1_w & UINT32_C(0xFF000000);
310  if (bias < UINT32_C(0x71000000)) {
311  bias = UINT32_C(0x71000000);
312  }
313 
314  base = fp32_from_bits((bias >> 1) + UINT32_C(0x07800000)) + base;
315  const uint32_t bits = fp32_to_bits(base);
316  const uint32_t exp_bits = (bits >> 13) & UINT32_C(0x00007C00);
317  const uint32_t mantissa_bits = bits & UINT32_C(0x00000FFF);
318  const uint32_t nonsign = exp_bits + mantissa_bits;
319  return (sign >> 16) | (shl1_w > UINT32_C(0xFF000000) ? UINT16_C(0x7E00) : nonsign);
320  }

430 // skip isnan and isinf check for integral types
431 template <typename To, typename From>
432 typename std::enable_if<std::is_integral<From>::value, bool>::type overflows(
433  From f) {
434  using limit = std::numeric_limits<typename scalar_value_type<To>::type>;
435  if (!limit::is_signed && std::numeric_limits<From>::is_signed) {
436  // allow for negative numbers to wrap using two's complement arithmetic.
437  // For example, with uint8, this allows for `a - b` to be treated as
438  // `a + 255 * b`.
439  return f > limit::max() ||
440  (f < 0 && -static_cast<uint64_t>(f) > limit::max());
441  } else {
442  return f < limit::lowest() || f > limit::max();
443  }
444 }

463 template <typename To, typename From>
464 typename std::enable_if<is_complex_t<From>::value, bool>::type overflows(
465  From f) {
466  // casts from complex to real are considered to overflow if the
467  // imaginary component is non-zero
468  if (!is_complex_t<To>::value && f.imag() != 0) {
469  return true;
470  }
471  // Check for overflow componentwise
472  // (Technically, the imag overflow check is guaranteed to be false
473  // when !is_complex_t<To>, but any optimizer worth its salt will be
474  // able to figure it out.)
475  return overflows<
476  typename scalar_value_type<To>::type,
477  typename From::value_type>(f.real()) ||
478  overflows<
479  typename scalar_value_type<To>::type,
480  typename From::value_type>(f.imag());
481 }
