546 inline Variable make_variable_view(
547  Variable base,
548  at::Tensor data,
549  bool is_differentiable = true,
550  bool allow_tensor_metadata_change = true,
551  Edge gradient_edge = Edge()) {
552  if (data.defined()) {
553  if (is_differentiable) {
555  auto data_impl_copy = data.getIntrusivePtr()->shallow_copy_and_detach();
556  data_impl_copy->set_allow_tensor_metadata_change(allow_tensor_metadata_change);
557  auto data_copy = at::Tensor(data_impl_copy);
558  auto diff_view_meta = c10::guts::make_unique<Variable::DifferentiableViewMeta>();
559  return Variable(c10::make_intrusive<Variable::DifferentiableViewImpl>(
560  std::move(base), std::move(data_copy), std::move(gradient_edge), std::move(diff_view_meta)));
561  } else {
563  auto data_impl_copy = data.getIntrusivePtr()->shallow_copy_and_detach();
564  data_impl_copy->set_allow_tensor_metadata_change(allow_tensor_metadata_change);
565  auto data_copy = at::Tensor(data_impl_copy);
566  auto autograd_meta = c10::guts::make_unique<Variable::AutogradMeta>();
567  auto var = Variable(c10::make_intrusive<Variable::Impl>(
568  std::move(data_copy), std::move(autograd_meta), false, std::move(gradient_edge)));
569  var.set_version_counter(base.version_counter());
570  return var;
571  }
572  }
573  return Variable();
574 }
