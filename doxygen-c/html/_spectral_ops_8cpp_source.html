19 // This is a pass-through wrapper function that does the size check and
20 // inferences. The actual forward implementation function is called
21 // at::_fft_with_size which dispatches to _fft_cufft (CUDA) or _fft_mkl (CPU).
22 static inline Tensor _fft(const Tensor &self, const int64_t signal_ndim,
23  const bool complex_input, const bool complex_output,
24  const bool inverse, IntArrayRef signal_sizes, const bool normalized,
25  const bool onesided) {
26 
27  AT_CHECK(signal_ndim >= 1 && signal_ndim <= 3,
28  "Expected signal_ndim to be 1, 2, or 3, but got signal_ndim=",
29  signal_ndim);
30  AT_CHECK(at::isFloatingType(self.scalar_type()),
31  "Expected an input tensor of floating types, but got input=",
32  self.type(), self.sizes());
33 
34  auto signal_tensor_ndim = signal_ndim + static_cast<int64_t>(complex_input); // add complex dim
35  if (self.dim() < signal_tensor_ndim) {
36  std::ostringstream ss;
37  ss << "Given signal_ndim=" << signal_ndim << ", expected an input tensor "
38  << "of at least " << signal_tensor_ndim << "D";
39  if (complex_input) {
40  ss << " (complex input adds an extra dimension)";
41  }
42  ss << ", but got input=" << self.type() << self.sizes();
43  AT_ERROR(ss.str());
44  }
45 
46  auto self_shape = self.sizes();
47  auto batch_ndim = self.dim() - signal_tensor_ndim;
48 
49  Tensor input = self;
50  // flatten the batch dims
51  if (batch_ndim == 0) {
52  // slightly faster path for non-batch mode
53  input = input.unsqueeze(0);
54  } else if (batch_ndim > 1) {
55  std::vector<int64_t> flatten_input_shape(signal_tensor_ndim + 1);
56  std::copy(self_shape.begin() + batch_ndim, self_shape.end(), flatten_input_shape.begin() + 1);
57  flatten_input_shape[0] = -1;
58  input = input.reshape(flatten_input_shape);
59 
60  }
61 
62  // now we assume that input is batched as [ B x signal_dims... ]
63 
64  if (complex_input) {
65  AT_CHECK(input.size(signal_ndim + 1) == 2,
66  "Expected an input tensor with a last dimension of size 2 "
67  "representing real + imaginary components, but got input ",
68  self.type(), self.sizes());
69  }
70 
71  // build signal_sizes and output_size
72  AT_CHECK(signal_sizes.size() == 0 || static_cast<int64_t>(signal_sizes.size()) == signal_ndim,
73  "Expected signal_sizes to be empty (default) or of signal_ndim=",
74  signal_ndim, "D, but got signal_sizes=", signal_sizes);
75  std::vector<int64_t> output_sizes(signal_ndim + 1 + static_cast<int64_t>(complex_output));
76  output_sizes[0] = input.size(0); // batch size
77  std::vector<int64_t> checked_signal_sizes(signal_ndim);
78  for (int64_t i = 0; i < signal_ndim; i++) {
79  int64_t input_size = input.size(i + 1);
80  if (i == signal_ndim - 1 && onesided && complex_input && !complex_output) {
81  // If last dim and complex-to-real onesided, input is only half of
82  // signal, and we need to infer basing on signal_sizes, if given
83  // See native/SpectralOpsUtils.h for detailed description.
84  int64_t inferred_size;
85  if (signal_sizes.size() > 0) {
86  inferred_size = infer_ft_complex_to_real_onesided_size(input_size, signal_sizes[i]);
87  } else {
88  inferred_size = infer_ft_complex_to_real_onesided_size(input_size);
89  }
90  checked_signal_sizes[i] = inferred_size;
91  output_sizes[i + 1] = inferred_size;
92  } else {
93  if (i == signal_ndim - 1 && onesided && !complex_input && complex_output) {
94  // if last dim and real-to-complex onesided, output should be only
95  // half of the signal, and we need to infer using input_size
96  output_sizes[i + 1] = infer_ft_real_to_complex_onesided_size(input_size);
97  } else {
98  output_sizes[i + 1] = input_size;
99  }
100  checked_signal_sizes[i] = input_size;
101  AT_CHECK(signal_sizes.size() == 0 || signal_sizes[i] == checked_signal_sizes[i],
102  "Expected given signal_sizes=", signal_sizes," to have same "
103  "shape with input at signal dimension ", i, ", but got "
104  "signal_sizes=", signal_sizes, " and input=", self.type(),
105  self.sizes());
106  }
107  }
108  if (complex_output) {
109  output_sizes[signal_ndim + 1] = 2;
110  }
111 
112  Tensor output = at::_fft_with_size(input, signal_ndim, complex_input,
113  complex_output, inverse,
114  checked_signal_sizes, normalized, onesided,
115  output_sizes);
116 
117  // unflatten the batch dims
118  if (batch_ndim == 0) {
119  // slightly faster path for non-batch mode
120  output = output.squeeze(0);
121  } else if (batch_ndim > 1) {
122  auto output_ndim = self.dim() + static_cast<int64_t>(complex_output) - static_cast<int64_t>(complex_input);
123  std::vector<int64_t> unflatten_output_shape(output_ndim);
124  std::copy(self_shape.begin(), self_shape.begin() + batch_ndim, unflatten_output_shape.begin());
125  std::copy(output_sizes.begin() + 1, output_sizes.end(), unflatten_output_shape.begin() + batch_ndim);
126  output = output.reshape(unflatten_output_shape);
127  }
128  return output;
129 }

176 Tensor stft(const Tensor& self, const int64_t n_fft, const optional<int64_t> hop_lengthOpt,
177  const optional<int64_t> win_lengthOpt, const Tensor& window,
178  const bool normalized, const bool onesided) {
179  #define REPR(SS) \
180  SS << "stft(" << self.type() << self.sizes() << ", n_fft=" << n_fft \
181  << ", hop_length=" << hop_length << ", win_length=" << win_length \
182  << ", window="; \
183  if (window.defined()) { \
184  SS << window.type() << "{" << window.sizes() << "}"; \
185  } else { \
186  SS << "None"; \
187  } \
188  SS << ", normalized=" << normalized << ", onesided=" << onesided << ")"
189 
190  // default_init hop_length and win_length
191  auto hop_length = hop_lengthOpt.value_or(n_fft >> 2);
192  auto win_length = win_lengthOpt.value_or(n_fft);
193 
194  if (!at::isFloatingType(self.scalar_type()) || self.dim() > 2 || self.dim() < 1) {
195  std::ostringstream ss;
196  REPR(ss) << ": expected a 1D or 2D tensor of floating types";
197  AT_ERROR(ss.str());
198  }
199  Tensor input = self;
200  if (self.dim() == 1) {
201  input = input.unsqueeze(0);
202  }
203  int64_t batch = input.size(0);
204  int64_t len = input.size(1);
205  if (n_fft <= 0 || n_fft > len) {
206  std::ostringstream ss;
207  REPR(ss) << ": expected 0 < n_fft < " << len
208  << ", but got n_fft=" << win_length;
209  AT_ERROR(ss.str());
210  }
211  if (hop_length <= 0) {
212  std::ostringstream ss;
213  REPR(ss) << ": expected hop_length > 0, but got hop_length=" << hop_length;
214  AT_ERROR(ss.str());
215  }
216  if (win_length <= 0 || win_length > n_fft) {
217  std::ostringstream ss;
218  REPR(ss) << ": expected 0 < win_length <= n_fft, but got win_length="
219  << win_length;
220  AT_ERROR(ss.str());
221  }
222  if (window.defined() && (window.dim() != 1 || window.size(0) != win_length)) {
223  std::ostringstream ss;
224  REPR(ss) << ": expected a 1D window tensor of size equal to win_length="
225  << win_length << ", but got window with size " << window.sizes();
226  AT_ERROR(ss.str());
227  }
228  #undef REPR
229  auto window_ = window;
230  if (win_length < n_fft) {
231  // pad center
232  window_ = at::zeros({n_fft}, self.options());
233  auto left = (n_fft - win_length) / 2;
234  if (window.defined()) {
235  window_.narrow(0, left, win_length).copy_(window);
236  } else {
237  window_.narrow(0, left, win_length).fill_(1);
238  }
239  }
240  int64_t n_frames = 1 + (len - n_fft) / hop_length;
241  // time2col
242  input = input.as_strided(
243  {batch, n_frames, n_fft},
244  {input.stride(0), hop_length * input.stride(1), input.stride(1)}
245  );
246  if (window_.defined()) {
247  input = input.mul(window_);
248  }
249  // rfft and transpose to get (batch x fft_size x num_frames)
250  auto out = input.rfft(1, normalized, onesided).transpose_(1, 2);
251  if (self.dim() == 1) {
252  return out.squeeze_(0);
253  } else {
254  return out;
255  }
256 }
