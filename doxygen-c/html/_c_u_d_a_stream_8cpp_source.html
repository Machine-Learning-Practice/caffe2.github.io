76 // Note [StreamId assignment]
77 // ~~~~~~~~~~~~~~~~~~~~~~~~~~
78 // How do we assign stream IDs?
79 //
80 // -- 25 bits -- -- 2 bits -- -- 5 bits -----
81 // zeros StreamIdType stream id index
82 //
83 // Where StreamIdType:
84 // 00 = default stream
85 // 01 = low priority stream
86 // 10 = high priority stream
87 //
88 // This is not really for efficiency; it's just easier to write the code
89 // to extract the index if we do this with bitmasks :)
90 //
91 // We are obligated to treat the stream ID 0 as the default stream, per the
92 // invariant specified in c10::Stream. However, all other numbers are entirely
93 // an internal implementation detail, we reserve the right to renumber streams
94 // however we like.
95 //
96 // Note that it is really important that the MSB is zero; StreamId is a
97 // *signed* integer, and unsigned to signed conversion outside of the
98 // bounds of signed integer representation is undefined behavior. You
99 // could work around this with something like
100 // https://stackoverflow.com/questions/13150449/efficient-unsigned-to-signed-cast-avoiding-implementation-defined-behavior
101 // but it seems a bit overkill for this.
102 
103 enum class StreamIdType : uint8_t {
104  DEFAULT = 0x0,
105  LOW = 0x1,
106  HIGH = 0x2,
107 };

150 static StreamId CUDAStream_getStreamId(const LeakyStreamInternals* ptr) {
151  // Hypothetically, we could store the stream ID in the stream. But that
152  // introduces a degree of freedom which could lead to bugs (where we
153  // misnumber streams in the pool, or overwrite the number). Better
154  // to just compute it based on the metric that actually matters,
155  // which is how we map IDs back into the vectors.
156 
157  DeviceIndex device_index = ptr->device_index;
158 
159  // Check if it's the default stream
160  if (ptr == &default_streams[device_index]) {
161  return makeStreamId(StreamIdType::DEFAULT, 0);
162  }
163 
164  // Check if it's a low priority stream
165  // NB: Because ptr may not necessarily lie within the array, we must use
166  // std::less and similar templates to avoid UB that arises when
167  // doing an operator< comparison.
168  if (pointer_within<LeakyStreamInternals>(
169  ptr, low_priority_streams[device_index])) {
170  return makeStreamId(
171  StreamIdType::LOW, ptr - low_priority_streams[device_index].data());
172  }
173 
174  // Check if it's a high priority stream
175  if (pointer_within<LeakyStreamInternals>(
176  ptr, high_priority_streams[device_index])) {
177  return makeStreamId(
178  StreamIdType::HIGH, ptr - high_priority_streams[device_index].data());
179  }
180 
181  AT_ASSERTM(
182  0,
183  "Could not compute stream ID for ",
184  ptr,
185  " on device ",
186  device_index,
187  " (something has gone horribly wrong!)");
188 }

193 // Populates global values and creates a default stream for each device.
194 // Note: the default stream on each device is signified by a nullptr,
195 // and so is not created as usual.
196 // In particular, we don't need to switch devices when creating the
197 // streams.
198 // Warning: this function must only be called once!
199 static void initGlobalStreamState() {
200  num_gpus = device_count();
201  // Check if the number of GPUs matches the expected compile-time max number
202  // of GPUs.
203  AT_ASSERTM(
204  num_gpus <= C10_COMPILE_TIME_MAX_GPUS,
205  "Number of CUDA devices on the machine is larger than the compiled "
206  "max number of gpus expected (",
207  C10_COMPILE_TIME_MAX_GPUS,
208  "). Increase that and recompile.");
209 
210  // Initializes default streams
211  for (auto i = decltype(num_gpus){0}; i < num_gpus; ++i) {
212  default_streams[i].device_index = i;
213  low_priority_counters[i] = 0;
214  high_priority_counters[i] = 0;
215  }
216 }

218 // Creates the low and high priority stream pools for the specified device
219 // Warning: only call once per device!
220 static void initDeviceStreamState(DeviceIndex device_index) {
221  // Switches to the requested device so streams are properly associated
222  // with it.
223  CUDAGuard device_guard{device_index};
224 
225  for (auto i = decltype(kStreamsPerPool){0}; i < kStreamsPerPool; ++i) {
226  auto& lowpri_stream = low_priority_streams[device_index][i];
227  auto& hipri_stream = high_priority_streams[device_index][i];
228 
229  lowpri_stream.device_index = device_index;
230  hipri_stream.device_index = device_index;
231 
232 #ifndef __HIP_PLATFORM_HCC__
233  C10_CUDA_CHECK(cudaStreamCreateWithPriority(
234  &lowpri_stream.stream, kDefaultFlags, kLowPriority));
235  C10_CUDA_CHECK(cudaStreamCreateWithPriority(
236  &hipri_stream.stream, kDefaultFlags, kHighPriority));
237 #else
238  C10_CUDA_CHECK(
239  cudaStreamCreateWithFlags(&lowpri_stream.stream, kDefaultFlags));
240  C10_CUDA_CHECK(
241  cudaStreamCreateWithFlags(&hipri_stream.stream, kDefaultFlags));
242 #endif // __HIP_PLATFORM_HCC__
243  }

246 // Init front-end to ensure initialization only occurs once
247 static void initCUDAStreamsOnce() {
248  // Inits default streams (once, globally)
249  std::call_once(init_flag, initGlobalStreamState);
250 
251  if (current_streams) {
252  return;
253  }
254 
255  // Inits current streams (thread local) to default streams
256  current_streams =
257  (LeakyStreamInternals**)malloc(num_gpus * sizeof(LeakyStreamInternals*));
258  for (auto i = decltype(num_gpus){0}; i < num_gpus; ++i) {
259  current_streams[i] = &default_streams[i];
260  }

276 LeakyStreamInternals* CUDAStream_internals(CUDAStream s) {
277  c10::DeviceIndex device_index = s.device_index();
278  StreamIdType st = streamIdType(s.unwrap().id());
279  size_t si = streamIdIndex(s.unwrap().id());
280  switch (st) {
281  case StreamIdType::DEFAULT:
282  AT_ASSERTM(
283  si == 0,
284  "Unrecognized stream ",
285  s.unwrap(),
286  " (I think this should be the default stream, but I got a non-zero index ",
287  si,
288  ").",
289  " Did you manufacture the StreamId yourself? Don't do that; use the",
290  " official API like c10::cuda::getStreamFromPool() to get a new stream.");
291  return &default_streams[device_index];
292  case StreamIdType::LOW:
293  return &low_priority_streams[device_index][si];
294  case StreamIdType::HIGH:
295  return &high_priority_streams[device_index][si];
296  default:
297  AT_ASSERTM(
298  0,
299  "Unrecognized stream ",
300  s.unwrap(),
301  " (I didn't recognize the stream type, ",
302  st,
303  ")");
304  }
305 }

324 // Returns a stream from the requested pool
325 // Note: when called the first time on a device, this will create the
326 // stream pools for that device.
327 CUDAStream getStreamFromPool(
328  const bool isHighPriority,
329  DeviceIndex device_index) {
330  initCUDAStreamsOnce();
331  if (device_index == -1)
332  device_index = current_device();
333  check_gpu(device_index);
334 
335  // Initializes the stream pools (once)
336  std::call_once(
337  device_flags[device_index], initDeviceStreamState, device_index);
338 
339  if (isHighPriority) {
340  const auto idx = get_idx(high_priority_counters[device_index]);
341  return CUDAStream_fromInternals(&high_priority_streams[device_index][idx]);
342  }
343 
344  const auto idx = get_idx(low_priority_counters[device_index]);
345  return CUDAStream_fromInternals(&low_priority_streams[device_index][idx]);
346 }
