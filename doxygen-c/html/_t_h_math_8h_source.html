88 static inline double TH_erfinv(double y) {
89 /* Function to calculate inverse error function. Rational approximation
90 is used to generate an initial approximation, which is then improved to
91 full accuracy by two steps of Newton's method. Code is a direct
92 translation of the erfinv m file in matlab version 2.0.
93 Author: Gary L. Pavlis, Indiana University
94 Date: February 1996
95 */
96  double x,z,num,dem; /*working variables */
97  /* coefficients in rational expansion */
98  double a[4]={ 0.886226899, -1.645349621, 0.914624893, -0.140543331};
99  double b[4]={-2.118377725, 1.442710462, -0.329097515, 0.012229801};
100  double c[4]={-1.970840454, -1.624906493, 3.429567803, 1.641345311};
101  double d[2]={ 3.543889200, 1.637067800};
102  if(fabs(y) > 1.0) return (atof("NaN")); /* This needs IEEE constant*/
103  if(fabs(y) == 1.0) return((copysign(1.0,y))*atof("INFINITY"));
104  if(fabs(y) <= CENTRAL_RANGE){
105  z = y*y;
106  num = (((a[3]*z + a[2])*z + a[1])*z + a[0]);
107  dem = ((((b[3]*z + b[2])*z + b[1])*z +b[0])*z + 1.0);
108  x = y*num/dem;
109  }
110  else{
111  z = sqrt(-log((1.0-fabs(y))/2.0));
112  num = ((c[3]*z + c[2])*z + c[1])*z + c[0];
113  dem = (d[1]*z + d[0])*z + 1.0;
114  x = (copysign(1.0,y))*num/dem;
115  }
116  /* Two steps of Newton-Raphson correction */
117  x = x - (erf(x) - y)/( (2.0/sqrt(M_PI))*exp(-x*x));
118  x = x - (erf(x) - y)/( (2.0/sqrt(M_PI))*exp(-x*x));
119 
120  return(x);
121 }
122 #undef CENTRAL_RANGE

243 static inline double TH_trigamma(double x) {
244  double sign = +1;
245  double result = 0;
246  if (x < 0.5) {
247  sign = -1;
248  const double sin_pi_x = sin(M_PI * x);
249  result -= (M_PI * M_PI) / (sin_pi_x * sin_pi_x);
250  x = 1 - x;
251  }
252  for (int i = 0; i < 6; ++i) {
253  result += 1 / (x * x);
254  x += 1;
255  }
256  const double ixx = 1 / (x*x);
257  result += (1 + 1 / (2*x) + ixx * (1./6 - ixx * (1./30 - ixx * (1./42)))) / x;
258  return sign * result;
259 }
