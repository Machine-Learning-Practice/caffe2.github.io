33 #define GENERATE_ALL_TYPES(type, func, args...) \

44  case ::at::ScalarType::Char: \

163 template <typename T>
164 void DataChannelGloo::allGatherT(
165  std::vector<at::Tensor>& output,
166  at::Tensor& input,
167  THDGroup group_id) {
168  auto input_device = getDeviceType(input);
169  for (auto& out : output) {
170  if (input_device != getDeviceType(out)) {
171  throw std::runtime_error(
172  "allGather got input and output on different devices");
173  }
174  }
175  uint64_t tensor_bytes = input.element_size() * input.numel();
176  uint64_t all_tensor_bytes = tensor_bytes * output.size();
177  auto ret = _cache->getAlgorithm<CollectiveType::ALL_GATHER, T>(
178  group_id,
179  _groups.at(group_id),
180  input_device,
181  tensor_bytes,
182  all_tensor_bytes,
183  input.numel());
184 
185  {
186  std::lock_guard<std::mutex> lock(*GlooCache::mutex(ret));
187  std::memcpy(
188  GlooCache::input_buffer(ret).get(), input.data_ptr(), tensor_bytes);
189  GlooCache::algorithm(ret)->run();
190  for (size_t i = 0; i < output.size(); i++) {
191  std::memcpy(
192  output.at(i).data_ptr(),
193  GlooCache::output_buffer(ret).get() + (i * tensor_bytes),
194  tensor_bytes);
195  }
196  }
197 }
