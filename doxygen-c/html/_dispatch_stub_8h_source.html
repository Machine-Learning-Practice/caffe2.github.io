8 // Implements instruction set specific function dispatch.
9 //
10 // Kernels that may make use of specialized instruction sets (e.g. AVX) are
11 // compiled multiple times with different compiler flags (e.g. -mavx). A
12 // DispatchStub contains a table of function pointers for a kernel. At runtime,
13 // the fastest available kernel is chosen based on the features reported by
14 // cpuinfo.
15 //
16 // Example:
17 //
18 // In native/MyKernel.h:
19 // using fn_type = void(*)(const Tensor& x);
20 // DECLARE_DISPATCH(fn_type, stub);
21 //
22 // In native/MyKernel.cpp
23 // DEFINE_DISPATCH(stub);
24 //
25 // In native/cpu/MyKernel.cpp:
26 // namespace {
27 // // use anonymous namespace so that different cpu versions won't conflict
28 // void kernel(const Tensor& x) { ... }
29 // }
30 // REGISTER_DISPATCH(stub, &kernel);
31 //
32 // To call:
33 // stub(kCPU, tensor);
34 //
35 // TODO: CPU instruction set selection should be folded into whatever
36 // the main dispatch mechanism is.
37

58 template <typename rT, typename T, typename... Args>
59 struct CAFFE2_API DispatchStub<rT (*)(Args...), T> {
60  using FnPtr = rT (*) (Args...);
61 
62  template <typename... ArgTypes>
63  rT operator()(DeviceType device_type, ArgTypes&&... args) {
64  if (device_type == DeviceType::CPU) {
65  if (!cpu_dispatch_ptr) {
66  cpu_dispatch_ptr = choose_cpu_impl();
67  }
68  return (*cpu_dispatch_ptr)(std::forward<ArgTypes>(args)...);
69  } else if (device_type == DeviceType::CUDA) {
70  AT_ASSERTM(cuda_dispatch_ptr, "DispatchStub: missing CUDA kernel");
71  return (*cuda_dispatch_ptr)(std::forward<ArgTypes>(args)...);
72  } else if (device_type == DeviceType::HIP) {
73  AT_ASSERTM(hip_dispatch_ptr, "DispatchStub: missing HIP kernel");
74  return (*hip_dispatch_ptr)(std::forward<ArgTypes>(args)...);
75  } else {
76  AT_ERROR("DispatchStub: unsupported device type", device_type);
77  }
78  }
79 
80  FnPtr choose_cpu_impl() {
81  auto capability = static_cast<int>(get_cpu_capability());
82  (void)capability;
83 #ifdef HAVE_AVX2_CPU_DEFINITION
84  if (capability >= static_cast<int>(CPUCapability::AVX2)) {
85  AT_ASSERTM(AVX2, "DispatchStub: missing AVX2 kernel");
86  return AVX2;
87  }
88 #endif
89 #ifdef HAVE_AVX_CPU_DEFINITION
90  if (capability >= static_cast<int>(CPUCapability::AVX)) {
91  AT_ASSERTM(AVX, "DispatchStub: missing AVX kernel");
92  return AVX;
93  }
94 #endif
95  AT_ASSERTM(DEFAULT, "DispatchStub: missing default kernel");
96  return DEFAULT;
97  }

128 // Compiler will complain if you put things like std::tuple<Tensor, Tensor> in
129 // the `fn` argument of DECLARE_DISPATCH. Some possible workarounds, e.g.,
130 // adding parentheses and using helper struct to get rid of the parentheses, do
131 // not work with MSVC. So do a `using`-declaration if you need to pass in such
132 // `fn`, e.g., grid_sampler_2d_backward_cpu_kernel in GridSampleKernel.h.
