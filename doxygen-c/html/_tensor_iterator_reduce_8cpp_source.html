16 void TensorIterator::parallel_reduce(const loop2d_t& loop) {
17  AT_CHECK(ntensors() == 2, "parallel_reduce only supports one input and one output");
18  int64_t numel = this->numel();
19  if (numel < at::internal::GRAIN_SIZE || at::get_max_threads() == 1 || at::in_parallel_region()) {
20  serial_for_each(loop, {0, numel});
21  } else if (use_two_pass_reduction(*this)) {
22  two_pass_reduction(*this, loop);
23  } else {

32 static void two_pass_reduction(TensorIterator& iter, const loop2d_t& loop) {
33  int max_threads = at::get_max_threads();
34 
35  auto& dst = iter.tensor(0);
36  auto buffer_shape = DimVector(dst.sizes());
37  buffer_shape.insert(buffer_shape.begin(), max_threads);
38  auto buffer = at::empty(buffer_shape, dst.type());
39 
40  std::unique_ptr<bool[]> written(new bool[max_threads]);
41  std::fill(written.get(), written.get() + max_threads, false);
42 
43  at::parallel_for(0, iter.numel(), internal::GRAIN_SIZE, [&](int64_t begin, int64_t end) {
44  int thread_num = at::get_thread_num();
45  written[thread_num] = true;
46  auto slice = buffer[thread_num];
47  slice.copy_(dst);
48 
49  auto sub_iter = TensorIterator::reduce_op(slice, iter.tensor(1));
50  sub_iter->serial_for_each(loop, {begin, end});
51  });
52 
53  // fill any unwritten slices of the buffer with the identity
54  for (int thread_num = 0; thread_num < max_threads; thread_num++) {
55  if (!written[thread_num]) {
56  buffer[thread_num].copy_(dst);
57  }
58  }
59 
60  auto unsqueezed = dst.unsqueeze(0);
61  auto final_reduce = TensorIterator::reduce_op(unsqueezed, buffer);
62  final_reduce->for_each(loop);
63 }

67 static int find_split_dim(TensorIterator& iter) {
68  int num_threads = at::get_max_threads();
69  auto shape = iter.shape();
70 
71  // start with the outer-most dimension
72  int best_dim = iter.ndim() - 1;
73  for (int dim = best_dim; dim >= 0 && !iter.is_dim_reduced(dim); dim--) {
74  if (shape[dim] >= num_threads) {
75  return dim;
76  } else if (shape[dim] > shape[best_dim]) {
77  best_dim = dim;
78  }
79  }
80 
81  AT_ASSERT(!iter.is_dim_reduced(best_dim));
82  return best_dim;
83 }

95 static void parallel_dim_reduction(TensorIterator& iter, const loop2d_t& loop) {
96  AT_ASSERT(iter.ndim() >= 1);
97  int dim = find_split_dim(iter);
98  int64_t cols = iter.shape()[dim];
99  int element_size = iter.element_size(/*arg=*/1);
100 
101  bool should_round_columns = iter.strides(1)[dim] == element_size;
102  at::parallel_for(0, cols, 1, [&](int64_t begin, int64_t end) {
103  if (should_round_columns) {
104  // round columns to multiples of 128 bytes if adjacent columns are
105  // contiguous in memory.
106  int64_t cols_per_128_bytes = 128 / element_size;
107  std::tie(begin, end) = round_columns(iter, dim, cols_per_128_bytes, begin, end);
108  }
109  if (begin == end) {
110  return;
111  }
112  auto sub_iter = TensorIterator(iter);
113  sub_iter.narrow(dim, begin, end - begin);
114  sub_iter.for_each(loop);
115  });
116 }

118 void TensorIterator::foreach_reduced_elt(const loop_subiter_t &loop, bool parallelize) {
119  AT_ASSERT(ntensors() == 2 && num_outputs_ == 1);
120 
121  auto shape = this->shape();
122  if (tensor(0).numel() == 0) {
123  return;
124  }
125  if (tensor(0).numel() == 1) {
126  loop(*this);
127  }
128  else if (numel() < at::internal::GRAIN_SIZE || at::get_max_threads() == 1 || at::in_parallel_region() || !parallelize) {
129  auto reduce_dims = num_reduce_dims();
130 
131  auto non_reduced_shape = shape.slice(reduce_dims, shape.size() - reduce_dims);
132 
133  int64_t non_reduced_numel = 1;
134  for (int i = 0; i < non_reduced_shape.size(); ++i) {
135  non_reduced_numel *= non_reduced_shape[i];
136  }
137  DimCounter dims {non_reduced_shape, {0, non_reduced_numel}};
138  while (!dims.is_done()) {
139  TensorIterator reduced = *this;
140  reduced.select_all_keeping_dim(reduce_dims, dims.values);
141  loop(reduced);
142  dims.increment({1, 1});
143  }
144  }
145  else {
146  int dim = find_split_dim(*this);
147  int64_t cols = shape[dim];
148  at::parallel_for(0, cols, 1, [&](int64_t begin, int64_t end) {
149  if (begin == end) {
150  return;
151  }
152 
153  auto sub_iter = *this;
154 
155  sub_iter.narrow(dim, begin, end - begin);
156  // On some broken setups, `#ifdef _OPENMP` is true,
157  // and `get_max_threads` returns > 1, but
158  // `#pragma omp parallel` is ignored.
159  // There is no API to check for this, so we need to explicitly
160  // stop trying to parallelize if we've already gotten here.
161  //
162  // (If we are on one of those broken setups, we will
163  // only have one thread here, and end - begin == cols.)
164  sub_iter.foreach_reduced_elt(loop, false);
165  });
166  }
167 }
