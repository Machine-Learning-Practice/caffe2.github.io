82  explicit File(
83  const std::string& path,
84  int flags,
85  std::chrono::milliseconds timeout) {
86  const auto start = std::chrono::steady_clock::now();
87  while (true) {
88  fd_ = syscall(std::bind(::open, path.c_str(), flags, 0644));
89  // Only retry when the file doesn't exist, since we are waiting for the
90  // file to be created in this case to address the following issue:
91  // https://github.com/pytorch/pytorch/issues/13750
92  if (fd_ >= 0 || (fd_ < 0 && errno != ENOENT)) {
93  break;
94  }
95  const auto elapsed = std::chrono::duration_cast<std::chrono::seconds>(
96  std::chrono::steady_clock::now() - start);
97  if (timeout != c10d::Store::kNoTimeout && elapsed > timeout) {
98  break;
99  }
100  std::this_thread::sleep_for(std::chrono::milliseconds(10));
101  }
102  SYSASSERT(fd_, "open(" + path + ")");
103  }

187 off_t refresh(
188  File& file,
189  off_t pos,
190  std::unordered_map<std::string, std::vector<uint8_t>>& cache) {
191  auto size = file.size();
192  if (size != pos) {
193  std::string tmpKey;
194  std::vector<uint8_t> tmpValue;
195  file.seek(pos, SEEK_SET);
196  while (size > pos) {
197  file.read(tmpKey);
198  file.read(tmpValue);
199  cache[tmpKey] = std::move(tmpValue);
200  pos = file.tell();
201  }
202  }
203  file.seek(0, SEEK_SET);
204  return pos;
205 }

242 std::vector<uint8_t> FileStore::get(const std::string& key) {
243  std::string regKey = regularPrefix_ + key;
244  const auto start = std::chrono::steady_clock::now();
245  while (true) {
246  File file(path_, O_RDONLY, timeout_);
247  auto lock = file.lockShared();
248  auto size = file.size();
249  if (cache_.count(regKey) == 0 && size == pos_) {
250  // No new entries; release the shared lock and sleep for a bit
251  lock.unlock();
252  const auto elapsed = std::chrono::duration_cast<std::chrono::seconds>(
253  std::chrono::steady_clock::now() - start);
254  if (timeout_ != kNoTimeout && elapsed > timeout_) {
255  throw std::runtime_error("Timeout waiting for key: " + key);
256  }
257  std::this_thread::sleep_for(std::chrono::milliseconds(10));
258  continue;
259  }
260  // Always refresh since even though the key exists in the cache,
261  // it might be outdated
262  pos_ = refresh(file, pos_, cache_);
263  if (cache_.count(regKey) != 0) {
264  break;
265  }
266  }
267 
268  return cache_[regKey];
269 }

271 int64_t FileStore::addHelper(const std::string& key, int64_t i) {
272  File file(path_, O_RDWR | O_CREAT, timeout_);
273  auto lock = file.lockExclusive();
274  pos_ = refresh(file, pos_, cache_);
275 
276  const auto& value = cache_[key];
277  int64_t ti = i;
278  if (!value.empty()) {
279  auto buf = reinterpret_cast<const char*>(value.data());
280  auto len = value.size();
281  ti += std::stoll(std::string(buf, len));
282  }
283  // Always seek to the end to write
284  file.seek(0, SEEK_END);
285  // File cursor is at the end of the file now, and we have an
286  // exclusive lock, so we can write the new value.
287  file.write(key);
288  file.write(std::to_string(ti));
289  return ti;
290 }

324  std::chrono::steady_clock::now() - start);

326  throw std::runtime_error("Wait timeout");
