174 void ProcessGroupNCCL::broadcastUniqueNCCLID(ncclUniqueId* ncclID) {
175  // Every time when we create a new unique NCCL ID, we need to use a new
176  // global key to access/update the store.
177  // The key is a combination of processGroupID_ and the current count of
178  // NCCL unique ID created
179  std::unique_lock<std::mutex> lock(pgTrackingLock_);
180  auto groupPgId = groupName_ + "_" + processGroupID_;
181  const auto uniqueNCCLIDCnt = ++pgUniqueNCCLIDCnt_[groupPgID_];
182 
183  lock.unlock();
184 
185  std::string storeKey =
186  processGroupID_ + "_" + std::to_string(uniqueNCCLIDCnt);
187 
188  // Rank 0 writes to the store as bcast
189  if (rank_ == 0) {
190  auto ncclIDVal = std::vector<uint8_t>(
191  reinterpret_cast<uint8_t*>(ncclID),
192  reinterpret_cast<uint8_t*>(ncclID) + NCCL_UNIQUE_ID_BYTES);
193  store_->set(storeKey, ncclIDVal);
194  // Other ranks get to the store
195  } else {
196  auto ncclIDVal = store_->get(storeKey);
197  // Just a sanity check
198  if (ncclIDVal.size() != NCCL_UNIQUE_ID_BYTES) {
199  throw std::runtime_error(
200  "Unexpected NCCL unique ID length received "
201  "from the store");
202  }
203  // Now put the data back to the input pointer
204  memcpy(ncclID, ncclIDVal.data(), NCCL_UNIQUE_ID_BYTES);
205  }
206 }

208 std::vector<std::shared_ptr<NCCLComm>>& ProcessGroupNCCL::getNCCLComm(
209  const std::string& devicesKey,
210  const std::vector<at::Device>& devices) {
211  // Sanity check
212  if (devicesKey.empty()) {
213  throw std::runtime_error(
214  "Not able to create/get the NCCL Communicator since "
215  "the GPU devices are not known");
216  }
217 
218  for (auto& device : devices) {
219  usedDeviceIdxs_.insert(device.index());
220  }
221 
222  if (devNCCLCommMap_.find(devicesKey) != devNCCLCommMap_.end()) {
223  // Reuse the cached communicator if there is one.
224  return devNCCLCommMap_[devicesKey];
225  }
226  // NCCL communicator not cached, create a new entry
227  std::vector<std::shared_ptr<NCCLComm>> ncclComms;
228  ncclComms.resize(devices.size());
229 
230  // Create the unique NCCL ID and broadcast it
231  ncclUniqueId ncclID;
232 
233  if (rank_ == 0) {
234  C10D_NCCL_CHECK(ncclGetUniqueId(&ncclID));
235  }
236 
237  // Broadcast so that each process can have a unique NCCL ID
238  broadcastUniqueNCCLID(&ncclID);
239 
240  at::cuda::OptionalCUDAGuard gpuGuard;
241 
242  std::vector<at::cuda::CUDAStream> streamVal;
243  streamVal.reserve(devices.size());
244 
245  // Create the NCCL communicators for each GPU
246  C10D_NCCL_CHECK(ncclGroupStart());
247 
248  for (size_t i = 0; i < devices.size(); ++i) {
249  // GPU world size and GPU rank
250  int numRanks = getSize() * devices.size();
251  int rank = getRank() * devices.size() + i;
252 
253  gpuGuard.set_index(devices[i].index());
254  ncclComms[i] = NCCLComm::create(numRanks, rank, ncclID);
255 
256  // Creates the NCCL streams
257  streamVal.push_back(at::cuda::getStreamFromPool());
258  }
259 
260  C10D_NCCL_CHECK(ncclGroupEnd());
261 
262  // Move the NCCL resource to cache
263  devNCCLCommMap_.emplace(devicesKey, std::move(ncclComms));
264  ncclStreams_.emplace(devicesKey, std::move(streamVal));
265 
266  // Note: these events are created with the (default) cudaEventDisableTiming
267  // flag This flag provides the best performance when used with
268  // cudaStreamWaitEvent() and cudaEventQuery(). Since we here don't measure the
269  // performance using cudaEvent, this should be set.
270  ncclEvents_.emplace(
271  std::piecewise_construct,
272  std::make_tuple(devicesKey),
273  std::make_tuple(devices.size()));
274 
275  return devNCCLCommMap_[devicesKey];
276 }

278 // Helper function that checks the input and output tensors for validity
279 void ProcessGroupNCCL::tensorCheckHelper(
280  const std::vector<at::Tensor>& input,
281  const std::vector<at::Tensor>& output,
282  int outputOverInput) {
283  if (input.size() * outputOverInput != output.size()) {
284  throw std::runtime_error(
285  "Input tensor sequence should have the same "
286  "number of tensors as the output tensor sequence");
287  }
288 
289  if (input.size() == 0) {
290  throw std::runtime_error("The number of input tensors should not be zero");
291  }
292 
293  if (input.size() > static_cast<size_t>(at::cuda::getNumGPUs())) {
294  throw std::runtime_error(
295  "The number of input tensors is larger than "
296  "the number of available GPUs");
297  }
298 
299  // To make sure each tensor is on separate devices
300  std::unordered_set<int> usedDevices;
301  usedDevices.reserve(input.size());
302 
303  auto inputNumElement = input[0].numel();
304  auto elementType = input[0].scalar_type();
305 
306  for (size_t i = 0; i < input.size(); ++i) {
307  // Check to make sure it's a GPU dense tensor
308  if (!(input[i].is_cuda() && !input[i].is_sparse() && output[i].is_cuda() &&
309  !output[i].is_sparse())) {
310  throw std::runtime_error(
311  "Only CUDA dense tensor is supported for NCCL "
312  "collective operations");
313  }
314  // Check the tensor type is identical
315  if (input[i].scalar_type() != elementType ||
316  output[i].scalar_type() != elementType) {
317  throw std::runtime_error(
318  "Expecting all GPU tensors to have identical "
319  "type");
320  }
321  // Check the input tensor size is identical
322  if (input[i].numel() != inputNumElement) {
323  throw std::runtime_error(
324  "Expecting all input tensors to have identical "
325  "number of elements");
326  }
327  // Check the output tensor size equals to input tensor size
328  if (output[i].numel() != inputNumElement) {
329  throw std::runtime_error(
330  "The number of elements of output tensor does "
331  "not match the number of elements of the input "
332  "tensor");
333  }
334  // Contiguous verification
335  if (!input[i].is_contiguous() || !output[i].is_contiguous()) {
336  throw std::runtime_error("Expecting all GPU tensors to be contiguous");
337  }
338 
339  bool inserted;
340  std::tie(std::ignore, inserted) = usedDevices.insert(input[i].get_device());
341  // Device verification, if the insertion didn't take place
342  if (!inserted) {
343  throw std::runtime_error("Expecting inputs on different GPU devices");
344  }
345 
346  // Now check the output device
347  if (input[i].get_device() != output[i].get_device()) {
348  throw std::runtime_error(
349  "Expecting input and output tensors to be on "
350  "the same device");
351  }
352  }
353 }

355 std::shared_ptr<ProcessGroup::Work> ProcessGroupNCCL::allreduce(
356  std::vector<at::Tensor>& tensors,
357  const AllreduceOptions& opts) {
358  tensorCheckHelper(tensors, tensors);
359 
360  auto devices = getDeviceList(tensors);
361  auto key = getKeyFromDevices(devices);
362  auto& ncclComms = getNCCLComm(key, devices);
363 
364  // First let NCCL streams wait for THC stream
365  syncStreams(devices, ncclEvents_[key], ncclStreams_[key]);
366 
367  // Work itself will create the CUDA events on all GPUs of tensors
368  auto work = std::make_shared<ProcessGroupNCCL::WorkNCCL>(devices);
369 
370  at::cuda::OptionalCUDAGuard gpuGuard;
371 
372  std::unique_lock<std::mutex> cudaFreeMutexLock(
373  *(c10::cuda::CUDACachingAllocator::getFreeMutex()));
374 
375  C10D_NCCL_CHECK(ncclGroupStart());
376 
377  for (size_t i = 0; i < tensors.size(); ++i) {
378  gpuGuard.set_index(devices[i].index());
379  at::cuda::CUDAStream& ncclStream = ncclStreams_[key][i];
380 
381  C10D_NCCL_CHECK(ncclAllReduce(
382  tensors[i].data_ptr(),
383  tensors[i].data_ptr(),
384  tensors[i].numel(),
385  getNcclDataType(tensors[i].scalar_type()),
386  ncclOp[opts.reduceOp],
387  ncclComms[i]->getNcclComm(),
388  ncclStream.stream()));
389  }
390 
391  C10D_NCCL_CHECK(ncclGroupEnd());
392 
393  // Event should only be recorded after the ncclGroupEnd()
394  for (size_t i = 0; i < tensors.size(); ++i) {
395  at::cuda::CUDAStream& ncclStream = ncclStreams_[key][i];
396  work->cudaEvents_[i].record(ncclStream);
397  }
398 
399  return work;

500 std::shared_ptr<ProcessGroup::Work> ProcessGroupNCCL::allgather(
501  std::vector<std::vector<at::Tensor>>& outputTensors,
502  std::vector<at::Tensor>& inputTensors,
503  const AllgatherOptions& opts) {
504  if (outputTensors.size() != inputTensors.size()) {
505  throw std::runtime_error("allgather: input and output size mismatch");
506  }
507  std::vector<at::Tensor> flattenOutputTensors;
508  flattenOutputTensors.resize(outputTensors.size());
509 
510  for (size_t i = 0; i < outputTensors.size(); ++i) {
511  tensorCheckHelper(
512  std::vector<at::Tensor>{inputTensors[i]},
513  outputTensors[i],
514  size_ * inputTensors.size());
515  // Flatten the output tensors (from all ranks) to a single big tensor
516  flattenOutputTensors[i] = newLikeFlat(outputTensors, i);
517 
518  if (static_cast<size_t>(flattenOutputTensors[i].numel()) !=
519  inputTensors[i].numel() * size_ * inputTensors.size()) {
520  throw std::runtime_error("Unexpected size for flatten tensor");
521  }
522  }
523 
524  auto devices = getDeviceList(inputTensors);
525  auto key = getKeyFromDevices(devices);
526  auto& ncclComms = getNCCLComm(key, devices);
527 
528  // First let NCCL streams wait for current streams
529  syncStreams(devices, ncclEvents_[key], ncclStreams_[key]);
530 
531  // Work itself will create the CUDA events on all GPUs of tensors
532  auto work = std::make_shared<ProcessGroupNCCL::WorkNCCL>(devices);
533 
534  at::cuda::OptionalCUDAGuard gpuGuard;
535 
536  std::unique_lock<std::mutex> cudaFreeMutexLock(
537  *(c10::cuda::CUDACachingAllocator::getFreeMutex()));
538 
539  C10D_NCCL_CHECK(ncclGroupStart());
540 
541  for (size_t i = 0; i < inputTensors.size(); ++i) {
542  gpuGuard.set_index(devices[i].index());
543 
544  at::cuda::CUDAStream& ncclStream = ncclStreams_[key][i];
545 
546  C10D_NCCL_CHECK(ncclAllGather(
547  inputTensors[i].data_ptr(),
548  flattenOutputTensors[i].data_ptr(),
549  inputTensors[i].numel(),
550  getNcclDataType(inputTensors[i].scalar_type()),
551  ncclComms[i]->getNcclComm(),
552  ncclStream.stream()));
553  }
554 
555  C10D_NCCL_CHECK(ncclGroupEnd());
556 
557  // Copy the flattened output tensors to the outputs
558  for (size_t i = 0; i < outputTensors.size(); ++i) {
559  at::cuda::CUDAStream& ncclStream = ncclStreams_[key][i];
560  at::cuda::CUDAStreamGuard guard(ncclStream);
561  for (size_t j = 0; j < outputTensors[0].size(); ++j) {
562  outputTensors[i][j].copy_(flattenOutputTensors[i][j], true);
563  }
564  }
565 
566  // Event should only be recorded after the ncclGroupEnd()
567  for (size_t i = 0; i < inputTensors.size(); ++i) {
568  at::cuda::CUDAStream& ncclStream = ncclStreams_[key][i];
569  work->cudaEvents_[i].record(ncclStream);
570  }
571  return work;
572 }

574 std::shared_ptr<ProcessGroup::Work> ProcessGroupNCCL::barrier(
575  const BarrierOptions& opts) {
576  std::vector<at::Device> devices;
577  if (usedDeviceIdxs_.empty()) {
578  // This means there is not yet a NCCL collective being called
579  // Here we have to use the best guesses and will use a single GPU to call
580  // allreduce to achieve barrier.
581  // In case the multiple processes fall into the same node, we use rank to
582  // ensure that each process is on a different GPU
583  auto numGPUs = at::cuda::getNumGPUs();
584  int16_t deviceIdx = static_cast<int16_t>(rank_ % numGPUs);
585  devices.push_back(at::Device(at::DeviceType::CUDA, deviceIdx));
586  } else {
587  for (auto usedDeviceIdx : usedDeviceIdxs_) {
588  devices.push_back(at::Device(at::DeviceType::CUDA, usedDeviceIdx));
589  }
590  }
591 
592  std::vector<at::Tensor> barrierTensors;
593  barrierTensors.reserve(devices.size());
594 
595  at::cuda::OptionalCUDAGuard gpuGuard;
596  for (auto& device : devices) {
597  gpuGuard.set_index(device.index());
598  barrierTensors.push_back(at::empty(
599  {1},
600  at::TensorOptions().device(at::DeviceType::CUDA).dtype(at::kByte)));
601  }
602 
603  // All reduce to achieve the barrier
604  auto work = allreduce(barrierTensors);

606  // Work will take over barrierTensors
