64  void reset_stream(Stream stream) {
65  // TODO: make a version that takes an impl argument. Unfortunately,
66  // that will require SFINAE because impl is only valid for the
67  // VirtualGuardImpl specialization.
68  if (stream.device() == this->current_device()) {
69  this->impl_.exchangeStream(stream);
70  current_stream_ = stream;
71  } else {
72  // Destruct and reconstruct the StreamGuard in-place
73  this->impl_.exchangeStream(original_stream_of_current_device_);
74  this->reset_device(stream.device());
75  original_stream_of_current_device_ = this->impl_.exchangeStream(stream);
76  current_stream_ = stream;
77  }
78  }
