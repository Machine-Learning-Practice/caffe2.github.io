461  // ~~~~~ Autograd API ~~~~~
462  // Some methods below are defined in TensorImpl.cpp because Tensor is an
463  // incomplete type.
464  //
465  // Note [Tensor versus Variable in C++]
466  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
467  // Autograd methods are only valid for the Variable::Impl subclass
468  // of Tensor. This is due to some questionable life choices, where
469  // a Variable has a Tensor (so they are not the same thing), but
470  // a Variable is a Tensor (they are subclassed, so that you can write
471  // code on Tensor that works both with Variables and Tensors. Poor
472  // man's polymorphism). Variable does NOT satisfy the Liskov Substitution
473  // Principle for Tensor; generally you want to work with all Variables,
474  // or all Tensors, but not a mix of both. We intend to fix this in
475  // the future.
476  //
477  // Note [We regret making Variable hold a Tensor]
478  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
479  // Tensor has a bunch of fields in it. Are those fields always valid?
480  // Not necessarily: the Variable::Impl subclass of a tensor doesn't use these
481  // fields; instead, it *forwards* them to a contained, inner tensor
482  // (the 'data' tensor). It doesn't even bother keeping the fields on the
483  // outer tensor up-to-date, because an end user could grab the inner
484  // tensor and directly, e.g., resize it (making any outer fields we track
485  // stale).
486  //
487  // As you might imagine, this is a TERRIBLE state of affairs to be in.
488  // It makes implementing everything on TensorImpl complicated: if
489  // you directly access a field on TensorImpl, you must *virtualize*
490  // the function, if you want it to work correctly when called from
491  // Variable (because we need to override the method to avoid looking
492  // in our fields, and look in the data tensor's fields.) Anything that
493  // isn't virtualized, won't work if called on a variable.
494  //
495  // The way to fix this is to make Variable::Impl stop holding a tensor;
496  // instead, it should just *be* a tensor.

756  void set_sizes_and_strides(IntArrayRef new_size, IntArrayRef new_stride) {
757  AT_CHECK(allow_tensor_metadata_change(), "set_sizes_and_strides is not allowed on Tensor created from .data or .detach()");
758  AT_ASSERT(!is_variable()); // TODO: remove this when Variable and Tensor are merged
759  AT_CHECK(
760  new_size.size() == new_stride.size(),
761  "dimensionality of sizes (",
762  new_size.size(),
763  ") must match dimensionality of strides (",
764  new_stride.size(),
765  ")");
766  auto new_dim = new_size.size();
767 
768  sizes_.resize(new_dim);
769  for (size_t dim = 0; dim < new_dim; ++dim) {
770  sizes_[dim] = new_size[dim];
771  }
772 
773  strides_.resize(new_dim);
774  if (new_dim > 0) {
775  for (size_t dim = new_dim - 1; ; dim--) {
776  if (new_stride[dim] >= 0) {
777  strides_[dim] = new_stride[dim];
778  } else {
779  // XXX: This behavior is surprising and may need to be removed to
780  // support negative strides. Some pytorch functions rely on it:
781  // for example, torch.cat (run TestTorch.test_cat_empty).
782  if (dim == new_dim - 1) {
783  strides_[dim] = 1;
784  } else {
785  // Keep stride monotonically increasing to match NumPy.
786  strides_[dim] = std::max<int64_t>(sizes_[dim + 1], 1) * strides_[dim + 1];
787  }
788  }
789  if (dim == 0) break;
790  }
791  }

913  void Extend(int64_t num, float growthPct) {
914  AT_ASSERT(sizes_.size() >= 1u);
915  AT_ASSERTM(num >= 0, "`num` must be non-negative for Extend");
916  AT_ASSERTM(
917  is_contiguous_,
918  "Right now Extend is only supported for contiguous Tensor.");
919  auto newDims = sizes_;
920  newDims[0] += num;
921  if (!storage_.data()) {
922  Resize(newDims);
923  return;
924  }
925  auto newNumel = std::accumulate(
926  newDims.begin(),
927  newDims.end(),
928  static_cast<int64_t>(1),
929  std::multiplies<int64_t>());
930  if (newNumel * storage_.itemsize() <= storage_.capacity()) {
931  sizes_ = newDims;
932  numel_ = newNumel;
933  return;
934  }
935  auto newCapacity = sizes_;
936  newCapacity[0] = std::max<size_t>(
937  newDims[0], std::ceil(sizes_[0] * (growthPct + 100) / 100));
938  auto oldData = std::move(storage_.data_ptr());
939  auto oldSize = numel_;
940  auto oldDims = sizes_;
941  Resize(newCapacity);
942  auto* newData = raw_mutable_data(data_type_);
943  if (data_type_.copy()) {
944  AT_ASSERTM(
945  device_type() == DeviceType::CPU,
946  "non-POD types work only on CPU");
947  data_type_.copy()(oldData.get(), newData, oldSize);
948  } else {
949  // The following copy uses the current (thread local) stream for copying
950  // and also takes the GPU id from the device() field passed in.
951  //
952  // TODO: Potentially more enforcements are necessary to avoid accidental
953  // switch to sync copy if the currently set device is wrong.
954  //
955  // Specifically, we might need to switch to a different context device
956  // here explicitly to avoid relying on user synchronizing things
957  // properly.
958  CopyBytes(
959  oldSize * itemsize(),
960  oldData.get(),
961  device(),
962  newData,
963  device(),
964  true); // non-blocking
965  }
966  reserved_ = true;
967  sizes_ = newDims;
968  numel_ = newNumel;
969  }

977  template <class T>
978  void ReserveSpace(const T& outer_dim) {
979  AT_ASSERTM(
980  is_contiguous_,
981  "Right now ReserveSpace is only supported for contiguous Tensor.");
982  AT_ASSERTM(
983  storage_.unique(), "Can't call ReserveSpace on shared storage.");
984  auto newCapacity = sizes_;
985  newCapacity[0] = outer_dim;
986  auto newNumel = std::accumulate(
987  newCapacity.begin(),
988  newCapacity.end(),
989  static_cast<int64_t>(1),
990  std::multiplies<int64_t>());
991  if (newNumel * storage_.itemsize() <= storage_.capacity()) {
992  return;
993  }
994  // Old data is discarded
995  storage_.data_ptr().clear();
996  auto oldSize = numel_;
997  auto oldDims = sizes_;
998  Resize(newCapacity);
999  // Allocate new memory but don't copy over the data
1000  raw_mutable_data(data_type_);
1001  sizes_ = oldDims;
1002  numel_ = oldSize;
1003  reserved_ = true;
1004  }

1022  template <typename... Ts>
1023  void Resize(Ts... dim_source) {
1024  bool size_changed = SetDims(dim_source...);
1025  if (size_changed) {
1026  // If needed, we will free the data. the next mutable_data() call
1027  // will create the data storage.
1028  bool reset_tensor = false;
1029  if (reserved_) {
1030  // If tensor is reserved then don't claim its memeory unless capacity()
1031  // is smaller than new size

1169  inline void* raw_mutable_data(const caffe2::TypeMeta& meta) {
1170  // For 0-size tensors it's fine to return any pointer (including nullptr)
1171  if (data_type_ == meta && storage_initialized()) {
1172  return static_cast<void*>(static_cast<char*>(storage_.data()) + storage_offset_ * meta.itemsize());
1173  } else {
1174  bool had_special_dtor = data_type_.placementDelete() != nullptr;
1175  storage_offset_ = 0;
1176  if (storage_.unique()) {
1177  storage_.set_dtype(meta);
1178  } else {
1179  if (data_type_ != meta) {
1180  storage_ = Storage::create_legacy(storage_.device(), meta);
1181  }
1182  }
1183  data_type_ = meta;
1184 
1185  // We can reuse the existing buffer if the current data does not have
1186  // a special destructor and the new data doesn't have a special
1187  // constructor.
1188  if (numel_ == 0 ||
1189  (meta.placementNew() == nullptr && !had_special_dtor &&
1190  storage_.numel() >= numel_)) {
1191  AT_ASSERT(storage_offset_ == 0); // because we just reallocated
1192  return storage_.data();
1193  }
1194  const Allocator* allocator = storage_.allocator();
1195  // Storage might have nullptr allocator in rare cases, for example, if
1196  // an external memory segment has been wrapped with Tensor and we don't
1197  // know how to reallocate it. However, in order to preserve legacy C2
1198  // behavior, we allow reallocating the memory using default allocator.
1199  if (allocator == nullptr) {
1200  allocator = GetAllocator(storage_.device_type());
1201  }
1202  if (meta.placementNew()) {
1203  // For types that need placement new, we will call it, as well as
1204  // making sure that when the data is freed, it calls the right
1205  // destruction procedure.
1206  auto size = numel_;
1207  auto dtor = data_type_.placementDelete();
1208  auto data_ptr = allocator->allocate(numel_ * storage_.itemsize());
1209  storage_.set_data_ptr(PlacementDeleteContext::makeDataPtr(
1210  std::move(data_ptr), dtor, size, storage_.device()));
1211  data_type_.placementNew()(storage_.data(), numel_);
1212  } else {
1213  // For fundamental type, new and delete is easier.
1214  storage_.set_data_ptr(
1215  allocator->allocate(numel_ * storage_.itemsize()));
1216  }
1217  storage_.set_numel(numel_);
1218  AT_ASSERT(storage_offset_ == 0); // because we just reallocated
1219  return storage_.data();
1220  }
