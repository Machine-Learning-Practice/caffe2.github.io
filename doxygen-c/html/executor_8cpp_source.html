25 // Returns the "map size" for this run, which is the common size for all
26 // intermediate tensors.
27 static c10::optional<std::vector<int64_t>> getMapSize(
28  const KernelSpec& spec,
29  at::TensorList args,
30  at::IntArrayRef arg_subset) {
31  // TODO: this keeps reallocating map_size at every iteration, but we know
32  // exactly how much storage do we need, so this could be fixed in-place at
33  // every step. We're just missing a few functions for ATen, but the fix
34  // should be straightforward.
35  // Note: left unitialized since empty shape is broadcastable to any shape
36  std::vector<int64_t> map_size;
37  map_size.reserve(8);
38  for (const auto arg_idx : arg_subset) {
39  auto& arg = args.at(arg_idx);
40  auto& chunk_desc = spec.inputChunks().at(arg_idx);
41  if (chunk_desc.nSubTensors() == 1) {
42  try {
43  map_size = at::infer_size(map_size, arg.sizes());
44  } catch (...) {
45  return c10::nullopt;
46  }
47  } else {
48  auto tensor_sizes = arg.sizes().vec();
49  const auto num_chunks = chunk_desc.nSubTensors();
50  const auto dim =
51  at::maybe_wrap_dim(chunk_desc.dim(), tensor_sizes.size());
52  if (tensor_sizes[dim] % num_chunks != 0) {
53  return c10::nullopt;
54  }
55  tensor_sizes[dim] /= num_chunks;
56  try {
57  map_size = at::infer_size(map_size, tensor_sizes);
58  } catch (...) {
59  return c10::nullopt;
60  }
61  }
62  }
63 
64  return {map_size};
65 }

67 // Tries to determine a map size for the instantiated kernel (see above)
68 static c10::optional<std::vector<int64_t>> canRunKernel(
69  const KernelSpec& spec,
70  at::TensorList args) {
71  // Short-circuits on size mismatch
72  AT_CHECK(
73  args.size() == spec.inputChunks().size(),
74  "Expected ",
75  spec.inputChunks().size(),
76  " arguments, but got ",
77  args.size());
78 
79  c10::optional<std::vector<int64_t>> map_size;
80  for (const auto& broadcast_group : spec.inputBroadcastGroups()) {
81  if (!map_size) {
82  map_size = getMapSize(spec, args, broadcast_group);
83  if (!map_size)
84  return c10::nullopt;
85  } else {
86  const auto group_map_size = getMapSize(spec, args, broadcast_group);
87  // Note: this checks that group_map_size is defined AND equal to map_size
88  if (map_size != group_map_size)
89  return c10::nullopt;
90  }
91  }
92 
93  return map_size;
94 }

96 // Arguments are expanded to a common shape, referred to as the "map size,"
97 // (see above).
98 // Note: Arguments are mutated by this call, although map_size is restored
99 // to its original value.
100 static bool expandArgs(
101  const KernelSpec& spec,
102  std::vector<at::Tensor>& args,
103  std::vector<int64_t>& map_size, bool dry_run) {
104  bool has_broadcast = false;
105  for (size_t i = 0; i < args.size(); ++i) {
106  auto& arg = args[i];
107  const auto& pdesc = spec.inputChunks()[i];
108  if (pdesc.nSubTensors() == 1) {
109  if (arg.sizes().equals(map_size))
110  continue;
111  if (!dry_run) {
112  arg = arg.expand(map_size);
113  has_broadcast = true;
114  } else {
115  return true;
116  }
117  } else {
118  map_size.at(pdesc.dim()) *= pdesc.nSubTensors();
119  if (!arg.sizes().equals(map_size)) {
120  if (!dry_run) {
121  arg = arg.expand(map_size);
122  has_broadcast = true;
123  } else {
124  return true;
125  }
126  }
127  map_size.at(pdesc.dim()) /= pdesc.nSubTensors();
128  }
129  }
130  return has_broadcast;
131 }

160 // Tries to compress sizes and strides according to cont. Emits the result t
161 // c_sizes, c_strides and throws an error on failure (if can't compress)
162 static void compressContiguous(
163  const at::IntArrayRef& sizes,
164  const at::IntArrayRef& strides,
165  const std::vector<bool>& cont,
166  uint32_t* c_sizes,
167  uint32_t* c_strides) {
168  size_t compressed_dims = 0;
169  size_t cur = 0;
170  size_t ndim = sizes.size();
171  while (cur < ndim) {
172  size_t total_size = sizes[cur];
173  cur++;
174  while (cont[cur - 1] && cur < ndim) {
175  AT_ASSERT(strides[cur - 1] == sizes[cur] * strides[cur]);
176  total_size *= sizes[cur];
177  cur++;
178  }
179  c_sizes[compressed_dims] = total_size;
180  c_strides[compressed_dims] = strides[cur - 1];
181  compressed_dims++;
182  }
183 
184  if (ndim > 0)
185  AT_ASSERT(!cont.back() || strides.back() == 1);
186 }

188 // Launches the requested fusion on the given device with the given inputs.
189 // Output pointers are stored in outputs (to be put on the stack later).
190 void launchFusion(
191  const FusedKernel& fusion,
192  const at::Device device,
193  const at::ArrayRef<at::Tensor>& inputs,
194  const at::ArrayRef<IValue>& all_inputs,
195  std::vector<at::Tensor>& outputs) {
196  // Fails if fusion and given inputs disagree
197  AT_ASSERT(inputs.size() == fusion.inputDesc().size());
198 
199  // Computes number of flattened inputs and outputs
200  size_t flat_inputs_size = 0;
201  size_t flat_outputs_size = 0;
202  for (const auto& c : fusion.chunkDesc())
203  flat_inputs_size += c.nSubTensors();
204  for (const auto& c : fusion.concatDesc())
205  flat_outputs_size += c.nSubTensors();
206 
207  // Fails if the elements of the first (any) tensor are not expressable as
208  // a 32-bit integer.
209  // Note: this code assumes that inputs are 32-bit addressable
210  // Note: this code assumes that all inputs are of the same size
211  AT_ASSERT(inputs[0].numel() <= std::numeric_limits<uint32_t>::max());
212 
213  // Computes map_size, numel from the first input
214  at::IntArrayRef map_size;
215  uint32_t numel;
216  std::vector<int64_t> keep_alive_size;
217  if (fusion.chunkDesc()[0].isNoop()) {
218  map_size = inputs[0].sizes();
219  numel = inputs[0].numel();
220  } else {
221  keep_alive_size = computeMapSize(inputs[0], fusion.chunkDesc()[0]);
222  map_size = keep_alive_size;
223  numel = computeNumel(map_size);
224  }
225 
226  // compute number of scalar inputs and convert them to float
227  std::vector<float> scalar_inputs;
228  scalar_inputs.reserve(all_inputs.size());
229  for (auto const &input: all_inputs){
230  if (input.isDouble()) scalar_inputs.push_back(input.to<float>());
231  }
232 
233  // Computes the storage needed to store TensorInfo structs for inputs and
234  // outputs.
235  size_t uncompressedDim = fusion.inputDesc().at(0).contiguity.size();
236  size_t maxPossibleTensorInfoSize =
237  sizeof(TensorInfo) + 2 * sizeof(uint32_t) * uncompressedDim;
238  size_t maxPossibleBufferSize =
239  maxPossibleTensorInfoSize * (flat_inputs_size + flat_outputs_size);
240  std::vector<char> buffer(maxPossibleBufferSize);
241  char* buffer_next = buffer.data();
242 
243  // A vector of arguments to the kernel (numel, *input_desc_s, *output_desc_s)
244  std::vector<void*> arguments;
245  arguments.reserve(3 + scalar_inputs.size() + flat_inputs_size + flat_outputs_size);
246  arguments.push_back(&numel);
247 
248  auto addTensorInfoRaw = [&](const TensorDesc& desc,
249  void* data_ptr,
250  at::IntArrayRef sizes,
251  at::IntArrayRef strides) {
252  const auto nDim = desc.nDim(); // NOTE: this is the compressed dim
253  AT_ASSERT(nDim <= uncompressedDim); // We'd overflow the space otherwise
254  auto ti = reinterpret_cast<TensorInfo*>(buffer_next);
255  ti->data = data_ptr;
256  compressContiguous(
257  sizes, strides, desc.contiguity, ti->sizes(nDim), ti->strides(nDim));
258  buffer_next += maxPossibleTensorInfoSize;
259  arguments.push_back(ti);
260  };
261 
262  // Asserts that t's dims can be compressed in the same way as in desc
263  // (that's what the kernel assumes), and appends it to the arguments vector.
264  auto addTensorInfo = [&](const TensorDesc& desc, const at::Tensor& t) {
265  addTensorInfoRaw(desc, t.data_ptr(), t.sizes(), t.strides());
266  };
267 
268  // Adds (flattened) input arguments
269  for (size_t i = 0; i < fusion.inputDesc().size(); ++i) {
270  const auto& chunk = fusion.chunkDesc()[i];
271  const at::Tensor& tensor = inputs[i];
272  if (chunk.isNoop()) {
273  addTensorInfo(fusion.inputDesc()[i], tensor);
274  } else {
275  size_t chunk_offset = map_size[chunk.dim()] * tensor.stride(chunk.dim()) *
276  elementSize(tensor.scalar_type());
277  char* data_ptr = reinterpret_cast<char*>(tensor.data_ptr());
278  for (size_t chunks = 0; chunks < chunk.nSubTensors(); ++chunks) {
279  addTensorInfoRaw(
280  *chunk.subTensorDesc(), data_ptr, map_size, tensor.strides());
281  data_ptr += chunk_offset;
282  }
283  }
284  }
285  // Adds scalar arguments
286  for (float &s: scalar_inputs){
287  arguments.push_back(&s);
288  }
289 
290  // Adds (flattened) output arguments
291  outputs.reserve(fusion.outputDesc().size());
292  const auto& ref_options = inputs[0].options();
293  for (size_t i = 0; i < fusion.outputDesc().size(); ++i) {
294  const auto& c = fusion.concatDesc()[i];
295  if (c.isNoop()) {
296  outputs.push_back(at::empty(
297  map_size, ref_options.dtype(fusion.outputDesc()[i].scalar_type)));
298  addTensorInfo(fusion.outputDesc()[i], outputs[i]);
299  } else {
300  size_t small_size = map_size[c.dim()];
301  std::vector<int64_t> concat_size(map_size.begin(), map_size.end());
302  concat_size[c.dim()] = small_size * c.nSubTensors();
303  outputs.push_back(at::empty(concat_size, ref_options));
304  const auto& o = outputs[i];
305  size_t offset = 0;
306  for (size_t j = 0; j < c.nSubTensors(); ++j) {
307  // because the concatenated_output stays live, the underlying data
308  // in this view remains live through the end of this function
309  // so there is not need to hold onto this tensor
310  const auto view = o.narrow(c.dim(), offset, small_size);
311  addTensorInfo(*c.subTensorDesc(), view);
312  offset += small_size;
313  }
314  }
315  }
316 
317  fusion.launch_raw(numel, arguments);
318 }

320 bool runFusion(const int64_t key, Stack& stack) {
321  // Short-circuits if fusion isn't enabled
322  if (!canFuseOnCPU() && !canFuseOnGPU())
323  return false;
324 
325  // Acquires the FusionSpec
326  auto maybe_spec = retrieve(key);
327  AT_ASSERT(maybe_spec);
328  auto& spec = *(*maybe_spec);
329 
330  // Acquires inputs from stack
331  auto all_inputs = last(stack, spec.nInputs());
332  std::vector<at::Tensor> inputs;
333  inputs.reserve(spec.nTensorInputs());
334  // we know that tensor inputs are first
335  for (int64_t i = 0; i < spec.nTensorInputs(); i++) {
336  inputs.emplace_back(all_inputs[i].toTensor());
337  }
338 
339  // Determines device to dispatch to. If there's a device mismatch in the
340  // inputs, we use the fallback (which should give a nice error message).
341  at::Device device = inputs.at(0).device();
342  for (const auto& t : at::TensorList(inputs).slice(1)) {
343  if (t.device() != device) {
344  return false;
345  }
346  }
347 
348  // Attempts to run fallback if device fusion is disabled
349  if (device.is_cuda() && !canFuseOnGPU())
350  return false;
351  if (device.is_cpu() && !canFuseOnCPU())
352  return false;
353 
354  // Validates sizes and expands inputs as needed
355  auto maybe_map_size = canRunKernel(spec, inputs);
356 
357  // Tries to run fallback if map size can't be computed
358  if (!maybe_map_size)
359  return false;
360  if (spec.hasRandom()) {
361  bool hasBroadcast = shouldExpandArgs(spec,inputs, *maybe_map_size);
362  if (hasBroadcast) return false;
363  }
364  expandArgs(spec, inputs, *maybe_map_size, /*dry_run=*/false);
365 
366  // Retrieves the kernel, compiling (and caching) if necessary
367  ArgSpec arg_spec{inputs, device.index()};
368  auto maybe_kernel = spec.findKernel(arg_spec);
369  if (!maybe_kernel) {
370  const auto kernel = compileKernel(spec, arg_spec, *maybe_map_size, device);
371  spec.cacheKernel(arg_spec, kernel);
372  }
373  maybe_kernel = spec.findKernel(arg_spec);
374  AT_ASSERT(maybe_kernel);
375 
376  // Launches fusion
377  std::vector<at::Tensor> raw_outputs;
378  launchFusion(*(*maybe_kernel), device, inputs, all_inputs, raw_outputs);
379 
380  auto outputs = fmap(spec.outputMapAndSizes(), [&](const OutputMapAndSize& omap) {
381  if (omap.needsSumToSize()) {
382  return at::sum_to(
383  raw_outputs[omap.offset()],
384  all_inputs[omap.sizeInput()].toIntList()->elements());
385  } else {
386  return raw_outputs[omap.offset()];
387  }
388  });
389 
390  // Updates stack
391  drop(stack, spec.nInputs());
392  stack.insert(
393  stack.end(),
394  std::make_move_iterator(outputs.begin()),
395  std::make_move_iterator(outputs.end()));
396 
397  return true;
398 }
