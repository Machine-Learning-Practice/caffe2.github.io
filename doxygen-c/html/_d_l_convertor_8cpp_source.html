87 ScalarType toScalarType(const DLDataType& dtype) {
88  ScalarType stype;

99  break;

101  switch (dtype.bits) {

163  atDLMTensor->tensor.dl_tensor.dtype = getDLDataType(src);

165  atDLMTensor->tensor.dl_tensor.strides = const_cast<int64_t*>(src.strides().data());
