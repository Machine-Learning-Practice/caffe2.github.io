72  FunctionSchema getSchema(const size_t n_args, const size_t n_binders) {
73  auto annotations = py::module::import("torch.jit.annotations");
74  auto signature = annotations.attr("get_signature")(self);
75  std::vector<Argument> args, rets;
76  // We may mutate this if we can determine the number of args from Python
77  // introspection.
78  size_t actual_n_args = n_args;
79  if (!signature.is_none()) {
80  std::vector<TypePtr> arg_types;
81  TypePtr ret_type;
82  std::tie(arg_types, ret_type) =
83  py::cast<std::pair<std::vector<TypePtr>, TypePtr>>(signature);
84  args.reserve(arg_types.size());
85  size_t idx = 0; // Fake argument names by putting in the index
86  for (auto& arg_type : arg_types) {
87  args.push_back(Argument(
88  std::to_string(idx++), std::move(arg_type), {}, {}, false));
89  }
90  rets.push_back(Argument("0", std::move(ret_type), {}, {}, false));
91  } else {
92  // Create a default signature using what information we have
93 
94  // First see if we can introspect the number of function parameters
95  // irrespective of the presence of explicit type annotations
96  auto num_params = annotations.attr("get_num_params")(self);
97  if (!num_params.is_none()) {
98  // Return a signature with the correct number of params according to the
99  // Python function. The error handling in call() will catch any mismatch
100  // later.
101  actual_n_args = py::cast<size_t>(num_params);
102  }
103  // Construct the default signature: all arguments and returns will be
104  // DynamicType
105  args.reserve(actual_n_args);
106  for (size_t i = 0; i < actual_n_args; ++i) {
107  args.push_back(
108  Argument(std::to_string(i), TensorType::get(), {}, {}, false));
109  }
110  TypePtr ret_type = TensorType::get();
111  if (n_binders == 0) {
112  ret_type = NoneType::get();
113  } else if (n_binders > 1) {
114  std::vector<TypePtr> tuple_values(n_binders, ret_type);
115  ret_type = TupleType::create(std::move(tuple_values));
116  }
117  rets.push_back(Argument("0", ret_type, {}, {}, false));
118  }
119  return FunctionSchema("", "", std::move(args), std::move(rets));
120  }

124  const SourceRange& loc,
125  Method& m,
126  at::ArrayRef<NamedValue> inputs_,
127  at::ArrayRef<NamedValue> attributes,
128  size_t n_binders) override {
129  auto inputs = toValues(*m.graph(), inputs_);
130  auto schema = getSchema(inputs.size(), n_binders);
131 
132  std::stringstream failure_messages;
133  c10::optional<MatchedSchema> matched_schema = tryMatchSchema(
134  schema,
135  loc,
136  *m.graph(),
137  c10::nullopt,
138  inputs_,
139  attributes,
140  failure_messages,
141  /*conv_tensor_to_num*/ true);
142  if (!matched_schema)
143  throw ErrorReport(loc) << failure_messages.str();

146  py::object func = self;
147  std::string cconv(inputs.size(), 'd');
148  Node* new_node = m.graph()->insertNode(m.graph()->createPythonOp(
149  THPObjectPtr(func.release().ptr()), cconv, {}));
150 
151  // Mark if function is ignored on export
152  if (py::cast<bool>(py::module::import("torch.jit")
153  .attr("_try_get_ignored_op")(self))) {
154  auto python_op = static_cast<PythonOp*>(new_node);
155  python_op->ignore_on_export = true;
156  }
157  new_node->setSourceLocation(std::make_shared<SourceRange>(loc));
158  for (auto& i : matched_schema->inputs)
159  new_node->addInput(i);
160 
161  Value* output =
162  new_node->addOutput()->setType(matched_schema->return_types.at(0));
163  return std::make_shared<SimpleValue>(output);
164  }

301  // select an attribute on it, e.g. `this.field`
302  std::shared_ptr<SugaredValue> attr(
303  const SourceRange& loc,
304  Method& m,
305  const std::string& field) override {
306  // workaround to make self.training work
307  // it adds a buffer 'training' to the model if one doesn't exist
308  // and then loads that parameter, casting it to bool
309  if (field == "training") {
310  NamedIValue* v = module->find_buffer(field);
311  if (!v) {
312  py::object py_module = py::cast(module);
313  bool training = py::cast<bool>(py::getattr(py_module, "training"));
314  auto t =
315  autograd::make_variable(at::full({}, training ? 1 : 0, at::kLong));
316  module->register_buffer("training", std::move(t));
317  v = module->find_buffer(field);
318  }
319  Value* the_tensor = m.get_or_add_parameter(v->slot());
320  Value* the_bool = m.graph()->insert(prim::Bool, {the_tensor});
321  return std::make_shared<SimpleValue>(the_bool);
322  }
323 
324  if (NamedModule* v = module->find_module(field)) {
325  return std::make_shared<ModuleValue>(v->module);
326  } else if (Method* v = module->find_method(field)) {
327  return std::make_shared<MethodValue>(shared_from_this(), *v);
328  } else if (NamedIValue* v = module->find_parameter(field)) {
329  return std::make_shared<SimpleValue>(m.get_or_add_parameter(v->slot()));
330  } else if (NamedIValue* v = module->find_attribute(field)) {
331  return std::make_shared<SimpleValue>(
332  m.get_or_add_attribute(v->type, v->slot()));
333  }
334 
335  // This can also be a call to a non-script module, or a plain
336  // python method. If so return this as a python value.
337  py::object py_module = py::cast(module);
338  if (py::object attr = py::getattr(py_module, field.c_str(), py::none())) {
339  if (py::isinstance<py::function>(attr) &&
340  py::hasattr(attr, "_is_parameter_list") &&
341  py::cast<bool>(py::getattr(attr, "_is_parameter_list"))) {
342  return std::make_shared<ConstantParameterList>(module);
343  }
344  if (py::isinstance<py::function>(attr) ||
345  py::isinstance(attr, py::module::import("torch.nn").attr("Module")) ||
346  py_module.attr("_constants_set").contains(field.c_str())) {
347  return toSugaredValue(attr, m, loc, true);
348  } else {
349  std::string hint = "did you forget to add it __constants__?";
350  if (py::isinstance(attr, py::module::import("torch").attr("Tensor"))) {
351  hint = "Tensors must be added to a module as a buffer or parameter";
352  }
353  throw ErrorReport(loc)
354  << "attribute '" << field << "' of type '" << typeString(attr)
355  << "' is not usable in a script method (" << hint << ")";
356  }
357  }
358  throw ErrorReport(loc) << "module has no attribute '" << field << "'";
359  }

488 std::shared_ptr<SugaredValue> toSugaredValue(
489  py::object obj,
490  Method& m,
491  SourceRange loc,
492  bool is_constant,
493  bool is_submodule) {
494  // directly create SimpleValues when possible, because they are first-class
495  // and can be re-assigned. Otherwise, this would be invalid:
496  // f = python_constant
497  // while ...
498  // f = f + 1
499  auto& g = *m.graph();
500  if (is_constant) {
501  if (py::isinstance<py::bool_>(obj)) {
502  return toSimple(g.insertConstant(py::cast<bool>(obj), nullptr, loc));
503  } else if (py::isinstance<py::int_>(obj)) {
504  return toSimple(g.insertConstant(py::cast<int64_t>(obj), nullptr, loc));
505  } else if (py::isinstance<py::float_>(obj)) {
506  return toSimple(g.insertConstant(py::cast<double>(obj), nullptr, loc));
507  } else if (py::isinstance<py::str>(obj)) {
508  return toSimple(
509  g.insertConstant(py::cast<std::string>(obj), nullptr, loc));
510  } else if (obj.is(py::none())) {
511  return toSimple(g.insertConstant(IValue(), nullptr, loc));
512  } else if (THPDevice_Check(obj.ptr())) {
513  auto device = reinterpret_cast<THPDevice*>(obj.ptr());
514  return toSimple(g.insertConstant(device->device));
515  } else if (THPLayout_Check(obj.ptr())) {
516  auto layout = reinterpret_cast<THPLayout*>(obj.ptr());
517  const auto v = static_cast<int64_t>(layout->layout);
518  return toSimple(g.insertConstant(v, nullptr, loc));
519  } else if (THPDtype_Check(obj.ptr())) {
520  auto dtype = reinterpret_cast<THPDtype*>(obj.ptr());
521  const auto v = static_cast<int64_t>(dtype->scalar_type);
522  return toSimple(g.insertConstant(v, nullptr, loc));
523  } else if (py::isinstance<py::tuple>(obj)) {
524  return std::make_shared<ConstantPythonTupleValue>(obj);
525  }
526  }
527 
528  auto weak_obj =
529  py::module::import("torch.jit").attr("_try_get_weak_module")(obj);
530  if (!weak_obj.is_none()) {
531  obj = weak_obj;
532  }
533  if (py::isinstance<Module>(obj)) {
534  auto mod = py::cast<std::shared_ptr<Module>>(obj);
535  // In the case that this Python object is not a submodule, inline *ONLY
536  // PURE* ScriptModules. This allows us to call arbitrary @script functions
537  // within a scripting context while still enforcing that parameters from
538  // stateful submodules are properly accounted for.
539  if (!is_submodule && mod->get_parameters().size() != 0) {
540  throw ErrorReport()
541  << "Attempted to inline a Module with parameters. "
542  "Stateful modules to be inlined must be submodules of the callee.";
543  }
544  return std::make_shared<ModuleValue>(mod);
545  } else if (py::isinstance<py::module>(obj)) {
546  return std::make_shared<PythonModuleValue>(obj);
547  } else if (obj.ptr() == py::module::import("torch.jit").attr("_fork").ptr()) {
548  return std::make_shared<ForkValue>();
549  } else if (
550  obj.ptr() == py::module::import("torch.jit").attr("annotate").ptr()) {
551  return std::make_shared<AnnotateValue>();
552  }
553 
554  py::object builtin_name =
555  py::module::import("torch.jit").attr("_find_builtin")(obj);
556  if (!builtin_name.is_none()) {
557  return std::make_shared<BuiltinFunction>(
558  Symbol::fromQualString(py::str(builtin_name)), c10::nullopt);
559  }
560 
561  if (py::isinstance<py::function>(obj)) {
562  auto compiled_fn =
563  py::module::import("torch.jit").attr("_try_compile_weak_script")(obj);
564  if (!compiled_fn.is(py::none())) {
565  auto mod = py::cast<std::shared_ptr<Module>>(compiled_fn);
566  return std::make_shared<ModuleValue>(mod);
567  }
568  }
569 
570  py::object dispatched_fn =
571  py::module::import("torch.jit").attr("_try_get_dispatched_fn")(obj);
572  if (!dispatched_fn.is_none()) {
573  return std::make_shared<BooleanDispatchValue>(std::move(dispatched_fn));
574  }
575 
576  py::object overloads =
577  py::module::import("torch.jit").attr("_try_get_overloaded_fn")(obj);
578  if (!overloads.is_none()) {
579  return std::make_shared<OverloadedFunctionValue>(std::move(overloads));
580  }
581 
582  return std::make_shared<PythonValue>(obj);
583 }

633 FunctionSchema getSchemaWithNameAndDefaults(
634  const SourceRange& range,
635  const FunctionSchema& schema,
636  const at::optional<std::string>& new_name,
637  const FunctionDefaults& default_args) {
638  std::vector<Argument> new_args;
639  for (auto& arg : schema.arguments()) {
640  auto it = default_args.find(arg.name());
641  if (it != default_args.end()) {
642  try {
643  IValue value;
644  auto n = arg.N();
645  auto list_type = arg.type()->cast<ListType>();
646  if (n && *n > 0 && list_type) {
647  // BroadcastingList, allow default values T for arg types List[T]
648  value = toIValue(it->second, list_type->getElementType());
649  } else {
650  value = toIValue(it->second, arg.type());
651  }
652  new_args.emplace_back(
653  arg.name(), arg.type(), arg.N(), value, arg.kwarg_only());
654  } catch (py::cast_error& e) {
655  throw ErrorReport(range)
656  << "Expected a default value of type " << arg.type()->str()
657  << " on parameter \"" << arg.name() << "\"";
658  }
659  } else {
660  new_args.push_back(arg);
661  }
662  }
663 
664  return FunctionSchema(
665  new_name.value_or(schema.name()),
666  schema.overload_name(),
667  new_args,
668  schema.returns(),
669  schema.is_vararg(),
670  schema.is_varret());
671 }
