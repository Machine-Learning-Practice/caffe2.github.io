134 Tensor sparse_coo_tensor(const Tensor& indices, const Tensor& values_, const TensorOptions& options) {
135  Tensor values = expand_values_if_needed(values_);
136 
137  // arg checking
138  AT_CHECK(!options.has_layout() || options.layout() == kSparse, "expected sparse layout, but got layout ", options.layout());
139  // the following checks are redundant because they are also checked in SparseTensorImpl::set_indices_and_values_unsafe
140  // but we need to ensure them in order to infer the shape.
141  AT_CHECK(indices.dim() == 2, "indices must be sparse_dim x nnz, but got: ", indices.sizes())
142  AT_CHECK(!indices.is_sparse(), "expected indices to be a dense tensor, but got indices of layout ", indices.layout());
143 
144  // If sizes are not given, it is inferred as max index of each dim.
145  int64_t sparse_dim = indices.size(0);
146  int64_t dense_dim = values.dim() - 1;
147 
148  std::vector<int64_t> computed_sizes(sparse_dim + dense_dim);
149  if (indices.numel() > 0) {
150  // If the indices has elements in it, we infer the minimum sparse dimension sizes
151  // as the max value of each dim in indices.
152  // NB: It used to keepdim. I think that was wrong.
153  LongTensor min_indices = std::get</* values */ 0>(indices.min(/* dim */ 1, /* keepdim */ false));
154  LongTensor computed_indices_sizes = std::get</* values */ 0>(indices.max(/* dim */ 1, /* keepdim */ false));
155  computed_indices_sizes.add_(1); // len = max_index + 1
156  LongTensor cpu_min_indices = min_indices.to(at::DeviceType::CPU);
157  LongTensor cpu_computed_indices_sizes = computed_indices_sizes.to(at::DeviceType::CPU);
158  auto cpu_min_indices_accessor = cpu_min_indices.accessor<int64_t, 1>();
159  auto cpu_computed_indices_sizes_accessor = cpu_computed_indices_sizes.accessor<int64_t, 1>();
160  for (int64_t d = 0; d < sparse_dim; d++) {
161  int64_t min_index_in_dim = cpu_min_indices_accessor[d];
162  AT_CHECK(min_index_in_dim >= 0,
163  "found negative index ", min_index_in_dim, " for dim ", d);
164  computed_sizes[static_cast<size_t>(d)] = cpu_computed_indices_sizes_accessor[d];
165  }
166  } else {
167  // If the indices doesn't have elements in it, there is not enough information
168  // to know what the minimum sparse dimension sizes should be, and in this case
169  // we set them to 0
170  for (int64_t d = 0; d < sparse_dim; d++) {
171  computed_sizes[static_cast<size_t>(d)] = 0;
172  }
173  }
174  for (int64_t d = 0; d < dense_dim; d++) {
175  computed_sizes[static_cast<size_t>(sparse_dim + d)] = values.size(d+1);
176  }
177 
178  return at::_sparse_coo_tensor_with_dims_and_tensors(
179  sparse_dim, dense_dim, computed_sizes, indices, values, values.options().layout(kSparse));
180 }

183 Tensor sparse_coo_tensor(const Tensor& indices, const Tensor& values_, ArrayRef<int64_t> size, const TensorOptions& options) {
184  Tensor values = expand_values_if_needed(values_);
185 
186  // arg checking
187  AT_CHECK(!options.has_layout() || options.layout() == kSparse, "expected sparse layout, but got layout ", options.layout());
188  // the following checks are redundant because they are also checked in SparseTensorImpl::set_indices_and_values_unsafe
189  // but we need to ensure them in order to infer the shape.
190  AT_CHECK(indices.dim() == 2, "indices must be sparse_dim x nnz, but got: ", indices.sizes())
191  AT_CHECK(!indices.is_sparse(), "expected indices to be a dense tensor, but got indices of layout ", indices.layout());
192  int64_t sparse_dim = indices.size(0);
193  int64_t dense_dim = values.dim() - 1;
194  AT_CHECK(size.size() == sparse_dim + dense_dim,
195  "number of dimensions must be sparse_dim (", sparse_dim, ") + dense_dim (", dense_dim, "), but got ", size.size());
196 
197  // Check to make sure all indices are within the boundaries of `size`
198  if (indices.numel() > 0) {
199  LongTensor min_indices = std::get</* values */ 0>(indices.min(/* dim */ 1, /* keepdim */ false));
200  LongTensor max_indices = std::get</* values */ 0>(indices.max(/* dim */ 1, /* keepdim */ false));
201  LongTensor cpu_min_indices, cpu_max_indices;
202  if (indices.is_cuda()) {
203  cpu_min_indices = min_indices.to(at::DeviceType::CPU);
204  cpu_max_indices = max_indices.to(at::DeviceType::CPU);
205  } else {
206  cpu_min_indices = min_indices;
207  cpu_max_indices = max_indices;
208  }
209  auto cpu_min_indices_accessor = cpu_min_indices.accessor<int64_t, 1>();
210  auto cpu_max_indices_accessor = cpu_max_indices.accessor<int64_t, 1>();
211  for (int64_t d = 0; d < sparse_dim; d++) {
212  // NB: This used to sync ndim times to access each entry; now we copy
213  // everything to CPU first and then access it.
214  int64_t min_index_in_dim = cpu_min_indices_accessor[d];
215  AT_CHECK(min_index_in_dim >= 0,
216  "found negative index ", min_index_in_dim, " for dim ", d);
217  int64_t max_index_in_dim = cpu_max_indices_accessor[d];
218  int64_t dim_size = size[static_cast<size_t>(d)];
219  AT_CHECK(max_index_in_dim < dim_size,
220  "size is inconsistent with indices: for dim ", d, ", size is ", dim_size, " but found index ", max_index_in_dim);
221  }
222  }
223 
224  return at::_sparse_coo_tensor_with_dims_and_tensors(
225  sparse_dim, dense_dim, size, indices, values, values.options().layout(kSparse));
226 }

285 SparseTensor dense_to_sparse(const Tensor& self, int64_t sparse_dim){
286  int64_t dims = self.dim();
287  // TODO: it seems like sparse_dim == 0 could be supported even if self.dim() > 0,
288  // but this would take some work and doesn't seem particularly useful.
289  AT_CHECK(sparse_dim > 0 || self.dim() == 0, "sparse_dim must be >0 if dimensionality > 0");
290  AT_CHECK(sparse_dim <= dims,
291  "sparse_dim must be less than or equal to self.dim()");
292  at::TensorOptions sparse_options = self.options().layout(kSparse);
293  std::vector<int64_t> sizes = self.sizes().vec();
294 
295  Tensor nz = self.nonzero().transpose(0, 1);
296  if (nz.size(1) == 0) {
297  return new_with_dims_sparse(sparse_dim, dims - sparse_dim, sizes, sparse_options);
298  }
299  LongTensor indices;
300  if (sparse_dim == dims) {
301  indices = nz.clone();
302  } else {
303  Tensor i = nz.narrow(0, 0, sparse_dim);
304  std::tie(indices, std::ignore) = _unique_dim(i, 1);
305  indices = indices.contiguous(); // many sparse CUDA kernels require contiguity, see issue #12633
306  }
307 
308  Tensor values;
309  if (self.dim() > 0) {
310  std::vector<Tensor> ix = indices.chunk(indices.size(0), 0);
311  values = self.index(ix).squeeze(0).clone();
312  } else {
313  AT_ASSERT(nz.sizes().equals({0, 1}));
314  // In this cases, indices is a clone of nz, which is a tensor of shape (0, 1).
315  // Given sparse tensor invariants, values should be shape (1,)
316  values = self.unsqueeze(0).clone();
317  }
318 
319  Tensor sparse = at::sparse_coo_tensor(indices, values, sizes, sparse_options);
320  return sparse._coalesced_(true);
321 }

337 SparseTensor coalesce_sparse_cpu(const SparseTensor& self) {
338  AT_ASSERT(self.defined());
339  AT_ASSERT(!self.is_variable()); // TODO: change this to check `.requires_grad()` and `GradMode::is_enabled()` when Variable and Tensor are merged
340  AT_ASSERT(self.is_sparse());
341 
342  if (self.is_coalesced()) {
343  return self;
344  }
345  // NOTE: Since `coalesce` is not an in-place operation when `is_coalesced` is false,
346  // we should keep the original tensor intact and do coalesce on a copy of the tensor
347  if (self._nnz() < 2) {
348  SparseTensor dst = self.clone();
349  dst._coalesced_(true);
350  return dst;
351  }
352 
353  LongTensor indices = self._indices();
354  Tensor values = self._values().contiguous();
355  int64_t sparse_dim = self.sparse_dim();
356  int64_t dense_dim = self.dense_dim();
357  int64_t nnz = self._nnz();
358 
359  LongTensor indices_scalar = flatten_indices(indices, self.sizes());
360 
361  SparseTensor dst = new_sparse(self.options());
362  get_sparse_impl(dst)->resize_(sparse_dim, dense_dim, self.sizes());
363  // TODO: is there a more idiomatic way to do this?
364  LongTensor newIndices = at::empty(indices.sizes(), indices.options());
365  Tensor newValues = at::empty(values.sizes(), values.options());
366  alias_into_sparse(dst, newIndices, newValues);
367 
368  LongTensor indicesBuffer;
369  LongTensor indicesPermutation;
370  std::tie(indicesBuffer, indicesPermutation) = indices_scalar.sort(0);
371  // NB: The accessor accesses here rely on self._nnz() > 0 (tested earlier in this function)
372  auto newIndicesAccessor = newIndices.accessor<int64_t, 2>();
373  auto indicesAccessor = indices.accessor<int64_t, 2>();
374  auto indicesPermutationAccessor = indicesPermutation.accessor<int64_t, 1>();
375  auto indicesBufferAccessor = indicesBuffer.accessor<int64_t, 1>();
376 
377  int64_t i = -1;
378  AT_DISPATCH_ALL_TYPES(
379  values.scalar_type(), "coalesce", [&] {
380  int64_t prev = -1;
381  int64_t blockSize = values.stride(0);
382  scalar_t* values_ptr = values.data<scalar_t>();
383  scalar_t* newValues_ptr = newValues.data<scalar_t>();
384  for (int64_t j = 0; j < nnz; j++) {
385  int64_t pos = indicesPermutationAccessor[j];
386  int64_t curr = indicesBufferAccessor[j];
387  if (curr == prev) {
388  if (values.numel() > 0) { // if values is an empty tensor, there are no elements to copy
389  THBlas_axpy<scalar_t>(blockSize, 1, values_ptr + pos * blockSize, 1, newValues_ptr + i * blockSize, 1);
390  }
391  } else {
392  ++i;
393  for (int64_t d = 0; d < sparse_dim; d++) {
394  newIndicesAccessor[d][i] = indicesAccessor[d][pos];
395  }
396  if (values.numel() > 0) { // if values is an empty tensor, there are no elements to copy
397  THBlas_copy<scalar_t>(blockSize, values_ptr + pos * blockSize, 1, newValues_ptr + i * blockSize, 1);
398  }
399  }
400  prev = curr;
401  }
402  });
403 
404  dst._coalesced_(true);
405  get_sparse_impl(dst)->set_nnz_and_narrow(i + 1);
406 
407  return dst;
408 }

418 template <typename scalar_t>
419 void inline sparse_mask_out_cpu_kernel(
420  Tensor& r_values,
421  const Tensor& t,
422  const int64_t r_nnz,
423  const int64_t sparse_dim,
424  const LongTensor& mask_indices
425 ) {
426  auto r_values_accessor = r_values.accessor<scalar_t, 1>();
427  auto mask_indices_accessor = mask_indices.accessor<int64_t, 2>();
428  scalar_t* t_ptr = t.data<scalar_t>();
429  int64_t i;
430 
431  #pragma omp parallel for private(i) if (r_nnz > 1000)
432  for (i = 0; i < r_nnz; i++) {
433  int64_t idx = 0;
434  for (int64_t d = 0; d < sparse_dim; d++) {
435  idx += mask_indices_accessor[d][i] * t.stride(d);
436  }
437  r_values_accessor[i] = t_ptr[idx];
438  }
439 }

441 SparseTensor& sparse_mask_out_cpu(SparseTensor& r, const Tensor& t, const SparseTensor& mask) {
442  AT_CHECK(mask.is_coalesced(), "sparse_mask: mask is uncoalesced");
443  AT_CHECK(mask.sizes().equals(t.sizes()), "sparse_mask: operands have incompatible sizes; self has size ",
444  t.sizes(), " but mask has size ", mask.sizes());
445  AT_ASSERT(!t.is_cuda()); // we were supposed to have dispatched on this
446  AT_CHECK(!r.is_cuda(), "sparse_mask: expected 'out' to be CPU, but got CUDA");
447  AT_CHECK(!mask.is_cuda(), "sparse_mask: expected 'mask' to be CPU, but got CUDA");
448  resize_as_sparse_(r, mask);
449  if (mask._nnz() == 0) {
450  return r.zero_();
451  }
452  int64_t dim = t.dim();
453  int64_t sparse_dim = mask.sparse_dim();
454  LongTensor mask_indices = mask._indices();
455  Tensor mask_values = mask._values();
456  Tensor r_values = at::empty(mask_values.sizes(), r._values().options());
457  alias_into_sparse(r, mask_indices.clone(), r_values);
458  r._coalesced_(mask.is_coalesced());
459  int64_t r_nnz = mask._nnz();
460  get_sparse_impl(r)->set_nnz_and_narrow(r_nnz);
461 
462  if (t.numel() == 0) { // if t is an empty tensor, there is no need to mask its elements
463  return r;
464  }
465 
466  if (dim > sparse_dim) {
467 
468  // Get a flattened sparse indices, similar to NOTE [ Flatten Sparse Indices ].
469  // Keeping this implementation because it is faster than flatten_indices()
470  LongTensor indices = at::zeros({mask._nnz()}, mask_indices.options());
471  for (int64_t d = 0; d < mask.sparse_dim(); d++) {
472  indices.mul_(mask.size(d));
473  indices.add_(mask_indices.select(0, d));
474  }
475 
476  std::vector<int64_t> view_size(1 + mask.dense_dim());
477  view_size[0] = -1;
478  for (int64_t d = 0; d < mask.dense_dim(); d++) {
479  view_size[d + 1] = mask.size(mask.sparse_dim() + d);
480  }
481 
482  Tensor t_view = t.view(view_size);
483  // TODO: Re-audit this; it used to be an indexSelect directly into r_values
484  at::index_select_out(r_values, t_view, 0, indices);
485  } else {
486  AT_DISPATCH_ALL_TYPES(r_values.scalar_type(), "sparse_mask", [&] {
487  sparse_mask_out_cpu_kernel<scalar_t>(
488  r_values,
489  t,
490  r_nnz,
491  sparse_dim,
492  mask_indices);
493  });
494  }
495  return r;
496 }
