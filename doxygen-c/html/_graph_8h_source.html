257  // Move a node from this graph to the destGraph
258  void moveNode(NodeRef node, Graph<T, U...>* destGraph) {
259  assert(hasNode(node));
260  for (auto it = nodes_.begin(); it != nodes_.end(); ++it) {
261  if (&(*it) == node) {
262  std::list<Node<T, U...>>& destNodes = destGraph->nodes_;
263  destNodes.splice(destNodes.end(), nodes_, it);
264  nodeRefs_.erase(node);
265  destGraph->nodeRefs_.insert(node);
266  break;
267  }
268  }
269  }

271  // Move an edge from this graph to the destGraph
272  void moveEdge(EdgeRef edge, Graph<T, U...>* destGraph) {
273  assert(hasEdge(edge));
274  assert(destGraph->hasNode(edge->tail()));
275  assert(destGraph->hasNode(edge->head()));
276  std::list<Edge<T, U...>>& destEdges = destGraph->edges_;
277  for (auto it = edges_.begin(); it != edges_.end(); ++it) {
278  if (&(*it) == edge) {
279  destEdges.splice(destEdges.end(), edges_, it);
280  break;
281  }

285  // Move entire subgraph to destGraph.
286  // Be sure to delete in/out edges from this graph first.
287  void moveSubgraph(
288  const Subgraph<T, U...>& subgraph,
289  Graph<T, U...>* destGraph) {
290  auto sg = subgraph; // Copy to check that all nodes and edges are matched
291  std::list<Edge<T, U...>>& destEdges = destGraph->edges_;
292  for (auto it = nodes_.begin(); it != nodes_.end(); ++it) {
293  auto node = &(*it);
294  if (sg.hasNode(node)) {
295  std::list<Node<T, U...>>& destNodes = destGraph->nodes_;
296  destNodes.splice(destNodes.end(), nodes_, it--);
297  nodeRefs_.erase(node);
298  destGraph->nodeRefs_.insert(node);
299  sg.removeNode(node);
300  }
301  }
302  for (auto it = edges_.begin(); it != edges_.end(); ++it) {
303  auto edge = &(*it);
304  if (sg.hasEdge(edge)) {
305  assert(destGraph->hasNode(edge->tail()));
306  assert(destGraph->hasNode(edge->head()));
307  destEdges.splice(destEdges.end(), edges_, it--);
308  sg.removeEdge(edge);
309  }
310  }
311  assert(sg.getNodes().size() == 0);
312  assert(sg.getEdges().size() == 0);
313  }

315  // Validates the graph. Returns true if the graph is valid
316  // and false if any node or edge referenced in the graph
317  // is not actually present in the graph.
318  bool isValid() {
319  for (auto& node : getMutableNodes()) {
320  for (auto& inEdge : node->getInEdges()) {
321  if (!hasEdge(inEdge)) {
322  DEBUG_PRINT("Invalid inEdge %p on node %p\n", inEdge, node);
323  return false;
324  }
325  }
326  for (auto& outEdge : node->getOutEdges()) {
327  if (!hasEdge(outEdge)) {
328  DEBUG_PRINT("invalid outEdge %p on node %p\n", outEdge, node);
329  return false;
330  }
331  }
332  // Check validity of nodeRefs_
333  if (!hasNode(node)) {
334  DEBUG_PRINT("Invalid node %p\n", node);
335  return false;
336  }
337  }
338  for (auto& edge : getMutableEdges()) {
339  if (!hasNode(edge->tail())) {
340  DEBUG_PRINT("Invalid tail on edge %p\n", edge);
341  return false;
342  }
343  if (!hasNode(edge->head())) {
344  DEBUG_PRINT("Invalid head on edge %p\n", edge);
345  return false;
346  }
347  }
348  return true;
349  }

351  // Swap two nodes.
352  // Any edge V -> N1 becomes V -> N2, and N1 -> V becomes N2 -> V.
353  void swapNodes(NodeRef n1, NodeRef n2) {
354  // First rectify the edges
355  for (auto& inEdge : n1->getInEdges()) {
356  inEdge->setHead(n2);
357  }
358  for (auto& outEdge : n1->getOutEdges()) {
359  outEdge->setTail(n2);
360  }
361  for (auto& inEdge : n2->getInEdges()) {
362  inEdge->setHead(n1);
363  }
364  for (auto& outEdge : n2->getOutEdges()) {
365  outEdge->setTail(n1);
366  }
367  // Then simply copy the edge vectors around
368  auto n1InEdges = n1->getInEdges();
369  auto n1OutEdges = n1->getOutEdges();
370  auto n2InEdges = n2->getInEdges();
371  auto n2OutEdges = n2->getOutEdges();
372 
373  n1->setOutEdges(n2OutEdges);
374  n1->setInEdges(n2InEdges);
375  n2->setOutEdges(n1OutEdges);
376  n2->setInEdges(n1InEdges);
