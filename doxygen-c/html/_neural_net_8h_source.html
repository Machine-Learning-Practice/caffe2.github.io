420 template <typename T, typename... Args>
421 void insertOp(
422  NNGraph& g,
423  NNGraph::NodeRef a,
424  NNGraph::NodeRef b,
425  Args... args) {
426  if (is<NeuralNetData>(a) && is<NeuralNetOperator>(b)) {
427  auto newNode = g.createNode(util::make_unique<T>(args...));
428  auto data = get<NeuralNetData>(a);
429  auto newData =
430  g.createNode(util::make_unique<Tensor>(data->getName() + "_"));
431  g.createEdge(a, newNode);
432  g.createEdge(newNode, newData);
433  g.createEdge(newData, b);
434  return;
435  }
436  if (is<NeuralNetOperator>(a) && is<NeuralNetData>(b)) {
437  auto newNode = g.createNode(util::make_unique<T>(args...));
438  auto data = get<NeuralNetData>(b);
439  auto newData =
440  g.createNode(util::make_unique<Tensor>(data->getName() + "_"));
441  g.createEdge(a, newData);
442  g.createEdge(newData, newNode);
443  g.createEdge(newNode, b);
444  return;
445  }
