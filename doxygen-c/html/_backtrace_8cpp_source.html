52 c10::optional<FrameInformation> parse_frame_information(
53  const std::string& frame_string) {
54  FrameInformation frame;
55 
56  // This is the function name in the CXX ABI mangled format, e.g. something
57  // like _Z1gv. Reference:
58  // https://itanium-cxx-abi.github.io/cxx-abi/abi.html#mangling
59  std::string mangled_function_name;
60 
61 #if defined(__GLIBCXX__)
62  // In GLIBCXX, `frame_string` follows the pattern
63  // `<object-file>(<mangled-function-name>+<offset-into-function>)
64  // [<return-address>]`
65 
66  auto function_name_start = frame_string.find("(");
67  if (function_name_start == std::string::npos) {
68  return c10::nullopt;
69  }
70  function_name_start += 1;
71 
72  auto offset_start = frame_string.find('+', function_name_start);
73  if (offset_start == std::string::npos) {
74  return c10::nullopt;
75  }
76  offset_start += 1;
77 
78  const auto offset_end = frame_string.find(')', offset_start);
79  if (offset_end == std::string::npos) {
80  return c10::nullopt;
81  }
82 
83  frame.object_file = frame_string.substr(0, function_name_start - 1);
84  frame.offset_into_function =
85  frame_string.substr(offset_start, offset_end - offset_start);
86 
87  // NOTE: We don't need to parse the return address because
88  // we already have it from the call to `backtrace()`.
89 
90  mangled_function_name = frame_string.substr(
91  function_name_start, (offset_start - 1) - function_name_start);
92 #elif defined(_LIBCPP_VERSION)
93  // In LIBCXX, The pattern is
94  // `<frame number> <object-file> <return-address> <mangled-function-name> +
95  // <offset-into-function>`
96  std::string skip;
97  std::istringstream input_stream(frame_string);
98  // operator>>() does not fail -- if the input stream is corrupted, the
99  // strings will simply be empty.
100  input_stream >> skip >> frame.object_file >> skip >> mangled_function_name >>
101  skip >> frame.offset_into_function;
102 #else
103 #warning Unknown standard library, backtraces may have incomplete debug information
104  return c10::nullopt;
105 #endif // defined(__GLIBCXX__)
106 
107  // Some system-level functions don't have sufficient debug information, so
108  // we'll display them as "<unknown function>". They'll still have a return
109  // address and other pieces of information.
110  if (mangled_function_name.empty()) {
111  frame.function_name = "<unknown function>";
112  return frame;
113  }
114 
115  frame.function_name = demangle(mangled_function_name.c_str());
116  return frame;
117 }

122 std::string get_backtrace(
123  size_t frames_to_skip,
124  size_t maximum_number_of_frames,
125  bool skip_python_frames) {
126 #if SUPPORTS_BACKTRACE
127 
128  // We always skip this frame (backtrace).
129  frames_to_skip += 1;
130 
131  std::vector<void*> callstack(
132  frames_to_skip + maximum_number_of_frames, nullptr);
133  // backtrace() gives us a list of return addresses in the current call stack.
134  // NOTE: As per man (3) backtrace it can never fail
135  // (http://man7.org/linux/man-pages/man3/backtrace.3.html).
136  auto number_of_frames =
137  ::backtrace(callstack.data(), static_cast<int>(callstack.size()));
138 
139  // Skip as many frames as requested. This is not efficient, but the sizes here
140  // are small and it makes the code nicer and safer.
141  for (; frames_to_skip > 0 && number_of_frames > 0;
142  --frames_to_skip, --number_of_frames) {
143  callstack.erase(callstack.begin());
144  }
145 
146  // `number_of_frames` is strictly less than the current capacity of
147  // `callstack`, so this is just a pointer subtraction and makes the subsequent
148  // code safer.
149  callstack.resize(static_cast<size_t>(number_of_frames));
150 
151  // `backtrace_symbols` takes the return addresses obtained from `backtrace()`
152  // and fetches string representations of each stack. Unfortunately it doesn't
153  // return a struct of individual pieces of information but a concatenated
154  // string, so we'll have to parse the string after. NOTE: The array returned
155  // by `backtrace_symbols` is malloc'd and must be manually freed, but not the
156  // strings inside the array.
157  std::unique_ptr<char*, std::function<void(char**)>> raw_symbols(
158  ::backtrace_symbols(callstack.data(), static_cast<int>(callstack.size())),
159  /*deleter=*/free);
160  const std::vector<std::string> symbols(
161  raw_symbols.get(), raw_symbols.get() + callstack.size());
162 
163  // The backtrace string goes into here.
164  std::ostringstream stream;
165 
166  // Toggles to true after the first skipped python frame.
167  bool has_skipped_python_frames = false;
168 
169  for (size_t frame_number = 0; frame_number < callstack.size();
170  ++frame_number) {
171  const auto frame = parse_frame_information(symbols[frame_number]);
172 
173  if (skip_python_frames && frame && is_python_frame(*frame)) {
174  if (!has_skipped_python_frames) {
175  stream << "<omitting python frames>\n";
176  has_skipped_python_frames = true;
177  }
178  continue;
179  }
180 
181  // frame #<number>:
182  stream << "frame #" << frame_number << ": ";
183 
184  if (frame) {
185  // <function_name> + <offset> (<return-address> in <object-file>)
186  stream << frame->function_name << " + " << frame->offset_into_function
187  << " (" << callstack[frame_number] << " in " << frame->object_file
188  << ")\n";
189  } else {
190  // In the edge-case where we couldn't parse the frame string, we can
191  // just use it directly (it may have a different format).
192  stream << symbols[frame_number] << "\n";
193  }
194  }
195 
196  return stream.str();
197 #else // !SUPPORTS_BACKTRACE
198  return "(no backtrace available)";
199 #endif // SUPPORTS_BACKTRACE
200 }
