29 def broadcast_tensors(*tensors):
30  r"""broadcast_tensors(*tensors) -> List of Tensors
31 
32  Broadcasts the given tensors according to :ref:`broadcasting-semantics`.
33 
34  Args:
35  *tensors: any number of tensors of the same type
36 
37  .. warning::
38 
39  More than one element of a broadcasted tensor may refer to a single
40  memory location. As a result, in-place operations (especially ones that
41  are vectorized) may result in incorrect behavior. If you need to write
42  to the tensors, please clone them first.
43 
44  Example::
45

52  tensor([[0, 1, 2],

54  """

83 def btriunpack(LU_data, LU_pivots, unpack_data=True, unpack_pivots=True):
84  r"""Unpacks the data and pivots from a batched LU factorization (btrifact) of a tensor.
85 
86  Returns a tuple of tensors as ``(the pivots, the L tensor, the U tensor)``.
87 
88  Arguments:
89  LU_data (Tensor): the packed LU factorization data
90  LU_pivots (Tensor): the packed LU factorization pivots
91  unpack_data (bool): flag indicating if the data should be unpacked
92  unpack_pivots (bool): flag indicating if the pivots should be unpacked
93 
94  Example::
95 
96  >>> A = torch.randn(2, 3, 3)
97  >>> A_LU, pivots = A.btrifact()
98  >>> P, A_L, A_U = torch.btriunpack(A_LU, pivots)
99  >>>
100  >>> # can recover A from factorization
101  >>> A_ = torch.bmm(P, torch.bmm(A_L, A_U))
102  """
103 
104  sz = LU_data.size(-1)
105 
106  if unpack_data:
107  U = LU_data.triu()
108  L = LU_data.tril()
109  L.diagonal(dim1=-2, dim2=-1).fill_(1)
110  else:
111  L = U = None
112 
113  if unpack_pivots:
114  P = torch.eye(sz, device=LU_data.device, dtype=LU_data.dtype).expand_as(LU_data).clone()
115  LU_pivots = LU_pivots - 1
116  for idx in product(*map(lambda x: list(range(x)), LU_data.shape[:-2])):
117  final_order = list(range(sz))
118  for k, j in enumerate(LU_pivots[idx]):
119  final_order[k], final_order[j] = final_order[j], final_order[k]
120  P[idx] = P[idx].index_select(1, torch.as_tensor(final_order, device=LU_pivots.device))
121  else:
122  P = None
123 
124  return P, L, U

127 def einsum(equation, *operands):
128  r"""einsum(equation, *operands) -> Tensor
129 
130 This function provides a way of computing multilinear expressions (i.e. sums of products) using the
131 Einstein summation convention.
132 
133 Args:
134  equation (string): The equation is given in terms of lower case letters (indices) to be associated
135  with each dimension of the operands and result. The left hand side lists the operands
136  dimensions, separated by commas. There should be one index letter per tensor dimension.
137  The right hand side follows after `->` and gives the indices for the output.
138  If the `->` and right hand side are omitted, it implicitly defined as the alphabetically
139  sorted list of all indices appearing exactly once in the left hand side.
140  The indices not apprearing in the output are summed over after multiplying the operands
141  entries.
142  If an index appears several times for the same operand, a diagonal is taken.
143  Ellipses `...` represent a fixed number of dimensions. If the right hand side is inferred,
144  the ellipsis dimensions are at the beginning of the output.
145  operands (list of Tensors): The operands to compute the Einstein sum of.
146 
147 Examples::
148 
149  >>> x = torch.randn(5)
150  >>> y = torch.randn(4)
151  >>> torch.einsum('i,j->ij', x, y) # outer product
152  tensor([[-0.0570, -0.0286, -0.0231, 0.0197],
153  [ 1.2616, 0.6335, 0.5113, -0.4351],
154  [ 1.4452, 0.7257, 0.5857, -0.4984],
155  [-0.4647, -0.2333, -0.1883, 0.1603],
156  [-1.1130, -0.5588, -0.4510, 0.3838]])
157 
158 
159  >>> A = torch.randn(3,5,4)
160  >>> l = torch.randn(2,5)
161  >>> r = torch.randn(2,4)
162  >>> torch.einsum('bn,anm,bm->ba', l, A, r) # compare torch.nn.functional.bilinear
163  tensor([[-0.3430, -5.2405, 0.4494],
164  [ 0.3311, 5.5201, -3.0356]])
165 
166 
167  >>> As = torch.randn(3,2,5)
168  >>> Bs = torch.randn(3,5,4)
169  >>> torch.einsum('bij,bjk->bik', As, Bs) # batch matrix multiplication
170  tensor([[[-1.0564, -1.5904, 3.2023, 3.1271],
171  [-1.6706, -0.8097, -0.8025, -2.1183]],
172 
173  [[ 4.2239, 0.3107, -0.5756, -0.2354],
174  [-1.4558, -0.3460, 1.5087, -0.8530]],
175 
176  [[ 2.8153, 1.8787, -4.3839, -1.2112],
177  [ 0.3728, -2.1131, 0.0921, 0.8305]]])
178 
179  >>> A = torch.randn(3, 3)
180  >>> torch.einsum('ii->i', A) # diagonal
181  tensor([-0.7825, 0.8291, -0.1936])
182 
183  >>> A = torch.randn(4, 3, 3)
184  >>> torch.einsum('...ii->...i', A) # batch diagonal
185  tensor([[-1.0864, 0.7292, 0.0569],
186  [-0.9725, -1.0270, 0.6493],
187  [ 0.5832, -1.1716, -1.5084],
188  [ 0.4041, -1.1690, 0.8570]])
189 
190  >>> A = torch.randn(2, 3, 4, 5)
191  >>> torch.einsum('...ij->...ji', A).shape # batch permute
192  torch.Size([2, 3, 5, 4])
193 """
194  if len(operands) == 1 and isinstance(operands[0], (list, tuple)):
195  # the old interface of passing the operands as one list argument
196  operands = operands[0]
197  return torch._C._VariableFunctions.einsum(equation, operands)

200 def isfinite(tensor):
201  r"""Returns a new tensor with boolean elements representing if each element is `Finite` or not.
202 
203  Arguments:
204  tensor (Tensor): A tensor to check
205 
206  Returns:
207  Tensor: A ``torch.ByteTensor`` containing a 1 at each location of finite elements and 0 otherwise
208 
209  Example::
210 
211  >>> torch.isfinite(torch.tensor([1, float('inf'), 2, float('-inf'), float('nan')]))
212  tensor([ 1, 0, 1, 0, 0], dtype=torch.uint8)
213  """
214  if not isinstance(tensor, torch.Tensor):
215  raise ValueError("The argument is not a tensor", str(tensor))
216 
217  # Support int input, nan and inf are concepts in floating point numbers.
218  # Numpy uses type 'Object' when the int overflows long, but we don't
219  # have a similar concept. It's safe to assume any created LongTensor doesn't
220  # overflow and it's finite.
221  if not tensor.is_floating_point():
222  return torch.ones_like(tensor, dtype=torch.uint8)
223  return (tensor == tensor) & (tensor.abs() != inf)

284 def stft(input, n_fft, hop_length=None, win_length=None, window=None,
285  center=True, pad_mode='reflect', normalized=False, onesided=True):
286  r"""Short-time Fourier transform (STFT).
287 
288  Ignoring the optional batch dimension, this method computes the following
289  expression:
290 
291  .. math::
292  X[m, \omega] = \sum_{k = 0}^{\text{win\_length-1}}%
293  \text{window}[k]\ \text{input}[m \times \text{hop\_length} + k]\ %
294  \exp\left(- j \frac{2 \pi \cdot \omega k}{\text{win\_length}}\right),
295 
296  where :math:`m` is the index of the sliding window, and :math:`\omega` is
297  the frequency that :math:`0 \leq \omega < \text{n\_fft}`. When
298  :attr:`onesided` is the default value ``True``,
299 
300  * :attr:`input` must be either a 1-D time sequence or a 2-D batch of time
301  sequences.
302 
303  * If :attr:`hop_length` is ``None`` (default), it is treated as equal to
304  ``floor(n_fft / 4)``.
305 
306  * If :attr:`win_length` is ``None`` (default), it is treated as equal to
307  :attr:`n_fft`.
308 
309  * :attr:`window` can be a 1-D tensor of size :attr:`win_length`, e.g., from
310  :meth:`torch.hann_window`. If :attr:`window` is ``None`` (default), it is
311  treated as if having :math:`1` everywhere in the window. If
312  :math:`\text{win\_length} < \text{n\_fft}`, :attr:`window` will be padded on
313  both sides to length :attr:`n_fft` before being applied.
314 
315  * If :attr:`center` is ``True`` (default), :attr:`input` will be padded on
316  both sides so that the :math:`t`-th frame is centered at time
317  :math:`t \times \text{hop\_length}`. Otherwise, the :math:`t`-th frame
318  begins at time :math:`t \times \text{hop\_length}`.
319 
320  * :attr:`pad_mode` determines the padding method used on :attr:`input` when
321  :attr:`center` is ``True``. See :meth:`torch.nn.functional.pad` for
322  all available options. Default is ``"reflect"``.
323 
324  * If :attr:`onesided` is ``True`` (default), only values for :math:`\omega`
325  in :math:`\left[0, 1, 2, \dots, \left\lfloor \frac{\text{n\_fft}}{2} \right\rfloor + 1\right]`
326  are returned because the real-to-complex Fourier transform satisfies the
327  conjugate symmetry, i.e., :math:`X[m, \omega] = X[m, \text{n\_fft} - \omega]^*`.
328 
329  * If :attr:`normalized` is ``True`` (default is ``False``), the function
330  returns the normalized STFT results, i.e., multiplied by :math:`(\text{frame\_length})^{-0.5}`.
331 
332  Returns the real and the imaginary parts together as one tensor of size
333  :math:`(* \times N \times T \times 2)`, where :math:`*` is the optional
334  batch size of :attr:`input`, :math:`N` is the number of frequencies where
335  STFT is applied, :math:`T` is the total number of frames used, and each pair
336  in the last dimension represents a complex number as the real part and the
337  imaginary part.
338 
339  .. warning::
340  This function changed signature at version 0.4.1. Calling with the
341  previous signature may cause error or return incorrect result.
342 
343  Arguments:
344  input (Tensor): the input tensor
345  n_fft (int): size of Fourier transform
346  hop_length (int, optional): the distance between neighboring sliding window
347  frames. Default: ``None`` (treated as equal to ``floor(n_fft / 4)``)
348  win_length (int, optional): the size of window frame and STFT filter.
349  Default: ``None`` (treated as equal to :attr:`n_fft`)
350  window (Tensor, optional): the optional window function.
351  Default: ``None`` (treated as window of all :math:`1` s)
352  center (bool, optional): whether to pad :attr:`input` on both sides so
353  that the :math:`t`-th frame is centered at time :math:`t \times \text{hop\_length}`.
354  Default: ``True``
355  pad_mode (string, optional): controls the padding method used when
356  :attr:`center` is ``True``. Default: ``"reflect"``
357  normalized (bool, optional): controls whether to return the normalized STFT results
358  Default: ``False``
359  onesided (bool, optional): controls whether to return half of results to
360  avoid redundancy Default: ``True``
361 
362  Returns:
363  Tensor: A tensor containing the STFT result with shape described above
364 
365  """
366  # TODO: after having proper ways to map Python strings to ATen Enum, move
367  # this and F.pad to ATen.
368  if center:
369  signal_dim = input.dim()
370  extended_shape = [1] * (3 - signal_dim) + list(input.size())
371  pad = int(n_fft // 2)
372  input = F.pad(input.view(extended_shape), (pad, pad), pad_mode)
373  input = input.view(input.shape[-signal_dim:])
374  return torch._C._VariableFunctions.stft(input, n_fft, hop_length, win_length, window, normalized, onesided)

454 def tensordot(a, b, dims=2):
455  r"""Returns a contraction of a and b over multiple dimensions.
456 
457  :attr:`tensordot` implements a generalizes the matrix product.
458 
459  Args:
460  a (Tensor): Left tensor to contract
461  b (Tensor): Right tensor to contract
462  dims (int or tuple of two lists of integers): number of dimensions to
463  contract or explicit lists of dimensions for :attr:`a` and
464  :attr:`b` respectively
465 
466  When called with an integer argument :attr:`dims` = :math:`d`, and the number of
467  dimensions of :attr:`a` and :attr:`b` is :math:`m` and :math:`n`, respectively,
468  it computes
469 
470  .. math::
471  r_{i_0,...,i_{m-d}, i_d,...,i_n}
472  = \sum_{k_0,...,k_{d-1}} a_{i_0,...,i_{m-d},k_0,...,k_{d-1}} \times b_{k_0,...,k_{d-1}, i_d,...,i_n}.
473 
474  When called with :attr:`dims` of the list form, the given dimensions will be contracted
475  in place of the last :math:`d` of :attr:`a` and the first :math:`d` of :math:`b`. The sizes
476  in these dimensions must match, but :attr:`tensordot` will deal with broadcasted
477  dimensions.
478 
479  Examples::
480 
481  >>> a = torch.arange(60.).reshape(3, 4, 5)
482  >>> b = torch.arange(24.).reshape(4, 3, 2)
483  >>> torch.tensordot(a, b, dims=([1, 0], [0, 1]))
484  tensor([[4400., 4730.],
485  [4532., 4874.],
486  [4664., 5018.],
487  [4796., 5162.],
488  [4928., 5306.]])
489 
490  >>> a = torch.randn(3, 4, 5, device='cuda')
491  >>> b = torch.randn(4, 5, 6, device='cuda')
492  >>> c = torch.tensordot(a, b, dims=2).cpu()
493  tensor([[ 8.3504, -2.5436, 6.2922, 2.7556, -1.0732, 3.2741],
494  [ 3.3161, 0.0704, 5.0187, -0.4079, -4.3126, 4.8744],
495  [ 0.8223, 3.9445, 3.2168, -0.2400, 3.4117, 1.7780]])
496 
497  """
498  if isinstance(dims, (list, tuple)) or \
499  (isinstance(dims, torch.Tensor) and dims.numel() > 1):
500  dims_a, dims_b = dims
501  else:
502  if isinstance(dims, torch.Tensor):
503  dims = dims.item()
504  dims_a = list(range(-dims, 0))
505  dims_b = list(range(dims))
506  return torch._C._VariableFunctions.tensordot(a, b, dims_a, dims_b)

540 def norm(input, p="fro", dim=None, keepdim=False, out=None, dtype=None):
541  r"""Returns the matrix norm or vector norm of a given tensor.
542 
543  Args:
544  input (Tensor): the input tensor
545  p (int, float, inf, -inf, 'fro', 'nuc', optional): the order of norm. Default: ``'fro'``
546  The following norms can be calculated:
547 
548  ===== ============================ ==========================
549  ord matrix norm vector norm
550  ===== ============================ ==========================
551  None Frobenius norm 2-norm
552  'fro' Frobenius norm --
553  'nuc' nuclear norm --
554  Other as vec norm when dim is None sum(abs(x)**ord)**(1./ord)
555  ===== ============================ ==========================
556 
557  dim (int, 2-tuple of ints, 2-list of ints, optional): If it is an int,
558  vector norm will be calculated, if it is 2-tuple of ints, matrix norm
559  will be calculated. If the value is None, matrix norm will be calculated
560  when the input tensor only has two dimensions, vector norm will be
561  calculated when the input tensor only has one dimension. If the input
562  tensor has more than two dimensions, the vector norm will be applied to
563  last dimension.
564  keepdim (bool, optional): whether the output tensors have :attr:`dim`
565  retained or not. Ignored if :attr:`dim` = ``None`` and
566  :attr:`out` = ``None``. Default: ``False``
