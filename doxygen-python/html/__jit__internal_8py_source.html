35 def createResolutionCallback(frames_up=0):
36  """
37  Creates a function which, given a string variable name,
38  returns the value of the variable in the scope of the caller of
39  the function which called createResolutionCallback (by default).
40 
41  This is used to enable access in-scope Python variables inside
42  TorchScript fragments.
43 
44  frames_up is number of additional frames to go up on the stack.
45  The default value is 0, which correspond to the frame of the caller
46  of createResolutionCallback. Also for example, if frames_up is set
47  to 1, then the frame of the caller's caller of createResolutionCallback
48  will be taken.
49 
50  For example, the following program prints 2::
51 
52  def bar():
53  cb = createResolutionCallback(1)
54  print(cb("foo"))
55 
56  def baz():
57  foo = 2
58  bar()
59 
60  baz()
61  """
62  frame = inspect.currentframe()
63  i = 0
64  while i < frames_up + 1:
65  frame = frame.f_back
66  i += 1
67 
68  f_locals = frame.f_locals
69  f_globals = frame.f_globals
70 
71  def env(key):
72  if key in f_locals:
73  return f_locals[key]
74  elif key in f_globals:
75  return f_globals[key]
76  elif hasattr(builtins, key):
77  return getattr(builtins, key)
78  else:
79  return None
80 
81  return env

114 def boolean_dispatch(arg_name, arg_index, default, if_true, if_false, module_name, func_name):
115  """
116  Dispatches to either of 2 weak script functions based on a boolean argument.
117  In TorchScript, the boolean argument must be constant so that the correct
118  function to use can be determined at compile time.
119  """
120  if compiled_weak_fns.get(if_true) is None or compiled_weak_fns.get(if_false) is None:
121  raise RuntimeError("both functions must be weak script")
122 
123  def fn(*args, **kwargs):
124  dispatch_flag = False
125  if arg_name in kwargs:
126  dispatch_flag = kwargs[arg_name]
127  elif arg_index < len(args):
128  dispatch_flag = args[arg_index]
129 
130  if dispatch_flag:
131  return if_true(*args, **kwargs)
132  else:
133  return if_false(*args, **kwargs)
134 
135  if if_true.__doc__ is None and if_false.__doc__ is not None:
136  doc = if_false.__doc__
137  if_true.__doc__ = doc
138  elif if_false.__doc__ is None and if_true.__doc__ is not None:
139  doc = if_true.__doc__
140  if_false.__doc__ = doc
141  elif if_false.__doc__ is None and if_true.__doc__ is None:
142  # neither function has a docstring
143  doc = None
144  else:
145  raise RuntimeError("only one function can have a docstring")
146  fn.__doc__ = doc
147 
148  if module_name is not None:
149  fn.__module__ = module_name
150  if func_name is not None:
151  fn.__name__ = func_name
152 
153  boolean_dispatched[fn] = {
154  "if_true": if_true,
155  "if_false": if_false,
156  "index": arg_index,
157  "default": default,
158  "arg_name": arg_name
