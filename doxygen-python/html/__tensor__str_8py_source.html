88  else:

92  # no valid number, do nothing
93  return

95  # Convert to double for easy calculation. HalfTensor overflows with 1e8, and there's no div() on CPU.
96  nonzero_finite_abs = nonzero_finite_vals.abs().double()
97  nonzero_finite_min = nonzero_finite_abs.min().double()
98  nonzero_finite_max = nonzero_finite_abs.max().double()
99 
100  for value in nonzero_finite_vals:
101  if value != torch.ceil(value):
102  self.int_mode = False
103  break

105  if self.int_mode:
106  # in int_mode for floats, all numbers are integers, and we append a decimal to nonfinites
107  # to indicate that the tensor is of floating type. add 1 to the len to account for this.
108  if nonzero_finite_max / nonzero_finite_min > 1000. or nonzero_finite_max > 1.e8:
109  self.sci_mode = True
110  for value in nonzero_finite_vals:
111  value_str = ('{{:.{}e}}').format(PRINT_OPTS.precision).format(value)
112  self.max_width = max(self.max_width, len(value_str))
113  else:
114  for value in nonzero_finite_vals:
115  value_str = ('{:.0f}').format(value)
116  self.max_width = max(self.max_width, len(value_str) + 1)
117  else:
118  # Check if scientific representation should be used.

121  or nonzero_finite_min < 1.e-4:

207 def _add_suffixes(tensor_str, suffixes, indent, force_newline):
208  tensor_strs = [tensor_str]
209  last_line_len = len(tensor_str) - tensor_str.rfind('\n') + 1
210  for suffix in suffixes:
211  suffix_len = len(suffix)
212  if force_newline or last_line_len + suffix_len + 2 > PRINT_OPTS.linewidth:
213  tensor_strs.append(',\n' + ' ' * indent + suffix)
214  last_line_len = indent + suffix_len
215  force_newline = False

246  if not torch._C._is_default_type_cuda():
247  if self.device.type == 'cuda':
248  suffixes.append('device=\'' + str(self.device) + '\'')
249  else:
250  if self.device.type == 'cpu' or torch.cuda.current_device() != self.device.index:
251  suffixes.append('device=\'' + str(self.device) + '\'')
252 
253  has_default_dtype = self.dtype == torch.get_default_dtype() or self.dtype == torch.int64
254 
255  if self.is_sparse:
256  suffixes.append('size=' + str(tuple(self.shape)))
257  suffixes.append('nnz=' + str(self._nnz()))
258  if not has_default_dtype:
259  suffixes.append('dtype=' + str(self.dtype))
260  indices_prefix = 'indices=tensor('
261  indices = self._indices().detach()
262  indices_str = _tensor_str(indices, indent + len(indices_prefix))
263  if indices.numel() == 0:
264  indices_str += ', size=' + str(tuple(indices.shape))
265  values_prefix = 'values=tensor('
266  values = self._values().detach()
267  values_str = _tensor_str(values, indent + len(values_prefix))
268  if values.numel() == 0:
269  values_str += ', size=' + str(tuple(values.shape))
270  tensor_str = indices_prefix + indices_str + '),\n' + ' ' * indent + values_prefix + values_str + ')'
271  else:
272  if self.numel() == 0 and not self.is_sparse:
273  # Explicitly print the shape if it is not (0,), to match NumPy behavior
274  if self.dim() != 1:
275  suffixes.append('size=' + str(tuple(self.shape)))
276 
277  # In an empty tensor, there are no elements to infer if the dtype
278  # should be int64, so it must be shown explicitly.
279  if self.dtype != torch.get_default_dtype():
280  suffixes.append('dtype=' + str(self.dtype))
281  tensor_str = '[]'
282  else:

284  suffixes.append('dtype=' + str(self.dtype))
285  tensor_str = _tensor_str(self, indent)
286 
287  if self.layout != torch.strided:
288  suffixes.append('layout=' + str(self.layout))
289 
290  if self.grad_fn is not None:
291  name = type(self.grad_fn).__name__
292  if name == 'CppFunction':
293  name = self.grad_fn.name().rsplit('::', 1)[-1]
294  suffixes.append('grad_fn=<{}>'.format(name))
295  elif self.requires_grad:
296  suffixes.append('requires_grad=True')
